027package gov.nist.secauto.metaschema.codegen;
028
029import com.squareup.javapoet.ClassName;
030import com.squareup.javapoet.TypeSpec;
031
032import gov.nist.secauto.metaschema.model.common.IAssemblyDefinition;
033import gov.nist.secauto.metaschema.model.common.IFieldDefinition;
034import gov.nist.secauto.metaschema.model.common.IFlagContainer;
035import gov.nist.secauto.metaschema.model.common.IFlagInstance;
036
037import java.io.IOException;
038import java.nio.file.Path;
039import java.util.Collection;
040
041import edu.umd.cs.findbugs.annotations.NonNull;
042import edu.umd.cs.findbugs.annotations.Nullable;
043
044public interface IModelDefinitionTypeInfo extends IDefinitionTypeInfo {
045
046  @NonNull
047  static IModelDefinitionTypeInfo newTypeInfo(@NonNull IFlagContainer definition,
048      @NonNull ITypeResolver typeResolver) {
049    IModelDefinitionTypeInfo retval;
050    switch (definition.getModelType()) {
051    case ASSEMBLY:
052      retval = IAssemblyDefinitionTypeInfo.newTypeInfo((IAssemblyDefinition) definition, typeResolver);
053      break;
054    case FIELD:
055      retval = IFieldDefinitionTypeInfo.newTypeInfo((IFieldDefinition) definition, typeResolver);
056      break;
057    default:
058      throw new UnsupportedOperationException(
059          String.format("Generation of child classes for %s definitions is unsupported",
060              definition.getModelType().name()));
061    }
062    return retval;
063  }
064
065  @Override
066  IFlagContainer getDefinition();
067
068  /**
069   * Get the class type information for the base class of the generated class, .
070   *
071   * @return the type information or {@code null} if no base class is configured
072   */
073  @Nullable
074  ClassName getBaseClassName();
075
076  /**
077   * Gets the class type information for the object definition for which this class is being
078   * generated.
079   *
080   * @return the class's type information
081   */
082  @NonNull
083  ClassName getClassName();
084
085  @Nullable
086  IFlagInstanceTypeInfo getFlagInstanceTypeInfo(@NonNull IFlagInstance instance);
087
088  @NonNull
089  Collection<IFlagInstanceTypeInfo> getFlagInstanceTypeInfos();
090
091  /**
092   * Generates the associated Java class and saves it using the provided file.
093   *
094   * @param dir
095   *          the directory to generate the class in
096   * @return the qualified class name for the generated class
097   * @throws IOException
098   *           if a build error occurred while generating the class
099   */
100  @NonNull
101  IGeneratedDefinitionClass generateClass(@NonNull Path dir) throws IOException;
102
103  /**
104   * This method is responsible for generating the Java class using a builder that is returned for
105   * further customization.
106   *
107   * @return the class definition for the generated class
108   * @throws IOException
109   *           if a build error occurred while generating the class
110   */
111  @NonNull
112  TypeSpec generateChildClass() throws IOException;
113}
