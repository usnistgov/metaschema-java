033import gov.nist.secauto.metaschema.model.common.MetaschemaException;
034
035import org.apache.logging.log4j.LogManager;
036import org.apache.logging.log4j.Logger;
037
038import java.io.IOException;
039import java.net.MalformedURLException;
040import java.net.URL;
041import java.net.URLClassLoader;
042import java.nio.file.Path;
043import java.security.AccessController;
044import java.security.PrivilegedAction;
045import java.util.ArrayList;
046import java.util.LinkedList;
047import java.util.List;
048import java.util.stream.Collectors;
049
050import javax.tools.DiagnosticCollector;
051import javax.tools.JavaCompiler;
052import javax.tools.JavaFileManager;
053import javax.tools.JavaFileObject;
054import javax.tools.StandardJavaFileManager;
055import javax.tools.ToolProvider;
056
057import edu.umd.cs.findbugs.annotations.NonNull;

059/**
060 * This class provides methods to generate and dynamically compile Java code based on a Metaschema.
061 * The {@link #getClassLoader(Path, ClassLoader)} method can be used to get a {@link ClassLoader}
062 * for Java code previously generated by this class.
063 */
064public final class MetaschemaCompilerHelper {
065  private static final Logger LOGGER = LogManager.getLogger(MetaschemaCompilerHelper.class);
066  private static final MetaschemaLoader LOADER = new MetaschemaLoader();
067
068  static {
069    LOADER.allowEntityResolution();
070  }
071
072  private MetaschemaCompilerHelper() {
073    // disable construction
074  }
075
076  @NonNull
077  public static IProduction compileMetaschema(@NonNull Path metaschemaPath, @NonNull Path classDir)
078      throws IOException, MetaschemaException {
079    return compileMetaschema(metaschemaPath, classDir, new DefaultBindingConfiguration());
080  }
081
082  @NonNull
083  public static IProduction compileMetaschema(@NonNull Path metaschemaPath, @NonNull Path classDir,
084      @NonNull IBindingConfiguration bindingConfiguration)
085      throws IOException, MetaschemaException {
086    IMetaschema metaschema = LOADER.load(metaschemaPath);
087    return compileMetaschema(metaschema, classDir, bindingConfiguration);
088  }
089
090  @NonNull
091  public static IProduction compileMetaschema(@NonNull IMetaschema metaschema, @NonNull Path classDir)
092      throws IOException {
093    return compileMetaschema(metaschema, classDir, new DefaultBindingConfiguration());
094  }
095
096  @NonNull
097  public static IProduction compileMetaschema(@NonNull IMetaschema metaschema, @NonNull Path classDir,
098      @NonNull IBindingConfiguration bindingConfiguration) throws IOException {
099    IProduction production = JavaGenerator.generate(metaschema, classDir, bindingConfiguration);
100    List<IGeneratedClass> classesToCompile = production.getGeneratedClasses().collect(Collectors.toList());
101
102    DiagnosticCollector<JavaFileObject> diagnostics = new DiagnosticCollector<>();
103    if (!compileGeneratedClasses(classesToCompile, diagnostics, classDir)) {
104      if (LOGGER.isErrorEnabled()) {
105        LOGGER.error(diagnostics.getDiagnostics().toString());
106      }
107      throw new IllegalStateException(String.format("failed to compile classes: %s",
108          classesToCompile.stream()
109              .map(clazz -> clazz.getClassName().canonicalName())
110              .collect(Collectors.joining(","))));
111    }
112    return production;
113  }
114
115  @SuppressWarnings("null")
116  @NonNull
117  public static ClassLoader getClassLoader(@NonNull final Path classDir, @NonNull final ClassLoader parent) {
118    return AccessController.doPrivileged(new PrivilegedAction<URLClassLoader>() {
119      @Override
120      public URLClassLoader run() {
121        try {
122          return new URLClassLoader(new URL[] { classDir.toUri().toURL() }, parent);
123        } catch (MalformedURLException ex) {
124          throw new IllegalStateException("unable to configure class loader", ex);
125        }
126      }
127    });
128  }
129
130  private static boolean compile(
131      JavaCompiler compiler,
132      JavaFileManager fileManager,
133      DiagnosticCollector<JavaFileObject> diagnostics,
134      List<JavaFileObject> compilationUnits,
135      Path classDir) {
136    List<String> options = new LinkedList<>();
137    // options.add("-verbose");
138    // options.add("-g");
139    options.add("-d");
140    options.add(classDir.toString());
141    options.add("-classpath");
142    options.add(System.getProperty("java.class.path"));
143
144    JavaCompiler.CompilationTask task
145        = compiler.getTask(null, fileManager, diagnostics, options, null, compilationUnits);
146    return task.call();
147  }
148
149  private static boolean compileGeneratedClasses(
150      List<IGeneratedClass> classesToCompile,
151      DiagnosticCollector<JavaFileObject> diagnostics,
152      Path classDir) throws IOException {
153    JavaCompiler compiler = ToolProvider.getSystemJavaCompiler();
154
155    try (StandardJavaFileManager fileManager = compiler.getStandardFileManager(diagnostics, null, null)) {
156
157      List<JavaFileObject> compilationUnits = new ArrayList<>(classesToCompile.size());
158      for (IGeneratedClass generatedClass : classesToCompile) {
159        compilationUnits.add(fileManager.getJavaFileObjects(generatedClass.getClassFile()).iterator().next());
160      }
161
162      return compile(compiler, fileManager, diagnostics, compilationUnits, classDir);
163    }
164  }
165}
