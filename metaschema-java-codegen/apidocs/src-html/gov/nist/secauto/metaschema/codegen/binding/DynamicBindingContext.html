027package gov.nist.secauto.metaschema.codegen.binding;
028
029import gov.nist.secauto.metaschema.binding.DefaultBindingContext;
030import gov.nist.secauto.metaschema.binding.IBindingMatcher;
031import gov.nist.secauto.metaschema.binding.model.IAssemblyClassBinding;
032import gov.nist.secauto.metaschema.codegen.IProduction;
033import gov.nist.secauto.metaschema.codegen.MetaschemaCompilerHelper;
034import gov.nist.secauto.metaschema.model.common.IAssemblyDefinition;
035import gov.nist.secauto.metaschema.model.common.IFlagContainer;
036import gov.nist.secauto.metaschema.model.common.IMetaschema;
037import gov.nist.secauto.metaschema.model.common.util.ObjectUtils;
038
039import java.io.IOException;
040import java.nio.file.Files;
041import java.nio.file.Path;
042
043import javax.xml.namespace.QName;
044
045import edu.umd.cs.findbugs.annotations.NonNull;
046import edu.umd.cs.findbugs.annotations.Nullable;
047
048public class DynamicBindingContext
049    extends DefaultBindingContext {
050
051  @SuppressWarnings("PMD.UseProperClassLoader") // false positive
052  @NonNull
053  public static DynamicBindingContext forMetaschema(
054      @NonNull IMetaschema metaschema,
055      @Nullable Path tempPath) throws IOException {
056    Path classDir;
057    if (tempPath == null) {
058      classDir = Files.createTempDirectory("classes-");
059    } else {
060      classDir = Files.createTempDirectory(tempPath, "classes-");
061    }
062    classDir.toFile().deleteOnExit();
063
064    IProduction production = MetaschemaCompilerHelper.compileMetaschema(metaschema, classDir);
065    return new DynamicBindingContext(production,
066        MetaschemaCompilerHelper.getClassLoader(classDir,
067            ObjectUtils.notNull(Thread.currentThread().getContextClassLoader())));
068  }
069
070  public DynamicBindingContext(@NonNull IProduction production, ClassLoader classLoader) {
071    production.getDefinitionProductionsAsStream()
072        .filter(
073            definitionProduction -> {
074              boolean retval = false;
075              IFlagContainer definition = definitionProduction.getDefinition();
076              if (definition instanceof IAssemblyDefinition) {
077                IAssemblyDefinition assembly = (IAssemblyDefinition) definition;
078                if (assembly.isRoot()) {
079                  retval = true;
080                }
081              }
082              return retval;
083            })
084        .map(
085            definitionProduction -> {
086              IAssemblyDefinition definition = (IAssemblyDefinition) definitionProduction.getDefinition();
087              try {
088                @SuppressWarnings("unchecked") Class<IAssemblyClassBinding> clazz
089                    = ObjectUtils.notNull(
090                        (Class<IAssemblyClassBinding>) classLoader
091                            .loadClass(
092                                definitionProduction.getGeneratedClass().getClassName().reflectionName()));
093                return new DynamicBindingMatcher(
094                    definition,
095                    clazz);
096              } catch (ClassNotFoundException ex) {
097                throw new IllegalStateException(
098                    ex);
099              }
100            })
101        .forEachOrdered(
102            matcher -> registerBindingMatcher(
103                ObjectUtils.notNull(
104                    matcher)));
105  }
106
107  private static class DynamicBindingMatcher implements IBindingMatcher {
108    private final IAssemblyDefinition definition;
109    private final Class<IAssemblyClassBinding> clazz;
110
111    public DynamicBindingMatcher(@NonNull IAssemblyDefinition definition, @NonNull Class<IAssemblyClassBinding> clazz) {
112      this.definition = definition;
113      this.clazz = clazz;
114    }
115
116    protected IAssemblyDefinition getDefinition() {
117      return definition;
118    }
119
120    protected Class<IAssemblyClassBinding> getClazz() {
121      return clazz;
122    }
123
124    @SuppressWarnings("null")
125    @NonNull
126    protected QName getRootQName() {
127      return getDefinition().getRootXmlQName();
128    }
129
130    @SuppressWarnings("null")
131    @NonNull
132    protected String getRootJsonName() {
133      return getDefinition().getRootJsonName();
134    }
135
136    @Override
137    public Class<?> getBoundClassForXmlQName(QName rootQName) {
138      return getRootQName().equals(
139          rootQName) ? getClazz() : null;
140    }
141
142    @Override
143    public Class<?> getBoundClassForJsonName(String rootName) {
144      return getRootJsonName().equals(
145          rootName) ? getClazz() : null;
146    }
147
148  }
149}
