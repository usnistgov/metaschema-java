77    public static Object getDefaultValue(Class<?> annotation, String member) {
78      Method method;
79      try {
80        method = annotation.getDeclaredMethod(member);
81      } catch (NoSuchMethodException ex) {
82        throw new IllegalArgumentException(ex);
83      }
84      Object retval;
85      try {
86        retval = method.getDefaultValue();
87      } catch (TypeNotPresentException ex) {
88        retval = null;
89      }
90      return retval;
91    }

93    private static void buildConstraint(Class<?> annotationType, AnnotationSpec.Builder annotation,
94        IConstraint constraint) {
95      String id = constraint.getId();
96      if (id != null) {
97        annotation.addMember("id", "$S", id);
98      }
99  
100     annotation.addMember("level", "$T.$L", IConstraint.Level.class, constraint.getLevel());
101 
102     MetapathExpression target = constraint.getTarget();
103     String path = target.getPath();
104     if (!path.equals(getDefaultValue(annotationType, "target"))) {
105       annotation.addMember("target", "$S", path);
106     }
107   }

109   public static void applyAllowedValuesConstraints(AnnotationSpec.Builder annotation,
110       List<@NotNull ? extends IAllowedValuesConstraint> constraints) {
111     for (IAllowedValuesConstraint constraint : constraints) {
112       AnnotationSpec.Builder constraintAnnotation = AnnotationSpec.builder(AllowedValues.class);
113       buildConstraint(AllowedValues.class, constraintAnnotation, constraint);
114 
115       boolean isAllowedOther = constraint.isAllowedOther();
116       if (!Boolean.valueOf(isAllowedOther).equals(getDefaultValue(AllowedValues.class, "allowOthers"))) {
117         constraintAnnotation.addMember("allowOthers", "$L", isAllowedOther);
118       }
119 
120       for (IAllowedValue value : constraint.getAllowedValues().values()) {
121         AnnotationSpec.Builder valueAnnotation = AnnotationSpec.builder(AllowedValue.class);
122 
123         valueAnnotation.addMember("value", "$S", value.getValue());
124         valueAnnotation.addMember("description", "$S", value.getDescription().toMarkdown());
125 
126         constraintAnnotation.addMember("values", "$L", valueAnnotation.build());
127       }
128       annotation.addMember("allowedValues", "$L", constraintAnnotation.build());
129     }
130   }

146   private static void buildKeyFields(@NotNull Builder constraintAnnotation,
147       @NotNull List<@NotNull ? extends IKeyField> keyFields) {
148     for (IKeyField key : keyFields) {
149       AnnotationSpec.Builder keyAnnotation = AnnotationSpec.builder(KeyField.class);
150 
151       MetapathExpression target = key.getTarget();
152       String path = target.getPath();
153       if (!path.equals(getDefaultValue(KeyField.class, "target"))) {
154         keyAnnotation.addMember("target", "$S", path);
155       }
156 
157       Pattern pattern = key.getPattern();
158       if (pattern != null) {
159         keyAnnotation.addMember("pattern", "$S", pattern.pattern());
160       }
161 
162       MarkupMultiline remarks = key.getRemarks();
163       if (remarks != null) {
164         keyAnnotation.addMember("remarks", "$S", remarks.toMarkdown());
165       }
166 
167       constraintAnnotation.addMember("keyFields", "$L", keyAnnotation.build());
168     }
169   }

171   public static void applyMatchesConstraints(AnnotationSpec.Builder annotation,
172       List<@NotNull ? extends IMatchesConstraint> constraints) {
173     for (IMatchesConstraint constraint : constraints) {
174       AnnotationSpec.Builder constraintAnnotation = AnnotationSpec.builder(Matches.class);
175       buildConstraint(Matches.class, constraintAnnotation, constraint);
176 
177       Pattern pattern = constraint.getPattern();
178       if (pattern != null) {
179         constraintAnnotation.addMember("pattern", "$S", pattern.pattern());
180       }
181 
182       IJavaTypeAdapter<?> dataType = constraint.getDataType();
183       if (dataType != null) {
184         constraintAnnotation.addMember("typeAdapter", "$T.class", dataType.getClass());
185       }
186       annotation.addMember("matches", "$L", constraintAnnotation.build());
187     }
188   }

190   public static void applyExpectConstraints(AnnotationSpec.Builder annotation,
191       List<@NotNull ? extends IExpectConstraint> constraints) {
192     for (IExpectConstraint constraint : constraints) {
193       AnnotationSpec.Builder constraintAnnotation = AnnotationSpec.builder(Expect.class);
194 
195       buildConstraint(Expect.class, constraintAnnotation, constraint);
196 
197       MetapathExpression test = constraint.getTest();
198       constraintAnnotation.addMember("test", "$S", test.getPath());
199 
200       if (constraint.getMessage() != null) {
201         constraintAnnotation.addMember("message", "$S", constraint.getMessage());
202       }
203 
204       annotation.addMember("expect", "$L", constraintAnnotation.build());
205     }
206   }

236   public static void applyHasCardinalityConstraints(IAssemblyDefinition definition, AnnotationSpec.Builder annotation,
237       List<@NotNull ? extends ICardinalityConstraint> constraints) {
238     for (ICardinalityConstraint constraint : constraints) {
239       Integer minOccurs = constraint.getMinOccurs();
240       Integer maxOccurs = constraint.getMaxOccurs();
241 
242       IInstanceSet instanceSet = definition.evaluateMetapathInstances(constraint.getTarget());
243       Collection<? extends IInstance> instances = instanceSet.getInstances();
244       if (instances.size() == 1) {
245         IInstance instance = instances.iterator().next();
246         if (instance instanceof IModelInstance) {
247           IModelInstance modelInstance = (IModelInstance) instance;
248 
249           if (minOccurs != null) {
250             if (minOccurs == modelInstance.getMinOccurs()) {
251               if (LOGGER.isWarnEnabled()) {
252                 LOGGER.warn(String.format(
253                     "Definition '%s' has min-occurs=%d cardinality constraint targeting '%s' that is redundant with a"
254                         + " targeted instance named '%s' that requires min-occurs=%d",
255                     definition.getName(), minOccurs, constraint.getTarget().getPath(),
256                     modelInstance instanceof INamedInstance ? ((INamedInstance) modelInstance).getName() : "no name",
257                     modelInstance.getMinOccurs()));
258               }
259             } else if (minOccurs < modelInstance.getMinOccurs()) {
260               if (LOGGER.isWarnEnabled()) {
261                 LOGGER.warn(String.format(
262                     "Definition '%s' has min-occurs=%d cardinality constraint targeting '%s' that conflicts with a"
263                         + " targeted instance named '%s' that requires min-occurs=%d",
264                     definition.getName(), minOccurs, constraint.getTarget().getPath(),
265                     modelInstance instanceof INamedInstance ? ((INamedInstance) modelInstance).getName() : "no name",
266                     modelInstance.getMinOccurs()));
267               }
268             }
269           }
270 
271           if (maxOccurs != null) {
272             if (maxOccurs == modelInstance.getMaxOccurs()) {
273               if (LOGGER.isWarnEnabled()) {
274                 LOGGER.warn(String.format(
275                     "Definition '%s' has max-occurs=%d cardinality constraint targeting '%s' that is redundant with a"
276                         + " targeted instance named '%s' that requires max-occurs=%d",
277                     definition.getName(), maxOccurs, constraint.getTarget().getPath(),
278                     modelInstance instanceof INamedInstance ? ((INamedInstance) modelInstance).getName() : "no name",
279                     modelInstance.getMaxOccurs()));
280               }
281             } else if (maxOccurs < modelInstance.getMaxOccurs()) {
282               if (LOGGER.isWarnEnabled()) {
283                 LOGGER.warn(String.format(
284                     "Definition '%s' has max-occurs=%d cardinality constraint targeting '%s' that conflicts with a"
285                         + " targeted instance named '%s' that requires max-occurs=%d",
286                     definition.getName(), maxOccurs, constraint.getTarget().getPath(),
287                     modelInstance instanceof INamedInstance ? ((INamedInstance) modelInstance).getName() : "no name",
288                     modelInstance.getMaxOccurs()));
289               }
290             }
291           }
292         } else if (LOGGER.isWarnEnabled()) {
293           LOGGER.warn(String.format(
294               "Definition '%s' has min-occurs=%d cardinality constraint targeting '%s' that is not a model instance",
295               definition.getName(), minOccurs, constraint.getTarget().getPath()));
296         }
297       }
298 
299       AnnotationSpec.Builder constraintAnnotation = AnnotationSpec.builder(HasCardinality.class);
300 
301       buildConstraint(HasCardinality.class, constraintAnnotation, constraint);
302 
303       if (minOccurs != null && !minOccurs.equals(getDefaultValue(HasCardinality.class, "minOccurs"))) {
304         constraintAnnotation.addMember("minOccurs", "$L", minOccurs);
305       }
306 
307       if (maxOccurs != null && !maxOccurs.equals(getDefaultValue(HasCardinality.class, "maxOccurs"))) {
308         constraintAnnotation.addMember("maxOccurs", "$L", maxOccurs);
309       }
310 
311       annotation.addMember("hasCardinality", "$L", constraintAnnotation.build());
312     }
