78    public static Object getDefaultValue(Class<?> annotation, String member) {
79      Method method;
80      try {
81        method = annotation.getDeclaredMethod(member);
82      } catch (NoSuchMethodException ex) {
83        throw new IllegalArgumentException(ex);
84      }
85      Object retval;
86      try {
87        retval = method.getDefaultValue();
88      } catch (TypeNotPresentException ex) {
89        retval = null;
90      }
91      return retval;
92    }

94    private static void buildConstraint(Class<?> annotationType, AnnotationSpec.Builder annotation,
95        IConstraint constraint) {
96      String id = constraint.getId();
97      if (id != null) {
98        annotation.addMember("id", "$S", id);
99      }
100 
101     annotation.addMember("level", "$T.$L", IConstraint.Level.class, constraint.getLevel());
102 
103     MetapathExpression target = constraint.getTarget();
104     String path = target.getPath();
105     if (!path.equals(getDefaultValue(annotationType, "target"))) {
106       annotation.addMember("target", "$S", path);
107     }
108   }

110   public static void applyAllowedValuesConstraints(AnnotationSpec.Builder annotation,
111       List<@NotNull ? extends IAllowedValuesConstraint> constraints) {
112     for (IAllowedValuesConstraint constraint : constraints) {
113       AnnotationSpec.Builder constraintAnnotation = AnnotationSpec.builder(AllowedValues.class);
114       buildConstraint(AllowedValues.class, constraintAnnotation, constraint);
115 
116       boolean isAllowedOther = constraint.isAllowedOther();
117       if (!Boolean.valueOf(isAllowedOther).equals(getDefaultValue(AllowedValues.class, "allowOthers"))) {
118         constraintAnnotation.addMember("allowOthers", "$L", isAllowedOther);
119       }
120 
121       for (IAllowedValue value : constraint.getAllowedValues().values()) {
122         AnnotationSpec.Builder valueAnnotation = AnnotationSpec.builder(AllowedValue.class);
123 
124         valueAnnotation.addMember("value", "$S", value.getValue());
125         valueAnnotation.addMember("description", "$S", value.getDescription().toMarkdown());
126 
127         constraintAnnotation.addMember("values", "$L", valueAnnotation.build());
128       }
129       annotation.addMember("allowedValues", "$L", constraintAnnotation.build());
130     }
131   }

143       annotation.addMember("indexHasKey", "$L", constraintAnnotation.build());
144     }

147   private static void buildKeyFields(@NotNull Builder constraintAnnotation,
148       @NotNull List<@NotNull ? extends IKeyField> keyFields) {
149     for (IKeyField key : keyFields) {
150       AnnotationSpec.Builder keyAnnotation = AnnotationSpec.builder(KeyField.class);
151 
152       MetapathExpression target = key.getTarget();
153       String path = target.getPath();
154       if (!path.equals(getDefaultValue(KeyField.class, "target"))) {
155         keyAnnotation.addMember("target", "$S", path);
156       }
157 
158       Pattern pattern = key.getPattern();
159       if (pattern != null) {
160         keyAnnotation.addMember("pattern", "$S", pattern.pattern());
161       }
162 
163       MarkupMultiline remarks = key.getRemarks();
164       if (remarks != null) {
165         keyAnnotation.addMember("remarks", "$S", remarks.toMarkdown());
166       }
167 
168       constraintAnnotation.addMember("keyFields", "$L", keyAnnotation.build());
169     }
170   }

172   public static void applyMatchesConstraints(AnnotationSpec.Builder annotation,
173       List<@NotNull ? extends IMatchesConstraint> constraints) {
174     for (IMatchesConstraint constraint : constraints) {
175       AnnotationSpec.Builder constraintAnnotation = AnnotationSpec.builder(Matches.class);
176       buildConstraint(Matches.class, constraintAnnotation, constraint);
177 
178       Pattern pattern = constraint.getPattern();
179       if (pattern != null) {
180         constraintAnnotation.addMember("pattern", "$S", pattern.pattern());
181       }
182 
183       IDataTypeAdapter<?> dataType = constraint.getDataType();
184       if (dataType != null) {
185         constraintAnnotation.addMember("typeAdapter", "$T.class", dataType.getClass());
186       }
187       annotation.addMember("matches", "$L", constraintAnnotation.build());
188     }
189   }

191   public static void applyExpectConstraints(AnnotationSpec.Builder annotation,
192       List<@NotNull ? extends IExpectConstraint> constraints) {
193     for (IExpectConstraint constraint : constraints) {
194       AnnotationSpec.Builder constraintAnnotation = AnnotationSpec.builder(Expect.class);
195 
196       buildConstraint(Expect.class, constraintAnnotation, constraint);
197 
198       MetapathExpression test = constraint.getTest();
199       constraintAnnotation.addMember("test", "$S", test.getPath());
200 
201       if (constraint.getMessage() != null) {
202         constraintAnnotation.addMember("message", "$S", constraint.getMessage());
203       }
204 
205       annotation.addMember("expect", "$L", constraintAnnotation.build());
206     }
207   }

237   public static void applyHasCardinalityConstraints(
238       @NotNull IAssemblyDefinition definition,
239       @NotNull AnnotationSpec.Builder annotation,
240       @NotNull List<@NotNull ? extends ICardinalityConstraint> constraints) {
241     for (ICardinalityConstraint constraint : constraints) {
242       Integer minOccurs = constraint.getMinOccurs();
243       Integer maxOccurs = constraint.getMaxOccurs();
244 
245       IAssemblyNodeItem definitionNodeItem = DefaultNodeItemFactory.instance().newAssemblyNodeItem(definition, null);
246 
247       ISequence<? extends IDefinitionNodeItem> instanceSet = constraint.matchTargets(definitionNodeItem);
248 
249       for (IDefinitionNodeItem item : instanceSet.asList()) {
250         INamedInstance instance = item.getInstance();
251         if (instance instanceof INamedModelInstance) {
252           INamedModelInstance modelInstance = (INamedModelInstance) instance;
253 
254           if (minOccurs != null) {
255             if (minOccurs == modelInstance.getMinOccurs()) {
256               if (LOGGER.isWarnEnabled()) {
257                 LOGGER.warn(String.format(
258                     "Definition '%s' has min-occurs=%d cardinality constraint targeting '%s' that is redundant with a"
259                         + " targeted instance named '%s' that requires min-occurs=%d",
260                     definition.getName(), minOccurs, constraint.getTarget().getPath(),
261                     modelInstance.getName(),
262                     modelInstance.getMinOccurs()));
263               }
264             } else if (minOccurs < modelInstance.getMinOccurs()) {
265               if (LOGGER.isWarnEnabled()) { // NOPMD - readability
266                 LOGGER.warn(String.format(
267                     "Definition '%s' has min-occurs=%d cardinality constraint targeting '%s' that conflicts with a"
268                         + " targeted instance named '%s' that requires min-occurs=%d",
269                     definition.getName(), minOccurs, constraint.getTarget().getPath(),
270                     modelInstance.getName(),
271                     modelInstance.getMinOccurs()));
272               }
273             }
274           }
275 
276           if (maxOccurs != null) {
277             if (maxOccurs == modelInstance.getMaxOccurs()) {
278               if (LOGGER.isWarnEnabled()) {
279                 LOGGER.warn(String.format(
280                     "Definition '%s' has max-occurs=%d cardinality constraint targeting '%s' that is redundant with a"
281                         + " targeted instance named '%s' that requires max-occurs=%d",
282                     definition.getName(), maxOccurs, constraint.getTarget().getPath(),
283                     modelInstance.getName(),
284                     modelInstance.getMaxOccurs()));
285               }
286             } else if (maxOccurs < modelInstance.getMaxOccurs()) {
287               if (LOGGER.isWarnEnabled()) { // NOPMD - readability
288                 LOGGER.warn(String.format(
289                     "Definition '%s' has max-occurs=%d cardinality constraint targeting '%s' that conflicts with a"
290                         + " targeted instance named '%s' that requires max-occurs=%d",
291                     definition.getName(), maxOccurs, constraint.getTarget().getPath(),
292                     modelInstance.getName(),
293                     modelInstance.getMaxOccurs()));
294               }
295             }
296           }
297         } else if (LOGGER.isWarnEnabled()) {
298           LOGGER.warn(String.format(
299               "Definition '%s' has min-occurs=%d cardinality constraint targeting '%s' that is not a model instance",
300               definition.getName(), minOccurs, constraint.getTarget().getPath()));
301         }
302       }
303 
304       AnnotationSpec.Builder constraintAnnotation = AnnotationSpec.builder(HasCardinality.class);
305 
306       buildConstraint(HasCardinality.class, constraintAnnotation, constraint);
307 
308       if (minOccurs != null && !minOccurs.equals(getDefaultValue(HasCardinality.class, "minOccurs"))) {
309         constraintAnnotation.addMember("minOccurs", "$L", minOccurs);
310       }
311 
312       if (maxOccurs != null && !maxOccurs.equals(getDefaultValue(HasCardinality.class, "maxOccurs"))) {
313         constraintAnnotation.addMember("maxOccurs", "$L", maxOccurs);
314       }
315 
316       annotation.addMember("hasCardinality", "$L", constraintAnnotation.build());
317     }
