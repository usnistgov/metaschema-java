71    /**
72     * Retrieve the binding configuration for the provided {@code definition}.
73     * 
74     * @param definition
75     *          the definition to get the config for
76     * @return the binding configuration or {@code null} if there is not configuration
77     */
78    @Nullable
79    public IDefinitionBindingConfiguration
80        getBindingConfigurationForDefinition(@NonNull IModelDefinition definition) {
81      String metaschemaUri = ObjectUtils.notNull(definition.getContainingMetaschema().getLocation().toString());
82      String definitionName = definition.getName();
83  
84      MetaschemaBindingConfiguration metaschemaConfig = getMetaschemaBindingConfiguration(metaschemaUri);
85  
86      IDefinitionBindingConfiguration retval = null;
87      if (metaschemaConfig != null) {
88        switch (definition.getModelType()) {
89        case ASSEMBLY:
90          retval = metaschemaConfig.getAssemblyDefinitionBindingConfig(definitionName);
91          break;
92        case FIELD:
93          retval = metaschemaConfig.getFieldDefinitionBindingConfig(definitionName);
94          break;
95        default:
96          throw new UnsupportedOperationException(
97              String.format("Unsupported definition type '%s'", definition.getModelType()));
98        }
99      }
100     return retval;
101   }

135   /**
136    * Binds an XML namespace, which is normally associated with one or more Metaschema, with a provided
137    * Java package name.
138    * 
139    * @param namespace
140    *          an XML namespace URI
141    * @param packageName
142    *          the package name to associate with the namespace
143    * @throws IllegalStateException
144    *           if the binding configuration is changing a previously changed namespace to package
145    *           binding
146    */
147   public void addModelBindingConfig(String namespace, String packageName) {
148     if (namespaceToPackageNameMap.containsKey(namespace)) {
149       String oldPackageName = namespaceToPackageNameMap.get(namespace);
150       if (!oldPackageName.equals(packageName)) {
151         throw new IllegalStateException(
152             String.format("Attempt to redefine existing package name '%s' to '%s' for namespace '%s'", oldPackageName,
153                 packageName, namespace));
154       } // else the same package name, so do nothing
155     } else {
156       namespaceToPackageNameMap.put(namespace, packageName);
157     }
158   }

160   /**
161    * Based on the current binding configuration, generate a Java package name for the provided
162    * namespace. If the namespace is already mapped, such as through the use of
163    * {@link #addModelBindingConfig(String, String)}, then the provided package name will be used. If
164    * the namespace is not mapped, then the namespace URI will be translated into a Java package name.
165    * 
166    * @param namespace
167    *          the namespace to generate a Java package name for
168    * @return a Java package name
169    */
170   @NonNull
171   protected String getPackageNameForNamespace(@NonNull String namespace) {
172     String packageName = namespaceToPackageNameMap.get(namespace);
173     if (packageName == null) {
174       packageName = ClassUtils.toPackageName(namespace);
175     }
176     return packageName;
177   }

247   /**
248    * Load the binding configuration from the provided {@code resource}.
249    * 
250    * @param resource
251    *          the configuration resource
252    * @throws IOException
253    *           if an error occurred while reading the {@code resource}
254    */
255   public void load(URL resource) throws IOException {
256     MetaschemaBindingsDocument xml;
257     try {
258       xml = MetaschemaBindingsDocument.Factory.parse(resource);
259     } catch (XmlException ex) {
260       throw new IOException(ex);
261     }
262 
263     MetaschemaBindingsType bindings = xml.getMetaschemaBindings();
264 
265     for (ModelBindingType model : bindings.getModelBindingList()) {
266       processModelBindingConfig(model);
267     }
268 
269     for (MetaschemaBindingType metaschema : bindings.getMetaschemaBindingList()) {
270       try {
271         processMetaschemaBindingConfig(resource, metaschema);
272       } catch (MalformedURLException | URISyntaxException ex) {
273         throw new IOException(ex);
274       }
275     }
276   }

278   private void processModelBindingConfig(ModelBindingType model) {
279     String namespace = model.getNamespace();
280 
281     if (model.isSetJava()) {
282       JavaModelBindingType java = model.getJava();
283       if (java.isSetUsePackageName()) {
284         addModelBindingConfig(namespace, java.getUsePackageName());
285       }
286     }
287   }

289   private void processMetaschemaBindingConfig(URL configResource, MetaschemaBindingType metaschema)
290       throws MalformedURLException, URISyntaxException {
291     String href = metaschema.getHref();
292     URL metaschemaUrl = new URL(configResource, href);
293     String metaschemaUri = ObjectUtils.notNull(metaschemaUrl.toURI().toString());
294 
295     MetaschemaBindingConfiguration metaschemaConfig = getMetaschemaBindingConfiguration(metaschemaUri);
296     if (metaschemaConfig == null) {
297       metaschemaConfig = new MetaschemaBindingConfiguration();
298       addMetaschemaBindingConfiguration(metaschemaUri, metaschemaConfig);
299     }
300     for (ObjectDefinitionBindingType assemblyBinding : metaschema.getDefineAssemblyBindingList()) {
301       String name = ObjectUtils.requireNonNull(assemblyBinding.getName());
302       IDefinitionBindingConfiguration config = metaschemaConfig.getAssemblyDefinitionBindingConfig(name);
303       config = processDefinitionBindingConfiguration(config, assemblyBinding);
304       metaschemaConfig.addAssemblyDefinitionBindingConfig(name, config);
305     }
306 
307     for (ObjectDefinitionBindingType fieldBinding : metaschema.getDefineFieldBindingList()) {
308       String name = ObjectUtils.requireNonNull(fieldBinding.getName());
309       IDefinitionBindingConfiguration config = metaschemaConfig.getFieldDefinitionBindingConfig(name);
310       config = processDefinitionBindingConfiguration(config, fieldBinding);
311       metaschemaConfig.addFieldDefinitionBindingConfig(name, config);
312     }
313   }

315   @NonNull
316   private static IMutableDefinitionBindingConfiguration processDefinitionBindingConfiguration(
317       @Nullable IDefinitionBindingConfiguration oldConfig,
318       @NonNull ObjectDefinitionBindingType objectDefinitionBinding) {
319     IMutableDefinitionBindingConfiguration config;
320     if (oldConfig != null) {
321       config = new DefaultDefinitionBindingConfiguration(oldConfig);
322     } else {
323       config = new DefaultDefinitionBindingConfiguration();
324     }
325 
326     if (objectDefinitionBinding.isSetJava()) {
327       JavaObjectDefinitionBindingType java = objectDefinitionBinding.getJava();
328       if (java.isSetUseClassName()) {
329         config.setClassName(ObjectUtils.notNull(java.getUseClassName()));
330       }
331 
332       if (java.isSetExtendBaseClass()) {
333         config.setQualifiedBaseClassName(ObjectUtils.notNull(java.getExtendBaseClass()));
334       }
335 
336       for (String interfaceName : java.getImplementInterfaceList()) {
337         config.addInterfaceToImplement(ObjectUtils.notNull(interfaceName));
338       }
339     }
340     return config;
341   }

343   public static class MetaschemaBindingConfiguration {
344     private final Map<String, IDefinitionBindingConfiguration> assemblyBindingConfigs = new ConcurrentHashMap<>();
345     private final Map<String, IDefinitionBindingConfiguration> fieldBindingConfigs = new ConcurrentHashMap<>();

347     private MetaschemaBindingConfiguration() {

350     /**
