27  package gov.nist.secauto.metaschema.codegen;
28  
29  import com.squareup.javapoet.FieldSpec;
30  import com.squareup.javapoet.MethodSpec;
31  import com.squareup.javapoet.ParameterSpec;
32  import com.squareup.javapoet.TypeSpec;
33  
34  import gov.nist.secauto.metaschema.model.common.IFlagContainer;
35  import gov.nist.secauto.metaschema.model.common.util.ObjectUtils;
36  
37  import java.util.Set;
38  
39  import javax.lang.model.element.Modifier;
40  
41  import edu.umd.cs.findbugs.annotations.NonNull;
42  
43  abstract class AbstractTypeInfo<PARENT extends IDefinitionTypeInfo> implements ITypeInfo {
44    @NonNull
45    private final PARENT parentDefinition;
46    private String propertyName;
47    private String fieldName;
48  
49    public AbstractTypeInfo(@NonNull PARENT parentDefinition) {
50      this.parentDefinition = parentDefinition;
51    }
52  
53    @NonNull
54    protected PARENT getParentDefinitionTypeInfo() {
55      return parentDefinition;
56    }
57  
58    /**
59     * The property name of the instance, which must be unique within the class.
60     *
61     * @return the name
62     */
63    @Override
64    @NonNull
65    public String getPropertyName() {
66      synchronized (this) {
67        if (this.propertyName == null) {
68          String name = ClassUtils.toPropertyName(getBaseName());
69          IDefinitionTypeInfo parent = getParentDefinitionTypeInfo();
70  
71          // first check if a property already exists with the same name
72  
73          if (parent.hasPropertyWithName(name)) {
74            // append an integer value to make the name unique
75            String newName;
76            int index = 1;
77            do {
78              newName = ClassUtils.toPropertyName(name + Integer.toString(index));
79              index++;
80            } while (parent.hasPropertyWithName(newName));
81            name = newName;
82          }
83          this.propertyName = name;
84        }
85        return ObjectUtils.notNull(this.propertyName);
86      }
87    }
88  
89    /**
90     * Gets the name of the Java field for this property item.
91     *
92     * @return the Java field name
93     */
94    @Override
95    @NonNull
96    public final String getJavaFieldName() {
97      synchronized (this) {
98        if (this.fieldName == null) {
99          this.fieldName = "_" + ClassUtils.toVariableName(getPropertyName());
100       }
101       return ObjectUtils.notNull(this.fieldName);
102     }
103   }
104 
105   @Override
106   public Set<IFlagContainer> build(@NonNull TypeSpec.Builder builder, ITypeResolver typeResolver) {
107     FieldSpec.Builder field = FieldSpec.builder(getJavaFieldType(), getJavaFieldName())
108         .addModifiers(Modifier.PRIVATE);
109     assert field != null;
110 
111     final Set<IFlagContainer> retval = buildField(field);
112 
113     FieldSpec valueField = ObjectUtils.notNull(field.build());
114     builder.addField(valueField);
115 
116     {
117       MethodSpec.Builder method = MethodSpec.methodBuilder("get" + getPropertyName())
118           .returns(getJavaFieldType())
119           .addModifiers(Modifier.PUBLIC);
120       assert method != null;
121       buildGetter(method, valueField);
122       builder.addMethod(method.build());
123     }
124 
125     {
126       ParameterSpec valueParam = ParameterSpec.builder(getJavaFieldType(), "value").build();
127 
128       MethodSpec.Builder method = MethodSpec.methodBuilder("set" + getPropertyName())
129           .addModifiers(Modifier.PUBLIC)
130           .addParameter(valueParam);
131       assert method != null;
132       buildSetter(method, valueParam, valueField);
133       builder.addMethod(method.build());
134     }
135 
136     buildExtraMethods(builder, valueField, typeResolver);
137     return retval;
138   }
139 
140   /**
141    * This method can be implemented by subclasses to create additional methods.
142    *
143    * @param builder
144    *          the class builder
145    * @param valueField
146    *          the field corresponding to this property
147    * @param typeResolver
148    *          the resolver used to get type information
149    */
150   protected void buildExtraMethods( // NOPMD - intentional
151       @NonNull TypeSpec.Builder builder,
152       @NonNull FieldSpec valueField,
153       @NonNull ITypeResolver typeResolver) {
154     // do nothing by default
155   }
156 
157   /**
158    * Generate the Java field associated with this property.
159    *
160    * @param builder
161    *          the field builder
162    * @return the set of definitions used by this field
163    */
164   @NonNull
165   protected abstract Set<IFlagContainer> buildField(@NonNull FieldSpec.Builder builder);
166 
167   /**
168    * Generate the getter for the property.
169    *
170    * @param builder
171    *          the method builder
172    * @param valueField
173    *          the field containing the value to get
174    */
175   protected void buildGetter(@NonNull MethodSpec.Builder builder, @NonNull FieldSpec valueField) {
176     builder.addStatement("return $N", valueField);
177   }
178 
179   /**
180    * Generate the setter for the property.
181    *
182    * @param builder
183    *          the method builder
184    * @param valueParam
185    *          the parameter value to set
186    * @param valueField
187    *          the field containing the value to set
188    */
189   protected void buildSetter(MethodSpec.Builder builder, ParameterSpec valueParam, FieldSpec valueField) {
190     builder.addStatement("$N = $N", valueField, valueParam);
191   }
192 }
