27  package gov.nist.secauto.metaschema.codegen;
28  
29  import com.squareup.javapoet.TypeName;
30  import com.squareup.javapoet.TypeSpec;
31  
32  import gov.nist.secauto.metaschema.model.common.IModelDefinition;
33  
34  import java.util.Set;
35  
36  import edu.umd.cs.findbugs.annotations.NonNull;
37  
38  public interface ITypeInfo {
39  
40    /**
41     * Get the name to use for the property. If the property is a collection type, then this will be the
42     * group-as name, else this will be the use name or the name if not use name is set.
43     * 
44     * @return the name
45     */
46    @NonNull
47    String getBaseName();
48  
49    /**
50     * The name to use for Java constructs that refer to the item. This is used for when a field is
51     * collection-based and there is a need to refer to a single item, such as in an add/remove method
52     * name.
53     * 
54     * @return the item base name
55     */
56    @NonNull
57    default String getItemBaseName() {
58      return getBaseName();
59    }
60  
61    /**
62     * Get the Java property name for the property.
63     * 
64     * @return the Java property name
65     */
66    @NonNull
67    String getPropertyName();
68  
69    /**
70     * Gets the name of the Java field for this property.
71     * 
72     * @return the Java field name
73     */
74    @NonNull
75    String getJavaFieldName();
76  
77    /**
78     * Gets the type of the associated Java field for the property.
79     * 
80     * @return the Java type for the field
81     */
82    @NonNull
83    TypeName getJavaFieldType();
84  
85    /**
86     * Gets the type of the property's item.
87     * 
88     * @return the Java type for the item
89     */
90    @NonNull
91    default TypeName getJavaItemType() {
92      return getJavaFieldType();
93    }
94  
95    /**
96     * Build the Java class data for the property.
97     * 
98     * @param builder
99     *          the class builder
100    * @param typeResolver
101    *          the resolver used to get type information
102    * @return the set of additional child definitions that need to be built
103    */
104   @NonNull
105   Set<IModelDefinition> build(@NonNull TypeSpec.Builder builder, @NonNull ITypeResolver typeResolver);
106 }
