063/**
064 * Goal which generates Java source files for a given set of Metaschema definitions.
065 */
066@Mojo(name = "generate-schemas", defaultPhase = LifecyclePhase.GENERATE_RESOURCES)
067public class GenerateSchemaMojo
068    extends AbstractMetaschemaMojo {
069  public enum SchemaFormat {
070    XSD,
071    JSON_SCHEMA;
072  }
073
074  @NonNull
075  private static final String STALE_FILE_NAME = "generateSschemaStaleFile";
076
077  @NonNull
078  private static final XmlSchemaGenerator XML_SCHEMA_GENERATOR = new XmlSchemaGenerator();
079  @NonNull
080  private static final JsonSchemaGenerator JSON_SCHEMA_GENERATOR = new JsonSchemaGenerator();
081
082  /**
083   * Specifies the formats of the schemas to generate. Multiple formats can be supplied and this
084   * plugin will generate a schema for each of the desired formats.
085   * <p>
086   * A format is specified by supplying one of the following values in a &lt;format&gt; subelement:
087   * <ul>
088   * <li><em>json</em> - Creates a JSON Schema</li>
089   * <li><em>xsd</em> - Creates an XML Schema Definition</li>
090   * </ul>
091   */
092  @Parameter
093  private List<String> formats;
094
095  /**
096   * If enabled, definitions that are defined inline will be generated as inline types. If disabled,
097   * definitions will always be generated as global types.
098   */
099  @Parameter(defaultValue = "true")
100  private boolean inlineDefinitions = true;
101
102  /**
103   * If enabled, child definitions of a choice that are defined inline will be generated as inline
104   * types. If disabled, child definitions of a choice will always be generated as global types. This
105   * option will only be used if <code>inlineDefinitions</code> is also enabled.
106   */
107  @Parameter(defaultValue = "false")
108  private boolean inlineChoiceDefinitions = false;
109
110  /**
111   * Determine if inlining definitions is required.
112   *
113   * @return {@code true} if inlining definitions is required, or {@code false} otherwise
114   */
115  protected boolean isInlineDefinitions() {
116    return inlineDefinitions;
117  }
118
119  /**
120   * Determine if inlining choice definitions is required.
121   *
122   * @return {@code true} if inlining choice definitions is required, or {@code false} otherwise
123   */
124  protected boolean isInlineChoiceDefinitions() {
125    return inlineChoiceDefinitions;
126  }
127
128  /**
129   * <p>
130   * Gets the last part of the stale filename.
131   * </p>
132   * <p>
133   * The full stale filename will be generated by pre-pending
134   * {@code "." + getExecution().getExecutionId()} to this staleFileName.
135   *
136   * @return the stale filename postfix
137   */
138  @Override
139  protected String getStaleFileName() {
140    return STALE_FILE_NAME;
141  }
142
143  /**
144   * Performs schema generation using the provided Metaschemas.
145   *
146   * @param metaschemaCollection
147   *          the Metaschemas to generate the schema for
148   * @throws MojoExecutionException
149   *           if an error occurred during generation
150   */
151  protected void generate(@NonNull Set<IMetaschema> metaschemaCollection) throws MojoExecutionException {
152    IMutableConfiguration<SchemaGenerationFeature<?>> schemaGenerationConfig
153        = new DefaultConfiguration<>();
154
155    if (isInlineDefinitions()) {
156      schemaGenerationConfig.enableFeature(SchemaGenerationFeature.INLINE_DEFINITIONS);
157    } else {
158      schemaGenerationConfig.disableFeature(SchemaGenerationFeature.INLINE_DEFINITIONS);
159    }
160
161    if (isInlineChoiceDefinitions()) {
162      schemaGenerationConfig.enableFeature(SchemaGenerationFeature.INLINE_CHOICE_DEFINITIONS);
163    } else {
164      schemaGenerationConfig.disableFeature(SchemaGenerationFeature.INLINE_CHOICE_DEFINITIONS);
165    }
166
167    Set<SchemaFormat> schemaFormats;
168    if (formats != null) {
169      schemaFormats = ObjectUtils.notNull(EnumSet.noneOf(SchemaFormat.class));
170      for (String format : formats) {
171        switch (format.toLowerCase(Locale.ROOT)) {
172        case "xsd":
173          schemaFormats.add(SchemaFormat.XSD);
174          break;
175        case "json":
176          schemaFormats.add(SchemaFormat.JSON_SCHEMA);
177          break;
178        default:
179          throw new IllegalStateException("Unsupported schema format: " + format);
180        }
181      }
182    } else {
183      schemaFormats = ObjectUtils.notNull(EnumSet.allOf(SchemaFormat.class));
184    }
185
186    Path outputDirectory = ObjectUtils.notNull(getOutputDirectory().toPath());
187    for (IMetaschema metaschema : metaschemaCollection) {
188      getLog().info(String.format("Processing metaschema: %s", metaschema.getLocation()));
189      if (metaschema.getExportedRootAssemblyDefinitions().isEmpty()) {
190        continue;
191      }
192
193      generateSchemas(metaschema, schemaGenerationConfig, outputDirectory, schemaFormats);
194    }
195  }
196
197  private static void generateSchemas(
198      @NonNull IMetaschema metaschema,
199      @NonNull IConfiguration<SchemaGenerationFeature<?>> schemaGenerationConfig,
200      @NonNull Path outputDirectory,
201      @NonNull Set<SchemaFormat> schemaFormats) throws MojoExecutionException {
202
203    String shortName = metaschema.getShortName();
204
205    if (schemaFormats.contains(SchemaFormat.XSD)) {
206      try { // XML Schema
207        String filename = String.format("%s_schema.xsd", shortName);
208        Path xmlSchema = ObjectUtils.notNull(outputDirectory.resolve(filename));
209        generateSchema(metaschema, schemaGenerationConfig, xmlSchema, XML_SCHEMA_GENERATOR);
210      } catch (Exception ex) {
211        throw new MojoExecutionException("Unable to generate XML schema.", ex);
212      }
213    }
214
215    if (schemaFormats.contains(SchemaFormat.JSON_SCHEMA)) {
216      try { // JSON Schema
217        String filename = String.format("%s_schema.json", shortName);
218        Path xmlSchema = ObjectUtils.notNull(outputDirectory.resolve(filename));
219        generateSchema(metaschema, schemaGenerationConfig, xmlSchema, JSON_SCHEMA_GENERATOR);
220      } catch (Exception ex) {
221        throw new MojoExecutionException("Unable to generate JSON schema.", ex);
222      }
223    }
224  }
225
226  private static void generateSchema(
227      @NonNull IMetaschema metaschema,
228      @NonNull IConfiguration<SchemaGenerationFeature<?>> schemaGenerationConfig,
229      @NonNull Path schemaPath,
230      @NonNull ISchemaGenerator generator) throws IOException {
231    try (@SuppressWarnings("resource") Writer writer = ObjectUtils.notNull(Files.newBufferedWriter(
232        schemaPath,
233        StandardCharsets.UTF_8,
234        StandardOpenOption.CREATE,
235        StandardOpenOption.WRITE,
236        StandardOpenOption.TRUNCATE_EXISTING))) {
237      generator.generateFromMetaschema(metaschema, writer, schemaGenerationConfig);
238    }
239  }
240
241  @Override
242  public void execute() throws MojoExecutionException {
243    File staleFile = getStaleFile();
244    try {
245      staleFile = staleFile.getCanonicalFile();
246    } catch (IOException ex) {
247      getLog().warn("Unable to resolve canonical path to stale file. Treating it as not existing.", ex);
248    }
249
250    boolean generate;
251    if (shouldExecutionBeSkipped()) {
252      getLog().debug(String.format("Schema generation is configured to be skipped. Skipping."));
253      generate = false;
254    } else if (!staleFile.exists()) {
255      getLog().info(String.format("Stale file '%s' doesn't exist! Generating source files.", staleFile.getPath()));
256      generate = true;
257    } else {
258      generate = isGenerationRequired();
259    }
260
261    if (generate) {
262      File outputDir = getOutputDirectory();
263      getLog().debug(String.format("Using outputDirectory: %s", outputDir.getPath()));
264
265      if (!outputDir.exists()) {
266        if (!outputDir.mkdirs()) {
267          throw new MojoExecutionException("Unable to create output directory: " + outputDir);
268        }
269      }
270
271      // generate Java sources based on provided Metaschema sources
272      final MetaschemaLoader loader = new MetaschemaLoader();
273      loader.allowEntityResolution();
274      final Set<IMetaschema> metaschemaCollection = new HashSet<>();
275      for (File source : getSources().collect(Collectors.toList())) {
276        getLog().info("Using metaschema source: " + source.getPath());
277        IMetaschema metaschema;
278        try {
279          metaschema = loader.load(source);
280        } catch (MetaschemaException | IOException ex) {
281          throw new MojoExecutionException("Loading of metaschema failed", ex);
282        }
283        metaschemaCollection.add(metaschema);
284      }
285
286      generate(metaschemaCollection);
287
288      // create the stale file
289      if (!staleFileDirectory.exists()) {
290        if (!staleFileDirectory.mkdirs()) {
291          throw new MojoExecutionException("Unable to create output directory: " + staleFileDirectory);
292        }
293      }
294      try (OutputStream os
295          = Files.newOutputStream(staleFile.toPath(), StandardOpenOption.CREATE, StandardOpenOption.WRITE,
296              StandardOpenOption.TRUNCATE_EXISTING)) {
297        os.close();
298        getLog().info("Created stale file: " + staleFile);
299      } catch (IOException ex) {
300        throw new MojoExecutionException("Failed to write stale file: " + staleFile.getPath(), ex);
301      }
302
303      // for m2e
304      getBuildContext().refresh(getOutputDirectory());
305    }
306
307    // // add generated sources to Maven
308    // try {
309    // getMavenProject()..addCompileSourceRoot(getOutputDirectory().getCanonicalFile().getPath());
310    // } catch (IOException ex) {
311    // throw new MojoExecutionException("Unable to add output directory to maven sources.", ex);
312    // }
313  }
