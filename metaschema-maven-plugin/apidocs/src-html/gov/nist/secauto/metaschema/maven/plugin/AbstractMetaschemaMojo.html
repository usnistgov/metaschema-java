027package gov.nist.secauto.metaschema.maven.plugin;
028
029import org.apache.maven.plugin.AbstractMojo;
030import org.apache.maven.plugin.MojoExecution;
031import org.apache.maven.plugins.annotations.Component;
032import org.apache.maven.plugins.annotations.Parameter;
033import org.apache.maven.project.MavenProject;
034import org.codehaus.plexus.util.DirectoryScanner;
035import org.sonatype.plexus.build.incremental.BuildContext;
036
037import java.io.File;
038import java.net.URI;
039import java.util.Objects;
040import java.util.stream.Collectors;
041import java.util.stream.Stream;
042
043import edu.umd.cs.findbugs.annotations.SuppressFBWarnings;
044
045public abstract class AbstractMetaschemaMojo
046    extends AbstractMojo {
047  private static final String SYSTEM_FILE_ENCODING_PROPERTY = "file.encoding";
048  private static final String[] DEFAULT_INCLUDES = { "**/*.xml" };
049
050  /**
051   * The Maven project context.
052   *
053   * @parameter default-value="${project}"
054   * @required
055   * @readonly
056   */
057  @Parameter(defaultValue = "${project}", required = true, readonly = true)
058  MavenProject mavenProject;
059
060  /**
061   * This will be injected if this plugin is executed as part of the standard Maven lifecycle. If the
062   * mojo is directly invoked, this parameter will not be injected.
063   */
064  @Parameter(defaultValue = "${mojoExecution}", readonly = true)
065  private MojoExecution mojoExecution;
066
067  @Component
068  private BuildContext buildContext;
069
070  /**
071   * <p>
072   * The directory where the staleFile is found. The staleFile is used to determine if re-generation
073   * of generated Java classes is needed, by recording when the last build occurred.
074   * </p>
075   * <p>
076   * This directory is expected to be located within the <code>${project.build.directory}</code>, to
077   * ensure that code (re)generation occurs after cleaning the project.
078   * </p>
079   */
080  @Parameter(defaultValue = "${project.build.directory}/metaschema", readonly = true, required = true)
081  protected File staleFileDirectory;
082
083  /**
084   * <p>
085   * Defines the encoding used for generating Java Source files.
086   * </p>
087   * <p>
088   * The algorithm for finding the encoding to use is as follows (where the first non-null value found
089   * is used for encoding):
090   * <ol>
091   * <li>If the configuration property is explicitly given within the plugin's configuration, use that
092   * value.</li>
093   * <li>If the Maven property <code>project.build.sourceEncoding</code> is defined, use its
094   * value.</li>
095   * <li>Otherwise use the value from the system property <code>file.encoding</code>.</li>
096   * </ol>
097   * </p>
098   *
099   * @see #getEncoding()
100   * @since 2.0
101   */
102  @Parameter(defaultValue = "${project.build.sourceEncoding}")
103  private String encoding;
104
105  /**
106   * Location to generate Java source files in.
107   */
108  @Parameter(defaultValue = "${project.build.directory}/generated-sources/metaschema", required = true)
109  private File outputDirectory;
110
111  /**
112   * The directory to read source metaschema from.
113   */
114  @Parameter(defaultValue = "${basedir}/src/main/metaschema")
115  private File metaschemaDir;
116
117  /**
118   * A set of inclusion patterns used to select which metaschema are to be processed. By default, all
119   * files are processed.
120   */
121  @Parameter
122  protected String[] includes;
123
124  /**
125   * A set of exclusion patterns used to prevent certain files from being processed. By default, this
126   * set is empty such that no files are excluded.
127   */
128  @Parameter
129  protected String[] excludes;
130
131  /**
132   * Indicate if the execution should be skipped.
133   */
134  @Parameter(property = "metaschema.skip", defaultValue = "false")
135  private boolean skip;
136
137  /**
138   * The BuildContext is used to identify which files or directories were modified since last build.
139   * This is used to determine if Metaschema-based generation must be performed again.
140   *
141   * @return the active Plexus BuildContext.
142   */
143  protected final BuildContext getBuildContext() {
144    return buildContext;
145  }
146
147  /**
148   * Retrieve the Maven project context.
149   *
150   * @return The active MavenProject.
151   */
152  protected final MavenProject getMavenProject() {
153    return mavenProject;
154  }
155
156  /**
157   * Retrieve the mojo execution context.
158   *
159   * @return The active MojoExecution.
160   */
161  @SuppressFBWarnings(value = "EI_EXPOSE_REP", justification = "this is a data holder")
162  public MojoExecution getMojoExecution() {
163    return mojoExecution;
164  }
165
166  /**
167   * Retrieve the directory where generated classes will be stored.
168   *
169   * @return the directory
170   */
171  protected File getOutputDirectory() {
172    return outputDirectory;
173  }
174
175  /**
176   * Set the directory where generated classes will be stored.
177   *
178   * @param outputDirectory
179   *          the directory to use
180   */
181  protected void setOutputDirectory(File outputDirectory) {
182    Objects.requireNonNull(outputDirectory, "outputDirectory");
183    this.outputDirectory = outputDirectory;
184  }
185
186  /**
187   * Gets the file encoding to use for generated classes.
188   * <p>
189   * The algorithm for finding the encoding to use is as follows (where the first non-null value found
190   * is used for encoding):
191   * </p>
192   * <ol>
193   * <li>If the configuration property is explicitly given within the plugin's configuration, use that
194   * value.</li>
195   * <li>If the Maven property <code>project.build.sourceEncoding</code> is defined, use its
196   * value.</li>
197   * <li>Otherwise use the value from the system property <code>file.encoding</code>.</li>
198   * </ol>
199   *
200   * @return The encoding to be used by this AbstractJaxbMojo and its tools.
201   */
202  protected final String getEncoding() {
203    String encoding;
204    if (this.encoding != null) {
205      // first try to use the provided encoding
206      encoding = this.encoding;
207      getLog().debug(String.format("Using configured encoding [%s].", encoding));
208    } else {
209      encoding = System.getProperty(SYSTEM_FILE_ENCODING_PROPERTY);
210      getLog().warn(String.format("Using system encoding [%s]. This build is platform dependent!", encoding));
211    }
212    return encoding;
213  }
214
215  /**
216   * Retrieve a stream of Metaschema file sources.
217   *
218   * @return the stream
219   */
220  protected Stream<File> getSources() {
221    DirectoryScanner ds = new DirectoryScanner();
222    ds.setBasedir(metaschemaDir);
223    ds.setIncludes(includes != null && includes.length > 0 ? includes : DEFAULT_INCLUDES);
224    ds.setExcludes(excludes != null && excludes.length > 0 ? excludes : null);
225    ds.addDefaultExcludes();
226    ds.setCaseSensitive(true);
227    ds.setFollowSymlinks(false);
228    ds.scan();
229    return Stream.of(ds.getIncludedFiles()).map(filename -> new File(metaschemaDir, filename)).distinct();
230  }
231
232  /**
233   * Determine if the execution of this mojo should be skipped.
234   *
235   * @return {@code true} if the mojo execution should be skipped, or {@code false} otherwise
236   */
237  protected boolean shouldExecutionBeSkipped() {
238    return skip;
239  }
240
241  /**
242   * Get the name of the file that is used to detect staleness.
243   *
244   * @return the name
245   */
246  protected abstract String getStaleFileName();
247
248  /**
249   * Gets the staleFile for this execution.
250   *
251   * @return the staleFile
252   */
253  protected final File getStaleFile() {
254    StringBuilder builder = new StringBuilder();
255    if (getMojoExecution() != null) {
256      builder.append(getMojoExecution().getExecutionId()).append('-');
257    }
258    builder.append(getStaleFileName());
259    return new File(staleFileDirectory, builder.toString());
260  }
261
262  /**
263   * Determine if code generation is required. This is done by comparing the last modified time of
264   * each Metaschema source file against the stale file managed by this plugin.
265   *
266   * @return {@code true} if the code generation is needed, or {@code false} otherwise
267   */
268  protected boolean isGenerationRequired() {
269    final File staleFile = getStaleFile();
270    boolean generate = !staleFile.exists();
271    if (generate) {
272      getLog().info(String.format("Stale file '%s' doesn't exist! Generating source files.", staleFile.getPath()));
273      generate = true;
274    } else {
275      generate = false;
276      // check for staleness
277      long staleLastModified = staleFile.lastModified();
278
279      BuildContext buildContext = getBuildContext();
280      URI metaschemaDirRelative = getMavenProject().getBasedir().toURI().relativize(metaschemaDir.toURI());
281
282      if (buildContext.isIncremental() && buildContext.hasDelta(metaschemaDirRelative.toString())) {
283        getLog().info("metaschemaDirRelative: " + metaschemaDirRelative.toString());
284        generate = true;
285      }
286
287      if (!generate) {
288        for (File sourceFile : getSources().collect(Collectors.toList())) {
289          getLog().info("Source file: " + sourceFile.getPath());
290          if (sourceFile.lastModified() > staleLastModified) {
291            generate = true;
292          }
293        }
294      }
295    }
296    return generate;
297  }
298}
