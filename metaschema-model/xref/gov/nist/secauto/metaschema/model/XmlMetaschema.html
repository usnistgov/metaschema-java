59  class XmlMetaschema
60      extends AbstractMetaschema {
61    private static final Logger LOGGER = LogManager.getLogger(XmlMetaschema.class);
62  
63    @NotNull
64    private final URI location;
65    @NotNull
66    private final METASCHEMADocument metaschema;
67    private final Map<@NotNull String, ? extends IFlagDefinition> flagDefinitions;
68    private final Map<@NotNull String, ? extends IFieldDefinition> fieldDefinitions;
69    private final Map<@NotNull String, ? extends IAssemblyDefinition> assemblyDefinitions;
70    private final Map<@NotNull String, ? extends IAssemblyDefinition> rootAssemblyDefinitions;

72    /**
73     * Constructs a new Metaschema instance.
74     * 
75     * @param resource
76     *          the resource from which the metaschema was loaded
77     * @param metaschemaXml
78     *          the XML source of the metaschema definition bound to Java objects
79     * @param importedMetaschema
80     *          the definitions for any metaschema imported by this metaschema
81     * @throws MetaschemaException
82     *           if a processing error occurs
83     */
84    XmlMetaschema( // NOPMD - unavoidable
85        @NotNull URI resource,
86        @NotNull METASCHEMADocument metaschemaXml,
87        @NotNull List<@NotNull IMetaschema> importedMetaschema) throws MetaschemaException {
88      super(importedMetaschema);
89      this.location = ObjectUtils.requireNonNull(resource, "resource");
90      this.metaschema = metaschemaXml;
91  
92      METASCHEMA metaschemaNode = metaschema.getMETASCHEMA();
93  
94      // handle definitions in this metaschema
95      {
96        // start with flag definitions
97        XmlCursor cursor = metaschemaNode.newCursor();
98        cursor.selectPath("declare namespace m='http://csrc.nist.gov/ns/oscal/metaschema/1.0';$this/m:define-flag");
99  
100       Map<@NotNull String, IFlagDefinition> flagDefinitions = new LinkedHashMap<>(); // NOPMD - intentional
101       while (cursor.toNextSelection()) {
102         GlobalFlagDefinitionType obj = ObjectUtils.notNull((GlobalFlagDefinitionType) cursor.getObject());
103         XmlGlobalFlagDefinition flag = new XmlGlobalFlagDefinition(obj, this); // NOPMD - intentional
104         if (LOGGER.isTraceEnabled()) {
105           LOGGER.trace("New flag definition '{}'", flag.toCoordinates());
106         }
107         flagDefinitions.put(flag.getName(), flag);
108       }
109       this.flagDefinitions
110           = flagDefinitions.isEmpty() ? Collections.emptyMap() : Collections.unmodifiableMap(flagDefinitions);
111     }
112 
113     {
114       // now field definitions
115       XmlCursor cursor = metaschemaNode.newCursor();
116       cursor.selectPath("declare namespace m='http://csrc.nist.gov/ns/oscal/metaschema/1.0';$this/m:define-field");
117 
118       Map<@NotNull String, IFieldDefinition> fieldDefinitions = new LinkedHashMap<>(); // NOPMD - intentional
119       while (cursor.toNextSelection()) {
120         GlobalFieldDefinitionType obj = ObjectUtils.notNull((GlobalFieldDefinitionType) cursor.getObject());
121         XmlGlobalFieldDefinition field = new XmlGlobalFieldDefinition(obj, this); // NOPMD - intentional
122         if (LOGGER.isTraceEnabled()) {
123           LOGGER.trace("New field definition '{}'", field.toCoordinates());
124         }
125         fieldDefinitions.put(field.getName(), field);
126       }
127       this.fieldDefinitions
128           = fieldDefinitions.isEmpty() ? Collections.emptyMap() : Collections.unmodifiableMap(fieldDefinitions);
129     }
130 
131     {
132       // finally assembly definitions
133       Map<@NotNull String, IAssemblyDefinition> assemblyDefinitions = new LinkedHashMap<>(); // NOPMD - intentional
134       Map<@NotNull String, IAssemblyDefinition> rootAssemblyDefinitions = new LinkedHashMap<>(); // NOPMD - intentional
135 
136       XmlCursor cursor = metaschemaNode.newCursor();
137       cursor.selectPath(
138           "declare namespace m='http://csrc.nist.gov/ns/oscal/metaschema/1.0';$this/m:define-assembly");
139 
140       while (cursor.toNextSelection()) {
141         GlobalAssemblyDefinitionType obj = ObjectUtils.notNull((GlobalAssemblyDefinitionType) cursor.getObject());
142         XmlGlobalAssemblyDefinition assembly = new XmlGlobalAssemblyDefinition(obj, this); // NOPMD - intentional
143         if (LOGGER.isTraceEnabled()) {
144           LOGGER.trace("New assembly definition '{}'", assembly.toCoordinates());
145         }
146         assemblyDefinitions.put(assembly.getName(), assembly);
147         if (assembly.isRoot()) {
148           rootAssemblyDefinitions.put(ObjectUtils.notNull(assembly.getRootName()), assembly);
149         }
150       }
151 
152       this.assemblyDefinitions
153           = assemblyDefinitions.isEmpty() ? Collections.emptyMap() : Collections.unmodifiableMap(assemblyDefinitions);
154       this.rootAssemblyDefinitions = rootAssemblyDefinitions.isEmpty() ? Collections.emptyMap()
155           : Collections.unmodifiableMap(rootAssemblyDefinitions);
156     }
157   }

168    * @return the XMLBean for the Metaschema

172     return metaschema.getMETASCHEMA();
