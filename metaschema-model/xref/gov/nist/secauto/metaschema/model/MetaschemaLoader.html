61  /**
62   * Provides methods to load a Metaschema expressed in XML.
63   * <p>
64   * Loaded Metaschema instances are cached to avoid the need to load them for every use. Any
65   * Metaschema imported is also loaded and cached automatically.
66   */
67  public class MetaschemaLoader
68      extends AbstractLoader<IMetaschema> {
69    private boolean resolveEntities; // = false;
70  
71    @NonNull
72    private final Set<IConstraintSet> registeredConstraintSets;
73  
74    /**
75     * Construct a new Metaschema loader.
76     */
77    public MetaschemaLoader() {
78      this(CollectionUtil.emptySet());
79    }
80  
81    /**
82     * Construct a new Metaschema loader, which will incorporate the additional provided constraints
83     * into matching loaded definitions.
84     * 
85     * @param additionalConstraintSets
86     *          additional constraints to associate with loaded definitions
87     */
88    public MetaschemaLoader(@NonNull Set<IConstraintSet> additionalConstraintSets) {
89      this.registeredConstraintSets = CollectionUtil.unmodifiableSet(additionalConstraintSets);
90    }
91  
92    /**
93     * Get the set of additional constraints associated with this loader.
94     * 
95     * @return the set of constraints
96     */
97    @NonNull
98    protected Set<IConstraintSet> getRegisteredConstraintSets() {
99      return registeredConstraintSets;
100   }
101 
102   /**
103    * Enable a mode that allows XML entity resolution. This may be needed to parse some resource files
104    * that contain entities. Enabling entity resolution is a less secure, which requires trust in the
105    * resource content being parsed.
106    */
107   public void allowEntityResolution() {
108     resolveEntities = true;
109   }
110 
111   /**
112    * Parse the {@code resource} based on the provided {@code xmlObject}.
113    * 
114    * @param resource
115    *          the URI of the resource being parsed
116    * @param xmlObject
117    *          the XML beans object to parse
118    * @param importedMetaschemas
119    *          previously parsed Metaschema resources imported by the provided {@code resource}
120    * @return the parsed resource as a Metaschema instance
121    * @throws MetaschemaException
122    *           if an error occurred while parsing the XML beans object
123    */
124   protected IMetaschema newXmlMetaschema(
125       @NonNull URI resource,
126       @NonNull METASCHEMADocument xmlObject,
127       @NonNull List<IMetaschema> importedMetaschemas) throws MetaschemaException {
128     IMetaschema retval = new XmlMetaschema(resource, xmlObject, importedMetaschemas);
129 
130     IConstraintSet.applyConstraintSetToMetaschema(getRegisteredConstraintSets(), retval);
131 
132     return retval;
133   }
134 
135   @Override
136   protected IMetaschema parseResource(@NonNull URI resource, @NonNull Stack<URI> visitedResources)
137       throws IOException {
138     // parse this metaschema
139     METASCHEMADocument xmlObject = parseMetaschema(resource);
140 
141     // now check if this Metaschema imports other metaschema
142     int size = xmlObject.getMETASCHEMA().sizeOfImportArray();
143     @NonNull
144     Map<URI, IMetaschema> importedMetaschema;
145     if (size == 0) {
146       importedMetaschema = ObjectUtils.notNull(Collections.emptyMap());
147     } else {
148       try {
149         importedMetaschema = new LinkedHashMap<>();
150         for (METASCHEMADocument.METASCHEMA.Import imported : xmlObject.getMETASCHEMA().getImportList()) {
151           URI importedResource = URI.create(imported.getHref());
152           importedResource = ObjectUtils.notNull(resource.resolve(importedResource));
153           importedMetaschema.put(importedResource, loadInternal(importedResource, visitedResources));
154         }
155       } catch (MetaschemaException ex) {
156         throw new IOException(ex);
157       }
158     }
159 
160     // now create this metaschema
161     Collection<IMetaschema> values = importedMetaschema.values();
162     try {
163       return newXmlMetaschema(resource, xmlObject, new ArrayList<>(values));
164     } catch (MetaschemaException ex) {
165       throw new IOException(ex);
166     }
167   }
168 
169   /**
170    * Parse the provided XML resource as a Metaschema.
171    * 
172    * @param resource
173    *          the resource to parse
174    * @return the XMLBeans representation of the Metaschema
175    * @throws IOException
176    *           if a parsing error occurred
177    */
178   protected METASCHEMADocument parseMetaschema(@NonNull URI resource) throws IOException {
179     METASCHEMADocument metaschemaXml;
180     try {
181       XmlOptions options = new XmlOptions();
182       if (resolveEntities) {
183         SAXParserFactory factory = SAXParserFactory.newInstance();
184 
185         try {
186           // factory.setFeature(XMLConstants.FEATURE_SECURE_PROCESSING, true);
187           factory.setFeature("http://apache.org/xml/features/disallow-doctype-decl", false);
188           factory.setFeature("http://xml.org/sax/features/external-general-entities", true);
189           factory.setFeature("http://xml.org/sax/features/external-parameter-entities", true);
190           SAXParser parser = factory.newSAXParser();
191           parser.setProperty(XMLConstants.ACCESS_EXTERNAL_DTD, "file"); // ,jar:file
192           XMLReader reader = parser.getXMLReader();
193           reader.setEntityResolver(new EntityResolver() {
194 
195             @Override
196             public InputSource resolveEntity(String publicId, String systemId) throws SAXException, IOException {
197               return null;
198             }
199 
200           });
201           options.setLoadUseXMLReader(reader);
202         } catch (SAXException | ParserConfigurationException ex) {
203           throw new IOException(ex);
204         }
205         // options.setLoadEntityBytesLimit(204800);
206         // options.setLoadUseDefaultResolver();
207         options.setEntityResolver(new EntityResolver() {
208 
209           @Override
210           public InputSource resolveEntity(String publicId, String systemId) throws SAXException, IOException {
211             String effectiveSystemId = systemId;
212             // TODO: It's very odd that the system id looks like this. Need to investigate.
213             if (effectiveSystemId.startsWith("file://file://")) {
214               effectiveSystemId = effectiveSystemId.substring(14);
215             }
216             URI resolvedSystemId = resource.resolve(effectiveSystemId);
217             return new InputSource(resolvedSystemId.toString());
218           }
219 
220         });
221         options.setLoadDTDGrammar(true);
222       }
223       options.setBaseURI(resource);
224       options.setLoadLineNumbers();
225       metaschemaXml = METASCHEMADocument.Factory.parse(resource.toURL(), options);
226     } catch (XmlException ex) {
227       throw new IOException(ex);
228     }
229     return metaschemaXml;
230   }
