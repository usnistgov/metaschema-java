27  package gov.nist.secauto.metaschema.model;
28  
29  import gov.nist.secauto.metaschema.model.common.AbstractAssemblyInstance;
30  import gov.nist.secauto.metaschema.model.common.IAssemblyDefinition;
31  import gov.nist.secauto.metaschema.model.common.JsonGroupAsBehavior;
32  import gov.nist.secauto.metaschema.model.common.MetaschemaModelConstants;
33  import gov.nist.secauto.metaschema.model.common.XmlGroupAsBehavior;
34  import gov.nist.secauto.metaschema.model.common.datatype.markup.MarkupMultiline;
35  import gov.nist.secauto.metaschema.model.common.util.ObjectUtils;
36  import gov.nist.secauto.metaschema.model.xmlbeans.AssemblyReferenceType;
37  
38  import org.jetbrains.annotations.NotNull;
39  
40  import java.math.BigInteger;
41  import java.util.Collection;
42  import java.util.Collections;
43  
44  class XmlAssemblyInstance
45      extends AbstractAssemblyInstance {
46    // private static final Logger logger = LogManager.getLogger(XmlAssemblyInstance.class);
47  
48    private final AssemblyReferenceType xmlAssembly;
49  
50    /**
51     * Constructs a new Metaschema Assembly instance definition from an XML representation bound to Java
52     * objects.
53     * 
54     * @param xmlAssembly
55     *          the XML representation bound to Java objects
56     * @param parent
57     *          the assembly definition this object is an instance of
58     */
59    public XmlAssemblyInstance(@NotNull AssemblyReferenceType xmlAssembly, @NotNull IAssemblyDefinition parent) {
60      super(parent);
61      this.xmlAssembly = xmlAssembly;
62    }
63  
64    /**
65     * Get the underlying XML data.
66     * 
67     * @return the underlying XML data
68     */
69    protected AssemblyReferenceType getXmlAssembly() {
70      return xmlAssembly;
71    }
72  
73    @Override
74    public IAssemblyDefinition getDefinition() {
75      // This will always be not null
76      return ObjectUtils.notNull(getContainingMetaschema()
77          .getScopedAssemblyDefinitionByName(getName()));
78    }
79  
80    @SuppressWarnings("null")
81    @Override
82    public String getName() {
83      return getXmlAssembly().getRef();
84    }
85  
86    @Override
87    public String getUseName() {
88      return getXmlAssembly().isSetUseName() ? getXmlAssembly().getUseName() : getDefinition().getUseName();
89    }
90  
91    @Override
92    public String getGroupAsName() {
93      return getXmlAssembly().isSetGroupAs() ? getXmlAssembly().getGroupAs().getName() : null;
94    }
95  
96    @Override
97    public int getMinOccurs() {
98      int retval = MetaschemaModelConstants.DEFAULT_GROUP_AS_MIN_OCCURS;
99      if (getXmlAssembly().isSetMinOccurs()) {
100       retval = getXmlAssembly().getMinOccurs().intValueExact();
101     }
102     return retval;
103   }
104 
105   @Override
106   public int getMaxOccurs() {
107     int retval = MetaschemaModelConstants.DEFAULT_GROUP_AS_MAX_OCCURS;
108     if (getXmlAssembly().isSetMaxOccurs()) {
109       Object value = getXmlAssembly().getMaxOccurs();
110       if (value instanceof String) {
111         // unbounded
112         retval = -1;
113       } else if (value instanceof BigInteger) {
114         retval = ((BigInteger) value).intValueExact();
115       }
116     }
117     return retval;
118   }
119 
120   @Override
121   public JsonGroupAsBehavior getJsonGroupAsBehavior() {
122     JsonGroupAsBehavior retval = MetaschemaModelConstants.DEFAULT_JSON_GROUP_AS_BEHAVIOR;
123     if (getXmlAssembly().isSetGroupAs() && getXmlAssembly().getGroupAs().isSetInJson()) {
124       retval = getXmlAssembly().getGroupAs().getInJson();
125     }
126     return retval;
127   }
128 
129   @Override
130   public XmlGroupAsBehavior getXmlGroupAsBehavior() {
131     XmlGroupAsBehavior retval = MetaschemaModelConstants.DEFAULT_XML_GROUP_AS_BEHAVIOR;
132     if (getXmlAssembly().isSetGroupAs() && getXmlAssembly().getGroupAs().isSetInXml()) {
133       retval = getXmlAssembly().getGroupAs().getInXml();
134     }
135     return retval;
136   }
137 
138   @SuppressWarnings("null")
139   @Override
140   public MarkupMultiline getRemarks() {
141     return getXmlAssembly().isSetRemarks() ? MarkupStringConverter.toMarkupString(getXmlAssembly().getRemarks()) : null;
142   }
143 
144   @Override
145   public Object getValue(@NotNull Object parentInstance) {
146     // there is no value
147     return null;
148   }
149 
150   @SuppressWarnings("null")
151   @Override
152   public Collection<@NotNull ?> getItemValues(Object instanceValue) {
153     // there are no item values
154     return Collections.emptyList();
155   }
156 }
