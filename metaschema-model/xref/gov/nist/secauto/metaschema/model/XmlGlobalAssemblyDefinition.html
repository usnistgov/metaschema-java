64  class XmlGlobalAssemblyDefinition implements IAssemblyDefinition { // NOPMD - intentional
65  
66    @NonNull
67    private final GlobalAssemblyDefinitionType xmlAssembly;
68    @NonNull
69    private final XmlMetaschema metaschema;
70    private Lazy<XmlFlagContainerSupport> flagContainer;
71    private Lazy<XmlModelContainerSupport> modelContainer;
72    private Lazy<AssemblyConstraintSupport> constraints;
73  
74    /**
75     * Constructs a new Metaschema Assembly definition from an XML representation bound to Java objects.
76     * 
77     * @param xmlAssembly
78     *          the XML representation bound to Java objects
79     * @param metaschema
80     *          the containing Metaschema
81     */
82    public XmlGlobalAssemblyDefinition(
83        @NonNull GlobalAssemblyDefinitionType xmlAssembly,
84        @NonNull XmlMetaschema metaschema) {
85      this.xmlAssembly = xmlAssembly;
86      this.metaschema = metaschema;
87      this.flagContainer = Lazy.lazy(() -> new XmlFlagContainerSupport(xmlAssembly, this));
88      this.modelContainer = Lazy.lazy(() -> new XmlModelContainerSupport(xmlAssembly, this));
89      this.constraints = Lazy.lazy(() -> AssemblyConstraintSupport.newInstance(
90          xmlAssembly,
91          ExternalModelSource.instance(metaschema.getLocation())));
92    }
