27  package gov.nist.secauto.metaschema.model.xml;
28  
29  import gov.nist.itl.metaschema.model.xml.AssemblyDocument;
30  import gov.nist.secauto.metaschema.datatypes.markup.MarkupLine;
31  import gov.nist.secauto.metaschema.model.info.definitions.InfoElementDefinition;
32  import gov.nist.secauto.metaschema.model.info.instances.AbstractAssemblyInstance;
33  import gov.nist.secauto.metaschema.model.info.instances.JsonGroupAsBehavior;
34  import gov.nist.secauto.metaschema.model.info.instances.XmlGroupAsBehavior;
35  
36  import java.math.BigInteger;
37  
38  public class XmlAssemblyInstance extends AbstractAssemblyInstance {
39    // private static final Logger logger = LogManager.getLogger(XmlAssemblyInstance.class);
40  
41    private final AssemblyDocument.Assembly xmlAssembly;
42  
43    /**
44     * Constructs a new Metaschema Assembly instance definition from an XML representation bound to Java
45     * objects.
46     * 
47     * @param xmlAssembly
48     *          the XML representation bound to Java objects
49     * @param parent
50     *          the assembly definition this object is an instance of
51     */
52    public XmlAssemblyInstance(AssemblyDocument.Assembly xmlAssembly, InfoElementDefinition parent) {
53      super(parent);
54      this.xmlAssembly = xmlAssembly;
55    }
56  
57    @Override
58    public String getName() {
59      return getXmlAssembly().getRef();
60    }
61  
62    @Override
63    public String getFormalName() {
64      return getDefinition().getFormalName();
65    }
66  
67    @Override
68    public MarkupLine getDescription() {
69      MarkupLine retval = null;
70      if (getXmlAssembly().isSetDescription()) {
71        retval = MarkupStringConverter.toMarkupString(getXmlAssembly().getDescription());
72      } else if (isReference()) {
73        retval = getDefinition().getDescription();
74      }
75      return retval;
76    }
77  
78    @Override
79    public int getMinOccurs() {
80      int retval = 0;
81      if (getXmlAssembly().isSetMinOccurs()) {
82        retval = getXmlAssembly().getMinOccurs().intValueExact();
83      }
84      return retval;
85    }
86  
87    @Override
88    public int getMaxOccurs() {
89      int retval = 1;
90      if (getXmlAssembly().isSetMaxOccurs()) {
91        Object value = getXmlAssembly().getMaxOccurs();
92        if (value instanceof String) {
93          // unbounded
94          retval = -1;
95        } else if (value instanceof BigInteger) {
96          retval = ((BigInteger) value).intValueExact();
97        }
98      }
99      return retval;
100   }
101 
102   @Override
103   public String getInstanceName() {
104     return getXmlAssembly().isSetGroupAs() ? getXmlAssembly().getGroupAs().getName() : getName();
105   }
106 
107   protected AssemblyDocument.Assembly getXmlAssembly() {
108     return xmlAssembly;
109   }
110 
111   @Override
112   public JsonGroupAsBehavior getJsonGroupAsBehavior() {
113     JsonGroupAsBehavior retval = JsonGroupAsBehavior.SINGLETON_OR_LIST;
114     if (getXmlAssembly().isSetGroupAs() && getXmlAssembly().getGroupAs().isSetInJson()) {
115       retval = JsonGroupAsBehavior.lookup(getXmlAssembly().getGroupAs().getInJson());
116     }
117     return retval;
118   }
119 
120   @Override
121   public XmlGroupAsBehavior getXmlGroupAsBehavior() {
122     XmlGroupAsBehavior retval = XmlGroupAsBehavior.UNGROUPED;
123     if (getXmlAssembly().isSetGroupAs() && getXmlAssembly().getGroupAs().isSetInXml()) {
124       retval = XmlGroupAsBehavior.lookup(getXmlAssembly().getGroupAs().getInXml());
125     }
126     return retval;
127   }
128 }
