68  public abstract class AbstractSchemaGeneratorTestSuite
69      extends AbstractTestSuite {
70    @NonNull
71    protected static final ISchemaGenerator XML_SCHEMA_GENERATOR = new XmlSchemaGenerator();
72    @NonNull
73    protected static final ISchemaGenerator JSON_SCHEMA_GENERATOR = new JsonSchemaGenerator();
74    @NonNull
75    protected static final IConfiguration<SchemaGenerationFeature<?>> SCHEMA_GENERATION_CONFIG;
76    @NonNull
77    protected static final BiFunction<IMetaschema, Writer, Void> XML_SCHEMA_PROVIDER;
78    @NonNull
79    protected static final BiFunction<IMetaschema, Writer, Void> JSON_SCHEMA_PROVIDER;
80    @NonNull
81    protected static final JsonSchemaContentValidator JSON_SCHEMA_VALIDATOR;
82    @NonNull
83    protected static final Function<Path, JsonSchemaContentValidator> JSON_CONTENT_VALIDATOR_PROVIDER;
84    @NonNull
85    protected static final Function<Path, XmlSchemaContentValidator> XML_CONTENT_VALIDATOR_PROVIDER;
86  
87    private static final String UNIT_TEST_CONFIG
88        = "../metaschema-model/metaschema/test-suite/schema-generation/unit-tests.xml";
89  
90    static {
91      IMutableConfiguration<SchemaGenerationFeature<?>> features = new DefaultConfiguration<>();
92      features.disableFeature(SchemaGenerationFeature.INLINE_DEFINITIONS);
93      SCHEMA_GENERATION_CONFIG = features;
94  
95      BiFunction<IMetaschema, Writer, Void> xmlProvider = (metaschema, writer) -> {
96        assert metaschema != null;
97        assert writer != null;
98        try {
99          XML_SCHEMA_GENERATOR.generateFromMetaschema(metaschema, writer, SCHEMA_GENERATION_CONFIG);
100       } catch (SchemaGenerationException ex) {
101         throw new JUnitException("IO error", ex);
102       }
103       return null;
104     };
105     XML_SCHEMA_PROVIDER = xmlProvider;
106 
107     BiFunction<IMetaschema, Writer, Void> jsonProvider = (metaschema, writer) -> {
108       assert metaschema != null;
109       assert writer != null;
110       try {
111         JSON_SCHEMA_GENERATOR.generateFromMetaschema(metaschema, writer, SCHEMA_GENERATION_CONFIG);
112       } catch (SchemaGenerationException ex) {
113         throw new JUnitException("IO error", ex);
114       }
115       return null;
116     };
117     JSON_SCHEMA_PROVIDER = jsonProvider;
118 
119     try (InputStream is = MetaschemaLoader.class.getResourceAsStream("/schema/json/json-schema.json")) {
120       assert is != null : "unable to get JSON schema resource";
121       JsonSchemaContentValidator schemaValidator = new JsonSchemaContentValidator(is);
122       JSON_SCHEMA_VALIDATOR = schemaValidator;
123     } catch (IOException ex) {
124       throw new IllegalStateException(ex);
125     }
126 
127     @SuppressWarnings("null")
128     @NonNull Function<Path, XmlSchemaContentValidator> xmlContentValidatorProvider = (path) -> {
129       try {
130         URL schemaResource = path.toUri().toURL();
131         @SuppressWarnings("resource") StreamSource source
132             = new StreamSource(schemaResource.openStream(), schemaResource.toString());
133         List<? extends Source> schemaSources = Collections.singletonList(source);
134         return new XmlSchemaContentValidator(schemaSources);
135       } catch (IOException | SAXException ex) {
136         throw new IllegalStateException(ex);
137       }
138     };
139     XML_CONTENT_VALIDATOR_PROVIDER = xmlContentValidatorProvider;
140 
141     @NonNull Function<Path, JsonSchemaContentValidator> jsonContentValidatorProvider = (path) -> {
142       try (InputStream is = Files.newInputStream(path, StandardOpenOption.READ)) {
143         assert is != null;
144         return new JsonSchemaContentValidator(is);
145       } catch (IOException ex) {
146         throw new JUnitException("Failed to create content validator for schema: " + path.toString(), ex);
147       }
148     };
149     JSON_CONTENT_VALIDATOR_PROVIDER = jsonContentValidatorProvider;
150   }

156   }

158   @Override

160     return ObjectUtils.notNull(Paths.get("target/test-schemagen"));
161   }

163   protected Path produceXmlSchema(@NonNull IMetaschema metaschema, @NonNull Path schemaPath) throws IOException {
164     produceSchema(metaschema, schemaPath, XML_SCHEMA_PROVIDER);
165     return schemaPath;
166   }
167 
168   protected Path produceJsonSchema(@NonNull IMetaschema metaschema, @NonNull Path schemaPath) throws IOException {
169     produceSchema(metaschema, schemaPath, JSON_SCHEMA_PROVIDER);
170     return schemaPath;
171   }
172 
173   @SuppressWarnings("null")
174   protected void doTest(
175       @NonNull String collectionName,
176       @NonNull String metaschemaName,
177       @NonNull String generatedSchemaName,
178       @NonNull ContentCase... contentCases) throws IOException, MetaschemaException {
179     Path generationDir = getGenerationPath();
180 
181     Path testSuite = Paths.get("../metaschema-model/metaschema/test-suite/schema-generation/");
182     Path collectionPath = testSuite.resolve(collectionName);
183 
184     MetaschemaLoader loader = new MetaschemaLoader();
185     loader.allowEntityResolution();
186     Path metaschemaPath = collectionPath.resolve(metaschemaName);
187     IMetaschema metaschema = loader.load(metaschemaPath);
188 
189     Path jsonSchema = produceJsonSchema(metaschema, generationDir.resolve(generatedSchemaName + ".json"));
190     assertEquals(true, validate(JSON_SCHEMA_VALIDATOR, jsonSchema),
191         String.format("JSON schema '%s' was invalid", jsonSchema.toString()));
192     Path xmlSchema = produceXmlSchema(metaschema, generationDir.resolve(generatedSchemaName + ".xsd"));
193 
194     Path schemaPath;
195     switch (getRequiredContentFormat()) {
196     case JSON:
197     case YAML:
198       schemaPath = jsonSchema;
199       break;
200     case XML:
201       schemaPath = xmlSchema;
202       break;
203     default:
204       throw new IllegalStateException();
205     }
206 
207     DynamicBindingContext context = DynamicBindingContext.forMetaschema(
208         metaschema,
209         generationDir);
210     for (ContentCase contentCase : contentCases) {
211       Path contentPath = collectionPath.resolve(contentCase.getName());
212 
213       if (!getRequiredContentFormat().equals(contentCase.getActualFormat())) {
214         contentPath = convertContent(contentPath.toUri(), generationDir, context);
215       }
216 
217       assertEquals(contentCase.isValid(),
218           validate(getContentValidatorSupplier().apply(schemaPath), contentPath),
219           String.format("validation of '%s' did not match expectation", contentPath));
220     }
221   }
222 
223   @NonNull
224   protected ContentCase contentCase(@NonNull Format actualFormat, @NonNull String contentName, boolean valid) {
225     return new ContentCase(contentName, actualFormat, valid);
226   }
227 
228   protected static class ContentCase {
229     @NonNull
230     private final String name;
231     @NonNull
232     private final Format actualFormat;
233     private final boolean valid;
234 
235     public ContentCase(@NonNull String name, @NonNull Format actualFormat, boolean valid) {
236       this.name = name;
237       this.actualFormat = actualFormat;
238       this.valid = valid;
239     }
240 
241     @NonNull
242     public String getName() {
243       return name;
244     }
245 
246     @NonNull
247     public Format getActualFormat() {
248       return actualFormat;
249     }
250 
251     public boolean isValid() {
252       return valid;
253     }
254   }
