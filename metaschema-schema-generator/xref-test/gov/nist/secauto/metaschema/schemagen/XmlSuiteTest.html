175   @Test
176   void testliboscalJavaIssue181() throws IOException, MetaschemaException, XMLStreamException, JDOMException {
177     ModuleLoader loader = new ModuleLoader();
178     loader.allowEntityResolution();
179 
180     IModule module = loader.load(new URL(
181         // "https://raw.githubusercontent.com/usnistgov/OSCAL/develop/src/metaschema/oscal_complete_metaschema.xml"));
182         "https://raw.githubusercontent.com/usnistgov/OSCAL/v1.1.1/src/metaschema/oscal_catalog_metaschema.xml"));
183     ISchemaGenerator schemaGenerator = new XmlSchemaGenerator();
184     IMutableConfiguration<SchemaGenerationFeature<?>> features = new DefaultConfiguration<>();
185     features.enableFeature(SchemaGenerationFeature.INLINE_DEFINITIONS);
186     features.disableFeature(SchemaGenerationFeature.INLINE_CHOICE_DEFINITIONS);
187 
188     Path schemaPath = Path.of("target/oscal-catalog_schema.xsd");
189     try (Writer writer = Files.newBufferedWriter(schemaPath, StandardCharsets.UTF_8, getWriteOpenOptions())) {
190       assert writer != null;
191       schemaGenerator.generateFromModule(module, writer, features);
192     }
193 
194     // check for missing attribute types per liboscal-java#181
195     XMLInputFactory factory = XMLInputFactory.newFactory();
196     try (Reader fileReader = new FileReader(schemaPath.toFile())) {
197       XMLEventReader reader = factory.createXMLEventReader(fileReader);
198       StAXEventBuilder builder = new StAXEventBuilder();
199       Document document = document = builder.build(reader);
200 
201       XPathExpression<Element> xpath = XPathFactory.instance()
202           .compile("//xs:attribute[not(@type or xs:simpleType)]",
203               Filters.element(),
204               null,
205               Namespace.getNamespace("xs", "http://www.w3.org/2001/XMLSchema"));
206       List<Element> result = xpath.evaluate(document);
207 
208       assertTrue(result.isEmpty());
209     }
210   }
