096    xmlOutputFactory.configureForSpeed();
097    xmlOutputFactory.setProperty(XMLOutputFactory.IS_REPAIRING_NAMESPACES, true);
098    return xmlOutputFactory;

114  @Override
115  protected AutoCloser<XMLStreamWriter2, SchemaGenerationException> newWriter(
116      Writer out) {
117    XMLStreamWriter2 writer;
118    try {
119      writer = ObjectUtils.notNull((XMLStreamWriter2) getXmlOutputFactory().createXMLStreamWriter(out));
120    } catch (XMLStreamException ex) {
121      throw new SchemaGenerationException(ex);
122    }
123    return AutoCloser.autoClose(writer, t -> {
124      try {
125        t.close();
126      } catch (XMLStreamException ex) {
127        throw new SchemaGenerationException(ex);
128      }
129    });
130  }

134      IMetaschema metaschema,
135      AutoCloser<XMLStreamWriter2, SchemaGenerationException> schemaWriter,
136      IConfiguration<SchemaGenerationFeature<?>> configuration) {
137    return new XmlGenerationState(metaschema, schemaWriter, configuration);

140  @Override
141  public void generateFromMetaschema(
142      @NonNull IMetaschema metaschema,
143      @NonNull Writer out,
144      @NonNull IConfiguration<SchemaGenerationFeature<?>> configuration) {
145    try (StringWriter stringWriter = new StringWriter()) {
146      super.generateFromMetaschema(metaschema, stringWriter, configuration);
147      stringWriter.flush();
148
149      try (StringReader stringReader = new StringReader(stringWriter.toString())) {
150        Processor processor = new Processor(false);
151        XsltCompiler compiler = processor.newXsltCompiler();
152        try (InputStream is = getClass().getResourceAsStream("/identity.xsl")) {
153          XsltExecutable stylesheet = compiler.compile(new StreamSource(is));
154          Xslt30Transformer transformer = stylesheet.load30();
155          Serializer serializer = processor.newSerializer(out);
156          StreamSource source = new StreamSource(stringReader);
157          transformer.transform(source, serializer);
158        }
159      }
160    } catch (IOException | SaxonApiException ex) {
161      throw new SchemaGenerationException(ex);
162    }
163  }

165  @Override
166  protected void generateSchema(XmlGenerationState state) {
167
168    try {
169      String targetNS = state.getDefaultNS();
170
171      // analyze all definitions
172      Map<String, String> prefixToNamespaceMap = new HashMap<>(); // NOPMD concurrency not needed
173      final List<IRootAssemblyDefinition> rootAssemblyDefinitions = analyzeDefinitions(
174          state,
175          (entry, definition) -> {
176            assert entry != null;
177            assert definition != null;
178            IXmlType type = state.getTypeForDefinition(definition);
179            if (!entry.isInline()) {
180              QName qname = type.getQName();
181              String namespace = qname.getNamespaceURI();
182              if (!targetNS.equals(namespace)) {
183                // collect namespaces and prefixes for definitions with a different namespace
184                prefixToNamespaceMap.computeIfAbsent(qname.getPrefix(), x -> namespace);
185              }
186            }
187          });
188
189      // write some root elements
190      XMLStreamWriter2 writer = state.getXMLStreamWriter();
191      writer.writeStartDocument("UTF-8", "1.0");
192      writer.writeStartElement(PREFIX_XML_SCHEMA, "schema", NS_XML_SCHEMA);
193      writer.writeDefaultNamespace(targetNS);
194      writer.writeNamespace(PREFIX_XML_SCHEMA_VERSIONING, NS_XML_SCHEMA_VERSIONING);
195
196      // write namespaces for all indexed definitions
197      for (Map.Entry<String, String> entry : prefixToNamespaceMap.entrySet()) {
198        state.writeNamespace(entry.getKey(), entry.getValue());
199      }
200
201      IMetaschema metaschema = state.getMetaschema();
202
203      // write remaining root attributes
204      writer.writeAttribute("targetNamespace", targetNS);
205      writer.writeAttribute("elementFormDefault", "qualified");
206      writer.writeAttribute(NS_XML_SCHEMA_VERSIONING, "minVersion", "1.0");
207      writer.writeAttribute(NS_XML_SCHEMA_VERSIONING, "maxVersion", "1.1");
208      writer.writeAttribute("version", metaschema.getVersion());
209
210      generateSchemaMetadata(metaschema, state);
211
212      for (IAssemblyDefinition definition : rootAssemblyDefinitions) {
213        QName xmlQName = definition.getRootXmlQName();
214        if (xmlQName != null
215            && (xmlQName.getNamespaceURI() == null || state.getDefaultNS().equals(xmlQName.getNamespaceURI()))) {
216          generateRootElement(definition, state);
217        }
218      }
219
220      state.generateXmlTypes();
221
222      writer.writeEndElement(); // xs:schema
223      writer.writeEndDocument();
224      writer.flush();
225    } catch (XMLStreamException ex) {
226      throw new SchemaGenerationException(ex);
227    }
228  }

230  protected static void generateSchemaMetadata(
231      @NonNull IMetaschema metaschema,
232      @NonNull XmlGenerationState state)
233      throws XMLStreamException {
234    String targetNS = ObjectUtils.notNull(metaschema.getXmlNamespace().toASCIIString());
235    state.writeStartElement(PREFIX_XML_SCHEMA, "annotation", NS_XML_SCHEMA);
236    state.writeStartElement(PREFIX_XML_SCHEMA, "appinfo", NS_XML_SCHEMA);
237
238    state.writeStartElement(targetNS, "schema-name");
239
240    metaschema.getName().writeXHtml(targetNS, state.getXMLStreamWriter());
241
242    state.writeEndElement();
243
244    state.writeStartElement(targetNS, "schema-version");
245    state.writeCharacters(metaschema.getVersion());
246    state.writeEndElement();
247
248    state.writeStartElement(targetNS, "short-name");
249    state.writeCharacters(metaschema.getShortName());
250    state.writeEndElement();
251
252    state.writeEndElement();
253
254    MarkupMultiline remarks = metaschema.getRemarks();
255    if (remarks != null) {
256      state.writeStartElement(PREFIX_XML_SCHEMA, "documentation", NS_XML_SCHEMA);
257
258      remarks.writeXHtml(targetNS, state.getXMLStreamWriter());
259      state.writeEndElement();
260    }
261
262    state.writeEndElement();
263  }
264
265  private static void generateRootElement(@NonNull IAssemblyDefinition definition, @NonNull XmlGenerationState state)
266      throws XMLStreamException {
267    assert definition.isRoot();
268
269    XMLStreamWriter2 writer = state.getXMLStreamWriter();
270    QName xmlQName = definition.getRootXmlQName();
271
272    writer.writeStartElement(PREFIX_XML_SCHEMA, "element", NS_XML_SCHEMA);
273    writer.writeAttribute("name", xmlQName.getLocalPart());
274    writer.writeAttribute("type", state.getTypeForDefinition(definition).getTypeReference());
275
276    writer.writeEndElement();
277  }
