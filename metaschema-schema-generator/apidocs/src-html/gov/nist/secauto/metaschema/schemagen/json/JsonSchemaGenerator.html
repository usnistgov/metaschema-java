027package gov.nist.secauto.metaschema.schemagen.json;
028
029import com.fasterxml.jackson.core.JsonFactory;
030import com.fasterxml.jackson.core.JsonGenerator;
031import com.fasterxml.jackson.core.JsonGenerator.Feature;
032import com.fasterxml.jackson.databind.ObjectMapper;
033import com.fasterxml.jackson.databind.node.JsonNodeFactory;
034import com.fasterxml.jackson.databind.node.ObjectNode;
035
036import gov.nist.secauto.metaschema.model.common.IMetaschema;
037import gov.nist.secauto.metaschema.model.common.IRootAssemblyDefinition;
038import gov.nist.secauto.metaschema.model.common.configuration.IConfiguration;
039import gov.nist.secauto.metaschema.model.common.util.ObjectUtils;
040import gov.nist.secauto.metaschema.schemagen.AbstractSchemaGenerator;
041import gov.nist.secauto.metaschema.schemagen.SchemaGenerationException;
042import gov.nist.secauto.metaschema.schemagen.SchemaGenerationFeature;
043import gov.nist.secauto.metaschema.schemagen.json.datatype.JsonDatatypeManager;
044
045import java.io.IOException;
046import java.io.Writer;
047import java.util.LinkedHashMap;
048import java.util.List;
049import java.util.Map;
050import java.util.stream.Collectors;
051
052import edu.umd.cs.findbugs.annotations.NonNull;
053
054public class JsonSchemaGenerator
055    extends AbstractSchemaGenerator<JsonGenerator, JsonDatatypeManager, JsonGenerationState> {
056  @NonNull
057  private final JsonFactory jsonFactory;
058
059  public JsonSchemaGenerator() {
060    this(new JsonFactory());
061  }
062
063  public JsonSchemaGenerator(@NonNull JsonFactory jsonFactory) {
064    this.jsonFactory = jsonFactory;
065  }
066
067  @NonNull
068  public JsonFactory getJsonFactory() {
069    return jsonFactory;
070  }
071
072  @SuppressWarnings("resource")
073  @Override
074  protected JsonGenerator newWriter(Writer out) {
075    try {
076      return ObjectUtils.notNull(getJsonFactory().createGenerator(out)
077          .setCodec(new ObjectMapper())
078          .useDefaultPrettyPrinter()
079          .disable(Feature.AUTO_CLOSE_TARGET));
080    } catch (IOException ex) {
081      throw new SchemaGenerationException(ex);
082    }
083  }
084
085  @Override
086  protected JsonGenerationState newGenerationState(
087      IMetaschema metaschema,
088      JsonGenerator schemaWriter,
089      IConfiguration<SchemaGenerationFeature<?>> configuration) {
090    return new JsonGenerationState(metaschema, schemaWriter, configuration);
091  }
092
093  @Override
094  protected void generateSchema(JsonGenerationState state) {
095    // analyze all definitions
096    List<IRootAssemblyDefinition> rootAssemblyDefinitions = analyzeDefinitions(
097        state,
098        (entry, definition) -> {
099          assert entry != null;
100          assert definition != null;
101
102          if (entry.isReferenced()) {
103            // ensure schema is generated
104            state.getSchema(definition);
105          }
106        });
107
108    if (rootAssemblyDefinitions.isEmpty()) {
109      throw new SchemaGenerationException("No root definitions found");
110    }
111
112    // generate the properties first to ensure all definitions are identified
113    List<RootPropertyEntry> rootEntries = rootAssemblyDefinitions.stream()
114        .map(root -> {
115          assert root != null;
116          return new RootPropertyEntry(root, state);
117        })
118        .collect(Collectors.toUnmodifiableList());
119
120    IMetaschema metaschema = state.getMetaschema();
121    try {
122      state.writeStartObject();
123
124      state.writeField("$schema", "http://json-schema.org/draft-07/schema#");
125      state.writeField("$id",
126          String.format("%s/%s-%s-schema.json",
127              metaschema.getXmlNamespace(),
128              metaschema.getShortName(),
129              metaschema.getVersion()));
130      state.writeField("$comment", metaschema.getName().toMarkdown());
131      state.writeField("type", "object");
132
133      ObjectNode definitionsObject = state.generateDefinitions();
134      if (!definitionsObject.isEmpty()) {
135        state.writeField("definitions", definitionsObject);
136      }
137
138      @SuppressWarnings("resource") JsonGenerator writer = state.getWriter(); // NOPMD not owned
139
140      if (rootEntries.size() == 1) {
141        rootEntries.iterator().next().write(writer);
142      } else {
143        writer.writeFieldName("oneOf");
144        writer.writeStartArray();
145
146        for (RootPropertyEntry root : rootEntries) {
147          assert root != null;
148          writer.writeStartObject();
149          root.write(writer);
150          writer.writeEndObject();
151        }
152
153        writer.writeEndArray();
154      }
155      state.writeEndObject();
156    } catch (IOException ex) {
157      throw new SchemaGenerationException(ex);
158    }
159  }
160
161  @NonNull
162  private static Map<String, ObjectNode> generateRootProperties(
163      @NonNull IRootAssemblyDefinition definition,
164      @NonNull JsonGenerationState state) {
165    Map<String, ObjectNode> properties = new LinkedHashMap<>(); // NOPMD no concurrent access
166
167    properties.put("$schema", JsonNodeFactory.instance.objectNode()
168        .put("type", "string")
169        .put("format", "uri-reference"));
170
171    ObjectNode rootObj = ObjectUtils.notNull(JsonNodeFactory.instance.objectNode());
172    state.getSchema(definition).generateSchemaOrRef(state, rootObj);
173
174    properties.put(definition.getRootJsonName(), rootObj);
175    return properties;
176  }
177
178  private static class RootPropertyEntry {
179    @NonNull
180    private final IRootAssemblyDefinition definition;
181    @NonNull
182    private final Map<String, ObjectNode> properties;
183
184    public RootPropertyEntry(
185        @NonNull IRootAssemblyDefinition definition,
186        @NonNull JsonGenerationState state) {
187      this.definition = definition;
188      this.properties = generateRootProperties(definition, state);
189    }
190
191    @NonNull
192    protected IRootAssemblyDefinition getDefinition() {
193      return definition;
194    }
195
196    @NonNull
197    protected Map<String, ObjectNode> getProperties() {
198      return properties;
199    }
200
201    public void write(JsonGenerator writer) throws IOException {
202      writer.writeFieldName("properties");
203      writer.writeStartObject();
204
205      for (Map.Entry<String, ObjectNode> entry : getProperties().entrySet()) {
206        writer.writeFieldName(entry.getKey());
207        writer.writeTree(entry.getValue());
208      }
209
210      writer.writeEndObject();
211
212      writer.writeFieldName("required");
213      writer.writeStartArray();
214      writer.writeString(getDefinition().getRootJsonName());
215      writer.writeEndArray();
216
217      writer.writeBooleanField("additionalProperties", false);
218    }
219  }
220}
