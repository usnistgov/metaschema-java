027package gov.nist.secauto.metaschema.schemagen.xml;
028
029import gov.nist.secauto.metaschema.model.common.IDefinition;
030import gov.nist.secauto.metaschema.model.common.IModelElement;
031import gov.nist.secauto.metaschema.model.common.INamedInstance;
032import gov.nist.secauto.metaschema.model.common.datatype.markup.MarkupLine;
033import gov.nist.secauto.metaschema.model.common.datatype.markup.MarkupMultiline;
034import gov.nist.secauto.metaschema.model.common.util.CollectionUtil;
035import gov.nist.secauto.metaschema.schemagen.SchemaGenerationException;
036
037import java.util.ArrayList;
038import java.util.List;
039
040import javax.xml.stream.XMLStreamException;
041
042import edu.umd.cs.findbugs.annotations.NonNull;
043import edu.umd.cs.findbugs.annotations.Nullable;
044
045public final class DocumentationGenerator {
046
047  private final @Nullable String formalName;
048  private final @Nullable MarkupLine description;
049  private final @NonNull List<MarkupMultiline> remarks;
050  private final @NonNull IModelElement modelElement;
051
052  private DocumentationGenerator(@NonNull IDefinition definition) {
053    this.formalName = definition.getEffectiveFormalName();
054    this.description = definition.getEffectiveDescription();
055
056    MarkupMultiline remarks = definition.getRemarks();
057    this.remarks = remarks == null ? CollectionUtil.emptyList() : CollectionUtil.singletonList(remarks);
058
059    this.modelElement = definition;
060  }
061
062  private DocumentationGenerator(@NonNull INamedInstance instance) {
063    this.formalName = instance.getEffectiveFormalName();
064    this.description = instance.getEffectiveDescription();
065
066    List<MarkupMultiline> remarks = new ArrayList<>(2);
067    MarkupMultiline remark = instance.getRemarks();
068    if (remark != null) {
069      remarks.add(remark);
070    }
071
072    remark = instance.getDefinition().getRemarks();
073    if (remark != null) {
074      remarks.add(remark);
075    }
076
077    this.remarks = CollectionUtil.listOrEmpty(remarks);
078
079    this.modelElement = instance;
080  }
081
082  @Nullable
083  public String getFormalName() {
084    return formalName;
085  }
086
087  @Nullable
088  public MarkupLine getDescription() {
089    return description;
090  }
091
092  @NonNull
093  public List<MarkupMultiline> getRemarks() {
094    return remarks;
095  }
096
097  @NonNull
098  public IModelElement getModelElement() {
099    return modelElement;
100  }
101
102  private void generate(@NonNull XmlGenerationState state) {
103    String formalName = getFormalName();
104    MarkupLine description = getDescription();
105    List<MarkupMultiline> remarks = getRemarks();
106
107    if (formalName != null || description != null || !remarks.isEmpty()) {
108      generateDocumentation(formalName, description, remarks, state.getNS(getModelElement()), state);
109    }
110  }
111
112  public static void generateDocumentation(
113      @NonNull IDefinition definition,
114      @NonNull XmlGenerationState state) {
115    new DocumentationGenerator(definition).generate(state);
116  }
117
118  public static void generateDocumentation(
119      @NonNull INamedInstance instance,
120      @NonNull XmlGenerationState state) {
121    new DocumentationGenerator(instance).generate(state);
122  }
123
124  public static void generateDocumentation( // NOPMD acceptable complexity
125      @Nullable String formalName,
126      @Nullable MarkupLine description,
127      @NonNull List<MarkupMultiline> remarks,
128      @NonNull String xmlNS, @NonNull XmlGenerationState state) {
129
130    try {
131      state.writeStartElement(XmlSchemaGenerator.PREFIX_XML_SCHEMA, "annotation", XmlSchemaGenerator.NS_XML_SCHEMA);
132      if (formalName != null || description != null) {
133        state.writeStartElement(XmlSchemaGenerator.PREFIX_XML_SCHEMA, "appinfo", XmlSchemaGenerator.NS_XML_SCHEMA);
134
135        if (formalName != null) {
136          state.writeStartElement(xmlNS, "formal-name");
137          state.writeCharacters(formalName);
138          state.writeEndElement();
139        }
140
141        if (description != null) {
142          state.writeStartElement(xmlNS, "description");
143          description.writeXHtml(xmlNS, state.getXMLStreamWriter());
144          state.writeEndElement();
145        }
146
147        state.writeEndElement(); // xs:appInfo
148      }
149
150      state.writeStartElement(XmlSchemaGenerator.PREFIX_XML_SCHEMA, "documentation", XmlSchemaGenerator.NS_XML_SCHEMA);
151      state.writeNamespace("", XmlSchemaGenerator.NS_XHTML);
152
153      if (description != null) {
154        // write description
155        state.writeStartElement(XmlSchemaGenerator.NS_XHTML, "p");
156
157        if (formalName != null) {
158          state.writeStartElement(XmlSchemaGenerator.NS_XHTML, "b");
159          state.writeCharacters(formalName);
160          state.writeEndElement();
161          state.writeCharacters(": ");
162        }
163
164        description.writeXHtml(XmlSchemaGenerator.NS_XHTML, state.getXMLStreamWriter());
165        state.writeEndElement(); // p
166      }
167
168      for (MarkupMultiline remark : remarks) {
169        remark.writeXHtml(XmlSchemaGenerator.NS_XHTML, state.getXMLStreamWriter());
170      }
171
172      state.writeEndElement(); // xs:documentation
173      state.writeEndElement(); // xs:annotation
174    } catch (XMLStreamException ex) {
175      throw new SchemaGenerationException(ex);
176    }
177  }
178}
