027package gov.nist.secauto.metaschema.schemagen.json.datatype;
028
029import com.fasterxml.jackson.databind.JsonNode;
030import com.fasterxml.jackson.databind.ObjectMapper;
031import com.fasterxml.jackson.databind.node.ObjectNode;
032
033import gov.nist.secauto.metaschema.model.MetaschemaLoader;
034import gov.nist.secauto.metaschema.model.common.util.CollectionUtil;
035import gov.nist.secauto.metaschema.schemagen.SchemaGenerationException;
036import gov.nist.secauto.metaschema.schemagen.datatype.AbstractDatatypeManager;
037
038import java.io.IOException;
039import java.io.InputStream;
040import java.util.List;
041import java.util.Map;
042import java.util.Set;
043import java.util.concurrent.ConcurrentHashMap;
044import java.util.regex.Matcher;
045import java.util.regex.Pattern;
046import java.util.stream.Collectors;
047import java.util.stream.Stream;
048
049import edu.umd.cs.findbugs.annotations.NonNull;
050
051public class JsonDatatypeManager
052    extends AbstractDatatypeManager {
053  private static final Map<String, List<String>> DATATYPE_DEPENDENCY_MAP = new ConcurrentHashMap<>();
054  private static final Pattern DEFINITION_REF_PATTERN = Pattern.compile("^#/definitions/(.+)$");
055  private static final Map<String, JsonNode> JSON_DATATYPES = new ConcurrentHashMap<>();
056
057  static {
058    JsonNode jsonData;
059    try (InputStream is
060        = MetaschemaLoader.class.getResourceAsStream("/schema/json/metaschema-datatypes.json")) {
061      ObjectMapper objectMapper = new ObjectMapper();
062      jsonData = objectMapper.readTree(is);
063    } catch (IOException ex) {
064      throw new IllegalStateException(ex);
065    }
066
067    // analyze datatypes for dependencies
068    for (String ref : getDatatypeTranslationMap().values()) {
069      JsonNode refNode = jsonData.at("/definitions/" + ref);
070      if (!refNode.isMissingNode()) {
071        JSON_DATATYPES.put(ref, refNode);
072
073        List<String> dependencies = getDependencies(refNode).collect(Collectors.toList());
074        if (!dependencies.isEmpty()) {
075          DATATYPE_DEPENDENCY_MAP.put(ref, dependencies);
076        }
077      }
078    }
079  }
080
081  private static Stream<String> getDependencies(@NonNull JsonNode node) {
082    Stream<String> retval = Stream.empty();
083    for (Map.Entry<String, JsonNode> entry : CollectionUtil.toIterable(node.fields())) {
084      JsonNode value = entry.getValue();
085      if ("$ref".equals(entry.getKey())) {
086        Matcher matcher = DEFINITION_REF_PATTERN.matcher(value.asText());
087        if (matcher.matches()) {
088          String dependency = matcher.group(1);
089          retval = Stream.concat(retval, Stream.of(dependency));
090        }
091      }
092
093      if (value.isArray()) {
094        for (JsonNode child : CollectionUtil.toIterable(value.elements())) {
095          assert child != null;
096          retval = Stream.concat(retval, getDependencies(child));
097        }
098      }
099    }
100    return retval;
101  }
102
103  public void generateDatatypes(@NonNull ObjectNode definitionsObject) {
104    Set<String> requiredJsonDatatypes = getUsedTypes();
105    // resolve dependencies
106    for (String datatype : CollectionUtil.toIterable(requiredJsonDatatypes.stream()
107        .flatMap(datatype -> {
108          Stream<String> result;
109          List<String> dependencies = DATATYPE_DEPENDENCY_MAP.get(datatype);
110          if (dependencies == null) {
111            result = Stream.of(datatype);
112          } else {
113            result = Stream.concat(Stream.of(datatype), dependencies.stream());
114          }
115          return result;
116        }).distinct()
117        .sorted()
118        .iterator())) {
119
120      JsonNode definition = JSON_DATATYPES.get(datatype);
121      if (definition == null) {
122        throw new SchemaGenerationException("Missing JSON datatype definition for: /definitions/" + datatype);
123      }
124      definitionsObject.set(datatype, definition);
125    }
126  }
127
128}
