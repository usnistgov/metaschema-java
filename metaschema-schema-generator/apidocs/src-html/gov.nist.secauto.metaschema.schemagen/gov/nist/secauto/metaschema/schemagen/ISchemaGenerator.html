027package gov.nist.secauto.metaschema.schemagen;
028
029import gov.nist.secauto.metaschema.core.configuration.IConfiguration;
030import gov.nist.secauto.metaschema.core.model.IModule;
031import gov.nist.secauto.metaschema.schemagen.json.JsonSchemaGenerator;
032import gov.nist.secauto.metaschema.schemagen.xml.XmlSchemaGenerator;
033
034import java.io.IOException;
035import java.io.OutputStream;
036import java.io.OutputStreamWriter;
037import java.io.Writer;
038import java.nio.charset.StandardCharsets;
039import java.nio.file.Files;
040import java.nio.file.Path;
041import java.nio.file.StandardOpenOption;
042
043import edu.umd.cs.findbugs.annotations.NonNull;
044
045public interface ISchemaGenerator {
046  /**
047   * Generate and write a schema for the provided {@code metaschema} to the
048   * {@link Writer} provided by {@code writer} using the provided
049   * {@code configuration}.
050   *
051   * @param metaschema
052   *          the Module to generate the schema for
053   * @param writer
054   *          the writer to use to write the schema
055   * @param configuration
056   *          the schema generation configuration
057   * @throws SchemaGenerationException
058   *           if an error occurred while writing the schema
059   */
060  void generateFromModule(
061      @NonNull IModule metaschema,
062      @NonNull Writer writer,
063      @NonNull IConfiguration<SchemaGenerationFeature<?>> configuration);
064
065  static void generateSchema(
066      @NonNull IModule module,
067      @NonNull Path destination,
068      @NonNull SchemaFormat asFormat,
069      @NonNull IConfiguration<SchemaGenerationFeature<?>> configuration)
070      throws IOException {
071    ISchemaGenerator schemaGenerator = asFormat.getSchemaGenerator();
072
073    try (Writer writer = Files.newBufferedWriter(
074        destination,
075        StandardCharsets.UTF_8,
076        StandardOpenOption.CREATE,
077        StandardOpenOption.WRITE,
078        StandardOpenOption.TRUNCATE_EXISTING)) {
079      assert writer != null;
080      schemaGenerator.generateFromModule(module, writer, configuration);
081      writer.flush();
082    }
083  }
084
085  static void generateSchema(
086      @NonNull IModule module,
087      @NonNull OutputStream os,
088      @NonNull SchemaFormat asFormat,
089      @NonNull IConfiguration<SchemaGenerationFeature<?>> configuration)
090      throws IOException {
091    ISchemaGenerator schemaGenerator = asFormat.getSchemaGenerator();
092
093    Writer writer = new OutputStreamWriter(os, StandardCharsets.UTF_8);
094    schemaGenerator.generateFromModule(module, writer, configuration);
095    writer.flush();
096    // we don't want to close os, since we do not own it
097  }
098
099  /**
100   * Identifies the supported schema generation formats.
101   */
102  enum SchemaFormat {
103    /**
104     * a JSON Schema.
105     */
106    JSON(new JsonSchemaGenerator()),
107    /**
108     * an XML Schema.
109     */
110    XML(new XmlSchemaGenerator());
111
112    private final ISchemaGenerator schemaGenerator;
113
114    SchemaFormat(@NonNull ISchemaGenerator schemaGenerator) {
115      this.schemaGenerator = schemaGenerator;
116    }
117
118    public ISchemaGenerator getSchemaGenerator() {
119      return schemaGenerator;
120    }
121  }
122}
