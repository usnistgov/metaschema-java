41  import java.util.stream.Stream;

43  import edu.umd.cs.findbugs.annotations.NonNull;

45  class OperationFunctionsTest {
46    private static Stream<Arguments> provideValuesForIntegerDivide() {
47      return Stream.of(
48          Arguments.of(integer(10), integer(3), integer(3)),
49          Arguments.of(integer(3), integer(-2), integer(-1)),
50          Arguments.of(integer(-3), integer(2), integer(-1)),
51          Arguments.of(integer(-3), integer(-2), integer(1)),
52          Arguments.of(decimal("9.0"), integer(3), integer(3)),
53          Arguments.of(decimal("-3.5"), integer(3), integer(-1)),
54          Arguments.of(decimal("3.0"), integer(4), integer(0)),
55          Arguments.of(decimal("3.1E1"), integer(6), integer(5)),
56          Arguments.of(decimal("3.1E1"), integer(7), integer(4)));
57    }
58  
59    @ParameterizedTest
60    @MethodSource("provideValuesForIntegerDivide")
61    void testIntegerDivide(@NonNull INumericItem dividend, @NonNull INumericItem divisor,
62        @NonNull IIntegerItem expected) {
63      INumericItem result = OperationFunctions.opNumericIntegerDivide(dividend, divisor);
64      assertEquals(expected, result);
65    }
66  
67    private static Stream<Arguments> provideValuesForMod() {
68      return Stream.of(
69          Arguments.of(integer(5), integer(3), decimal("2")),
70          Arguments.of(integer(6), integer(-2), decimal("0")),
71          Arguments.of(decimal("4.5"), decimal("1.2"), decimal("0.9")),
72          Arguments.of(integer(123), integer(6), decimal("3")));
73    }
74  
75    @ParameterizedTest
76    @MethodSource("provideValuesForMod")
77    void test(@NonNull INumericItem dividend, @NonNull INumericItem divisor, @NonNull INumericItem expected) {
78      INumericItem result = OperationFunctions.opNumericMod(dividend, divisor);
79      assertEquals(expected, result);
80    }
