56  class ListPropertyValueHandlerTest {
57  
58    @RegisterExtension
59    JUnit5Mockery context = new JUnit5Mockery();
60  
61    @Mock
62    ClassBinding<?> classBinding;
63    @Mock
64    JsonParsingContext parsingContext;
65    @Mock
66    PropertyBinding propertyBinding;
67    @Mock
68    PropertyInfo propertyInfo;
69  
70    PropertyItemHandler newPropertyItemHandler(PropertyBinding propertyBinding) {
71      return new PropertyItemHandler() {
72        private int count = 0;
73  
74        @Override
75        public PropertyBinding getPropertyBinding() {
76          return propertyBinding;
77        }
78  
79        @Override
80        public List<Object> parse(JsonParsingContext parsingContext, PropertyBindingFilter filter)
81            throws BindingException, IOException {
82          JsonParser parser = parsingContext.getEventReader();
83          assertEquals(JsonToken.START_OBJECT, parser.currentToken());
84          JsonUtil.readNextToken(parser, JsonToken.FIELD_NAME);
85          assertEquals("test", parser.currentName());
86          JsonUtil.readNextToken(parser, JsonToken.VALUE_STRING);
87          assertEquals("data", parser.getText());
88          JsonUtil.readNextToken(parser, JsonToken.END_OBJECT);
89          parser.nextToken();
90          return Collections.singletonList(Integer.valueOf(count++));
91        }
92  
93        @Override
94        public void writeValue(Object value, JsonWritingContext writingContext, PropertyBindingFilter filter)
95            throws BindingException, IOException {
96          throw new UnsupportedOperationException();
97        }
98  
99      };
100   }
101 
102   private void parseProperty(JsonParser parser, PropertyValueHandler propertyValueHandler, int count)
103       throws IOException, BindingException {
104     JsonUtil.readNextToken(parser, JsonToken.START_OBJECT);
105     JsonUtil.readNextToken(parser, JsonToken.FIELD_NAME);
106     assertEquals("property", parser.currentName());
107 
108     // advance to value
109     parser.nextToken();
110 
111     for (int i = 0; i < count; i++) {
112       assertEquals(count != i + 1, propertyValueHandler.parseNextFieldValue(parsingContext), "when parsing item #" + i);
113     }
114 
115     @SuppressWarnings("unchecked")
116     List<Integer> objects = (List<Integer>) propertyValueHandler.getObjectSupplier().get();
117     for (int i = 0; i < count; i++) {
118       assertEquals(i, objects.get(i));
119     }
120 
121     assertEquals(JsonToken.END_OBJECT, parser.currentToken());
122     assertNull(parser.nextToken());
123   }

125   @Test
126   void testSingleton() throws BindingException, IOException {
127 
128     JsonParser jsonParser
129         = new JsonFactory().createParser(ListPropertyValueHandlerTest.class.getResourceAsStream("list-singleton.json"));
130 
131     context.checking(new Expectations() {
132       {
133         allowing(parsingContext).getEventReader();
134         will(returnValue(jsonParser));
135         allowing(classBinding).getClazz();
136         will(returnValue(Object.class));
137         allowing(propertyBinding).getPropertyInfo();
138         will(returnValue(propertyInfo));
139         allowing(propertyInfo).getSimpleName();
140         will(returnValue("_property"));
141       }
142     });
143 
144     PropertyItemHandler propertyItemHandler = newPropertyItemHandler(propertyBinding);
145     ListPropertyValueHandler handler = new ListPropertyValueHandler(classBinding, propertyItemHandler, true);
146     parseProperty(jsonParser, handler, 1);
147     context.assertIsSatisfied();
148   }

150   @Test
151   void testSingletonFail() throws IOException {
152 
153     JsonParser jsonParser
154         = new JsonFactory().createParser(ListPropertyValueHandlerTest.class.getResourceAsStream("list-singleton.json"));
155 
156     context.checking(new Expectations() {
157       {
158         allowing(parsingContext).getEventReader();
159         will(returnValue(jsonParser));
160         allowing(classBinding).getClazz();
161         will(returnValue(Object.class));
162         allowing(propertyBinding).getPropertyInfo();
163         will(returnValue(propertyInfo));
164         allowing(propertyInfo).getSimpleName();
165         will(returnValue("_property"));
166       }
167     });
168 
169     PropertyItemHandler propertyItemHandler = newPropertyItemHandler(propertyBinding);
170     ListPropertyValueHandler handler = new ListPropertyValueHandler(classBinding, propertyItemHandler, false);
171     Assertions.assertThrows(BindingException.class, () -> {
172       parseProperty(jsonParser, handler, 1);
173     }, "Found unexpected 'START_OBJECT' token when parsing property '_property' on class 'Object' at location '2:15'. This list doesn't allow a singleton object.");
174     context.assertIsSatisfied();
175   }
