111   private static List<Entry> generateTestVectors() throws JsonParseException, IOException {
112     ObjectMapper mapper = new ObjectMapper();
113     // mapper.disable(DeserializationFeature.FAIL_ON_UNKNOWN_PROPERTIES);
114 
115     try (InputStream is = CommonmarkConformanceTest.class.getResourceAsStream("/commonmark-spec.json")) {
116       try (JsonParser parser = mapper.getFactory().createParser(is)) {
117         if (parser.nextToken() != JsonToken.START_ARRAY) {
118           throw new IllegalStateException();
119         }
120 
121         List<Entry> entries = new LinkedList<>();
122         while (parser.nextToken() != JsonToken.END_ARRAY) {
123           Entry entry = mapper
124               .readerFor(Entry.class)
125               .readValue(parser);
126           entries.add(entry);
127         }
128         return entries;
129       }
130     }
131   }

156   @NonNull
157   public String generateXmlInstance(@NonNull String html) {
158 
159     StringBuilder builder = new StringBuilder(206);
160     builder.append("<?xml version=\"1.0\" encoding=\"UTF-8\"?>\r\n"); // NOPMD
161 
162     String topLevelElementName;
163     if (isBlockElement(html)) {
164       topLevelElementName = "multiline";
165     } else {
166       topLevelElementName = "line";
167     }
168 
169     builder
170         .append(
171             "<!DOCTYPE " + topLevelElementName + " [\r\n" // NOPMD
172                 + "<!ENTITY nbsp \"&#160;\">\r\n"
173                 + "]>\r\n")
174         .append('<')
175         .append(topLevelElementName)
176         .append(" xmlns=\"" + METASCHEMA_NS + "\"") // NOPMD
177         .append(" xmlns:zoop=\"http://csrc.nist.gov/ns/oscal/metaschema-zoop/1.0\"")
178         .append('>')
179         .append(html)
180         .append("</")
181         .append(topLevelElementName)
182         .append(">\r\n");
183 
184     return ObjectUtils.notNull(builder.toString());
185   }

187   @NonNull
188   public String generateXmlInstance(@NonNull IMarkupString<?> content) throws XMLStreamException {
189 
190     WstxOutputFactory factory = new WstxOutputFactory();
191     factory.setProperty(XMLOutputFactory.IS_REPAIRING_NAMESPACES, true);
192     StringWriter strWriter = new StringWriter();
193 
194     XMLStreamWriter2 xmlWriter = (XMLStreamWriter2) factory.createXMLStreamWriter(strWriter);
195     xmlWriter.writeStartDocument();
196 
197     String topLevelElementName;
198     if (content instanceof MarkupMultiline) {
199       topLevelElementName = "multiline";
200     } else {
201       topLevelElementName = "line";
202     }
203 
204     xmlWriter.writeDTD("<!DOCTYPE " + topLevelElementName + " [\r\n" // NOPMD
205         + "<!ENTITY nbsp \"&#160;\">\r\n"
206         + "]>\r\n");
207 
208     xmlWriter.setPrefix("", METASCHEMA_NS);
209     xmlWriter.writeStartElement(METASCHEMA_NS, topLevelElementName);
210     xmlWriter.writeNamespace("", METASCHEMA_NS);
211     xmlWriter.writeNamespace("zoop", "http://csrc.nist.gov/ns/oscal/metaschema-zoop/1.0");
212 
213     content.writeXHtml(METASCHEMA_NS, xmlWriter);
214 
215     xmlWriter.writeEndElement();
216     xmlWriter.writeEndDocument();
217     xmlWriter.flush();
218 
219     return ObjectUtils.notNull(strWriter.toString());
220   }

225   Stream<DynamicNode> generateConversionTests() // NOPMD
226       throws IOException, URISyntaxException, SAXException {
227     Schema schema = loadDataTypeSchema();
228 
229     List<Entry> entries = generateTestVectors();
230 
231     return entries.stream()
232         .filter(Entry::isEnabled)
233         .collect(Collectors.groupingBy(
234             Entry::getSection,
235             Collectors.toList()))
236         .entrySet().stream()
237         .map(entry -> {
238           String sectionName = entry.getKey();
239           List<Entry> testVectors = entry.getValue();
240           return DynamicContainer.dynamicContainer(
241               sectionName,
242               testVectors.stream()
243                   .map(testVector -> {
244                     String testName = new StringBuilder()
245                         .append(testVector.getSection())
246                         .append(" (")
247                         .append(testVector.getExampleNumber())
248                         .append(')')
249                         .toString();
250                     return DynamicContainer.dynamicContainer(
251                         testName,
252                         Stream.of(
253                             DynamicTest.dynamicTest(
254                                 testVector.getMarkdown(),
255                                 () -> {
256                                   // do nothing
257                                 }),
258                             DynamicContainer.dynamicContainer(
259                                 "Markdown To XML/HTML Conversion",
260                                 Stream.of(
261                                     DynamicTest.dynamicTest(
262                                         "Validate Markdown To XML/HTML",
263                                         () -> {
264                                           String markdown = testVector.getMarkdown();
265                                           AbstractMarkupString<?> content;
266                                           content = MarkupDataTypeProvider.MARKUP_MULTILINE.parse(markdown);
267 
268                                           String convertedHtmlInstance = generateXmlInstance(content);
269 
270                                           // validate the produced content
271                                           StringReader reader = new StringReader(convertedHtmlInstance);
272                                           StreamSource source = new StreamSource(reader);
273                                           Validator validator = schema.newValidator();
274                                           validator.setErrorHandler(new Handler());
275                                           try {
276                                             validator.validate(source);
277                                           } catch (SAXParseException ex) {
278                                             fail(String.format("Invalid content '%s'. %s",
279                                                 content,
280                                                 ex.getLocalizedMessage()));
281                                           }
282                                         }),
283                                     DynamicTest.dynamicTest(
284                                         "Convert Markdown to HTML/XHTML and Match with Test Vector",
285                                         () -> {
286                                           String markdown = testVector.getMarkdown();
287                                           IMarkupString<?> content;
288 
289                                           content = MarkupDataTypeProvider.MARKUP_MULTILINE.parse(markdown);
290                                           String convertedHtmlInstance = generateXmlInstance(content);
291 
292                                           // extract the generated HTML
293                                           String topLevelElementName = "multiline";
294                                           convertedHtmlInstance = convertedHtmlInstance.substring(
295                                               convertedHtmlInstance.indexOf("<" + topLevelElementName),
296                                               convertedHtmlInstance.indexOf("</" + topLevelElementName + ">"));
297                                           Pattern pattern = Pattern.compile(
298                                               String.format("^<%s[^>]+>(.+)$",
299                                                   topLevelElementName),
300                                               Pattern.DOTALL);
301                                           Matcher matcher = pattern.matcher(convertedHtmlInstance);
302 
303                                           if (!matcher.matches()) {
304                                             fail(String.format(
305                                                 "Unable to extract converted content using pattern '%s' from: %s",
306                                                 pattern.pattern(),
307                                                 convertedHtmlInstance));
308                                           }
309 
310                                           // compare the generated XML
311                                           String actualXml = matcher.group(1);
312 
313                                           // replace q tags
314                                           actualXml = QUOTE_TAG_REPLACEMENT_PATTERN.matcher(actualXml)
315                                               .replaceAll("\"");
316 
317                                           assertEquals(testVector.getXml(), actualXml);
318 
319                                           // compare the generated HTML
320                                           String actualHtml = content.toHtml();
321 
322                                           assertEquals(testVector.getRawHtml(), actualHtml);
323                                         }))),
324                             DynamicTest.dynamicTest(
325                                 "Validate XML Test Vector",
326                                 () -> {
327                                   // Document data = testVector.getHtmlDocument();
328                                   // String html = data.body().html();
329                                   String html = testVector.getHtml();
330                                   String xmlTestVector = generateXmlInstance(html);
331 
332                                   // System.out.println(xmlTestVector);

334 
335                                   StreamSource source = new StreamSource(reader);

339                                     validator.validate(source);
340                                   } catch (SAXParseException ex) {
341                                     fail(String.format("Invalid vector '%s'. %s",

343                                         ex.getLocalizedMessage()));
344                                   }

350   @JsonIgnoreProperties({ "comment", "start_line", "end_line" })
351   @SuppressWarnings("PMD.DataClass")
352   private static class Entry {
353     @NonNull
354     private final String markdown;
355     @NonNull
356     private final String rawHtml;
357     @NonNull
358     private final Document html;
359     @NonNull
360     private final String xml;
361     private final int exampleNumber;
362     @NonNull
363     private final String section;
364     private final boolean enabled;
365 
366     @JsonCreator
367     public Entry(
368         @JsonProperty("markdown") @NonNull String markdown,
369         @JsonProperty("html") @NonNull String html,
370         @JsonProperty("xml") @Nullable String xml,
371         @JsonProperty("example") int exampleNumber,
372         @JsonProperty("section") @NonNull String section,
373         @JsonProperty("enabled") @Nullable Boolean enabled) {
374       this.markdown = markdown;
375       this.rawHtml = ObjectUtils.notNull(html.trim());
376       Document document = Jsoup.parseBodyFragment(html);
377       document.outputSettings().syntax(Document.OutputSettings.Syntax.xml);
378       this.html = document;
379       this.xml = ObjectUtils.notNull(
380           (xml == null ? html : xml) // allow for XML overrides
381               .replace(" />", "/>") // the XML stack produces tighter closing elements
382               .replaceAll("(?<!>|\")&gt;", ">") // the XML stack doesn't need these entities
383               .stripTrailing()); // this implementation removes trailing whitespace
384 
385       this.exampleNumber = exampleNumber;
386       this.section = section;
387       this.enabled = enabled == null || enabled;
388     }
389 
390     @NonNull
391     public String getMarkdown() {
392       return markdown;
393     }
394 
395     @NonNull
396     protected Document getHtmlDocument() {
397       return html;
398     }
399 
400     @NonNull
401     public String getRawHtml() {
402       return rawHtml;
403     }
404 
405     @SuppressWarnings("null")
406     @NonNull
407     public String getHtml() {
408       return getHtmlDocument().body().html();
409     }
410 
411     @NonNull
412     public String getXml() {
413       return xml;
414     }
415 
416     public int getExampleNumber() {
417       return exampleNumber;
418     }
419 
420     @NonNull
421     public String getSection() {
422       return section;
423     }
424 
425     public boolean isEnabled() {
426       return enabled;
427     }
