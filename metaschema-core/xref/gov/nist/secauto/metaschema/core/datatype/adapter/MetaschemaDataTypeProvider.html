27  package gov.nist.secauto.metaschema.core.datatype.adapter;
28  
29  import com.google.auto.service.AutoService;
30  
31  import gov.nist.secauto.metaschema.core.datatype.AbstractDataTypeProvider;
32  import gov.nist.secauto.metaschema.core.datatype.IDataTypeProvider;
33  
34  import edu.umd.cs.findbugs.annotations.NonNull;
35  
36  /**
37   * Provides for runtime discovery of built-in implementations of the core
38   * Metaschema data types.
39   */
40  @AutoService(IDataTypeProvider.class)
41  public final class MetaschemaDataTypeProvider // NOPMD - Used for service initialization
42      extends AbstractDataTypeProvider {
43    @NonNull
44    public static final Base64Adapter BASE64 = new Base64Adapter();
45    @NonNull
46    public static final BooleanAdapter BOOLEAN = new BooleanAdapter();
47    @NonNull
48    public static final DateAdapter DATE = new DateAdapter();
49    @NonNull
50    public static final DateWithTZAdapter DATE_WITH_TZ = new DateWithTZAdapter();
51    @NonNull
52    public static final DateTimeAdapter DATE_TIME = new DateTimeAdapter();
53    @NonNull
54    public static final DateTimeWithTZAdapter DATE_TIME_WITH_TZ = new DateTimeWithTZAdapter();
55    @NonNull
56    public static final IPv4AddressAdapter IP_V4_ADDRESS = new IPv4AddressAdapter();
57    @NonNull
58    public static final IPv6AddressAdapter IP_V6_ADDRESS = new IPv6AddressAdapter();
59    @NonNull
60    public static final UriAdapter URI = new UriAdapter();
61    @NonNull
62    public static final UriReferenceAdapter URI_REFERENCE = new UriReferenceAdapter();
63    @NonNull
64    public static final UuidAdapter UUID = new UuidAdapter();
65  
66    @NonNull
67    public static final DayTimeAdapter DAY_TIME_DURATION = new DayTimeAdapter();
68    @NonNull
69    public static final YearMonthAdapter YEAR_MONTH_DURATION = new YearMonthAdapter();
70  
71    @NonNull
72    public static final DecimalAdapter DECIMAL = new DecimalAdapter();
73    @NonNull
74    public static final IntegerAdapter INTEGER = new IntegerAdapter();
75    @NonNull
76    public static final NonNegativeIntegerAdapter NON_NEGATIVE_INTEGER = new NonNegativeIntegerAdapter();
77    @NonNull
78    public static final PositiveIntegerAdapter POSITIVE_INTEGER = new PositiveIntegerAdapter();
79  
80    @NonNull
81    public static final EmailAddressAdapter EMAIL_ADDRESS = new EmailAddressAdapter();
82    @NonNull
83    public static final HostnameAdapter HOSTNAME = new HostnameAdapter();
84    @NonNull
85    public static final NcNameAdapter NCNAME = new NcNameAdapter();
86    @NonNull
87    public static final StringAdapter STRING = new StringAdapter();
88    @NonNull
89    public static final TokenAdapter TOKEN = new TokenAdapter();
90  
91    @NonNull
92    public static final StringAdapter DEFAULT_DATA_TYPE = STRING;
93  
94    /**
95     * Initialize the built-in data types.
96     */
97    public MetaschemaDataTypeProvider() {
98      // The data type "string" must be first since this is the default data type for
99      // the {@link String}
100     // Java type. This ensures that when a data type is resolved that this data type
101     // is matched first
102     // before other String-based data types.
103     registerDatatype(STRING);
104 
105     registerDatatype(BASE64);
106     registerDatatype(BOOLEAN);
107     registerDatatype(DATE);
108     registerDatatype(DATE_WITH_TZ);
109     registerDatatype(DATE_TIME);
110     registerDatatype(DATE_TIME_WITH_TZ);
111     registerDatatype(DAY_TIME_DURATION);
112     registerDatatype(DECIMAL);
113     registerDatatype(EMAIL_ADDRESS);
114     registerDatatype(HOSTNAME);
115     registerDatatype(INTEGER);
116     registerDatatype(IP_V4_ADDRESS);
117     registerDatatype(IP_V6_ADDRESS);
118 
119     registerDatatype(NON_NEGATIVE_INTEGER);
120     registerDatatype(POSITIVE_INTEGER);
121     registerDatatype(TOKEN);
122     registerDatatype(URI);
123     registerDatatype(URI_REFERENCE);
124     registerDatatype(UUID);
125     registerDatatype(YEAR_MONTH_DURATION);
126   }
127 }
