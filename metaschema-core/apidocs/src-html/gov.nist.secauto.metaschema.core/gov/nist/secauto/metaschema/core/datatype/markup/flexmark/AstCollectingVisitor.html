027package gov.nist.secauto.metaschema.core.datatype.markup.flexmark;
028
029import com.vladsch.flexmark.util.ast.Node;
030import com.vladsch.flexmark.util.ast.NodeVisitorBase;
031
032import edu.umd.cs.findbugs.annotations.NonNull;
033
034public class AstCollectingVisitor
035    extends NodeVisitorBase {
036  public static final String EOL = "\n";
037  @NonNull
038  protected StringBuilder output = new StringBuilder(); // NOPMD - intentional
039  protected int indent; // 0;
040  protected boolean eolPending; // false;
041
042  @NonNull
043  public static String asString(@NonNull Node node) {
044    AstCollectingVisitor visitor = new AstCollectingVisitor();
045    visitor.collect(node);
046    return visitor.getAst();
047  }
048
049  @SuppressWarnings("null")
050  @NonNull
051  public String getAst() {
052    return output.toString();
053  }
054
055  public void clear() {
056    output = new StringBuilder();
057    indent = 0;
058    eolPending = false;
059  }
060
061  protected void appendIndent() {
062    for (int i = 0; i < indent * 2; i++) {
063      output.append(' ');
064    }
065    eolPending = true;
066  }
067
068  protected void appendEOL() {
069    output.append(EOL);
070    eolPending = false;
071  }
072
073  protected void appendPendingEOL() {
074    if (eolPending) {
075      appendEOL();
076    }
077  }
078
079  public void collect(@NonNull Node node) {
080    visit(node);
081  }
082
083  @Override
084  protected void visit(Node node) {
085    assert node != null;
086    appendIndent();
087    node.astString(output, true);
088    output.append(EOL);
089    indent++;
090
091    try {
092      super.visitChildren(node);
093    } finally {
094      indent--;
095    }
096  }
097}
