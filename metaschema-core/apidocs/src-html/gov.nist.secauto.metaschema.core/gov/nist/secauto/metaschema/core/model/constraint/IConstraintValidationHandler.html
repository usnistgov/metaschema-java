027package gov.nist.secauto.metaschema.core.model.constraint;
028
029import gov.nist.secauto.metaschema.core.metapath.DynamicContext;
030import gov.nist.secauto.metaschema.core.metapath.ISequence;
031import gov.nist.secauto.metaschema.core.metapath.MetapathException;
032import gov.nist.secauto.metaschema.core.metapath.item.node.INodeItem;
033
034import java.util.List;
035
036import edu.umd.cs.findbugs.annotations.NonNull;
037
038public interface IConstraintValidationHandler {
039
040  void handleCardinalityMinimumViolation(
041      @NonNull ICardinalityConstraint constraint,
042      @NonNull INodeItem node,
043      @NonNull ISequence<? extends INodeItem> targets);
044
045  void handleCardinalityMaximumViolation(
046      @NonNull ICardinalityConstraint constraint,
047      @NonNull INodeItem node,
048      @NonNull ISequence<? extends INodeItem> targets);
049
050  void handleIndexDuplicateViolation(
051      @NonNull IIndexConstraint constraint,
052      @NonNull INodeItem node);
053
054  void handleIndexDuplicateKeyViolation(
055      @NonNull IIndexConstraint constraint,
056      @NonNull INodeItem node,
057      @NonNull INodeItem oldItem,
058      @NonNull INodeItem target);
059
060  void handleUniqueKeyViolation(
061      @NonNull IUniqueConstraint constraint,
062      @NonNull INodeItem node,
063      @NonNull INodeItem oldItem,
064      @NonNull INodeItem target);
065
066  void handleMatchPatternViolation(
067      @NonNull IMatchesConstraint constraint,
068      @NonNull INodeItem node,
069      @NonNull INodeItem target,
070      @NonNull String value);
071
072  void handleMatchDatatypeViolation(
073      @NonNull IMatchesConstraint constraint,
074      @NonNull INodeItem node,
075      @NonNull INodeItem target,
076      @NonNull String value,
077      @NonNull IllegalArgumentException cause);
078
079  void handleExpectViolation(
080      @NonNull IExpectConstraint constraint,
081      @NonNull INodeItem node,
082      @NonNull INodeItem target,
083      @NonNull DynamicContext metapathContext);
084
085  void handleKeyMatchError(
086      @NonNull IKeyConstraint constraint,
087      @NonNull INodeItem node,
088      @NonNull INodeItem target,
089      @NonNull MetapathException ex);
090
091  void handleIndexMiss(
092      @NonNull IIndexHasKeyConstraint constraint,
093      @NonNull INodeItem node,
094      @NonNull INodeItem target,
095      @NonNull List<String> key);
096
097  void handleAllowedValuesViolation(
098      @NonNull List<IAllowedValuesConstraint> failedConstraints,
099      @NonNull INodeItem target);
100}
