027package gov.nist.secauto.metaschema.core.metapath.function.library;
028
029import gov.nist.secauto.metaschema.core.metapath.DynamicContext;
030import gov.nist.secauto.metaschema.core.metapath.ISequence;
031import gov.nist.secauto.metaschema.core.metapath.MetapathConstants;
032import gov.nist.secauto.metaschema.core.metapath.function.IArgument;
033import gov.nist.secauto.metaschema.core.metapath.function.IFunction;
034import gov.nist.secauto.metaschema.core.metapath.item.IItem;
035import gov.nist.secauto.metaschema.core.metapath.item.atomic.IBooleanItem;
036import gov.nist.secauto.metaschema.core.util.ObjectUtils;
037
038import java.util.List;
039
040import edu.umd.cs.findbugs.annotations.NonNull;
041
042public final class FnNot {
043  @NonNull
044  static final IFunction SIGNATURE = IFunction.builder()
045      .name("not")
046      .namespace(MetapathConstants.NS_XPATH_FUNCTIONS)
047      .deterministic()
048      .contextIndependent()
049      .focusIndependent()
050      .argument(IArgument.newBuilder()
051          .name("arg")
052          .type(IItem.class)
053          .zeroOrMore()
054          .build())
055      .returnType(IBooleanItem.class)
056      .returnOne()
057      .functionHandler(FnNot::execute)
058      .build();
059
060  private FnNot() {
061    // disable construction
062  }
063
064  @SuppressWarnings("unused")
065  @NonNull
066  private static ISequence<IBooleanItem> execute(@NonNull IFunction function,
067      @NonNull List<ISequence<?>> arguments,
068      @NonNull DynamicContext dynamicContext,
069      IItem focus) {
070
071    ISequence<?> items = ObjectUtils.requireNonNull(arguments.iterator().next());
072
073    IBooleanItem result = fnNot(items);
074    return ISequence.of(result);
075  }
076
077  /**
078   * Get the negated, effective boolean value of the provided item.
079   * <p>
080   * Based on the XPath 3.1
081   * <a href="https://www.w3.org/TR/xpath-functions-31/#func-not">fn:not</a>
082   * function.
083   *
084   * @param item
085   *          the item to get the negated, effective boolean value for
086   * @return the negated boolean value
087   */
088  @NonNull
089  public static IBooleanItem fnNot(@NonNull IItem item) {
090    return IBooleanItem.valueOf(!FnBoolean.fnBooleanAsPrimitive(item));
091  }
092
093  /**
094   * Get the negated, effective boolean value of the provided item.
095   * <p>
096   * Based on the XPath 3.1
097   * <a href="https://www.w3.org/TR/xpath-functions-31/#func-not">fn:not</a>
098   * function.
099   *
100   * @param sequence
101   *          the sequence to get the negated, effective boolean value for
102   * @return the negated boolean value
103   */
104  @NonNull
105  public static IBooleanItem fnNot(ISequence<?> sequence) {
106    return FnBoolean.fnBoolean(sequence).negate();
107  }
108}
