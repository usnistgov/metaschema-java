027package gov.nist.secauto.metaschema.core.metapath;
028
029import gov.nist.secauto.metaschema.core.metapath.item.node.IDocumentNodeItem;
030import gov.nist.secauto.metaschema.core.model.IResourceLoader;
031import gov.nist.secauto.metaschema.core.model.IUriResolver;
032import gov.nist.secauto.metaschema.core.util.ObjectUtils;
033
034import java.io.File;
035import java.io.IOException;
036import java.io.InputStream;
037import java.net.URI;
038import java.net.URISyntaxException;
039import java.net.URL;
040import java.nio.file.Files;
041import java.nio.file.Path;
042
043import edu.umd.cs.findbugs.annotations.NonNull;
044
045public interface IDocumentLoader extends IResourceLoader {
046  void setUriResolver(@NonNull IUriResolver resolver);
047
048  @NonNull
049  default IDocumentNodeItem loadAsNodeItem(@NonNull File file) throws IOException {
050    return loadAsNodeItem(ObjectUtils.notNull(file.toPath()));
051  }
052
053  @NonNull
054  default IDocumentNodeItem loadAsNodeItem(@NonNull Path path) throws IOException {
055    try (InputStream is = ObjectUtils.notNull(Files.newInputStream(path))) {
056      return loadAsNodeItem(is, ObjectUtils.notNull(path.toUri()));
057    }
058  }
059
060  @NonNull
061  default IDocumentNodeItem loadAsNodeItem(@NonNull URL url) throws IOException, URISyntaxException {
062    return loadAsNodeItem(ObjectUtils.notNull(url.toURI()));
063  }
064
065  @NonNull
066  IDocumentNodeItem loadAsNodeItem(@NonNull URI uri) throws IOException;
067
068  @NonNull
069  IDocumentNodeItem loadAsNodeItem(@NonNull InputStream is, @NonNull URI documentUri) throws IOException;
070}
