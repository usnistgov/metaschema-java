027package gov.nist.secauto.metaschema.core.datatype.adapter;
028
029import com.fasterxml.jackson.databind.jsonFormatVisitors.JsonFormatTypes;
030
031import gov.nist.secauto.metaschema.core.datatype.AbstractDataTypeAdapter;
032import gov.nist.secauto.metaschema.core.metapath.item.atomic.IDateItem;
033import gov.nist.secauto.metaschema.core.util.ObjectUtils;
034
035import java.time.ZonedDateTime;
036import java.time.format.DateTimeParseException;
037import java.util.List;
038import java.util.regex.Matcher;
039import java.util.regex.Pattern;
040
041import edu.umd.cs.findbugs.annotations.NonNull;
042
043public class DateWithTZAdapter
044    extends AbstractDataTypeAdapter<ZonedDateTime, IDateItem> {
045  @NonNull
046  private static final List<String> NAMES = ObjectUtils.notNull(
047      List.of(
048          "date-with-timezone"));
049  private static final Pattern DATE_TIMEZONE = Pattern.compile("^("
050      + "^(?:(?:2000|2400|2800|(?:19|2[0-9](?:0[48]|[2468][048]|[13579][26])))-02-29)"
051      + "|(?:(?:(?:19|2[0-9])[0-9]{2})-02-(?:0[1-9]|1[0-9]|2[0-8]))"
052      + "|(?:(?:(?:19|2[0-9])[0-9]{2})-(?:0[13578]|10|12)-(?:0[1-9]|[12][0-9]|3[01]))"
053      + "|(?:(?:(?:19|2[0-9])[0-9]{2})-(?:0[469]|11)-(?:0[1-9]|[12][0-9]|30))"
054      + ")"
055      + "(Z|[+-][0-9]{2}:[0-9]{2})$");
056
057  DateWithTZAdapter() {
058    super(ZonedDateTime.class);
059  }
060
061  @Override
062  public List<String> getNames() {
063    return NAMES;
064  }
065
066  @Override
067  public JsonFormatTypes getJsonRawType() {
068    return JsonFormatTypes.STRING;
069  }
070
071  @SuppressWarnings("null")
072  @Override
073  public ZonedDateTime parse(String value) {
074    Matcher matcher = DATE_TIMEZONE.matcher(value);
075    if (!matcher.matches()) {
076      throw new IllegalArgumentException("Invalid date: " + value);
077    }
078    String parseValue = String.format("%sT00:00:00%s", matcher.group(1), matcher.group(2));
079    try {
080      return ZonedDateTime.from(DateFormats.DATE_TIME_WITH_TZ.parse(parseValue));
081    } catch (DateTimeParseException ex) {
082      throw new IllegalArgumentException(ex.getLocalizedMessage(), ex);
083    }
084  }
085
086  @SuppressWarnings("null")
087  @Override
088  public String asString(Object value) {
089    return DateFormats.DATE_WITH_TZ.format((ZonedDateTime) value);
090  }
091
092  @SuppressWarnings("null")
093  @Override
094  public ZonedDateTime copy(Object obj) {
095    return ZonedDateTime.from((ZonedDateTime) obj);
096  }
097
098  @Override
099  public Class<IDateItem> getItemClass() {
100    return IDateItem.class;
101  }
102
103  @Override
104  public IDateItem newItem(Object value) {
105    ZonedDateTime item = toValue(value);
106    return IDateItem.valueOf(item);
107  }
108}
