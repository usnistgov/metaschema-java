90    @Test
91    void testJsonRead()
92        throws JsonParseException, IOException, NoSuchFieldException {
93      String json = "{ \"field1\": \"field1value\", \"fields2\": [ \"field2value\" ] } }";
94      JsonFactory factory = new JsonFactory();
95      try (JsonParser jsonParser = factory.createParser(json)) {
96        Class<?> theClass = TestField.class;
97  
98        context.checking(new Expectations() {
99          { // NOPMD - intentional
100           allowing(bindingContext).getJavaTypeAdapterInstance(StringAdapter.class);
101           will(returnValue(MetaschemaDataTypeProvider.STRING));
102           allowing(bindingContext).getClassBinding(String.class);
103           will(returnValue(null));
104 
105           allowing(classBinding).getBoundClass();
106           will(returnValue(theClass));
107           allowing(classBinding).getBindingContext();
108           will(returnValue(bindingContext));
109 
110           allowing(jsonParsingContext).getReader();
111           will(returnValue(jsonParser));
112         }
113       });
114 
115       java.lang.reflect.Field field1 = ObjectUtils.requireNonNull(theClass.getDeclaredField("field1"));
116       IBoundFieldInstance field1Property = IBoundFieldInstance.newInstance(
117           field1,
118           ObjectUtils.notNull(classBinding));
119 
120       java.lang.reflect.Field field2 = ObjectUtils.requireNonNull(theClass.getDeclaredField("_field2"));
121       IBoundFieldInstance field2Property = IBoundFieldInstance.newInstance(
122           field2,
123           ObjectUtils.notNull(classBinding));
124 
125       TestField obj = new TestField();
126 
127       assertAll(
128           () -> assertEquals(JsonToken.START_OBJECT, jsonParser.nextToken()),
129           () -> assertEquals("field1", jsonParser.nextFieldName()),
130           () -> assertTrue(field1Property.read(obj, jsonParsingContext)),
131           () -> assertEquals("field1value", obj.getField1()),
132 
133           () -> assertEquals(JsonToken.FIELD_NAME, jsonParser.currentToken()),
134           () -> assertEquals("fields2", jsonParser.currentName()),
135 
136           () -> assertTrue(field2Property.read(obj, jsonParsingContext)),
137           () -> assertTrue(obj.getField2() instanceof LinkedList),
138           () -> assertIterableEquals(Collections.singleton("field2value"), obj.getField2()));
139 
140       // assertEquals(JsonToken.START_OBJECT, jsonParser.nextToken());
141       // assertEquals(JsonToken.FIELD_NAME, jsonParser.nextToken());
142       // assertEquals("id", jsonParser.currentName());
143     }
144   }

263   @Test
264   void testJsonReadMissingFieldValue()
265       throws JsonParseException, IOException, NoSuchFieldException {
266     String json = "{ \"test\":\n" + "  { \"fields2\": [\n" + "    \"field2value\"\n" + "    ]\n" + "  }\n" + "}\n";
267     JsonFactory factory = new JsonFactory();
268     JsonParser jsonParser = factory.createParser(json);
269     Class<?> theClass = TestField.class;
270 
271     context.checking(new Expectations() {
272       { // NOPMD - intentional
273         allowing(bindingContext).getJavaTypeAdapterInstance(StringAdapter.class);
274         will(returnValue(MetaschemaDataTypeProvider.STRING));
275         allowing(bindingContext).getClassBinding(String.class);
276         will(returnValue(null));
277 
278         allowing(classBinding).getBoundClass();
279         will(returnValue(theClass));
280         allowing(classBinding).getBindingContext();
281         will(returnValue(bindingContext));
282 
283         allowing(jsonParsingContext).getReader();
284         will(returnValue(jsonParser));
285       }
286     });
287 
288     java.lang.reflect.Field field1 = theClass.getDeclaredField("field1");
289     IBoundFieldInstance field1Property = IBoundFieldInstance.newInstance(
290         field1,
291         ObjectUtils.notNull(classBinding));
292     java.lang.reflect.Field field2 = theClass.getDeclaredField("_field2");
293     IBoundFieldInstance field2Property = IBoundFieldInstance.newInstance(
294         field2,
295         ObjectUtils.notNull(classBinding));
296 
297     TestField obj = new TestField();
298 
299     // Advance to first property to parse
300     assertEquals(JsonToken.START_OBJECT, jsonParser.nextToken());
301     assertEquals("test", jsonParser.nextFieldName());
302     assertEquals(JsonToken.START_OBJECT, jsonParser.nextToken());
303     assertEquals("fields2", jsonParser.nextFieldName());
304 
305     // attempt to parse the missing property
306     assertFalse(field1Property.read(obj, jsonParsingContext));
307     assertEquals(null, obj.getField1());
308 
309     // attempt to parse the existing property
310     assertEquals(JsonToken.FIELD_NAME, jsonParser.currentToken());
311     assertEquals("fields2", jsonParser.currentName());
312 
313     assertTrue(field2Property.read(obj, jsonParsingContext));
314     assertTrue(obj.getField2() instanceof LinkedList);
315     assertIterableEquals(Collections.singleton("field2value"), obj.getField2());
316   }
