027package gov.nist.secauto.metaschema.binding.io;
028
029import gov.nist.secauto.metaschema.model.common.configuration.IConfiguration;
030import gov.nist.secauto.metaschema.model.common.configuration.IMutableConfiguration;
031import gov.nist.secauto.metaschema.model.common.util.ObjectUtils;
032
033import java.io.File;
034import java.io.IOException;
035import java.io.OutputStream;
036import java.io.OutputStreamWriter;
037import java.io.Writer;
038import java.nio.charset.StandardCharsets;
039import java.nio.file.Files;
040import java.nio.file.OpenOption;
041import java.nio.file.Path;
042import java.nio.file.StandardOpenOption;
043
044import edu.umd.cs.findbugs.annotations.NonNull;
045
046/**
047 * Implementations of this interface are able to write data in a bound object instance of the
048 * parameterized type to a structured data format.
049 *
050 * @param <CLASS>
051 *          the Java type from which data can be written
052 */
053public interface ISerializer<CLASS> extends IMutableConfiguration<SerializationFeature<?>> {
054
055  @Override
056  ISerializer<CLASS> enableFeature(SerializationFeature<?> feature);
057
058  @Override
059  ISerializer<CLASS> disableFeature(SerializationFeature<?> feature);
060
061  @Override
062  ISerializer<CLASS> applyConfiguration(IConfiguration<SerializationFeature<?>> other);
063
064  @Override
065  ISerializer<CLASS> set(SerializationFeature<?> feature, Object value);
066
067  /**
068   * Write data from a bound class instance to the {@link OutputStream}.
069   * <p>
070   * This method does not have ownership of the the provided output stream and will not close it.
071   *
072   * @param data
073   *          the instance data
074   * @param os
075   *          the output stream to write to
076   * @throws IOException
077   *           if an error occurred while writing data to the stream
078   */
079  default void serialize(@NonNull CLASS data, @NonNull OutputStream os) throws IOException {
080    OutputStreamWriter writer = new OutputStreamWriter(os, StandardCharsets.UTF_8);
081    serialize(data, writer);
082    writer.flush();
083  }
084
085  /**
086   * Write data from a bound class instance to the {@link File}.
087   *
088   * @param data
089   *          the instance data
090   * @param path
091   *          the file to write to
092   * @param openOptions
093   *          options specifying how the file is opened
094   * @throws IOException
095   *           if an error occurred while writing data to the file indicated by the {@code path}
096   *           parameter
097   */
098  default void serialize(@NonNull CLASS data, @NonNull Path path, OpenOption... openOptions) throws IOException {
099    try (Writer writer = Files.newBufferedWriter(path, StandardCharsets.UTF_8, openOptions)) {
100      assert writer != null;
101      serialize(data, writer);
102    }
103  }
104
105  /**
106   * Write data from a bound class instance to the {@link File}.
107   *
108   * @param data
109   *          the instance data
110   * @param file
111   *          the file to write to
112   * @throws IOException
113   *           if an error occurred while writing data to the stream
114   */
115  default void serialize(@NonNull CLASS data, @NonNull File file) throws IOException {
116    serialize(data, ObjectUtils.notNull(file.toPath()), StandardOpenOption.CREATE, StandardOpenOption.WRITE,
117        StandardOpenOption.TRUNCATE_EXISTING);
118  }
119
120  /**
121   * Write data from a bound class instance to the {@link Writer}.
122   *
123   * @param data
124   *          the instance data
125   * @param writer
126   *          the writer to write to
127   * @throws IOException
128   *           if an error occurred while writing data to the stream
129   */
130  void serialize(@NonNull CLASS data, @NonNull Writer writer) throws IOException;
131}
