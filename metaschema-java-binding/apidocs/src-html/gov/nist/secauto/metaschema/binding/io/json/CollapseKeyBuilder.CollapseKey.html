049public class CollapseKeyBuilder {
050  @NonNull
051  private final IFieldClassBinding classBinding;
052  @NonNull
053  private final List<IBoundFlagInstance> flagProperties;
054  @NonNull
055  private final Map<CollapseKey, List<Object>> keyToValuesMap;
056
057  public CollapseKeyBuilder(@NonNull IFieldClassBinding classBinding) {
058    this.classBinding = classBinding;
059    this.flagProperties = new ArrayList<>(classBinding.getFlagInstances());
060    this.keyToValuesMap = new LinkedHashMap<>();
061  }
062
063  @NonNull
064  protected IFieldClassBinding getClassBinding() {
065    return classBinding;
066  }
067
068  @NonNull
069  protected List<IBoundFlagInstance> getFlagProperties() {
070    return flagProperties;
071  }
072
073  public void addAll(@NonNull Collection<? extends Object> instances) {
074    for (Object instance : instances) {
075      add(ObjectUtils.notNull(instance));
076    }
077  }
078
079  // TODO: check and handle nullness of values
080  public void add(@NonNull Object instance) {
081    int size = getFlagProperties().size();
082    Object[] flagValues = new Object[size];
083    int index = 0;
084    for (IBoundFlagInstance flag : getFlagProperties()) {
085      flagValues[index++] = flag.getValue(instance);
086    }
087
088    CollapseKey key = new CollapseKey(flagValues);
089    List<Object> values = this.keyToValuesMap.get(key);
090    if (values == null) {
091      values = new LinkedList<>();
092      this.keyToValuesMap.put(key, values);
093    }
094
095    Object value = getClassBinding().getFieldValueInstance().getValue(instance);
096    values.add(value);
097  }
098
099  public void write(boolean writeObjectWrapper, @NonNull IJsonWritingContext context) throws IOException {
100    IFieldClassBinding classBinding = getClassBinding();
101    IBoundFlagInstance jsonKey = classBinding.getJsonKeyFlagInstance();
102    IBoundFlagInstance jsonValueKey = classBinding.getJsonValueKeyFlagInstance();
103    IBoundFieldValueInstance fieldValue = classBinding.getFieldValueInstance();
104    List<IBoundFlagInstance> flagProperties = getFlagProperties();
105
106    // first build an index of the flag properties
107    List<Integer> flagIndex;
108    Integer jsonKeyIndex = null;
109    Integer jsonValueKeyIndex = null;
110    if (flagProperties.isEmpty()) {
111      flagIndex = Collections.emptyList();
112    } else {
113      flagIndex = new ArrayList<>(flagProperties.size());
114      int index = 0;
115      for (IBoundFlagInstance flag : flagProperties) {
116        if (jsonKey != null && jsonKey.equals(flag)) {
117          jsonKeyIndex = index++;
118        } else if (jsonValueKey != null && jsonValueKey.equals(flag)) {
119          jsonValueKeyIndex = index++;
120        } else {
121          // regular properties
122          flagIndex.add(index++);
123        }
124      }
125    }
126
127    @SuppressWarnings("resource") // not owned
128    JsonGenerator writer = context.getWriter(); // NOPMD - intentional
129    // for each key, we need to write the properties
130    for (Map.Entry<CollapseKey, List<Object>> entry : keyToValuesMap.entrySet()) {
131      CollapseKey key = entry.getKey();
132
133      Object[] flagValues = key.getFlagValues();
134
135      if (writeObjectWrapper) {
136        writer.writeStartObject();
137      }
138
139      // first write the JSON key if it is configured
140      if (jsonKey != null && jsonKeyIndex != null) {
141        // the field
142        writer.writeFieldName(jsonKey.getValueAsString(flagValues[jsonKeyIndex]));
143
144        writer.writeStartObject();
145      }
146
147      // next, write the flags
148      for (int index : flagIndex) {
149        IBoundFlagInstance flag = flagProperties.get(index);
150        Object flagValue = flagValues[index];
151
152        if (flagValue != null) {
153          writer.writeFieldName(flag.getJsonName());
154          flag.writeValue(flagValue, context);
155        }
156      }
157
158      // finally write the field value
159      List<Object> fieldValues = entry.getValue();
160      if (!fieldValues.isEmpty()) {
161        String valueKeyName;
162        if (jsonValueKey == null || jsonValueKeyIndex == null) {
163          valueKeyName = fieldValue.getJsonValueKeyName();
164        } else {
165          valueKeyName = jsonValueKey.getValueAsString(flagValues[jsonValueKeyIndex]);
166        }
167        writer.writeFieldName(valueKeyName);
168        if (fieldValues.size() > 1) {
169          writer.writeStartArray();
170        }
171        for (Object value : fieldValues) {
172          fieldValue.writeValue(value, context);
173        }
174        if (fieldValues.size() > 1) {
175          writer.writeEndArray();
176        }
177      }
178
179      if (jsonKey != null) {
180        // close the JSON key's object
181        writer.writeEndObject();
182      }
183
184      if (writeObjectWrapper) {
185        writer.writeEndObject();
186      }
187    }
188  }

190  public class CollapseKey {
191    private final Object[] flagValues;
192    private Integer cachedHashCode;
193
194    @SuppressFBWarnings(value = "EI_EXPOSE_REP2", justification = "access is restricted")
195    public CollapseKey(Object... values) {
196      this.flagValues = Arrays.copyOf(values, values.length);
197    }
198
199    protected Object[] getFlagValues() {
200      return Arrays.copyOf(flagValues, flagValues.length);
201    }
202
203    @Override
204    public int hashCode() {
205      synchronized (this) {
206        if (cachedHashCode == null) {
207          final int prime = 31;
208          cachedHashCode = 1;
209          cachedHashCode = prime * cachedHashCode + getEnclosingInstance().hashCode();
210          cachedHashCode = prime * cachedHashCode + Arrays.hashCode(flagValues);
211        }
212        return cachedHashCode;
213      }
214    }
215
216    @Override
217    public boolean equals(Object obj) {
218      if (this == obj) {
219        return true; // NOPMD - readability
220      }
221      if (!(obj instanceof CollapseKey)) {
222        return false; // NOPMD - readability
223      }
224      CollapseKey other = (CollapseKey) obj;
225      if (!getEnclosingInstance().equals(other.getEnclosingInstance())) {
226        return false; // NOPMD - readability
227      }
228      return Arrays.equals(flagValues, other.flagValues);
229    }
230
231    private CollapseKeyBuilder getEnclosingInstance() {
232      return CollapseKeyBuilder.this;
233    }
234  }
