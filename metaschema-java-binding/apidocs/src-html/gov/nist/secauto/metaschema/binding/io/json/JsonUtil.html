048public final class JsonUtil {
049  private static final Logger LOGGER = LogManager.getLogger(JsonUtil.class);
050
051  private JsonUtil() {
052    // disable construction
053  }
054
055  public static String toString(@NonNull JsonParser parser) throws IOException {
056    StringBuilder builder = new StringBuilder(32);
057    builder
058        .append(parser.currentToken().name())
059        .append(" '")
060        .append(parser.getText())
061        .append("' at location '")
062        .append(toString(ObjectUtils.notNull(parser.getCurrentLocation())))
063        .append('\'');
064    return builder.toString();
065  }
066
067  public static String toString(@NonNull JsonLocation location) {
068    StringBuilder builder = new StringBuilder();
069    builder
070        .append(location.getLineNr())
071        .append(':')
072        .append(location.getColumnNr());
073    return builder.toString();
074  }
075
076  public static JsonToken advanceTo(@NonNull JsonParser parser, JsonToken token) throws IOException {
077    JsonToken currentToken = null;
078    while (parser.hasCurrentToken() && !token.equals(currentToken = parser.currentToken())) {
079      currentToken = parser.nextToken();
080      if (LOGGER.isWarnEnabled()) {
081        LOGGER.warn("skipping over: {}", toString(parser));
082      }
083    }
084    return currentToken;
085  }
086
087  @SuppressWarnings({
088      "resource", // parser not owned
089      "PMD.CyclomaticComplexity" // acceptable
090  })
091  public static JsonToken skipNextValue(@NonNull JsonParser parser) throws IOException {
092
093    JsonToken currentToken = parser.currentToken();
094    // skip the field name
095    if (JsonToken.FIELD_NAME.equals(currentToken)) {
096      currentToken = parser.nextToken();
097    }
098
099    switch (currentToken) {
100    case START_ARRAY:
101    case START_OBJECT:
102      parser.skipChildren();
103      break;
104    case VALUE_FALSE:
105    case VALUE_NULL:
106    case VALUE_NUMBER_FLOAT:
107    case VALUE_NUMBER_INT:
108    case VALUE_STRING:
109    case VALUE_TRUE:
110      // do nothing
111      break;
112    default:
113      // error
114      String msg = String.format("Unhandled JsonToken %s", toString(parser));
115      LOGGER.error(msg);
116      throw new UnsupportedOperationException(msg);
117    }
118
119    // advance past the value
120    return parser.nextToken();
121  }
122
123  @SuppressWarnings("PMD.CyclomaticComplexity") // acceptable
124  public static boolean checkEndOfValue(@NonNull JsonParser parser, @NonNull JsonToken startToken) {
125    JsonToken currentToken = parser.getCurrentToken();
126
127    boolean retval;
128    switch (startToken) { // NOPMD - intentional fall through
129    case START_OBJECT:
130      retval = JsonToken.END_OBJECT.equals(currentToken);
131      break;
132    case START_ARRAY:
133      retval = JsonToken.END_ARRAY.equals(currentToken);
134      break;
135    case VALUE_EMBEDDED_OBJECT:
136    case VALUE_FALSE:
137    case VALUE_NULL:
138    case VALUE_NUMBER_FLOAT:
139    case VALUE_NUMBER_INT:
140    case VALUE_STRING:
141    case VALUE_TRUE:
142      retval = true;
143      break;
144    default:
145      retval = false;
146    }
147    return retval;
148  }
149
150  public static void assertCurrent(@NonNull JsonParser parser, @NonNull JsonToken... expectedTokens) {
151    JsonToken current = parser.currentToken();
152    assert Arrays.stream(expectedTokens).anyMatch(expected -> expected.equals(current)) : getAssertMessage(
153        expectedTokens, parser.currentToken(), ObjectUtils.notNull(parser.getCurrentLocation()));
154  }
155
156  public static void assertCurrentIsFieldValue(@NonNull JsonParser parser) {
157    JsonToken token = parser.currentToken();
158    assert token.isStructStart() || token.isScalarValue() : String.format(
159        "Expected a START_OBJECT, START_ARRAY, or VALUE_xxx token, but found JsonToken '%s' at '%s'.",
160        token,
161        JsonUtil.toString(ObjectUtils.notNull(parser.getCurrentLocation())));
162  }
163
164  public static JsonToken assertAndAdvance(@NonNull JsonParser parser, @NonNull JsonToken expectedToken)
165      throws IOException {
166    JsonToken token = parser.currentToken();
167    assert expectedToken.equals(token) : getAssertMessage(expectedToken, token,
168        ObjectUtils.notNull(parser.getCurrentLocation()));
169    return parser.nextToken();
170  }
171
172  public static JsonToken advanceAndAssert(@NonNull JsonParser parser, @NonNull JsonToken expectedToken)
173      throws IOException {
174    JsonToken token = parser.nextToken();
175    assert expectedToken.equals(token) : getAssertMessage(expectedToken, token,
176        ObjectUtils.notNull(parser.getCurrentLocation()));
177    return token;
178  }
179
180  @NonNull
181  public static String getAssertMessage(@NonNull JsonToken expected, JsonToken actual, @NonNull JsonLocation location) {
182    return ObjectUtils.notNull(
183        String.format("Expected JsonToken '%s', but found JsonToken '%s' at '%s'.",
184            expected,
185            actual,
186            JsonUtil.toString(location)));
187  }
188
189  @NonNull
190  public static String getAssertMessage(@NonNull JsonToken[] expected, JsonToken actual,
191      @NonNull JsonLocation location) {
192    List<JsonToken> expectedTokens = ObjectUtils.notNull(Arrays.asList(expected));
193    return getAssertMessage(expectedTokens, actual, location);
194  }
195
196  @NonNull
197  public static String getAssertMessage(@NonNull Collection<JsonToken> expected, JsonToken actual,
198      @NonNull JsonLocation location) {
199    return ObjectUtils.notNull(
200        String.format("Expected JsonToken(s) '%s', but found JsonToken '%s' at '%s'.",
201            expected.stream().map(token -> token.name()).collect(CustomCollectors.joiningWithOxfordComma("and")),
202            actual,
203            JsonUtil.toString(location)));
204  }
205
206  @NonNull
207  public static String toLocationContext(@NonNull JsonParser parser, @NonNull IClassBinding classBinding,
208      IBoundNamedInstance property) {
209    StringBuilder builder = new StringBuilder(64);
210    builder
211        .append("property '")
212        .append(property.getEffectiveName())
213        .append("' on class '")
214        .append(classBinding.getBoundClass().getName())
215        .append("' at location '");
216    JsonLocation location = parser.getCurrentLocation();
217    builder
218        .append(location.getLineNr())
219        .append(':')
220        .append(location.getColumnNr())
221        .append('\'');
222    return ObjectUtils.notNull(builder.toString());
223  }
