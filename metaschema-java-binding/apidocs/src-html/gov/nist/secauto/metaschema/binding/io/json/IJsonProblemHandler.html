027package gov.nist.secauto.metaschema.binding.io.json;
028
029import gov.nist.secauto.metaschema.binding.io.BindingException;
030import gov.nist.secauto.metaschema.binding.io.IProblemHandler;
031import gov.nist.secauto.metaschema.binding.model.IAssemblyClassBinding;
032import gov.nist.secauto.metaschema.binding.model.IBoundNamedInstance;
033import gov.nist.secauto.metaschema.binding.model.IClassBinding;
034import gov.nist.secauto.metaschema.binding.model.IJsonBindingSupplier;
035
036import java.io.IOException;
037import java.util.Map;
038
039import edu.umd.cs.findbugs.annotations.NonNull;
040
041public interface IJsonProblemHandler extends IProblemHandler {
042  boolean handleUnknownRootProperty(@NonNull IAssemblyClassBinding classBinding, @NonNull String fieldName,
043      @NonNull IJsonParsingContext parsingContext) throws IOException;
044
045  boolean canHandleUnknownProperty(@NonNull IClassBinding classBinding, @NonNull String propertyName,
046      @NonNull IJsonParsingContext parsingContext)
047      throws IOException;
048
049  boolean handleUnknownProperty(@NonNull IClassBinding classBinding, @NonNull String propertyName,
050      IJsonParsingContext parsingContext) throws IOException;
051
052  /**
053   * A callback used to handle bound properties for which no data was found when the content was
054   * parsed.
055   * <p>
056   * This can be used to supply default or prescribed values based on application logic.
057   *
058   * @param classBinding
059   *          the bound class on which the missing properties are found
060   * @param missingPropertyBindings
061   *          a map of field names to property bindings for missing fields
062   * @param context
063   *          the parser context used for deserialziation
064   * @return a mapping of property to suppliers for any properties handled by this method
065   * @throws BindingException
066   *           if an unhandled binding error has occurred for any reason
067   */
068  Map<IBoundNamedInstance, IJsonBindingSupplier> handleMissingFields(@NonNull IClassBinding classBinding,
069      @NonNull Map<String, IBoundNamedInstance> missingPropertyBindings, @NonNull IJsonParsingContext context)
070      throws BindingException;
071}
