107  /**
108   * Construct a new {@link IClassBinding} for a Java bean annotated with the {@link BoundAssembly}
109   * annotation.
110   *
111   * @param clazz
112   *          the Java bean class
113   * @param bindingContext
114   *          the class binding context for which this class is participating
115   */
116  protected DefaultAssemblyClassBinding(@NonNull Class<?> clazz, @NonNull IBindingContext bindingContext) {
117    super(clazz, bindingContext);
118    Objects.requireNonNull(clazz, "clazz");
119    if (!clazz.isAnnotationPresent(MetaschemaAssembly.class)) {
120      throw new IllegalArgumentException(
121          String.format("Class '%s' is missing the '%s' annotation.",
122              clazz.getName(),
123              MetaschemaAssembly.class.getName())); // NOPMD
124    }
125    this.metaschemaAssembly = ObjectUtils.notNull(clazz.getAnnotation(MetaschemaAssembly.class));
126    String namespace = ObjectUtils.notNull(ModelUtil.resolveNamespace(this.metaschemaAssembly.rootNamespace(), this));
127    String localName = ModelUtil.resolveLocalName(this.metaschemaAssembly.rootName(), null);
128
129    this.xmlRootQName = localName == null ? null : new QName(namespace, localName);
130
131    this.constraints = Lazy.lazy(() -> new AssemblyConstraintSupport(
132        clazz.getAnnotation(ValueConstraints.class),
133        clazz.getAnnotation(AssemblyConstraints.class),
134        InternalModelSource.instance()));
135  }

195  protected Stream<IBoundNamedModelInstance> getModelInstanceFieldStream(Class<?> clazz) {
196    Stream<IBoundNamedModelInstance> superInstances;
197    Class<?> superClass = clazz.getSuperclass();
198    if (superClass == null) {
199      superInstances = Stream.empty();
200    } else {
201      // get instances from superclass
202      superInstances = getModelInstanceFieldStream(superClass);
203    }
204
205    return Stream.concat(superInstances, Arrays.stream(clazz.getDeclaredFields())
206        // skip this field, since it is ignored
207        .filter(field -> !field.isAnnotationPresent(Ignore.class))
208        // skip fields that aren't a Metaschema field or assembly instance
209        .filter(field -> field.isAnnotationPresent(BoundField.class) || field.isAnnotationPresent(BoundAssembly.class))
210        .map(field -> {
211          assert field != null;
212          return newModelInstance(clazz, field);
213        })
214        .filter(Objects::nonNull)
215        .map(ObjectUtils::notNull));
216  }
217
218  protected IBoundNamedModelInstance newModelInstance(@NonNull Class<?> clazz, @NonNull java.lang.reflect.Field field) {
219    IBoundNamedModelInstance retval;
220    if (field.isAnnotationPresent(BoundAssembly.class)
221        && getBindingContext().getClassBinding(IBoundNamedModelInstance.getItemType(field)) != null) {
222      retval = IBoundAssemblyInstance.newInstance(field, this);
223    } else if (field.isAnnotationPresent(BoundField.class)) {
224      retval = IBoundFieldInstance.newInstance(field, this);
225    } else {
226      throw new IllegalStateException(
227          String.format("The field '%s' on class '%s' is not bound", field.getName(), clazz.getName()));
228    }
229    // TODO: handle choice
230    return retval;
231  }

240            .collect(Collectors.toMap(instance -> instance.getEffectiveName(), Function.identity(),
241                CustomCollectors.useLastMapper(),
242                LinkedHashMap::new));
243      }
244    }

337  @SuppressWarnings("resource") // not owned
338  @Override
339  public Object readObject(IJsonParsingContext context) throws IOException {
340    JsonParser parser = context.getReader(); // NOPMD - intentional
341
342    JsonUtil.assertAndAdvance(parser, JsonToken.START_OBJECT);
343
344    try {
345      Object instance = newInstance();
346
347      readInternal(instance, null, context);
348
349      // advance past the end object
350      JsonUtil.assertAndAdvance(parser, JsonToken.END_OBJECT);
351      return instance;
352    } catch (BindingException ex) {
353      throw new IOException(String.format("Failed to parse JSON object for '%s'", getBoundClass().getName()), ex);
354    }
355  }

357  @Override
358  protected void readBody(Object instance, StartElement start, IXmlParsingContext context)
359      throws IOException, XMLStreamException {
360    Set<IBoundNamedModelInstance> unhandledProperties = new HashSet<>();
361    for (IBoundNamedModelInstance modelProperty : getModelInstances()) {
362      if (!modelProperty.read(instance, start, context)) {
363        unhandledProperties.add(modelProperty);
364      }
365    }
366
367    // process all properties that did not get a value
368    for (IBoundNamedModelInstance property : unhandledProperties) {
369      // use the default value of the collector
370      property.setValue(instance, property.newPropertyCollector().getValue());
371    }
372  }

392  /**
393   * Parses the JSON field contents related to the bound assembly.
394   * <p>
395   * This method expects the parser's current token to be at the first field name to parse.
396   * <p>
397   * After parsing the current token will be the token at the end object immediately after all the
398   * fields and values.
399   *
400   * @param instance
401   *          the bound object to read data into
402   * @param parentInstance
403   *          the parent object used for deserialization callbacks
404   * @param context
405   *          the JSON parser
406   * @throws IOException
407   *           if an error occurred while reading the JSON
408   */
409  @SuppressWarnings("resource") // not owned
410  protected void readInternal( // NOPMD - ok
411      @NonNull Object instance,
412      @Nullable Object parentInstance,
413      @NonNull IJsonParsingContext context) throws IOException {
414    JsonParser parser = context.getReader(); // NOPMD - intentional
415
416    JsonUtil.assertCurrent(parser, JsonToken.FIELD_NAME, JsonToken.END_OBJECT);
417
418    try {
419      callBeforeDeserialize(instance, parentInstance);
420    } catch (BindingException ex) {
421      throw new IOException("an error occured calling the beforeDeserialize() method", ex);
422    }
423
424    IBoundFlagInstance jsonKey = getJsonKeyFlagInstance();
425    Map<String, ? extends IBoundNamedInstance> properties;
426    if (jsonKey == null) {
427      properties = getNamedInstances(null);
428    } else {
429      properties = getNamedInstances((flag) -> {
430        return !jsonKey.equals(flag);
431      });
432
433      // if there is a json key, the first field will be the key
434      String key = ObjectUtils.notNull(parser.getCurrentName());
435
436      Object value = jsonKey.readValueFromString(key);
437      jsonKey.setValue(instance, value.toString());
438
439      // advance past the FIELD_NAME
440      // next the value will be a start object
441      JsonUtil.assertAndAdvance(parser, JsonToken.FIELD_NAME);
442      //
443      // JsonUtil.assertAndAdvance(jsonParser, JsonToken.START_OBJECT);
444    }
445
446    Set<String> handledProperties = new HashSet<>();
447    while (!JsonToken.END_OBJECT.equals(parser.currentToken())) {
448      String propertyName = parser.getCurrentName();
449      IBoundNamedInstance property = properties.get(propertyName);
450
451      boolean handled = false;
452      if (property != null) {
453        handled = property.read(instance, context);
454      }
455
456      if (handled) {
457        handledProperties.add(propertyName);
458      } else {
459        if (LOGGER.isWarnEnabled()) {
460          LOGGER.warn("Unrecognized property named '{}' at '{}'", propertyName,
461              JsonUtil.toString(ObjectUtils.notNull(parser.getCurrentLocation())));
462        }
463        JsonUtil.assertAndAdvance(parser, JsonToken.FIELD_NAME);
464        JsonUtil.skipNextValue(parser);
465      }
466    }
467
468    // set undefined properties
469    // TODO: re-implement this by removing the parsed properties from the properties map to speed up
470    for (Map.Entry<String, ? extends IBoundNamedInstance> entry : properties.entrySet()) {
471      if (!handledProperties.contains(entry.getKey())) {
472        // use the default value of the collector
473        IBoundNamedInstance property = ObjectUtils.notNull(entry.getValue());
474        property.setValue(instance, property.newPropertyCollector().getValue());
475      }
476    }
477
478    if (jsonKey != null) {
479      // read the END_OBJECT for the JSON key value
480      JsonUtil.assertAndAdvance(parser, JsonToken.END_OBJECT);
481    }
482
483    try {
484      callAfterDeserialize(instance, parentInstance);
485    } catch (BindingException ex) {
486      throw new IOException("an error occured calling the afterDeserialize() method", ex);
487    }
488
489    JsonUtil.assertCurrent(parser, JsonToken.END_OBJECT);
490  }

492  /**
493   * Serializes the provided instance in JSON.
494   *
495   * @param instance
496   *          the instance to serialize
497   * @param writeObjectWrapper
498   *          {@code true} if the start and end object should be written, or {@code false} otherwise
499   * @param context
500   *          the JSON writing context used to generate output
501   * @throws IOException
502   *           if an error occurs while writing to the output context
503   * @throws NullPointerException
504   *           if there is a JSON key configured and the key property's value is {@code null}
505   */
506  @SuppressWarnings("resource") // not owned
507  protected void writeInternal(@NonNull Object instance, boolean writeObjectWrapper,
508      @NonNull IJsonWritingContext context)
509      throws IOException {
510    JsonGenerator writer = context.getWriter(); // NOPMD - intentional
511
512    if (writeObjectWrapper) {
513      writer.writeStartObject();
514    }
515
516    IBoundFlagInstance jsonKey = getJsonKeyFlagInstance();
517    Map<String, ? extends IBoundNamedInstance> properties;
518    if (jsonKey == null) {
519      properties = getNamedInstances(null);
520    } else {
521      properties = getNamedInstances((flag) -> {
522        return !jsonKey.equals(flag);
523      });
524
525      // if there is a json key, the first field will be the key
526      Object flagValue = jsonKey.getValue(instance);
527      String key = jsonKey.getValueAsString(flagValue);
528      if (key == null) {
529        throw new IOException(new NullPointerException("Null key value"));
530      }
531      writer.writeFieldName(key);
532
533      // next the value will be a start object
534      writer.writeStartObject();
535    }
536
537    for (IBoundNamedInstance property : properties.values()) {
538      ObjectUtils.notNull(property).write(instance, context);
539    }
540
541    if (jsonKey != null) {
542      // write the END_OBJECT for the JSON key value
543      writer.writeEndObject();
544    }
545
546    if (writeObjectWrapper) {
547      writer.writeEndObject();
548    }
549  }
