027package gov.nist.secauto.metaschema.binding.model;
028
029import com.fasterxml.jackson.core.JsonToken;
030
031import gov.nist.secauto.metaschema.binding.io.json.IJsonParsingContext;
032import gov.nist.secauto.metaschema.binding.io.json.IJsonProblemHandler;
033import gov.nist.secauto.metaschema.binding.io.json.IJsonWritingContext;
034import gov.nist.secauto.metaschema.binding.io.xml.IXmlParsingContext;
035import gov.nist.secauto.metaschema.binding.io.xml.IXmlWritingContext;
036import gov.nist.secauto.metaschema.model.common.IRootAssemblyDefinition;
037
038import java.io.IOException;
039
040import javax.xml.stream.XMLStreamException;
041
042import edu.umd.cs.findbugs.annotations.NonNull;
043
044public interface IRootAssemblyClassBinding extends IAssemblyClassBinding, IRootAssemblyDefinition {
045
046  /**
047   * Parses JSON into a bound object. This assembly must be a root assembly for which a call to
048   * {@link IAssemblyClassBinding#isRoot()} will return {@code true}.
049   * <p>
050   * This method expects the parser's current token to be:
051   * <ul>
052   * <li>{@code null} indicating that the parser has not yet parsed a JSON node;</li>
053   * <li>a {@link JsonToken#START_OBJECT} which represents the object wrapper containing the root
054   * field,</li>
055   * <li>a {@link JsonToken#FIELD_NAME} representing the root field to parse, or</li>
056   * <li>a peer field to the root field that will be handled by the
057   * {@link IJsonProblemHandler#handleUnknownRootProperty(IAssemblyClassBinding, String, IJsonParsingContext)}
058   * method.</li>
059   * </ul>
060   * <p>
061   * After parsing the current token will be:
062   * <ul>
063   * <li>the next token after the {@link JsonToken#END_OBJECT} corresponding to the initial
064   * {@link JsonToken#START_OBJECT} parsed by this method;</li>
065   * <li>the next token after the {@link JsonToken#END_OBJECT} for the root field's value; or</li>
066   * <li>the next token after all fields and associated values have been parsed looking for the root
067   * field. This next token will be the {@link JsonToken#END_OBJECT} for the object containing the
068   * fields. In this case the method will throw an {@link IOException} indicating the root was not
069   * found.</li>
070   * </ul>
071   *
072   * @param context
073   *          the JSON parser
074   * @return the bound object instance representing the JSON object
075   * @throws IOException
076   *           if an error occurred while reading the JSON
077   */
078  @NonNull
079  Object readRoot(@NonNull IJsonParsingContext context) throws IOException;
080
081  /**
082   * Parses XML into a bound object. This assembly must be a root assembly for which a call to
083   * {@link IAssemblyClassBinding#isRoot()} will return {@code true}.
084   *
085   * @param context
086   *          the XML parser
087   * @return the bound object instance representing the JSON object
088   * @throws XMLStreamException
089   *           if an error occurred while parsing into XML
090   * @throws IOException
091   *           if an error occurred while reading the input
092   */
093  // TODO: merge the XMLStreamException into IOException
094  @NonNull
095  Object readRoot(@NonNull IXmlParsingContext context) throws XMLStreamException, IOException;
096
097  /**
098   * Writes data in a bound object to JSON. This assembly must be a root assembly for which a call to
099   * {@link IAssemblyClassBinding#isRoot()} will return {@code true}.
100   *
101   * @param instance
102   *          the bound object
103   * @param context
104   *          the JSON serializer
105   * @throws IOException
106   *           if an error occurred while reading the JSON
107   */
108  void writeRoot(@NonNull Object instance, @NonNull IJsonWritingContext context) throws IOException;
109
110  /**
111   * Writes data in a bound object to XML. This assembly must be a root assembly for which a call to
112   * {@link IAssemblyClassBinding#isRoot()} will return {@code true}.
113   *
114   * @param instance
115   *          the bound object
116   * @param context
117   *          the XML serializer
118   * @throws XMLStreamException
119   *           if an error occurred while parsing into XML
120   * @throws IOException
121   *           if an error occurred while writing the output
122   */
123  // TODO: merge the XMLStreamException into IOException
124  void writeRoot(@NonNull Object instance, @NonNull IXmlWritingContext context) throws XMLStreamException, IOException;
125}
