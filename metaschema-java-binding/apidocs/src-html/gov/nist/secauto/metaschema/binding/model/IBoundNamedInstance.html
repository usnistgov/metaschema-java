053public interface IBoundNamedInstance extends INamedInstance {
054
055  @Override
056  default String getName() {
057    // delegate to the definition
058    return getDefinition().getName();
059  }
060
061  @Override
062  default @NonNull Map<QName, Set<String>> getProperties() {
063    // TODO: implement
064    throw new UnsupportedOperationException();
065  }
066
067  /**
068   * Get the {@link IClassBinding} for the Java class within which this property exists.
069   *
070   * @return the containing class's binding
071   */
072  @NonNull
073  IClassBinding getParentClassBinding();
074
075  /**
076   * Gets the bound Java field associated with this property.
077   *
078   * @return the Java field
079   */
080  @NonNull
081  Field getField();
082
083  @NonNull
084  default String getJavaFieldName() {
085    return ObjectUtils.notNull(getField().getName());
086  }
087
088  /**
089   * Get the actual Java type of the underlying bound object.
090   * <p>
091   * This may be the same as the what is returned by {@link #getItemType()}, or may be a Java
092   * collection class.
093   *
094   * @return the raw type of the bound object
095   */
096  @SuppressWarnings("null")
097  @NonNull
098  default Type getType() {
099    return getField().getGenericType();
100  }
101
102  /**
103   * Get the item type of the bound object. An item type is the primitive or specialized type that
104   * represents that data associated with this binding.
105   *
106   * @return the item type of the bound object
107   */
108  @NonNull
109  default Class<?> getItemType() {
110    return (Class<?>) getType();
111  }
112
113  /**
114   * Get the current value from the provided {@code parentInstance} object. The provided object must
115   * be of the type associated with the definition containing this property.
116   *
117   * @param parentInstance
118   *          the object associated with the definition containing this property
119   * @return the value if available, or {@code null} otherwise
120   */
121  @Override
122  default Object getValue(@NonNull Object parentInstance) {
123    Field field = getField();
124    boolean accessable = field.canAccess(parentInstance);
125    field.setAccessible(true); // NOPMD - intentional
126    Object retval;
127    try {
128      Object result = field.get(parentInstance);
129      retval = result;
130    } catch (IllegalArgumentException | IllegalAccessException ex) {
131      throw new IllegalArgumentException(
132          String.format("Unable to get the value of field '%s' in class '%s'.", field.getName(),
133              field.getDeclaringClass().getName()),
134          ex);
135    } finally {
136      field.setAccessible(accessable); // NOPMD - intentional
137    }
138    return retval;
139  }
140
141  /**
142   * Set the provided value on the provided object. The provided object must be of the item's type
143   * associated with this property.
144   *
145   * @param parentInstance
146   *          the object
147   * @param value
148   *          a value, which may be a simple {@link Type} or a {@link ParameterizedType} for a
149   *          collection
150   */
151  default void setValue(@NonNull Object parentInstance, Object value) {
152    Field field = getField();
153    boolean accessable = field.canAccess(parentInstance);
154    field.setAccessible(true); // NOPMD - intentional
155    try {
156      field.set(parentInstance, value);
157    } catch (IllegalArgumentException | IllegalAccessException ex) {
158      throw new IllegalArgumentException(
159          String.format("Unable to set the value of field '%s' in class '%s'.", field.getName(),
160              field.getDeclaringClass().getName()),
161          ex);
162    } finally {
163      field.setAccessible(accessable); // NOPMD - intentional
164    }
165  }
166
167  @NonNull
168  IPropertyCollector newPropertyCollector();
169
170  /**
171   * Read JSON data associated with this property and apply it to the provided {@code objectInstance}
172   * on which this property exists.
173   * <p>
174   * The parser's current token is expected to be the {@link JsonToken#FIELD_NAME} for the field value
175   * being parsed.
176   * <p>
177   * After parsing the parser's current token will be the next token after the field's value.
178   *
179   * @param objectInstance
180   *          an instance of the class on which this property exists
181   * @param context
182   *          the JSON parsing context
183   * @return {@code true} if the property was parsed, or {@code false} if the data did not contain
184   *         information for this property
185   * @throws IOException
186   *           if there was an error when reading JSON data
187   */
188  boolean read(@NonNull Object objectInstance, @NonNull IJsonParsingContext context) throws IOException;
189
190  /**
191   * Read JSON data associated with this property and return it.
192   * <p>
193   * The parser's current token is expected to be the {@link JsonToken#FIELD_NAME} for the field value
194   * being parsed.
195   * <p>
196   * After parsing the parser's current token will be the next token after the field's value.
197   *
198   * @param context
199   *          the JSON parsing context
200   * @return the instance value or {@code null} if no data was available to read
201   * @throws IOException
202   *           if there was an error when reading JSON data
203   */
204  Object read(@NonNull IJsonParsingContext context) throws IOException;
205
206  /**
207   * Read the XML data associated with this property and apply it to the provided
208   * {@code objectInstance} on which this property exists.
209   *
210   * @param objectInstance
211   *          an instance of the class on which this property exists
212   * @param parent
213   *          the containing XML element that was previously parsed
214   * @param context
215   *          the XML parsing context
216   * @return {@code true} if the property was parsed, or {@code false} if the data did not contain
217   *         information for this property
218   * @throws IOException
219   *           if there was an error when reading XML data
220   * @throws XMLStreamException
221   *           if there was an error generating an {@link XMLEvent} from the XML
222   */
223  boolean read(@NonNull Object objectInstance, @NonNull StartElement parent, @NonNull IXmlParsingContext context)
224      throws IOException, XMLStreamException;

252   * Copy this instance from one parent object to another.
253   *
254   * @param fromInstance
255   *          the object to copy from
256   * @param toInstance
257   *          the object to copy to
258   * @throws BindingException
259   *           if an error occurred while processing the object bindings
260   */
261  void copyBoundObject(@NonNull Object fromInstance, @NonNull Object toInstance) throws BindingException;
