027package gov.nist.secauto.metaschema.binding.model.annotations;
028
029import static java.lang.annotation.ElementType.FIELD;
030import static java.lang.annotation.ElementType.METHOD;
031import static java.lang.annotation.RetentionPolicy.RUNTIME;
032
033import gov.nist.secauto.metaschema.model.common.datatype.IDataTypeAdapter;
034
035import java.lang.annotation.Documented;
036import java.lang.annotation.Retention;
037import java.lang.annotation.Target;
038
039import edu.umd.cs.findbugs.annotations.NonNull;
040
041/**
042 * Identifies that the annotation target is a bound property that represents a Metaschema flag.
043 */
044@Documented
045@Retention(RUNTIME)
046@Target({ FIELD, METHOD })
047public @interface BoundFlag {
048  /**
049   * Get the documentary formal name of the flag.
050   * <p>
051   * If the value is "##none", then the description will be considered {@code null}.
052   *
053   * @return a markdown string or {@code "##none"} if no formal name is provided
054   */
055  @NonNull
056  String formalName() default Constants.NO_STRING_VALUE;
057
058  /**
059   * Get the documentary description of the flag.
060   * <p>
061   * If the value is "##none", then the description will be considered {@code null}.
062   *
063   * @return a markdown string or {@code "##none"} if no description is provided
064   */
065  @NonNull
066  String description() default Constants.NO_STRING_VALUE;
067
068  /**
069   * The model name to use for singleton values. This name will be used for associated XML attributes
070   * and JSON properties.
071   * <p>
072   * If the value is "##none", then element name is derived from the JavaBean property name.
073   *
074   * @return the name
075   */
076  @NonNull
077  String useName() default Constants.NO_STRING_VALUE;
078
079  /**
080   * XML target namespace of the XML Schema element.
081   * <p>
082   * If the value is "##default", then namespace is derived from the namespace provided in the
083   * package-info. If the value is "##none", the namespace will be {@code null}.
084   *
085   * @return the namespace
086   */
087  @NonNull
088  String namespace() default Constants.NO_STRING_VALUE;
089
090  /**
091   * The default value of the flag represented as a string.
092   * <p>
093   * The value {@link Constants#NULL_VALUE} is used to indicate if no default value is provided.
094   *
095   * @return the default value
096   */
097  @NonNull
098  String defaultValue() default Constants.NULL_VALUE;
099
100  /**
101   * Specifies if the XML Schema attribute is optional or required. If true, then the JavaBean
102   * property is mapped to a XML Schema attribute that is required. Otherwise it is mapped to a XML
103   * Schema attribute that is optional.
104   *
105   * @return {@code true} if the flag must occur, or {@code false} otherwise
106   */
107  boolean required() default false;
108
109  /**
110   * The Metaschema data type adapter for the field's value.
111   *
112   * @return the data type adapter
113   */
114  @NonNull
115  Class<? extends IDataTypeAdapter<?>> typeAdapter() default NullJavaTypeAdapter.class;
116
117  /**
118   * Get any remarks for this flag.
119   *
120   * @return a markdown string or {@code "##none"} if no remarks are provided
121   */
122  @NonNull
123  String remarks() default Constants.NO_STRING_VALUE;
124}
