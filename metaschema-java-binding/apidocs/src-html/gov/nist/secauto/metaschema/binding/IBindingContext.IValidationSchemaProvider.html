071/**
072 * Provides information supporting a binding between a set of Metaschema models and corresponding
073 * Java classes.
074 */
075public interface IBindingContext extends IMetaschemaLoaderStrategy {
076
077  /**
078   * Get the singleton {@link IBindingContext} instance, which can be used to load information that
079   * binds a model to a set of Java classes.
080   *
081   * @return a new binding context
082   */
083  @NonNull
084  static IBindingContext instance() {
085    return DefaultBindingContext.instance();
086  }

088  /**
089   * Register a matcher used to identify a bound class by the content's root name.

091   * @param matcher
092   *          the matcher implementation

096  /**
097   * Determine the bound class for the provided XML {@link QName}.
098   *

108   * Determine the bound class for the provided JSON/YAML property/item name using any registered

123   * formats. Adapters are used to support bindings for simple data objects (e.g., {@link String},

126   * @param <TYPE>

129   *          the Java {@link Class} for the bound type
130   * @return the adapter instance or {@code null} if the provided class is not bound

132  @Nullable

249  /**
250   * Get a new single use constraint validator.
251   *
252   * @param handler
253   *          the validation handler to use to process the validation results
254   *
255   * @return the validator
256   */
257  default IConstraintValidator newValidator(@NonNull IConstraintValidationHandler handler) {
258    IBoundLoader loader = newBoundLoader();
259    loader.disableFeature(DeserializationFeature.DESERIALIZE_VALIDATE_CONSTRAINTS);
260
261    DynamicContext context = new StaticContext().newDynamicContext();
262    context.setDocumentLoader(loader);
263
264    return new DefaultConstraintValidator(context, handler);
265  }
266
267  /**
268   * Perform constraint validation on the provided bound object represented as an {@link INodeItem}.
269   * The bound object can be turned into a {@link INodeItem} using {@link #toNodeItem(Object, URI)}.
270   *
271   * @param nodeItem
272   *          the node item to validate
273   * @return the validation result
274   * @throws IllegalArgumentException
275   *           if the provided class is not bound to a Metaschema assembly or field
276   */
277  default IValidationResult validate(@NonNull INodeItem nodeItem) {
278    FindingCollectingConstraintValidationHandler handler = new FindingCollectingConstraintValidationHandler();
279    IConstraintValidator validator = newValidator(handler);
280    validator.validate(nodeItem);
281    validator.finalizeValidation();
282    return handler;
283  }
284
285  /**
286   * Perform schema and constraint validation on the target. The constraint validation will only be
287   * performed if the schema validation is passes.
288   *
289   * @param target
290   *          the target to validate
291   * @param asFormat
292   *          the schema format to use to validate the target
293   * @param schemaProvider
294   *          provides callbacks to get the appropriate schemas
295   * @return the validation result
296   * @throws IOException
297   *           if an error occurred while reading the target
298   * @throws SAXException
299   *           if an error occurred when parsing the target as XML
300   */
301  default IValidationResult validate(
302      @NonNull Path target,
303      @NonNull Format asFormat,
304      @NonNull IValidationSchemaProvider schemaProvider) throws IOException, SAXException {
305    IValidationResult retval;
306    switch (asFormat) {
307    case JSON:
308      retval = new JsonSchemaContentValidator(schemaProvider.getJsonSchema()).validate(target);
309      break;
310    case XML:
311      List<Source> schemaSources = schemaProvider.getXmlSchemas();
312      retval = new XmlSchemaContentValidator(schemaSources).validate(target);
313      break;
314    case YAML:
315      JSONObject json = YamlOperations.yamlToJson(YamlOperations.parseYaml(target));
316      assert json != null;
317      retval = new JsonSchemaContentValidator(schemaProvider.getJsonSchema())
318          .validate(json, ObjectUtils.notNull(target.toUri()));
319      break;
320    default:
321      throw new UnsupportedOperationException("Unsupported format: " + asFormat.name());
322    }
323
324    if (retval.isPassing()) {
325      IValidationResult constraintValidationResult = validateWithConstraints(target);
326      retval = AggregateValidationResult.aggregate(retval, constraintValidationResult);
327    }
328    return retval;
329  }
330
331  default IValidationResult validateWithConstraints(@NonNull Path target) throws IOException {
332    IBoundLoader loader = newBoundLoader();
333    loader.disableFeature(DeserializationFeature.DESERIALIZE_VALIDATE_CONSTRAINTS);
334
335    DynamicContext dynamicContext = new StaticContext().newDynamicContext();
336    dynamicContext.setDocumentLoader(loader);
337    IDocumentNodeItem nodeItem = loader.loadAsNodeItem(target);
338
339    return validate(nodeItem);
340  }
341
342  interface IValidationSchemaProvider {
343    @NonNull
344    JSONObject getJsonSchema() throws IOException;
345
346    @NonNull
347    List<Source> getXmlSchemas() throws IOException;
348  }
