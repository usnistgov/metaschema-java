059  public AbstractFieldProperty(@NonNull Field field, @NonNull IAssemblyClassBinding parentClassBinding) {
060    super(field, parentClassBinding);
061
062    BoundField fieldAnnotation = field.getAnnotation(BoundField.class);
063    if (fieldAnnotation == null) {
064      throw new IllegalStateException(String.format("Field '%s' on class '%s' is missing the '%s' annotation.",
065          field.getName(), parentClassBinding.getBoundClass().getName(), BoundField.class.getName()));
066    }
067    this.fieldAnnotation = fieldAnnotation;
068  }

126  @Override
127  public boolean isNextProperty(IXmlParsingContext context) throws XMLStreamException {
128    boolean retval = super.isNextProperty(context);
129    if (!retval) {
130      XMLEventReader2 eventReader = context.getReader();
131      XMLEvent event = eventReader.peek();
132      if (event.isStartElement()) {
133        QName qname = ObjectUtils.notNull(event.asStartElement().getName());
134        IDataTypeAdapter<?> adapter = getDefinition().getJavaTypeAdapter();
135        retval = !isInXmlWrapped() && adapter.isUnrappedValueAllowedInXml() && adapter.canHandleQName(qname);
136      }
137    }
138    return retval;
139  }
140
141  @Override
142  public Object readItem(Object parentInstance, StartElement start,
143      IXmlParsingContext context) throws XMLStreamException, IOException {
144    // figure out if we need to parse the wrapper or not
145    IDataTypeAdapter<?> adapter = getDefinition().getJavaTypeAdapter();
146    boolean parseWrapper = true;
147    if (!isInXmlWrapped() && adapter.isUnrappedValueAllowedInXml()) {
148      parseWrapper = false;
149    }
150
151    XMLEventReader2 eventReader = context.getReader();
152
153    StartElement currentStart = start;
154    if (parseWrapper) {
155      // TODO: not sure this is needed, since there is a peek just before this
156      // parse any whitespace before the element
157      XmlEventUtil.skipWhitespace(eventReader);
158
159      XMLEvent event = eventReader.peek();
160      if (event.isStartElement() && getXmlQName().equals(event.asStartElement().getName())) {
161        // Consume the start element
162        currentStart
163            = ObjectUtils.notNull(
164                XmlEventUtil.consumeAndAssert(eventReader, XMLStreamConstants.START_ELEMENT, getXmlQName())
165                    .asStartElement());
166      } else {
167        throw new IOException(String.format("Did not find expected element '%s'.", getXmlQName()));
168      }
169    }
170
171    // figure out how to parse the item
172    IXmlBindingSupplier supplier = getDataTypeHandler();
173
174    // consume the value
175    Object retval = supplier.get(parentInstance, currentStart, context);
176
177    if (parseWrapper) {
178      // consume the end element
179      XmlEventUtil.consumeAndAssert(context.getReader(), XMLStreamConstants.END_ELEMENT, currentStart.getName());
180    }
181
182    return retval;
183  }
184
185  @Override
186  public void writeItem(Object item, QName parentName, IXmlWritingContext context)
187      throws XMLStreamException, IOException {
188    // figure out how to parse the item
189    IDataTypeHandler handler = getDataTypeHandler();
190
191    // figure out if we need to parse the wrapper or not
192    boolean writeWrapper = isInXmlWrapped() || !handler.isUnwrappedValueAllowedInXml();
193
194    XMLStreamWriter2 writer = context.getWriter();
195
196    QName currentParentName;
197    if (writeWrapper) {
198      currentParentName = getXmlQName();
199      writer.writeStartElement(currentParentName.getNamespaceURI(), currentParentName.getLocalPart());
200    } else {
201      currentParentName = parentName;
202    }
203
204    // write the value
205    handler.accept(item, currentParentName, context);
206
207    if (writeWrapper) {
208      writer.writeEndElement();
209    }
210  }

222        getParentClassBinding().getBoundClass().getName(),
223        getName());
224  }
