027package gov.nist.secauto.metaschema.binding.model.annotations;
028
029import static java.lang.annotation.ElementType.TYPE;
030import static java.lang.annotation.RetentionPolicy.RUNTIME;
031
032import gov.nist.secauto.metaschema.model.common.IMetaschema;
033
034import java.lang.annotation.Documented;
035import java.lang.annotation.Retention;
036import java.lang.annotation.Target;
037
038import edu.umd.cs.findbugs.annotations.NonNull;
039
040/**
041 * This annotation indicates that the target class represents a Metaschema assembly.
042 */
043@Documented
044@Retention(RUNTIME)
045@Target(TYPE)
046public @interface MetaschemaAssembly {
047  /**
048   * Get the documentary formal name of the assembly.
049   * <p>
050   * If the value is "##none", then the description will be considered {@code null}.
051   *
052   * @return a markdown string or {@code "##none"} if no formal name is provided
053   */
054  @NonNull
055  String formalName() default Constants.NO_STRING_VALUE;
056
057  /**
058   * Get the documentary description of the assembly.
059   * <p>
060   * If the value is "##none", then the description will be considered {@code null}.
061   *
062   * @return a markdown string or {@code "##none"} if no description is provided
063   */
064  @NonNull
065  String description() default Constants.NO_STRING_VALUE;
066
067  /**
068   * Get the metaschema class that "owns" this assembly, which is the concrete implementation of the
069   * metaschema containing the assembly.
070   *
071   * @return the {@link IMetaschema} class
072   */
073  Class<? extends IMetaschema> metaschema();
074
075  /**
076   * Name of the assembly.
077   *
078   * @return the name
079   */
080  @NonNull
081  String name();
082
083  /**
084   * Name of the root XML element or the JSON/YAML property.
085   * <p>
086   * If the value is "##none", then there is no root name.
087   *
088   * @return the name
089   */
090  @NonNull
091  String rootName() default Constants.NO_STRING_VALUE;
092
093  /**
094   * XML target namespace of the XML element.
095   * <p>
096   * If the value is "##default", then namespace is derived from the namespace provided in the
097   * package-info.
098   *
099   * @return the namespace
100   */
101  @NonNull
102  String rootNamespace() default Constants.DEFAULT_STRING_VALUE;
103
104  /**
105   * Get any remarks for this assembly.
106   *
107   * @return a markdown string or {@code "##none"} if no remarks are provided
108   */
109  @NonNull
110  String remarks() default Constants.NO_STRING_VALUE;
111}
