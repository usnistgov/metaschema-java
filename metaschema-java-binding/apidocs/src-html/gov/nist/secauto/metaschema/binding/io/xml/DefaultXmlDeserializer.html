027package gov.nist.secauto.metaschema.binding.io.xml;
028
029import com.ctc.wstx.stax.WstxInputFactory;
030
031import gov.nist.secauto.metaschema.binding.IBindingContext;
032import gov.nist.secauto.metaschema.binding.io.AbstractDeserializer;
033import gov.nist.secauto.metaschema.binding.model.IAssemblyClassBinding;
034import gov.nist.secauto.metaschema.binding.model.RootAssemblyDefinition;
035import gov.nist.secauto.metaschema.model.common.metapath.item.DefaultNodeItemFactory;
036import gov.nist.secauto.metaschema.model.common.metapath.item.IDocumentNodeItem;
037import gov.nist.secauto.metaschema.model.common.util.AutoCloser;
038import gov.nist.secauto.metaschema.model.common.util.ObjectUtils;
039
040import org.codehaus.stax2.XMLEventReader2;
041import org.codehaus.stax2.XMLInputFactory2;
042
043import java.io.IOException;
044import java.io.Reader;
045import java.net.URI;
046
047import javax.xml.stream.EventFilter;
048import javax.xml.stream.XMLEventReader;
049import javax.xml.stream.XMLInputFactory;
050import javax.xml.stream.XMLStreamException;
051
052import edu.umd.cs.findbugs.annotations.NonNull;
053
054public class DefaultXmlDeserializer<CLASS>
055    extends AbstractDeserializer<CLASS> {
056  private XMLInputFactory2 xmlInputFactory;
057
058  public DefaultXmlDeserializer(@NonNull IBindingContext bindingContext, @NonNull IAssemblyClassBinding classBinding) {
059    super(bindingContext, classBinding);
060  }
061
062  // @Override
063  // public Format supportedFromat() {
064  // return Format.XML;
065  // }
066
067  @NonNull
068  protected XMLInputFactory2 getXMLInputFactory() {
069    synchronized (this) {
070      if (xmlInputFactory == null) {
071        xmlInputFactory = (XMLInputFactory2) XMLInputFactory.newInstance();
072        assert xmlInputFactory instanceof WstxInputFactory;
073        xmlInputFactory.configureForXmlConformance();
074        xmlInputFactory.setProperty(XMLInputFactory.IS_COALESCING, false);
075        // xmlInputFactory.configureForSpeed();
076      }
077      return ObjectUtils.notNull(xmlInputFactory);
078    }
079  }
080
081  protected void setXMLInputFactory(@NonNull XMLInputFactory2 factory) {
082    synchronized (this) {
083      this.xmlInputFactory = factory;
084    }
085  }
086
087  @NonNull
088  protected XMLEventReader2 newXMLEventReader2(@NonNull Reader reader) throws XMLStreamException {
089    XMLEventReader eventReader = getXMLInputFactory().createXMLEventReader(reader);
090    EventFilter filter = new CommentFilter();
091    return ObjectUtils.notNull((XMLEventReader2) getXMLInputFactory().createFilteredReader(eventReader, filter));
092  }
093
094  @Override
095  protected IDocumentNodeItem deserializeToNodeItemInternal(Reader reader, URI documentUri) throws IOException {
096    // doesn't auto close the underlying reader
097    try (AutoCloser<XMLEventReader2, XMLStreamException> closer
098        = new AutoCloser<>(newXMLEventReader2(reader), event -> event.close())) {
099      return parseXmlInternal(closer.getResource(), documentUri);
100    } catch (XMLStreamException ex) {
101      throw new IOException("Unable to create a new XMLEventReader2 instance.", ex);
102    }
103  }
104
105  @NonNull
106  protected IDocumentNodeItem parseXmlInternal(@NonNull XMLEventReader2 reader, @NonNull URI documentUri)
107      throws IOException, XMLStreamException {
108
109    IAssemblyClassBinding classBinding = getClassBinding();
110    if (!classBinding.isRoot()) {
111      throw new IOException(
112          String.format("The assembly '%s' is not a root assembly.", classBinding.getBoundClass().getName()));
113    }
114
115    DefaultXmlParsingContext parsingContext = new DefaultXmlParsingContext(reader, new DefaultXmlProblemHandler());
116
117    RootAssemblyDefinition root = new RootAssemblyDefinition(classBinding);
118
119    return DefaultNodeItemFactory.instance().newDocumentNodeItem(root, root.readRoot(parsingContext), documentUri);
120  }
121}
