063/**
064 * The implementation of a {@link IBindingContext} provided by this library.
065 * <p>
066 * This implementation caches Metaschema information, which can dramatically improve read and write
067 * performance at the cost of some memory use. Thus, using the same singleton of this class across
068 * multiple I/O operations will improve overall read and write performance when processing the same
069 * types of data.
070 * <p>
071 * Serializers and deserializers provided by this class using the
072 * {@link #newSerializer(Format, Class)} and {@link #newDeserializer(Format, Class)} methods will
073 * <p>
074 * This class is synchronized and is thread-safe.
075 */
076public class DefaultBindingContext implements IBindingContext {
077  private static DefaultBindingContext singleton;
078  @NonNull
079  private final IMetaschemaLoaderStrategy metaschemaLoaderStrategy;
080  @NonNull
081  private final List<IBindingMatcher> bindingMatchers = new LinkedList<>();
082
083  @NonNull
084  public static DefaultBindingContext instance() {
085    synchronized (DefaultBindingContext.class) {
086      if (singleton == null) {
087        singleton = new DefaultBindingContext();
088      }
089    }
090    return ObjectUtils.notNull(singleton);
091  }
092
093  /**
094   * Construct a new binding context.
095   *
096   * @param externalConstraintSets
097   *          the set of external constraints to configure this binding to use
098   */
099  public DefaultBindingContext(@NonNull Set<IConstraintSet> externalConstraintSets) {
100    // only allow extended classes
101    metaschemaLoaderStrategy = new ExternalConstraintsMetaschemaLoaderStrategy(this, externalConstraintSets);
102  }
103
104  /**
105   * Construct a new binding context.
106   */
107  protected DefaultBindingContext() {
108    // only allow extended classes
109    metaschemaLoaderStrategy = new SimpleMetaschemaLoaderStrategy(this);
110  }
111
112  @Override
113  public IMetaschema getMetaschemaInstanceByClass(@NonNull Class<? extends IMetaschema> clazz) {
114    return metaschemaLoaderStrategy.getMetaschemaInstanceByClass(clazz);
115  }
116
117  @Override
118  public IClassBinding getClassBinding(@NonNull Class<?> clazz) {
119    return metaschemaLoaderStrategy.getClassBinding(clazz);
120  }
121
122  @Override
123  public Map<Class<?>, IClassBinding> getClassBindingsByClass() {
124    return metaschemaLoaderStrategy.getClassBindingsByClass();
125  }
126
127  @Override
128  public <TYPE extends IDataTypeAdapter<?>> TYPE getJavaTypeAdapterInstance(@NonNull Class<TYPE> clazz) {
129    return DataTypeService.getInstance().getJavaTypeAdapterByClass(clazz);
130  }
131
132  /**
133   * {@inheritDoc}
134   * <p>
135   * A serializer returned by this method is thread-safe.
136   */
137  @Override
138  public <CLASS> ISerializer<CLASS> newSerializer(@NonNull Format format, @NonNull Class<CLASS> clazz) {
139    Objects.requireNonNull(format, "format");
140    IAssemblyClassBinding classBinding = (IAssemblyClassBinding) getClassBinding(clazz);
141    if (classBinding == null) {
142      throw new IllegalStateException(String.format("Class '%s' is not bound", clazz.getClass().getName()));
143    }
144    ISerializer<CLASS> retval;
145    switch (format) {
146    case JSON:
147      retval = new DefaultJsonSerializer<>(this, classBinding);
148      break;
149    case XML:
150      retval = new DefaultXmlSerializer<>(this, classBinding);
151      break;
152    case YAML:
153      retval = new DefaultYamlSerializer<>(this, classBinding);
154      break;
155    default:
156      throw new UnsupportedOperationException(String.format("Unsupported format '%s'", format));
157    }
158    return retval;
159  }
160
161  /**
162   * {@inheritDoc}
163   * <p>
164   * A deserializer returned by this method is thread-safe.
165   */
166  @Override
167  public <CLASS> IDeserializer<CLASS> newDeserializer(@NonNull Format format, @NonNull Class<CLASS> clazz) {
168    IAssemblyClassBinding classBinding = (IAssemblyClassBinding) getClassBinding(clazz);
169    if (classBinding == null) {
170      throw new IllegalStateException(String.format("Class '%s' is not bound", clazz.getClass().getName()));
171    }
172    IDeserializer<CLASS> retval;
173    switch (format) {
174    case JSON:
175      retval = new DefaultJsonDeserializer<>(this, classBinding);
176      break;
177    case XML:
178      retval = new DefaultXmlDeserializer<>(this, classBinding);
179      break;
180    case YAML:
181      retval = new DefaultYamlDeserializer<>(this, classBinding);
182      break;
183    default:
184      throw new UnsupportedOperationException(String.format("Unsupported format '%s'", format));
185    }
186
187    return retval;
188  }
189
190  @Override
191  public void registerBindingMatcher(@NonNull IBindingMatcher matcher) {
192    synchronized (this) {
193      bindingMatchers.add(matcher);
194    }
195  }
196
197  @NonNull
198  protected List<? extends IBindingMatcher> getBindingMatchers() {
199    synchronized (this) {
200      return CollectionUtil.unmodifiableList(bindingMatchers);
201    }
202  }
203
204  @Override
205  public Class<?> getBoundClassForXmlQName(@NonNull QName rootQName) {
206    Class<?> retval = null;
207    for (IBindingMatcher matcher : getBindingMatchers()) {
208      retval = matcher.getBoundClassForXmlQName(rootQName);
209      if (retval != null) {
210        break;
211      }
212    }
213    return retval;
214  }
215
216  @Override
217  public Class<?> getBoundClassForJsonName(@NonNull String rootName) {
218    Class<?> retval = null;
219    for (IBindingMatcher matcher : getBindingMatchers()) {
220      retval = matcher.getBoundClassForJsonName(rootName);
221      if (retval != null) {
222        break;
223      }
224    }
225    return retval;
226  }
227
228  @Override
229  public IBoundLoader newBoundLoader() {
230    return new DefaultBoundLoader(this);
231  }
232
233  @Override
234  public <CLASS> CLASS copyBoundObject(@NonNull CLASS other, Object parentInstance) throws BindingException {
235    IClassBinding classBinding = getClassBinding(other.getClass());
236    if (classBinding == null) {
237      throw new IllegalStateException(String.format("Class '%s' is not bound", other.getClass().getName()));
238    }
239    @SuppressWarnings("unchecked") CLASS retval = (CLASS) classBinding.copyBoundObject(other, parentInstance);
240    return retval;
241  }
242
243  @Override
244  public INodeItem toNodeItem(@NonNull Object boundObject, URI baseUri, boolean rootNode) {
245    IClassBinding classBinding = getClassBinding(boundObject.getClass());
246    if (classBinding == null) {
247      throw new IllegalStateException(String.format("Class '%s' is not bound", boundObject.getClass().getName()));
248    }
249    return DefaultNodeItemFactory.instance().newNodeItem(classBinding, boundObject, baseUri, rootNode);
250  }
