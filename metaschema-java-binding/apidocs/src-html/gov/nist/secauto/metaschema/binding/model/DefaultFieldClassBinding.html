092  /**
093   * Create a new {@link IClassBinding} for a Java bean annotated with the {@link BoundField}
094   * annotation.
095   *
096   * @param clazz
097   *          the Java bean class
098   * @param bindingContext
099   *          the Metaschema binding environment context
100   * @return the Metaschema field binding for the class

102  @NonNull
103  public static DefaultFieldClassBinding createInstance(@NonNull Class<?> clazz,
104      @NonNull IBindingContext bindingContext) {
105    Objects.requireNonNull(clazz, "clazz");
106    if (!clazz.isAnnotationPresent(MetaschemaField.class)) {
107      throw new IllegalArgumentException(
108          String.format("Class '%s' is missing the '%s' annotation.",
109              clazz.getName(),
110              MetaschemaField.class.getName()));
111    }
112    return new DefaultFieldClassBinding(clazz, bindingContext);
113  }

167  /**
168   * Collect all fields that are part of the model for this class.
169   *
170   * @param clazz
171   *          the class
172   * @return the field value instances if found or {@code null} otherwise
173   */
174  protected java.lang.reflect.Field getFieldValueField(Class<?> clazz) {
175    java.lang.reflect.Field[] fields = clazz.getDeclaredFields();
176
177    java.lang.reflect.Field retval = null;
178
179    Class<?> superClass = clazz.getSuperclass();
180    if (superClass != null) {
181      // get instances from superclass
182      retval = getFieldValueField(superClass);
183    }
184
185    if (retval == null) {
186      for (java.lang.reflect.Field field : fields) {
187        if (!field.isAnnotationPresent(MetaschemaFieldValue.class)) {
188          // skip fields that aren't a field or assembly instance
189          continue;
190        }
191
192        if (field.isAnnotationPresent(Ignore.class)) {
193          // skip this field, since it is ignored
194          continue;
195        }
196        retval = field;
197      }
198    }
199    return retval;
200  }

202  /**
203   * Initialize the flag instances for this class.
204   *
205   * @return the field value instance
206   */
207  protected IBoundFieldValueInstance initalizeFieldValueInstance() {
208    synchronized (this) {
209      if (this.fieldValue == null) {
210        java.lang.reflect.Field field = getFieldValueField(getBoundClass());
211        if (field == null) {
212          throw new IllegalArgumentException(
213              String.format("Class '%s' is missing the '%s' annotation on one of its fields.",
214                  getBoundClass().getName(),
215                  MetaschemaFieldValue.class.getName()));
216        }
217
218        this.fieldValue = new DefaultFieldValueProperty(this, field);
219      }
220      return this.fieldValue;
221    }
222  }

327  @SuppressWarnings("resource") // not owned
328  @NonNull
329  private Object readNormal(@Nullable Object parentInstance, boolean requiresJsonKey,
330      @NonNull IJsonParsingContext context)
331      throws IOException {
332    Predicate<IBoundFlagInstance> flagFilter = null;
333
334    IBoundFlagInstance jsonKey = null;
335    if (requiresJsonKey) {
336      IBoundFlagInstance instance = getJsonKeyFlagInstance();
337      if (instance == null) {
338        throw new IOException("This property is configured to use a JSON key, but no JSON key was found");
339      }
340
341      flagFilter = (flag) -> {
342        return !instance.equals(flag);
343      };
344      jsonKey = instance;
345    }
346
347    IBoundFlagInstance jsonValueKey = getJsonValueKeyFlagInstance();
348    if (jsonValueKey != null) {
349      if (flagFilter == null) {
350        flagFilter = (flag) -> {
351          return !jsonValueKey.equals(flag);
352        };
353      } else {
354        flagFilter = flagFilter.and((flag) -> {
355          return !jsonValueKey.equals(flag);
356        });
357      }
358    }
359
360    Map<String, ? extends IBoundNamedInstance> properties = getNamedInstances(flagFilter);
361
362    try {
363      Object instance = newInstance();
364
365      callBeforeDeserialize(instance, parentInstance);
366
367      JsonParser jsonParser = context.getReader(); // NOPMD - intentional
368
369      if (jsonKey != null) {
370        // if there is a json key, the first field will be the key
371        String key = jsonParser.currentName();
372        assert key != null;
373        jsonKey.setValue(instance, jsonKey.readValueFromString(key));
374
375        // advance past the field name
376        if (properties.isEmpty()) {
377          // the value will be a standard value
378          // advance past the field name
379          JsonUtil.assertAndAdvance(jsonParser, JsonToken.FIELD_NAME);
380        } else {
381          // the value will be a start object
382          JsonUtil.advanceAndAssert(jsonParser, JsonToken.START_OBJECT);
383          // advance past the start object
384          jsonParser.nextToken();
385        }
386      }
387
388      Set<String> handledProperties = new HashSet<>();
389      if (properties.isEmpty()) {
390        // this may be a value key value, an unrecognized flag, or the field value
391        IBoundFieldValueInstance fieldValue = getFieldValueInstance();
392        Object value = fieldValue.readValue(context);
393        if (value != null) {
394          fieldValue.setValue(instance, value);
395        }
396        handledProperties.add(fieldValue.getJsonValueKeyName());
397      } else {
398        // This could be an empty assembly signified by a END_OBJECT, or a series of properties signified by
399        // a FIELD_NAME
400        JsonUtil.assertCurrent(jsonParser, JsonToken.FIELD_NAME, JsonToken.END_OBJECT);
401
402        boolean parsedValueKey = false;
403        // now parse each property until the end object is reached
404        while (!jsonParser.hasTokenId(JsonToken.END_OBJECT.id())) {
405          String propertyName = jsonParser.getCurrentName();
406          assert propertyName != null;
407          // JsonUtil.assertAndAdvance(jsonParser, JsonToken.FIELD_NAME);
408
409          IBoundNamedInstance namedProperty = properties.get(propertyName);
410
411          boolean handled = false;
412          if (namedProperty != null) {
413            // this is a recognized flag
414
415            if (namedProperty.equals(jsonValueKey)) {
416              throw new IOException(
417                  String.format("JSON value key configured, but found standard flag for the value key '%s'",
418                      namedProperty.toCoordinates()));
419            }
420
421            // Now parse
422            Object value = namedProperty.read(context);
423            if (value != null) {
424              namedProperty.setValue(instance, value);
425              handled = true;
426            }
427          }
428
429          if (namedProperty == null && !parsedValueKey) {
430            // this may be a value key value, an unrecognized flag, or the field value
431            parsedValueKey = getFieldValueInstance().read(instance, context);
432
433            if (parsedValueKey) {
434              handled = true;
435            } else {
436              if (context.getProblemHandler().canHandleUnknownProperty(this, propertyName, context)) {
437                handled = context.getProblemHandler().handleUnknownProperty(this, propertyName, context);
438              }
439            }
440          }
441
442          if (handled) {
443            handledProperties.add(propertyName);
444          } else {
445            if (LOGGER.isWarnEnabled()) {
446              LOGGER.warn("Unrecognized property named '{}' at '{}'", propertyName,
447                  JsonUtil.toString(ObjectUtils.notNull(jsonParser.getCurrentLocation())));
448            }
449            JsonUtil.skipNextValue(jsonParser);
450          }
451        }
452      }
453
454      // set undefined properties
455      for (Map.Entry<String, ? extends IBoundNamedInstance> entry : properties.entrySet()) {
456        if (!handledProperties.contains(entry.getKey())) {
457          IBoundNamedInstance property = ObjectUtils.notNull(entry.getValue());
458          // use the default value of the collector
459          property.setValue(instance, property.newPropertyCollector().getValue());
460        }
461
462      }
463
464      if (jsonKey != null && !properties.isEmpty()) {
465        // read the END_OBJECT for the JSON key value
466        JsonUtil.assertAndAdvance(jsonParser, JsonToken.END_OBJECT);
467      }
468
469      if (properties.isEmpty()) {
470        // this is the next field or the end of the containing object of this field
471        JsonUtil.assertCurrent(jsonParser, JsonToken.FIELD_NAME, JsonToken.END_OBJECT);
472      } else {
473        // this is the current end element, but we are not responsible for parsing it.
474        JsonUtil.assertCurrent(jsonParser, JsonToken.END_OBJECT);
475      }
476
477      callAfterDeserialize(instance, parentInstance);
478      return instance;
479    } catch (BindingException ex) {
480      throw new IOException(ex);
481    }
482  }

484  @SuppressWarnings("resource") // not owned
485  @NonNull
486  private List<Object> readCollapsed(@Nullable Object parentInstance, boolean requiresJsonKey,
487      IJsonParsingContext context) throws IOException {
488
489    Predicate<IBoundFlagInstance> flagFilter = null;
490
491    IBoundFlagInstance jsonKey = null;
492    if (requiresJsonKey) {
493      IBoundFlagInstance instance = getJsonKeyFlagInstance();
494      if (instance == null) {
495        throw new IOException("This property is configured to use a JSON key, but no JSON key was found");
496      }
497
498      flagFilter = (flag) -> {
499        return !instance.equals(flag);
500      };
501      jsonKey = instance;
502    }
503
504    IBoundFlagInstance jsonValueKey = getJsonValueKeyFlagInstance();
505    if (jsonValueKey != null) {
506      if (flagFilter == null) {
507        flagFilter = (flag) -> {
508          return !jsonValueKey.equals(flag);
509        };
510      } else {
511        flagFilter = flagFilter.and((flag) -> {
512          return !jsonValueKey.equals(flag);
513        });
514      }
515    }
516
517    Map<String, ? extends IBoundNamedInstance> properties = getNamedInstances(flagFilter);
518
519    Map<IBoundNamedInstance, Supplier<? extends Object>> parsedProperties = new HashMap<>(); // NOPMD - intentional
520
521    JsonParser jsonParser = context.getReader(); // NOPMD - intentional
522    // This could be an empty assembly signified by a END_OBJECT, or a series of properties signified by
523    // a FIELD_NAME
524    JsonUtil.assertCurrent(jsonParser, JsonToken.FIELD_NAME, JsonToken.END_OBJECT);
525
526    if (jsonKey != null) {
527      // if there is a json key, the first field will be the key
528      String key = jsonParser.currentName();
529      assert key != null;
530      Supplier<? extends Object> jsonKeySupplier = jsonKey.readValueAndSupply(key);
531      parsedProperties.put(jsonKey, jsonKeySupplier);
532
533      // next the value will be a start object
534      JsonUtil.advanceAndAssert(jsonParser, JsonToken.START_OBJECT);
535      // advance to the next field name
536      jsonParser.nextToken();
537    }
538
539    List<? extends Object> values = null;
540    Set<String> handledProperties = new HashSet<>();
541    // now parse each property until the end object is reached
542    while (!jsonParser.hasTokenId(JsonToken.END_OBJECT.id())) {
543      String propertyName = jsonParser.getCurrentName();
544      // // advance past the field name
545      // jsonParser.nextToken();
546
547      IBoundNamedInstance namedProperty = properties.get(propertyName);
548
549      boolean handled = false;
550      if (namedProperty != null) {
551        // this is a recognized flag
552
553        if (namedProperty.equals(jsonValueKey)) {
554          throw new IOException(String.format(
555              "JSON value key configured, but found standard flag for the value key '%s'",
556              namedProperty.toCoordinates()));
557        }
558
559        // Now parse
560        Supplier<? extends Object> supplier = ((IBoundFlagInstance) namedProperty).readValueAndSupply(context);
561        parsedProperties.put(namedProperty, supplier);
562        handled = true;
563      } else {
564        // this may be a value key value, an unrecognized flag, or the field value
565        IBoundFieldValueInstance fieldValue = getFieldValueInstance();
566        if (jsonValueKey != null) {
567          // treat this as the value key
568          String key = jsonParser.nextFieldName();
569          assert key != null;
570          Supplier<? extends Object> supplier = jsonValueKey.readValueAndSupply(key);
571          parsedProperties.put(jsonValueKey, supplier);
572
573          values = handleCollapsedValues(context);
574          handled = true;
575        } else {
576          String valueKeyName = fieldValue.getJsonValueKeyName();
577          if (propertyName.equals(valueKeyName)) {
578            // advance past the field
579            jsonParser.nextToken();
580            // treat this as the field value
581            values = handleCollapsedValues(context);
582            handled = true;
583          }
584        }
585      }
586
587      if (handled) {
588        handledProperties.add(propertyName);
589      } else {
590        if (LOGGER.isWarnEnabled()) {
591          LOGGER.warn("Unrecognized property named '{}' at '{}'", propertyName,
592              JsonUtil.toString(ObjectUtils.notNull(jsonParser.getCurrentLocation())));
593        }
594        JsonUtil.skipNextValue(jsonParser);
595      }
596    }
597
598    // set undefined properties
599    for (Map.Entry<String, ? extends IBoundNamedInstance> entry : properties.entrySet()) {
600      if (!handledProperties.contains(entry.getKey())) {
601        // use the default value of the collector
602        IBoundNamedInstance property = ObjectUtils.notNull(entry.getValue());
603        parsedProperties.put(property, () -> {
604          return property.newPropertyCollector().getValue();
605        });
606      }
607    }
608
609    if (jsonKey != null) {
610      // read the END_OBJECT for the JSON key value
611      JsonUtil.assertAndAdvance(jsonParser, JsonToken.END_OBJECT);
612    }
613
614    // this is the current end element, but we are not responsible for parsing it.
615    JsonUtil.assertCurrent(jsonParser, JsonToken.END_OBJECT);
616
617    // now we need to clone one item per value
618    // TODO: handle the case where there are no values
619    List<Object> retval;
620    if (values == null) {
621      try {
622        Object item = newInstance();
623
624        callBeforeDeserialize(item, parentInstance);
625
626        for (Map.Entry<IBoundNamedInstance, Supplier<? extends Object>> entry : parsedProperties.entrySet()) {
627          IBoundNamedInstance property = entry.getKey();
628          Supplier<? extends Object> supplier = entry.getValue();
629
630          property.setValue(item, supplier.get());
631        }
632
633        callAfterDeserialize(item, parentInstance);
634
635        retval = CollectionUtil.singletonList(item);
636      } catch (BindingException ex) {
637        throw new IOException(ex);
638      }
639    } else {
640      List<Object> items = new ArrayList<>(values.size());
641      for (Object value : values) {
642        try {
643          Object item = newInstance();
644
645          callBeforeDeserialize(item, parentInstance);
646
647          initalizeFieldValueInstance().setValue(item, value);
648
649          for (Map.Entry<IBoundNamedInstance, Supplier<? extends Object>> entry : parsedProperties.entrySet()) {
650            IBoundNamedInstance property = entry.getKey();
651            Supplier<? extends Object> supplier = entry.getValue();
652
653            property.setValue(item, supplier.get());
654          }
655
656          callAfterDeserialize(item, parentInstance);
657          items.add(item);
658        } catch (BindingException ex) {
659          throw new IOException(ex);
660        }
661      }
662      retval = items;
663    }
664    return retval;
665  }

667  @SuppressWarnings("resource") // not owned
668  @NonNull
669  private List<? extends Object> handleCollapsedValues(@NonNull IJsonParsingContext context)
670      throws IOException {
671    IBoundFieldValueInstance fieldValue = getFieldValueInstance();
672
673    JsonParser jsonParser = context.getReader(); // NOPMD - intentional
674
675    ListPropertyCollector collector = new ListPropertyCollector();
676    if (jsonParser.hasToken(JsonToken.START_ARRAY)) {
677      JsonUtil.assertAndAdvance(jsonParser, JsonToken.START_ARRAY);
678      while (!jsonParser.hasToken(JsonToken.END_ARRAY)) {
679        Object value = fieldValue.readValue(context);
680        if (value != null) {
681          collector.add(value);
682        }
683      }
684      JsonUtil.assertAndAdvance(jsonParser, JsonToken.END_ARRAY);
685    } else {
686      Object value = fieldValue.readValue(context);
687      if (value != null) {
688        collector.add(value);
689      }
690    }
691
692    return collector.getValue();
693  }

701  @Override
702  public void writeItems(Collection<? extends Object> items, boolean writeObjectWrapper, IJsonWritingContext context)
703      throws IOException {
704    if (isCollapsible()) {
705      writeCollapsed(items, writeObjectWrapper, context);
706    } else {
707      writeNormal(items, writeObjectWrapper, context);
708    }
709  }

711  private void writeCollapsed(@NonNull Collection<? extends Object> items, boolean writeObjectWrapper,
712      @NonNull IJsonWritingContext context) throws IOException {
713    CollapseKeyBuilder builder = new CollapseKeyBuilder(this);
714    builder.addAll(items);
715
716    builder.write(writeObjectWrapper, context);
717  }
718
719  @SuppressWarnings("resource") // not owned
720  private void writeNormal(Collection<? extends Object> items, boolean writeObjectWrapper,
721      IJsonWritingContext context)
722      throws IOException {
723    if (items.isEmpty()) {
724      return;
725    }
726
727    Predicate<IBoundFlagInstance> flagFilter = null;
728
729    IBoundFlagInstance jsonKey = getJsonKeyFlagInstance();
730    if (jsonKey != null) {
731      flagFilter = (flag) -> {
732        return !jsonKey.equals(flag);
733      };
734    }
735
736    IBoundFlagInstance jsonValueKey = getJsonValueKeyFlagInstance();
737    if (jsonValueKey != null) {
738      if (flagFilter == null) {
739        flagFilter = (flag) -> {
740          return !jsonValueKey.equals(flag);
741        };
742      } else {
743        flagFilter = flagFilter.and((flag) -> {
744          return !jsonValueKey.equals(flag);
745        });
746      }
747    }
748
749    Map<String, ? extends IBoundNamedInstance> properties = getNamedInstances(flagFilter);
750
751    JsonGenerator writer = context.getWriter(); // NOPMD - intentional
752
753    for (Object item : items) {
754      assert item != null;
755      if (writeObjectWrapper) {
756        writer.writeStartObject();
757      }
758
759      if (jsonKey != null) {
760        // if there is a json key, the first field will be the key
761        Object flagValue = jsonKey.getValue(item);
762        String key = jsonKey.getValueAsString(flagValue);
763        if (key == null) {
764          throw new IOException(new NullPointerException("Null key value")); // NOPMD - intentional
765        }
766        writer.writeFieldName(key);
767
768        // next the value will be a start object
769        writer.writeStartObject();
770      }
771
772      for (IBoundNamedInstance property : properties.values()) {
773        ObjectUtils.notNull(property).write(item, context);
774      }
775
776      Object fieldValue = getFieldValueInstance().getValue(item);
777      if (fieldValue != null) {
778        String valueKeyName;
779        if (jsonValueKey != null) {
780          valueKeyName = jsonValueKey.getValueAsString(jsonValueKey.getValue(item));
781        } else {
782          valueKeyName = getFieldValueInstance().getJsonValueKeyName();
783        }
784        writer.writeFieldName(valueKeyName);
785        getFieldValueInstance().writeValue(fieldValue, context);
786      }
787
788      if (jsonKey != null) {
789        writer.writeEndObject();
790      }
791
792      if (writeObjectWrapper) {
793        writer.writeEndObject();
794      }
795    }
796  }
