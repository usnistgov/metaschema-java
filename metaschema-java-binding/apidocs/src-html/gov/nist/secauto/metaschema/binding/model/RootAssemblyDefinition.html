176  @Override
177  public Object readRoot(IXmlParsingContext context) throws IOException, XMLStreamException {
178
179    XMLEventReader2 reader = context.getReader();
180
181    // we may be at the START_DOCUMENT
182    if (reader.peek().isStartDocument()) {
183      XmlEventUtil.consumeAndAssert(reader, XMLStreamConstants.START_DOCUMENT);
184    }
185
186    XmlEventUtil.skipEvents(reader, XMLStreamConstants.CHARACTERS, XMLStreamConstants.PROCESSING_INSTRUCTION);
187
188    QName rootQName = getRootXmlQName();
189    if (!reader.peek().isStartElement()) {
190      throw new IOException(
191          String.format("Expected an element named '%s', but found a '%s' instead.",
192              rootQName,
193              XmlEventUtil.toString(reader.peek())));
194    }
195
196    XmlEventUtil.assertNext(reader, XMLStreamConstants.START_ELEMENT, rootQName);
197
198    StartElement start = ObjectUtils.notNull(reader.nextEvent().asStartElement());
199    Object result = ObjectUtils.requireNonNull(readItem(null, start, context));
200
201    XmlEventUtil.consumeAndAssert(reader, XMLStreamConstants.END_ELEMENT, rootQName);
202
203    // if (reader.hasNext() && LOGGER.isDebugEnabled()) {
204    // LOGGER.debug("After Parse: {}", XmlEventUtil.toString(reader.peek()));
205    // }
206
207    return result;
208  }

210  @SuppressWarnings("resource") // not owned
211  @Override
212  public Object readRoot(IJsonParsingContext context) throws IOException {
213    if (!isRoot()) {
214      throw new IOException(
215          String.format("The bound assembly '%s' does not have a root defined in the '%s' annotation.",
216              getBoundClass().getName(),
217              MetaschemaAssembly.class.getName()));
218    }
219
220    JsonParser parser = context.getReader(); // NOPMD - intentional
221
222    boolean objectWrapper = false;
223    if (parser.currentToken() == null) {
224      parser.nextToken();
225    }
226
227    if (JsonToken.START_OBJECT.equals(parser.currentToken())) {
228      // advance past the start object to the field name
229      JsonUtil.assertAndAdvance(parser, JsonToken.START_OBJECT);
230      objectWrapper = true;
231    }
232
233    String rootFieldName = getRootJsonName();
234    JsonToken token;
235    Object instance = null;
236    while (!(JsonToken.END_OBJECT.equals(token = parser.currentToken()) || token == null)) {
237      if (!JsonToken.FIELD_NAME.equals(token)) {
238        throw new IOException(String.format("Expected FIELD_NAME token, found '%s'", token.toString()));
239      }
240
241      String fieldName = parser.currentName();
242      if (fieldName.equals(rootFieldName)) {
243        // process the object value, bound to the requested class
244        JsonUtil.assertAndAdvance(parser, JsonToken.FIELD_NAME);
245        instance = readObject(context);
246
247        // stop now, since we found the root field
248        break;
249      }
250
251      if (!context.getProblemHandler().handleUnknownRootProperty(this, fieldName, context)) {
252        LOGGER.warn("Skipping unhandled top-level JSON field '{}'.", fieldName);
253        JsonUtil.skipNextValue(parser);
254      }
255    }
256
257    if (instance == null) {
258      throw new IOException(String.format("Failed to find root field '%s'.", rootFieldName));
259    }
260
261    if (objectWrapper) {
262      // advance past the end object
263      JsonUtil.assertAndAdvance(parser, JsonToken.END_OBJECT);
264    }
265
266    return instance;
267  }

269  @Override
270  public void writeRoot(Object instance, IXmlWritingContext context) throws XMLStreamException, IOException {
271
272    XMLStreamWriter2 writer = context.getWriter();
273
274    writer.writeStartDocument("UTF-8", "1.0");
275
276    QName rootQName = getRootXmlQName();
277
278    NamespaceContext nsContext = writer.getNamespaceContext();
279    String prefix = nsContext.getPrefix(rootQName.getNamespaceURI());
280    if (prefix == null) {
281      prefix = "";
282    }
283
284    writer.writeStartElement(prefix, rootQName.getLocalPart(), rootQName.getNamespaceURI());
285
286    writeItem(instance, rootQName, context);
287
288    writer.writeEndElement();
289  }
290
291  @Override
292  public void writeRoot(Object instance, IJsonWritingContext context) throws IOException {
293
294    @SuppressWarnings("resource") JsonGenerator writer = context.getWriter(); // NOPMD - intentional
295
296    // first read the initial START_OBJECT
297    writer.writeStartObject();
298
299    writer.writeFieldName(getRootJsonName());
300
301    writeItems(CollectionUtil.singleton(instance), true, context);
302
303    // end of root object
304    writer.writeEndObject();
305  }
