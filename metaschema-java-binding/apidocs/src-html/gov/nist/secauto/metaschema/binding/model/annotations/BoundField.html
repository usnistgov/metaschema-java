027package gov.nist.secauto.metaschema.binding.model.annotations;
028
029import static java.lang.annotation.ElementType.FIELD;
030import static java.lang.annotation.RetentionPolicy.RUNTIME;
031
032import gov.nist.secauto.metaschema.model.common.MetaschemaModelConstants;
033
034import java.lang.annotation.Documented;
035import java.lang.annotation.Retention;
036import java.lang.annotation.Target;
037
038import edu.umd.cs.findbugs.annotations.NonNull;
039
040/**
041 * Identifies that the annotation target is a bound property that references a Metaschema field.
042 * <p>
043 * For XML serialization, the {@link #useName()} identifies the name of the element to use and the
044 * {@link #namespace()} identifies the namespace of this element.
045 * <p>
046 * For JSON and YAML serializations, the {@link #useName()} identifies the property/item name to
047 * use.
048 * <p>
049 * The field must be either:
050 * <ol>
051 * <li>A Metaschema data type or a collection whose item value is Metaschema data type, with a
052 * {@link BoundFieldValue} annotation on the field.
053 * <li>A type or a collection whose item value is a type based on a class with a
054 * {@link MetaschemaField} annotation.</li>
055 * </ol>
056 */
057@Documented
058@Retention(RUNTIME)
059@Target({ FIELD })
060public @interface BoundField {
061  /**
062   * Get the documentary formal name of the field.
063   * <p>
064   * If the value is "##none", then the description will be considered {@code null}.
065   *
066   * @return a markdown string or {@code "##none"} if no formal name is provided
067   */
068  @NonNull
069  String formalName() default Constants.NO_STRING_VALUE;
070
071  /**
072   * Get the documentary description of the field.
073   * <p>
074   * If the value is "##none", then the description will be considered {@code null}.
075   *
076   * @return a markdown string or {@code "##none"} if no description is provided
077   */
078  @NonNull
079  String description() default Constants.NO_STRING_VALUE;
080
081  /**
082   * The model name to use for JSON/YAML singleton values and associated XML elements.
083   * <p>
084   * If the value is "##none", then the use name will be provided by the definition or by the field
085   * name if the item value class is missing the {@link MetaschemaField} annotation.
086   *
087   * @return the name
088   */
089  @NonNull
090  String useName() default Constants.NO_STRING_VALUE;
091
092  /**
093   * The namespace to use for associated XML elements.
094   * <p>
095   * If the value is "##default", then element name is derived from the namespace provided in the
096   * package-info.
097   *
098   * @return the namespace
099   */
100  String namespace() default Constants.DEFAULT_STRING_VALUE;
101
102  /**
103   * If the data type allows it, determines if the field's value must be wrapped with an XML element.
104   *
105   * @return {@code true} if the field must be wrapped, or {@code false} otherwise
106   */
107  boolean inXmlWrapped() default MetaschemaModelConstants.DEFAULT_FIELD_IN_XML_WRAPPED;
108
109  /**
110   * A non-negative number that indicates the minimum occurrence of the model instance.
111   *
112   * @return a non-negative number
113   */
114  int minOccurs() default MetaschemaModelConstants.DEFAULT_GROUP_AS_MIN_OCCURS;
115
116  /**
117   * A number that indicates the maximum occurrence of the model instance.
118   *
119   * @return a positive number or {@code -1} to indicate "unbounded"
120   */
121  int maxOccurs() default MetaschemaModelConstants.DEFAULT_GROUP_AS_MAX_OCCURS;
122
123  /**
124   * Get any remarks for this field.
125   *
126   * @return a markdown string or {@code "##none"} if no remarks are provided
127   */
128  @NonNull
129  String remarks() default Constants.NO_STRING_VALUE;
130}
