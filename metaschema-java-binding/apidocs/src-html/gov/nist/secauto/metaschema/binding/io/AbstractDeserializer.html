027package gov.nist.secauto.metaschema.binding.io;

029import gov.nist.secauto.metaschema.binding.IBindingContext;
030import gov.nist.secauto.metaschema.binding.model.IAssemblyClassBinding;
031import gov.nist.secauto.metaschema.model.common.configuration.IConfiguration;
032import gov.nist.secauto.metaschema.model.common.configuration.IMutableConfiguration;
033import gov.nist.secauto.metaschema.model.common.constraint.DefaultConstraintValidator;
034import gov.nist.secauto.metaschema.model.common.constraint.IConstraintValidationHandler;
035import gov.nist.secauto.metaschema.model.common.constraint.LoggingConstraintValidationHandler;
036import gov.nist.secauto.metaschema.model.common.metapath.DynamicContext;
037import gov.nist.secauto.metaschema.model.common.metapath.StaticContext;
038import gov.nist.secauto.metaschema.model.common.metapath.item.INodeItem;

040
041import java.io.IOException;
042import java.io.Reader;
043import java.net.URI;
044
045import edu.umd.cs.findbugs.annotations.NonNull;
046
047/**
048 * The base class of all format-specific deserializers.
049 *
050 * @param <CLASS>
051 *          the bound class to deserialize to
052 */
053public abstract class AbstractDeserializer<CLASS>
054    extends AbstractSerializationBase<DeserializationFeature<?>>
055    implements IDeserializer<CLASS> {
056
057  private IConstraintValidationHandler constraintValidationHandler;
058
059  /**
060   * Construct a new deserializer.
061   *
062   * @param bindingContext
063   *          the binding context used to supply bound Java classes while writing
064   * @param classBinding
065   *          the bound class information for the Java type this deserializer is operating on
066   */
067  protected AbstractDeserializer(@NonNull IBindingContext bindingContext, @NonNull IAssemblyClassBinding classBinding) {
068    super(bindingContext, classBinding);
069  }
070
071  @Override
072  public IConstraintValidationHandler getConstraintValidationHandler() {
073    synchronized (this) {
074      if (constraintValidationHandler == null) {
075        constraintValidationHandler = new LoggingConstraintValidationHandler();
076      }
077      return ObjectUtils.notNull(constraintValidationHandler);
078    }
079  }
080
081  @Override
082  public void setConstraintValidationHandler(@NonNull IConstraintValidationHandler constraintValidationHandler) {
083    synchronized (this) {
084      this.constraintValidationHandler = constraintValidationHandler;
085    }
086  }
087
088  @Override
089  public INodeItem deserializeToNodeItem(Reader reader, URI documentUri) throws IOException {
090
091    INodeItem nodeItem;
092    try {
093      nodeItem = deserializeToNodeItemInternal(reader, documentUri);
094    } catch (Exception ex) { // NOPMD - this is intentional
095      throw new IOException(ex);
096    }
097
098    if (isValidating()) {
099      StaticContext staticContext = new StaticContext();
100      DynamicContext dynamicContext = staticContext.newDynamicContext();
101      dynamicContext.setDocumentLoader(getBindingContext().newBoundLoader());
102      DefaultConstraintValidator validator = new DefaultConstraintValidator(
103          dynamicContext,
104          getConstraintValidationHandler());
105      validator.validate(nodeItem);
106      validator.finalizeValidation();
107    }
108    return nodeItem;
109  }
110
111  /**
112   * This abstract method delegates parsing to the concrete implementation.
113   *
114   * @param reader
115   *          the reader instance to read data from
116   * @param documentUri
117   *          the URI of the document that is being read
118   * @return a new node item containing the read contents
119   * @throws IOException
120   *           if an error occurred while reading data from the stream
121   */
122  @NonNull
123  protected abstract INodeItem deserializeToNodeItemInternal(@NonNull Reader reader, @NonNull URI documentUri)
124      throws IOException;
125
126  @Override
127  public IDeserializer<CLASS> enableFeature(DeserializationFeature<?> feature) {
128    return set(feature, true);
129  }
130
131  @Override
132  public IDeserializer<CLASS> disableFeature(DeserializationFeature<?> feature) {
133    return set(feature, false);
134  }
135
136  @Override
137  public IDeserializer<CLASS> applyConfiguration(
138      @NonNull IConfiguration<DeserializationFeature<?>> other) {
139    IMutableConfiguration<DeserializationFeature<?>> config = getConfiguration();
140    config.applyConfiguration(other);
141    configurationChanged(config);
142    return this;
143  }
144
145  @Override
146  public IDeserializer<CLASS> set(DeserializationFeature<?> feature, Object value) {
147    IMutableConfiguration<DeserializationFeature<?>> config = getConfiguration();
148    config.set(feature, value);
149    configurationChanged(config);
150    return this;
151  }
152}
