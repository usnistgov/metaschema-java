173  @Override
174  public Format detectFormat(InputSource source) throws IOException {
175    Format retval;
176    if (source.getCharacterStream() != null) {
177      throw new UnsupportedOperationException("Character streams are not supported");
178    } else if (source.getByteStream() != null) {
179      // attempt to use a provided byte stream stream
180      retval = detectFormatInternal(ObjectUtils.notNull(source.getByteStream()));
181    } else {
182      // fall back to a URL-based connection
183      URI uri = ObjectUtils.notNull(URI.create(source.getSystemId()));
184      URL url = uri.toURL();
185      try (InputStream is = url.openStream()) {
186        retval = detectFormatInternal(ObjectUtils.notNull(is));
187      }
188    }
189    return retval;
190  }
191
192  @NonNull
193  protected Format detectFormatInternal(@NonNull InputStream is) throws IOException {
194    return detectFormatInternal(is, LOOK_AHEAD_BYTES - 1);
195  }

197  @NonNull
198  protected Format detectFormatInternal(@NonNull InputStream is, int lookAheadBytes) throws IOException {
199    DataFormatMatcher matcher = matchFormat(is, lookAheadBytes);
200    return formatFromMatcher(matcher);
201  }

203  private JsonFactory[] getDetectorFactory() {
204    if (detectorFactory == null) {
205      detectorFactory = new JsonFactory[3];
206      detectorFactory[0] = YamlFactoryFactory.newParserFactoryInstance(getConfiguration());
207      detectorFactory[1] = JsonFactoryFactory.instance();
208      detectorFactory[2] = new XmlFactory();
209    }
210    return detectorFactory;
211  }

213  @NonNull
214  protected DataFormatMatcher matchFormat(@NonNull InputStream is, int lookAheadBytes) throws IOException {
215    DataFormatDetector det = new DataFormatDetector(getDetectorFactory());
216    det = det.withMinimalMatch(MatchStrength.INCONCLUSIVE).withOptimalMatch(MatchStrength.SOLID_MATCH)
217        .withMaxInputLookahead(lookAheadBytes);
218
219    DataFormatMatcher matcher = det.findFormat(is);
220    switch (matcher.getMatchStrength()) {
221    case FULL_MATCH:
222    case SOLID_MATCH:
223    case WEAK_MATCH:
224    case INCONCLUSIVE:
225      return matcher;
226    case NO_MATCH:
227    default:
228      throw new UnsupportedOperationException("Unable to identify format");
229    }
230  }
231
232  @NonNull
233  protected Format formatFromMatcher(@NonNull DataFormatMatcher matcher) {
234    Format retval;
235    String formatName = matcher.getMatchedFormatName();
236    if (YAMLFactory.FORMAT_NAME_YAML.equals(formatName)) {
237      retval = Format.YAML;
238    } else if (JsonFactory.FORMAT_NAME_JSON.equals(formatName)) {
239      retval = Format.JSON;
240    } else if (XmlFactory.FORMAT_NAME_XML.equals(formatName)) {
241      retval = Format.XML;
242    } else {
243      throw new UnsupportedOperationException(String.format("The detected format '%s' is not supported", formatName));
244    }
245    return retval;
246  }

250    BufferedInputStream bis = new BufferedInputStream(is, LOOK_AHEAD_BYTES); // NOPMD - stream not owned

255  @Override
256  public IDocumentNodeItem loadAsNodeItem(InputSource source) throws IOException {
257    URI uri = ObjectUtils.notNull(URI.create(source.getSystemId()));
258
259    IDocumentNodeItem retval;
260    if (source.getByteStream() != null) {
261      // attempt to use a provided byte stream stream
262      try (BufferedInputStream bis = toBufferedInputStream(ObjectUtils.notNull(source.getByteStream()))) {
263        retval = loadAsNodeItemInternal(bis, uri);
264      }
265    } else {
266      // fall back to a URL-based connection
267      URL url = uri.toURL();
268      try (InputStream is = url.openStream()) {
269        try (BufferedInputStream bis = toBufferedInputStream(ObjectUtils.notNull(is))) {
270          retval = loadAsNodeItemInternal(bis, uri);
271        }
272      }
273    }
274    return retval;
275  }

277  @Override
278  public IDocumentNodeItem loadAsNodeItem(@NonNull Format format, @NonNull InputSource source)
279      throws IOException {
280    URI uri = ObjectUtils.notNull(URI.create(source.getSystemId()));
281
282    IDocumentNodeItem retval;
283    if (source.getByteStream() != null) {
284      // attempt to use a provided byte stream stream
285      try (@SuppressWarnings("resource") BufferedInputStream bis = toBufferedInputStream(
286          ObjectUtils.requireNonNull(source.getByteStream()))) {
287        Class<?> clazz = detectModel(bis, format); // NOPMD - must be called before reset
288        retval = deserializeToNodeItem(clazz, format, bis, uri);
289      }
290    } else {
291      // fall back to a URL-based connection
292      URL url = uri.toURL();
293      try (InputStream is = url.openStream()) {
294        try (BufferedInputStream bis = toBufferedInputStream(ObjectUtils.notNull(is))) {
295          Class<?> clazz = detectModel(bis, format); // NOPMD - must be called before reset
296          retval = deserializeToNodeItem(clazz, format, bis, uri);
297        }
298      }
299    }
300    return retval;
301  }

303  @NonNull
304  protected IDocumentNodeItem loadAsNodeItemInternal(@NonNull BufferedInputStream bis, @NonNull URI documentUri)
305      throws IOException {
306    DataFormatMatcher matcher = matchFormat(bis, LOOK_AHEAD_BYTES - 1);
307    Format format = formatFromMatcher(matcher);
308    Class<?> clazz = detectModel(matcher, format); // NOPMD - must be called before reset
309    return deserializeToNodeItem(clazz, format, bis, documentUri);
310  }

312  @NonNull
313  protected IDocumentNodeItem deserializeToNodeItem(@NonNull Class<?> clazz, @NonNull Format format,
314      @NonNull BufferedInputStream bis, @NonNull URI documentUri) throws IOException {
315    try {
316      bis.reset();
317    } catch (IOException ex) {
318      throw new IOException("Unable to reset input stream before parsing", ex);
319    }
320
321    IDeserializer<?> deserializer = getDeserializer(clazz, format, getConfiguration());
322    return (IDocumentNodeItem) deserializer.deserializeToNodeItem(bis, documentUri);
323  }

325  @NonNull
326  protected Class<?> detectModel(@NonNull BufferedInputStream bis, @NonNull Format format)
327      throws IOException {
328    Class<?> clazz;
329    switch (format) {
330    case JSON:
331      clazz = detectModelJsonClass(ObjectUtils.notNull(JsonFactoryFactory.instance().createParser(bis)));
332      if (clazz == null) {
333        throw new IllegalStateException(
334            String.format("Detected format '%s', but unable to detect the bound data type", format.name()));
335      }
336      break;
337    case YAML:
338      YAMLFactory factory = YamlFactoryFactory.newParserFactoryInstance(getConfiguration());
339      clazz = detectModelJsonClass(ObjectUtils.notNull(factory.createParser(bis)));
340      if (clazz == null) {
341        throw new IllegalStateException(
342            String.format("Detected format '%s', but unable to detect the bound data type", format.name()));
343      }
344      break;
345    case XML:
346      clazz = detectModelXmlClass(ObjectUtils.notNull(bis));
347      break;
348    default:
349      throw new UnsupportedOperationException(
350          String.format("The format '%s' is not supported", format));
351    }
352
353    try {
354      bis.reset();
355    } catch (IOException ex) {
356      throw new IOException("Unable to reset input stream before parsing", ex);
357    }
358    return clazz;
359  }
360
361  @NonNull
362  protected Class<?> detectModel(@NonNull DataFormatMatcher matcher, @NonNull Format format)
363      throws IOException {
364    Class<?> clazz;
365    switch (format) {
366    case JSON:
367    case YAML:
368      clazz = detectModelJsonClass(ObjectUtils.notNull(matcher.createParserWithMatch()));
369      if (clazz == null) {
370        throw new IllegalStateException(
371            String.format("Detected format '%s', but unable to detect the bound data type", format.name()));
372      }
373      break;
374    case XML:
375      clazz = detectModelXmlClass(ObjectUtils.notNull(matcher.getDataStream()));
376      break;
377    default:
378      throw new UnsupportedOperationException(
379          String.format("The detected format '%s' is not supported", matcher.getMatchedFormatName()));
380    }
381    return clazz;
382  }

384  @NonNull
385  protected Class<?> detectModelXmlClass(@NonNull InputStream is) throws IOException {
386
387    QName startElementQName;
388    try {
389      XMLInputFactory2 xmlInputFactory = (XMLInputFactory2) XMLInputFactory.newInstance();
390      assert xmlInputFactory instanceof WstxInputFactory;
391      xmlInputFactory.configureForXmlConformance();
392      xmlInputFactory.setProperty(XMLInputFactory.IS_COALESCING, false);
393
394      Reader reader = new InputStreamReader(is, Charset.forName("UTF8"));
395      XMLEventReader2 eventReader = (XMLEventReader2) xmlInputFactory.createXMLEventReader(reader);
396      if (eventReader.peek().isStartDocument()) {
397        while (eventReader.hasNext() && !eventReader.peek().isStartElement()) {
398          eventReader.nextEvent();
399        }
400      }
401
402      if (!eventReader.peek().isStartElement()) {
403        throw new UnsupportedOperationException("Unable to detect a start element");
404      }
405
406      StartElement start = eventReader.nextEvent().asStartElement();
407      startElementQName = ObjectUtils.notNull(start.getName());
408    } catch (XMLStreamException ex) {
409      throw new IOException(ex);
410    }
411
412    Class<?> clazz = getBoundClassForXmlQName(startElementQName);
413
414    if (clazz == null) {
415      throw new UnsupportedOperationException("Unrecognized element name: " + startElementQName.toString());
416    }
417    return clazz;
418  }

424  @Nullable
425  protected Class<?> detectModelJsonClass(@NonNull JsonParser parser) throws IOException {
426    Class<?> retval = null;
427    JsonUtil.advanceAndAssert(parser, JsonToken.START_OBJECT);
428    outer: while (JsonToken.FIELD_NAME.equals(parser.nextToken())) {
429      String name = ObjectUtils.notNull(parser.getCurrentName());
430      if ("$schema".equals(name)) {
431        // do nothing
432        parser.nextToken();
433        // JsonUtil.skipNextValue(parser);
434      } else {
435        retval = getBoundClassForJsonName(name);
436        break outer;
437      }
438    }
439    return retval;
440  }

446  @Override
447  public <CLASS> CLASS load(Class<CLASS> clazz, InputSource source) throws IOException {
448    URI uri = ObjectUtils.notNull(URI.create(source.getSystemId()));
449
450    CLASS retval;
451    if (source.getCharacterStream() != null) {
452      throw new UnsupportedOperationException("Character streams are not supported");
453    } else if (source.getByteStream() != null) {
454      // attempt to use a provided byte stream stream
455      try (BufferedInputStream bis = new BufferedInputStream(source.getByteStream(), LOOK_AHEAD_BYTES)) {
456        retval = loadInternal(clazz, bis, uri);
457      }
458    } else {
459      // fall back to a URL-based connection
460      URL url = uri.toURL();
461      try (InputStream is = url.openStream()) {
462        try (BufferedInputStream bis = new BufferedInputStream(is, LOOK_AHEAD_BYTES)) {
463          retval = loadInternal(clazz, bis, uri);
464        }
465      }
466    }
467    return retval;
468  }

470  @SuppressWarnings("unchecked")
471  @NonNull
472  protected <CLASS> CLASS loadInternal(@NonNull Class<CLASS> clazz, @NonNull BufferedInputStream bis,
473      @NonNull URI documentUri) throws IOException {
474    // we cannot close this stream, since it will cause the underlying stream to be closed
475    bis.mark(LOOK_AHEAD_BYTES);
476
477    Format format = detectFormatInternal(bis);
478    bis.reset();
479
480    IDeserializer<CLASS> deserializer = getDeserializer(clazz, format, getConfiguration());
481    INodeItem nodeItem = deserializer.deserializeToNodeItem(bis, documentUri);
482    return (CLASS) ObjectUtils.requireNonNull(nodeItem.getValue());
483  }

485  @NonNull
486  protected <CLASS> IDeserializer<CLASS> getDeserializer(
487      @NonNull Class<CLASS> clazz,
488      @NonNull Format format,
489      @NonNull IConfiguration<DeserializationFeature<?>> config) {
490    IDeserializer<CLASS> retval = getBindingContext().newDeserializer(format, clazz);
491    retval.applyConfiguration(config);
492    return retval;
493  }
