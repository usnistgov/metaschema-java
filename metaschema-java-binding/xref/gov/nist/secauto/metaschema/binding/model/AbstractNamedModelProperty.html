185   public boolean isNextProperty(@NonNull IXmlParsingContext context) throws XMLStreamException {
186     XMLEventReader2 eventReader = context.getReader();
187 
188     XmlEventUtil.skipWhitespace(eventReader);
189 
190     boolean handled = false;
191     QName groupQName = getXmlGroupAsQName();
192     if (groupQName != null) {
193       // we are to parse the grouping element, if the next token matches
194       XMLEvent event = eventReader.peek();
195       if (event.isStartElement() && groupQName.equals(event.asStartElement().getName())) {
196         handled = true;
197       }
198     }
199 
200     if (!handled) {
201       XMLEvent event = eventReader.peek();
202       QName xmlQName = getXmlQName();
203       if (xmlQName != null && event.isStartElement() && xmlQName.equals(event.asStartElement().getName())) {
204         handled = true;
205       }
206     }
207     return handled;
208   }

210   @Override
211   public boolean read(Object parentInstance, StartElement start, IXmlParsingContext context)
212       throws IOException, XMLStreamException {
213     boolean handled = isNextProperty(context);
214     if (handled) {
215       Object value = readInternal(parentInstance, start, context);
216       setValue(parentInstance, value);
217     }
218     return handled;
219   }

221   @SuppressWarnings("resource")
222   @Override
223   protected Object readInternal(Object parentInstance, IJsonParsingContext context)
224       throws IOException {
225     JsonParser parser = context.getReader(); // NOPMD - intentional
226 
227     // the parser's current token should be the JSON field name
228     // advance past the property name
229     JsonUtil.assertAndAdvance(parser, JsonToken.FIELD_NAME);
230 
231     // parse the value
232     IPropertyCollector collector = newPropertyCollector();
233     IModelPropertyInfo info = getPropertyInfo();
234     info.readValue(collector, parentInstance, context);
235 
236     JsonUtil.assertCurrent(context.getReader(), JsonToken.FIELD_NAME, JsonToken.END_OBJECT);
237 
238     return collector.getValue();
239   }

241   protected Object readInternal(@Nullable Object parentInstance, @NonNull StartElement start,
242       @NonNull IXmlParsingContext context)
243       throws IOException, XMLStreamException {
244     XMLEventReader2 eventReader = context.getReader();
245 
246     XmlEventUtil.skipWhitespace(eventReader);
247 
248     StartElement currentStart = start;
249 
250     QName groupQName = getXmlGroupAsQName();
251     if (groupQName != null) {
252       // we are to parse the grouping element, if the next token matches
253       XMLEvent groupEvent = XmlEventUtil.consumeAndAssert(eventReader, XMLStreamConstants.START_ELEMENT, groupQName);
254       currentStart = ObjectUtils.notNull(groupEvent.asStartElement());
255     }
256 
257     IPropertyCollector collector = newPropertyCollector();
258     // There are zero or more named values based on cardinality
259     getPropertyInfo().readValue(collector, parentInstance, currentStart, context);
260 
261     Object value = collector.getValue();
262 
263     // consume extra whitespace between elements
264     XmlEventUtil.skipWhitespace(eventReader);
265 
266     if (groupQName != null) {
267       // consume the end of the group
268       XmlEventUtil.consumeAndAssert(eventReader, XMLStreamConstants.END_ELEMENT, groupQName);
269     }
270 
271     return value;
272   }

286   @Override
287   public boolean write(Object parentInstance, QName parentName, IXmlWritingContext context)
288       throws XMLStreamException, IOException {
289     Object value = getValue(parentInstance);
290     if (value == null) {
291       return false; // NOPMD - intentional
292     }
293 
294     QName currentStart = parentName;
295     XMLStreamWriter2 writer = context.getWriter();
296     QName groupQName = getXmlGroupAsQName();
297     if (groupQName != null) {
298       // write the grouping element
299       writer.writeStartElement(groupQName.getNamespaceURI(), groupQName.getLocalPart());
300       currentStart = groupQName;
301     }
302 
303     // There are one or more named values based on cardinality
304     getPropertyInfo().writeValue(value, currentStart, context);
305 
306     if (groupQName != null) {
307       writer.writeEndElement();
308     }
309     return true;
310   }

326   public void write(Object parentInstance, IJsonWritingContext context) throws IOException {
327     if (getPropertyInfo().isValueSet(parentInstance)) {
328       // write the field name

331       // dispatch to the property info implementation to address cardinality
332       getPropertyInfo().writeValue(parentInstance, context);

336   @Override
337   public void copyBoundObject(@NonNull Object fromInstance, @NonNull Object toInstance) throws BindingException {
338     Object value = getValue(fromInstance);
339     if (value != null) {
340       IModelPropertyInfo propertyInfo = getPropertyInfo();
341       IPropertyCollector collector = newPropertyCollector();
342 
343       propertyInfo.copy(fromInstance, toInstance, collector);
344 
345       value = collector.getValue();
346     }
347     setValue(toInstance, value);
348   }

372   private static class SimpleGroupAs implements IGroupAs {
373     @NonNull
374     private final String name;
375     @NonNull
376     private final Lazy<String> namespace;
377     @NonNull
378     private final GroupAs annotation;
379 
380     private SimpleGroupAs(@NonNull GroupAs annotation, @NonNull IClassBinding parentDefinition) {
381       this.annotation = annotation;
382       {
383         String value = ModelUtil.resolveLocalName(annotation.name(), null);
384         if (value == null) {
385           throw new IllegalStateException(
386               String.format("The %s#groupName value '%s' resulted in an invalid null value",
387                   GroupAs.class.getName(),
388                   annotation.name()));
389         }
390         this.name = value;
391       }
392       this.namespace = ObjectUtils.notNull(
393           Lazy.lazy(() -> ModelUtil.resolveNamespace(annotation.namespace(), parentDefinition)));
394     }
395 
396     @Override
397     public String getGroupAsName() {
398       return name;
399     }
400 
401     @Override
402     public String getGroupAsXmlNamespace() {
403       return namespace.get();
404     }
405 
406     @Override
407     public JsonGroupAsBehavior getJsonGroupAsBehavior() {
408       return annotation.inJson();
409     }
410 
411     @Override
412     public XmlGroupAsBehavior getXmlGroupAsBehavior() {
413       return annotation.inXml();
414     }
415   }
