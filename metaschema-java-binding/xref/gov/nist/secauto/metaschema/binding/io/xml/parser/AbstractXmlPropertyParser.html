27  package gov.nist.secauto.metaschema.binding.io.xml.parser;
28  
29  import gov.nist.secauto.metaschema.binding.BindingContext;
30  import gov.nist.secauto.metaschema.binding.BindingException;
31  import gov.nist.secauto.metaschema.binding.model.property.PropertyBinding;
32  
33  import org.apache.logging.log4j.LogManager;
34  import org.apache.logging.log4j.Logger;
35  import org.codehaus.stax2.XMLEventReader2;
36  
37  import javax.xml.namespace.QName;
38  import javax.xml.stream.XMLStreamException;
39  import javax.xml.stream.events.EndElement;
40  import javax.xml.stream.events.StartElement;
41  import javax.xml.stream.events.XMLEvent;
42  
43  public abstract class AbstractXmlPropertyParser<BINDING extends PropertyBinding> implements XmlPropertyParser {
44    private static final Logger logger = LogManager.getLogger(AbstractXmlPropertyParser.class);
45  
46    private final BINDING propertyBinding;
47    private final BindingContext bindingContext;
48  
49    public AbstractXmlPropertyParser(BINDING propertyBinding, BindingContext bindingContext) {
50      this.propertyBinding = propertyBinding;
51      this.bindingContext = bindingContext;
52    }
53  
54    @Override
55    public BINDING getPropertyBinding() {
56      return propertyBinding;
57    }
58  
59    protected BindingContext getBindingContext() {
60      return bindingContext;
61    }
62  
63    protected StartElement consumeStartElement(XMLEventReader2 reader, QName name) throws BindingException {
64      XMLEvent event;
65      try {
66        event = reader.nextEvent();
67      } catch (XMLStreamException ex) {
68        throw new BindingException(ex);
69      }
70      if (logger.isDebugEnabled()) {
71        logger.debug("Consume: {}", XmlEventUtil.toString(event));
72      }
73      StartElement startElement = event.asStartElement();
74      QName actualName = startElement.getName();
75      if (!actualName.equals(name)) {
76        throw new BindingException(String.format("Unexpected START ELEMENT '%s' at '%s'. Expected'%s'.", actualName,
77            XmlEventUtil.toString(startElement.getLocation()), name));
78      }
79      return startElement;
80    }
81  
82    protected EndElement consumeEndElement(XMLEventReader2 reader, QName name) throws BindingException {
83      XMLEvent event;
84      try {
85        event = reader.nextEvent();
86      } catch (XMLStreamException ex) {
87        throw new BindingException(ex);
88      }
89      if (logger.isDebugEnabled()) {
90        logger.debug("Consume: {}", XmlEventUtil.toString(event));
91      }
92      try {
93        EndElement element = event.asEndElement();
94        QName actualName = element.getName();
95        if (!actualName.equals(name)) {
96          throw new BindingException(String.format("Unexpected END ELEMENT name '%s' at '%s'. Expected'%s'.", actualName,
97              XmlEventUtil.toString(element.getLocation()), name));
98        }
99        return element;
100     } catch (ClassCastException ex) {
101       throw new BindingException(String.format("Expected END ELEMENT, but found '%s' at '%s'.",
102           XmlEventUtil.toString(event), XmlEventUtil.toString(event.getLocation()), name));
103     }
104   }
