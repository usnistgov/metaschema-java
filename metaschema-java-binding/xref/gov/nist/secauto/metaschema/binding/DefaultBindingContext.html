69  /**
70   * The implementation of a {@link IBindingContext} provided by this library.
71   * <p>
72   * This implementation caches Metaschema information, which can dramatically improve read and write
73   * performance at the cost of some memory use. Thus, using the same singleton of this class across
74   * multiple I/O operations will improve overall read and write performance when processing the same
75   * types of data.
76   * <p>
77   * Serializers and deserializers provided by this class using the
78   * {@link #newSerializer(Format, Class)} and {@link #newDeserializer(Format, Class)} methods will
79   * <p>
80   * This class is synchronized and is thread-safe.
81   */
82  public class DefaultBindingContext implements IBindingContext {
83    private static DefaultBindingContext singleton;
84  
85    private final Map<Class<?>, IClassBinding> classBindingsByClass = new HashMap<>();
86    private final Map<Class<? extends IJavaTypeAdapter<?>>, IJavaTypeAdapter<?>> javaTypeAdapterMap = new HashMap<>();
87    private final List<IBindingMatcher> bindingMatchers = new LinkedList<>();
88  
89    public static DefaultBindingContext instance() {
90      synchronized (DefaultBindingContext.class) {
91        if (singleton == null) {
92          singleton = new DefaultBindingContext();
93        }
94      }
95      return singleton;
96    }

99     * Construct a new binding context.
100    */
101   protected DefaultBindingContext() {
102     // only allow extended classes
103   }

105   @Override
106   public synchronized IClassBinding getClassBinding(@NotNull Class<?> clazz) {
107     IClassBinding retval = classBindingsByClass.get(clazz);
108     if (retval == null) {
109       if (clazz.isAnnotationPresent(MetaschemaAssembly.class)) {
110         retval = DefaultAssemblyClassBinding.createInstance(clazz, this);
111       } else if (clazz.isAnnotationPresent(MetaschemaField.class)) {
112         retval = DefaultFieldClassBinding.createInstance(clazz, this);
113       } else {
114         throw new IllegalArgumentException(String.format(
115             "Class '%s' does not represent a Metaschema definition" + " since it is missing a '%s' or '%s' annotation.",
116             clazz.getName(), MetaschemaAssembly.class.getName(), MetaschemaField.class.getName()));
117       }
118       if (retval != null) {
119         classBindingsByClass.put(clazz, retval);
120       }
121     }
122     return retval;
123   }
124 
125   @Override
126   public synchronized <TYPE extends IJavaTypeAdapter<?>> IJavaTypeAdapter<TYPE>
127       getJavaTypeAdapterInstance(@NotNull Class<TYPE> clazz) {
128     @SuppressWarnings("unchecked")
129     IJavaTypeAdapter<TYPE> retval
130         = (IJavaTypeAdapter<TYPE>) javaTypeAdapterMap.get(clazz);
131     if (retval == null) {
132       Constructor<TYPE> constructor;
133       try {
134         constructor = clazz.getDeclaredConstructor();
135       } catch (NoSuchMethodException ex) {
136         throw new IllegalArgumentException(ex);
137       }
138 
139       try {
140         @SuppressWarnings("unchecked")
141         IJavaTypeAdapter<TYPE> instance
142             = (IJavaTypeAdapter<TYPE>) constructor.newInstance();
143         retval = instance;
144       } catch (InstantiationException | IllegalAccessException | InvocationTargetException ex) {
145         throw new IllegalArgumentException(ex);
146       }
147       javaTypeAdapterMap.put(clazz, retval);
148     }
149     return retval;
150   }

152   /**
153    * {@inheritDoc}
154    * <p>
155    * A serializer returned by this method is thread-safe.
156    */
157   @Override
158   public <CLASS> ISerializer<CLASS> newSerializer(@NotNull Format format, @NotNull Class<CLASS> clazz) {
159     Objects.requireNonNull(format, "format");
160     IAssemblyClassBinding classBinding = (IAssemblyClassBinding) getClassBinding(clazz);
161 
162     ISerializer<CLASS> retval;
163     switch (format) {
164     case JSON:
165       retval = new DefaultJsonSerializer<>(this, classBinding);
166       break;
167     case XML:
168       retval = new DefaultXmlSerializer<>(this, classBinding);
169       break;
170     case YAML:
171       retval = new DefaultYamlSerializer<>(this, classBinding);
172       break;
173     default:
174       throw new UnsupportedOperationException(String.format("Unsupported format '%s'", format));
175     }
176 
177     return retval;
178   }

181    * {@inheritDoc}
182    * <p>
183    * A deserializer returned by this method is thread-safe.
184    */
185   @Override
186   public <CLASS> IDeserializer<CLASS> newDeserializer(@NotNull Format format, @NotNull Class<CLASS> clazz) {
187     Objects.requireNonNull(format, "format");
188     IAssemblyClassBinding classBinding = (IAssemblyClassBinding) getClassBinding(clazz);
189 
190     IDeserializer<CLASS> retval;
191     switch (format) {
192     case JSON:
193       retval = new DefaultJsonDeserializer<>(this, classBinding);
194       break;
195     case XML:
196       retval = new DefaultXmlDeserializer<>(this, classBinding);
197       break;
198     case YAML:
199       retval = new DefaultYamlDeserializer<>(this, classBinding);
200       break;
201     default:
202       throw new UnsupportedOperationException(String.format("Unsupported format '%s'", format));
203     }
204 
205     return retval;
206   }

208   @Override
209   public synchronized void registerBindingMatcher(@NotNull IBindingMatcher matcher) {
210     bindingMatchers.add(matcher);
211   }

213   protected synchronized List<IBindingMatcher> getBindingMatchers() {
214     return Collections.unmodifiableList(bindingMatchers);
215   }
216 
217   public synchronized Map<Class<?>, IClassBinding> getClassBindingsByClass() {
218     return Collections.unmodifiableMap(classBindingsByClass);
219   }
220 
221   public Map<Class<? extends IJavaTypeAdapter<?>>, IJavaTypeAdapter<?>> getJavaTypeAdaptersByClass() {
222     return Collections.unmodifiableMap(javaTypeAdapterMap);
223   }

225   @Override
226   public Class<?> getBoundClassForXmlQName(@NotNull QName rootQName) {
227     Class<?> retval = null;
228     for (IBindingMatcher matcher : getBindingMatchers()) {
229       retval = matcher.getBoundClassForXmlQName(rootQName);
230       if (retval != null) {
231         break;
232       }
233     }
234     return retval;
235   }

237   @Override
238   public Class<?> getBoundClassForJsonName(@NotNull String rootName) {
239     Class<?> retval = null;
240     for (IBindingMatcher matcher : getBindingMatchers()) {
241       retval = matcher.getBoundClassForJsonName(rootName);
242       if (retval != null) {
243         break;
244       }
245     }
246     return retval;
247   }

254   @Override
255   public <CLASS> CLASS copyBoundObject(@NotNull CLASS other, Object parentInstance) throws BindingException {
256     IClassBinding classBinding = getClassBinding(other.getClass());
257     @SuppressWarnings("unchecked")
258     CLASS retval = (CLASS) classBinding.copyBoundObject(other, parentInstance);
259     return retval;
260   }

262   @Override
263   public IBoundXdmNodeItem toNodeItem(@NotNull Object boundObject, URI baseUri, boolean rootNode) {
264     IClassBinding binding = getClassBinding(boundObject.getClass());
265     return IXdmFactory.INSTANCE.newNodeItem(binding, boundObject, baseUri, rootNode);
266   }
267 
268   @Override
269   public void validate(@NotNull Object boundObject, URI baseUri, boolean rootNode,
270       IConstraintValidationHandler handler) {
271     IBoundXdmNodeItem nodeItem = toNodeItem(boundObject, baseUri, rootNode);
272 
273     StaticContext staticContext = new StaticContext();
274     DynamicContext dynamicContext = staticContext.newDynamicContext();
275     dynamicContext.setDocumentLoader(newBoundLoader());
276     DefaultConstraintValidator validator = new DefaultConstraintValidator(dynamicContext);
277     if (handler != null) {
278       validator.setConstraintValidationHandler(handler);
279     }
280     nodeItem.validate(validator);
281     validator.finalizeValidation();
282   }
