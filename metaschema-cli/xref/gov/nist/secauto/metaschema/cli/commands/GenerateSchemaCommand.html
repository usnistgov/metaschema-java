66  public class GenerateSchemaCommand
67      extends AbstractTerminalCommand {
68    private static final Logger LOGGER = LogManager.getLogger(GenerateSchemaCommand.class);
69  
70    @NonNull
71    private static final String COMMAND = "generate-schema";
72    @NonNull
73    private static final List<ExtraArgument> EXTRA_ARGUMENTS;
74    @NonNull
75    private static final Option OVERWRITE_OPTION = ObjectUtils.notNull(
76        Option.builder()
77            .longOpt("overwrite")
78            .desc("overwrite the destination if it exists")
79            .build());
80    @NonNull
81    private static final Option AS_OPTION = ObjectUtils.notNull(
82        Option.builder()
83            .longOpt("as")
84            .hasArg()
85            .argName("FORMAT")
86            .desc("source format: xml, json, or yaml")
87            .build());
88    @NonNull
89    private static final Option INLINE_TYPES_OPTION = ObjectUtils.notNull(
90        Option.builder()
91            .longOpt("inline-types")
92            .desc("definitions declared inline will be generated as inline types")
93            .build());
94    static {
95      EXTRA_ARGUMENTS = ObjectUtils.notNull(List.of(
96          new DefaultExtraArgument("metaschema file", true),
97          new DefaultExtraArgument("destination schema file", false)));
98    }
99  
100   @Override
101   public String getName() {
102     return COMMAND;
103   }
104 
105   @Override
106   public String getDescription() {
107     return "Generate a schema for the specified Metaschema";
108   }
109 
110   @SuppressWarnings("null")
111   @Override
112   public Collection<? extends Option> gatherOptions() {
113     return List.of(
114         OVERWRITE_OPTION,
115         AS_OPTION,
116         INLINE_TYPES_OPTION);
117   }
118 
119   @Override
120   public List<ExtraArgument> getExtraArguments() {
121     return EXTRA_ARGUMENTS;
122   }

124   @SuppressWarnings("PMD.PreserveStackTrace") // intended
125   @Override
126   public void validateOptions(CallingContext callingContext, CommandLine cmdLine) throws InvalidArgumentException {
127     try {
128       String toFormatText = cmdLine.getOptionValue(AS_OPTION);
129       SchemaFormat.valueOf(toFormatText.toUpperCase(Locale.ROOT));
130     } catch (IllegalArgumentException ex) {
131       InvalidArgumentException newEx = new InvalidArgumentException( // NOPMD - intentional
132           String.format("Invalid '%s' argument. The format must be one of: %s.",
133               OptionUtils.toArgument(AS_OPTION),
134               Arrays.asList(Format.values()).stream()
135                   .map(format -> format.name())
136                   .collect(CustomCollectors.joiningWithOxfordComma("and"))));
137       newEx.setOption(AS_OPTION);
138       newEx.addSuppressed(ex);
139       throw newEx;
140     }
141 
142     List<String> extraArgs = cmdLine.getArgList();
143     if (extraArgs.isEmpty() || extraArgs.size() > 2) {
144       throw new InvalidArgumentException("Illegal number of arguments.");
145     }
146 
147     Path metaschema = Paths.get(extraArgs.get(0));
148     if (!Files.exists(metaschema)) {
149       throw new InvalidArgumentException("The provided metaschema '" + metaschema + "' does not exist.");
150     }
151     if (!Files.isReadable(metaschema)) {
152       throw new InvalidArgumentException("The provided metaschema '" + metaschema + "' is not readable.");
153     }
154   }
155 
156   @Override
157   public ExitStatus executeCommand(CallingContext callingContext, CommandLine cmdLine) {
158     List<String> extraArgs = cmdLine.getArgList();
159 
160     Path destination = null;
161     if (extraArgs.size() > 1) {
162       destination = Paths.get(extraArgs.get(1)).toAbsolutePath();
163     }
164 
165     if (destination != null) {
166       if (Files.exists(destination)) {
167         if (!cmdLine.hasOption(OVERWRITE_OPTION)) {
168           return ExitCode.FAIL.exitMessage( // NOPMD readability
169               String.format("The provided destination '%s' already exists and the '%s' option was not provided.",
170                   destination,
171                   OptionUtils.toArgument(OVERWRITE_OPTION)));
172         }
173         if (!Files.isWritable(destination)) {
174           return ExitCode.FAIL.exitMessage( // NOPMD readability
175               "The provided destination '" + destination + "' is not writable.");
176         }
177       } else {
178         Path parent = destination.getParent();
179         if (parent != null) {
180           try {
181             Files.createDirectories(parent);
182           } catch (IOException ex) {
183             return ExitCode.INVALID_TARGET.exit().withThrowable(ex); // NOPMD readability
184           }
185         }
186       }
187     }
188 
189     String asFormatText = cmdLine.getOptionValue(AS_OPTION);
190     SchemaFormat asFormat = SchemaFormat.valueOf(asFormatText.toUpperCase(Locale.ROOT));
191 
192     IMutableConfiguration<SchemaGenerationFeature> configuration
193         = new DefaultConfiguration<>(SchemaGenerationFeature.class);
194     if (cmdLine.hasOption(INLINE_TYPES_OPTION)) {
195       configuration.enableFeature(SchemaGenerationFeature.INLINE_DEFINITIONS);
196       if (SchemaFormat.JSON.equals(asFormat)) {
197         configuration.disableFeature(SchemaGenerationFeature.INLINE_CHOICE_DEFINITIONS);
198       } else {
199         configuration.enableFeature(SchemaGenerationFeature.INLINE_CHOICE_DEFINITIONS);
200       }
201     }
202 
203     Path input = Paths.get(extraArgs.get(0));
204     assert input != null;
205     try {
206       IMetaschema metaschema = new MetaschemaLoader().load(input);
207       if (destination == null) {
208         @SuppressWarnings({ "resource", "PMD.CloseResource" }) // not owned
209         OutputStream os = ObjectUtils.notNull(System.out);
210         ISchemaGenerator.generateSchema(metaschema, os, asFormat, configuration);
211       } else {
212         ISchemaGenerator.generateSchema(metaschema, destination, asFormat, configuration);
213       }
214     } catch (IOException | MetaschemaException ex) {
215       return ExitCode.FAIL.exit().withThrowable(ex); // NOPMD readability
216     }
217     if (destination != null && LOGGER.isInfoEnabled()) {
218       LOGGER.info("Generated {} schema file: {}", asFormat.toString(), destination);
219     }
220     return ExitCode.OK.exit();
221   }
