27  package gov.nist.secauto.metaschema.databind.codegen.typeinfo;
28  
29  import gov.nist.secauto.metaschema.core.model.IDefinition;
30  import gov.nist.secauto.metaschema.core.model.INamedInstance;
31  
32  import java.util.Collection;
33  
34  import edu.umd.cs.findbugs.annotations.NonNull;
35  import edu.umd.cs.findbugs.annotations.Nullable;
36  
37  public interface IDefinitionTypeInfo {
38    /**
39     * Get the definition associated with this type info.
40     *
41     * @return the definition
42     */
43    @NonNull
44    IDefinition getDefinition();
45  
46    /**
47     * Gets the resolver which can be used to lookup Java type information for
48     * Module objects.
49     *
50     * @return the type resolver
51     */
52    @NonNull
53    ITypeResolver getTypeResolver();
54  
55    /**
56     * Check's if the Java class to be generated will have a property with the given
57     * name.
58     *
59     * @param propertyName
60     *          the property name to look for
61     * @return {@code true} if there is an associated property with the name or
62     *         {@code false} otherwise
63     */
64    boolean hasPropertyWithName(@NonNull String propertyName);
65  
66    /**
67     * Get the type information for the provided {@code instance} value.
68     *
69     * @param instance
70     *          the instance to get type information for
71     * @return the type information
72     */
73    @Nullable
74    IInstanceTypeInfo getInstanceTypeInfo(@NonNull INamedInstance instance);
75  
76    /**
77     * Get the type information for all instance values on this definition.
78     *
79     * @return the type information
80     */
81    @NonNull
82    Collection<IInstanceTypeInfo> getInstanceTypeInfos();
83  
84    /**
85     * Get the type information for all Java class properties associated with a
86     * given instance.
87     *
88     * @return the type information
89     */
90    @NonNull
91    Collection<ITypeInfo> getPropertyTypeInfos();
92  }
