027package gov.nist.secauto.metaschema.databind.codegen;
028
029import gov.nist.secauto.metaschema.core.model.IModule;
030import gov.nist.secauto.metaschema.core.util.CollectionUtil;
031import gov.nist.secauto.metaschema.databind.codegen.config.IBindingConfiguration;
032import gov.nist.secauto.metaschema.databind.codegen.typeinfo.ITypeResolver;
033
034import org.apache.logging.log4j.LogManager;
035import org.apache.logging.log4j.Logger;
036
037import java.io.IOException;
038import java.nio.file.Path;
039import java.util.Collection;
040import java.util.Objects;
041
042import edu.umd.cs.findbugs.annotations.NonNull;
043
044/**
045 * Provides methods for generating Java classes based on a single or a
046 * collection of Metaschemas.
047 */
048public final class JavaGenerator {
049  private static final Logger LOGGER = LogManager.getLogger(JavaGenerator.class);
050
051  private JavaGenerator() {
052    // disable construction
053  }
054
055  /**
056   * Generate Java sources for the provided Metaschema module.
057   *
058   * @param module
059   *          the Metaschema module to generate Java sources for
060   * @param targetDir
061   *          the directory to generate sources in
062   * @param bindingConfiguration
063   *          the binding customizations to use when generating the Java classes
064   * @return information about all the produced classes
065   * @throws IOException
066   *           if an error occurred while generating the class
067   */
068  public static IProduction generate(
069      @NonNull IModule module,
070      @NonNull Path targetDir,
071      @NonNull IBindingConfiguration bindingConfiguration) throws IOException {
072    return generate(CollectionUtil.singletonList(module), targetDir, bindingConfiguration);
073  }
074
075  /**
076   * Generates Java classes for Module fields and flags.
077   *
078   * @param modules
079   *          the Metaschema modules to build classes for
080   * @param targetDirectory
081   *          the directory to generate classes in
082   * @param bindingConfiguration
083   *          binding customizations that can be used to set namespaces, class
084   *          names, and other aspects of generated classes
085   * @return information about all the produced classes
086   * @throws IOException
087   *           if a build error occurred while generating the class
088   */
089  @NonNull
090  public static IProduction generate(
091      @NonNull Collection<? extends IModule> modules,
092      @NonNull Path targetDirectory,
093      @NonNull IBindingConfiguration bindingConfiguration) throws IOException {
094    Objects.requireNonNull(modules, "metaschemas");
095    Objects.requireNonNull(targetDirectory, "generationTargetDirectory");
096    Objects.requireNonNull(bindingConfiguration, "bindingConfiguration");
097    if (LOGGER.isInfoEnabled()) {
098      LOGGER.info("Generating Java classes in: {}", targetDirectory);
099    }
100
101    ITypeResolver typeResolver = ITypeResolver.newTypeResolver(bindingConfiguration);
102
103    return IProduction.of(modules, typeResolver, targetDirectory);
104  }
105}
