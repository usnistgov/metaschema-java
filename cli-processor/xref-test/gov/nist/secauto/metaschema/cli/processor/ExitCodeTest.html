32  import org.apache.logging.log4j.LogManager;
33  import org.apache.logging.log4j.core.LogEvent;
34  import org.apache.logging.log4j.core.Logger;
35  import org.apache.logging.log4j.core.appender.AbstractAppender;
36  import org.junit.jupiter.api.BeforeAll;
37  import org.junit.jupiter.api.BeforeEach;
38  import org.junit.jupiter.api.Test;
39  
40  import java.util.LinkedList;
41  import java.util.List;
42  
43  /**
44   * Logging solution based on
45   * https://stackoverflow.com/questions/24205093/how-to-create-a-custom-appender-in-log4j2
46   */
47  class ExitCodeTest {
48    private static MockedAppender mockedAppender;
49    private static Logger logger;
50  
51    @BeforeEach
52    public void setup() {
53      mockedAppender.events.clear();
54    }
55  
56    @BeforeAll
57    public static void setupClass() {
58      mockedAppender = new MockedAppender();
59      logger = (Logger) LogManager.getLogger(AbstractExitStatus.class);
60      logger.addAppender(mockedAppender);
61      // logger.setLevel(Level.INFO);
62      mockedAppender.start();
63    }
64  
65    @Test
66    void testExitMessage() {
67      Throwable ex = new IllegalStateException("a message");
68      ExitStatus exitStatus = ExitCode.FAIL.exit().withThrowable(ex);
69      exitStatus.generateMessage(false);
70  
71      List<LogEvent> events = mockedAppender.getEvents();
72      assertAll(
73          () -> assertEquals(1, events.size()),
74          () -> assertEquals("a message", events.get(0).getMessage().getFormattedMessage()));
75    }
76  
77    @Test
78    void testExitThrown() {
79      Throwable ex = new IllegalStateException("a message");
80      ExitStatus exitStatus = ExitCode.FAIL.exit().withThrowable(ex);
81      exitStatus.generateMessage(true);
82  
83      List<LogEvent> events = mockedAppender.getEvents();
84      assertAll(
85          () -> assertEquals(1, events.size()),
86          () -> assertEquals(ex, events.get(0).getThrown()),
87          () -> assertEquals("a message", events.get(0).getMessage().getFormattedMessage()));
88    }
89  
90    private static class MockedAppender
91        extends AbstractAppender {
92  
93      private final List<LogEvent> events = new LinkedList<>();
94  
95      protected MockedAppender() {
96        super("MockedAppender", null, null, false, null);
97      }
98  
99      public List<LogEvent> getEvents() {
100       return events;
101     }
102 
103     @Override
104     public void append(LogEvent event) {
105       synchronized (this) {
106         events.add(event.toImmutable());
107       }
108     }
109   }
110 }
