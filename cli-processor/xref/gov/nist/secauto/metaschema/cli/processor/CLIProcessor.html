117   public static void main(String... args) {
118     System.setProperty("java.util.logging.manager", "org.apache.logging.log4j.jul.LogManager");
119     CLIProcessor processor = new CLIProcessor("metaschema-cli");
120 
121     CommandService.getInstance().getCommands().stream().forEach(command -> {
122       assert command != null;
123       processor.addCommandHandler(command);
124     });
125     System.exit(processor.process(args).getExitCode().getStatusCode());
126   }

177   @NonNull
178   private ExitStatus parseCommand(String... args) {
179     List<String> commandArgs = Arrays.asList(args);
180     assert commandArgs != null;
181     CallingContext callingContext = new CallingContext(commandArgs);
182 
183     ExitStatus status;
184     // the first two arguments should be the <command> and <operation>, where <type> is the object type
185     // the <operation> is performed against.
186     if (commandArgs.isEmpty()) {
187       status = ExitCode.INVALID_COMMAND.exit();
188       callingContext.showHelp();
189     } else {
190       status = callingContext.processCommand();
191     }
192     return status;
193   }

206   public static void handleQuiet() {
207     @SuppressWarnings("resource") LoggerContext ctx = (LoggerContext) LogManager.getContext(false); // NOPMD not
208                                                                                                     // closable here
209     Configuration config = ctx.getConfiguration();
210     LoggerConfig loggerConfig = config.getLoggerConfig(LogManager.ROOT_LOGGER_NAME);
211     Level oldLevel = loggerConfig.getLevel();
212     if (oldLevel.isLessSpecificThan(Level.ERROR)) {
213       loggerConfig.setLevel(Level.ERROR);
214       ctx.updateLoggers();
215     }
216   }

218   protected void showVersion() {
219     @SuppressWarnings("resource") PrintStream out = AnsiConsole.out(); // NOPMD - not owner
220     getVersionInfos().stream().forEach((info) -> {
221       out.println(ansi()
222           .bold().a(info.getName()).boldOff()
223           .a(" ")
224           .bold().a(info.getVersion()).boldOff()
225           .a(" built at ")
226           .bold().a(info.getBuildTimestamp()).boldOff()
227           .a(" from branch ")
228           .bold().a(info.getGitBranch()).boldOff()
229           .a(" (")
230           .bold().a(info.getGitCommit()).boldOff()
231           .a(") at ")
232           .bold().a(info.getGitOriginUrl()).boldOff()
233           .reset());
234     });
235     out.flush();
236   }

244   public class CallingContext {
245     @NonNull
246     private final List<Option> options;
247     @NonNull
248     private final Deque<Command> calledCommands;
249     @NonNull
250     private final List<String> extraArgs;
251 
252     public CallingContext(@NonNull List<String> args) {
253       Map<String, Command> topLevelCommandMap = getTopLevelCommands().stream()
254           .collect(Collectors.toUnmodifiableMap(Command::getName, Function.identity()));
255 
256       List<Option> options = new LinkedList<>(OPTIONS);
257       Deque<Command> calledCommands = new LinkedList<>();
258       List<String> extraArgs = new LinkedList<>();
259 
260       boolean endArgs = false;
261       for (String arg : args) {
262         if (endArgs) {
263           extraArgs.add(arg);
264         } else {
265           if (arg.startsWith("-")) {
266             extraArgs.add(arg);
267           } else if ("--".equals(arg)) {
268             endArgs = true;
269           } else {
270             Command command;
271             if (calledCommands.isEmpty()) {
272               command = topLevelCommandMap.get(arg);
273             } else {
274               command = calledCommands.getLast();
275               command = command.getSubCommandByName(arg);
276             }
277 
278             if (command == null) {
279               extraArgs.add(arg);
280               endArgs = true;
281             } else {
282               calledCommands.add(command);
283             }
284           }
285         }
286       }
287 
288       if (LOGGER.isDebugEnabled()) {
289         String commandChain = calledCommands.stream()
290             .map(command -> command.getName())
291             .collect(Collectors.joining(" -> "));
292         LOGGER.debug("Processing command chain: {}", commandChain);
293       }
294 
295       for (Command cmd : calledCommands) {
296         options.addAll(cmd.gatherOptions());
297       }
298 
299       options = Collections.unmodifiableList(options);
300       extraArgs = Collections.unmodifiableList(extraArgs);
301 
302       assert options != null;
303       assert extraArgs != null;
304 
305       this.options = options;
306       this.calledCommands = calledCommands;
307       this.extraArgs = extraArgs;
308     }
309

338     @SuppressWarnings("PMD.OnlyOneReturn") // readability
339     @NonNull
340     public ExitStatus processCommand() {
341       CommandLineParser parser = new DefaultParser();
342       CommandLine cmdLine;
343       try {
344         cmdLine = parser.parse(toOptions(), getExtraArgs().toArray(new String[0]));
345       } catch (ParseException ex) {
346         String msg = ex.getMessage();
347         assert msg != null;
348         return handleInvalidCommand(msg);
349       }
350 
351       if (cmdLine.hasOption(NO_COLOR_OPTION)) {
352         handleNoColor();
353       }
354 
355       if (cmdLine.hasOption(QUIET_OPTION)) {
356         handleQuiet();
357       }
358 
359       ExitStatus retval = null;
360       if (cmdLine.hasOption(VERSION_OPTION)) {
361         showVersion();
362         retval = ExitCode.OK.exit();
363       } else if (cmdLine.hasOption(HELP_OPTION)) {
364         showHelp();
365         retval = ExitCode.OK.exit();
366         // } else {
367         // retval = handleInvalidCommand(commandResult, options,
368         // "Invalid command arguments: " + cmdLine.getArgList().stream().collect(Collectors.joining(" ")));
369       }
370 
371       if (retval == null) {
372         retval = invokeCommand(cmdLine);
373       }
374 
375       retval.generateMessage(cmdLine.hasOption(SHOW_STACK_TRACE_OPTION));
376       return retval;
377     }
378 
379     @SuppressWarnings("PMD.OnlyOneReturn") // readability
380     protected ExitStatus invokeCommand(@NonNull CommandLine cmdLine) {
381       for (Command cmd : getCalledCommands()) {
382         try {
383           cmd.validateOptions(this, cmdLine);
384         } catch (InvalidArgumentException ex) {
385           String msg = ex.getMessage();
386           assert msg != null;
387           return handleInvalidCommand(msg);
388         }
389       }
390 
391       Command targetCommand = getTargetCommand();
392       ExitStatus retval;
393       if (targetCommand == null) {
394         retval = ExitCode.INVALID_COMMAND.exit();
395       } else {
396         retval = targetCommand.executeCommand(this, cmdLine);
397       }
398 
399       if (ExitCode.INVALID_COMMAND.equals(retval.getExitCode())) {
400         showHelp();
401       }
402       return retval;
403     }

405     @NonNull
406     public ExitStatus handleInvalidCommand(
407         @NonNull String message) {
408       showHelp();
409 
410       ExitStatus retval = ExitCode.INVALID_COMMAND.exitMessage(message);
411       retval.generateMessage(false);
412       return retval;
413     }

426     /**
427      * Callback for providing a help footer.
428      *
429      * @param exec
430      *          the executable name
431      *
432      * @return the footer or {@code null}
433      */
434     @NonNull
435     private String buildHelpFooter() {
436 
437       Command targetCommand = getTargetCommand();
438       Collection<Command> subCommands;
439       if (targetCommand == null) {
440         subCommands = getTopLevelCommands();
441       } else {
442         subCommands = targetCommand.getSubCommands();
443       }
444 
445       String retval;
446       if (subCommands.isEmpty()) {
447         retval = "";
448       } else {
449         StringBuilder builder = new StringBuilder(64);
450         builder
451             .append(System.lineSeparator())
452             .append("The following are available commands:")
453             .append(System.lineSeparator());
454 
455         int length = subCommands.stream()
456             .mapToInt(command -> command.getName().length())
457             .max().orElse(0);
458 
459         for (Command command : subCommands) {
460           builder.append(
461               ansi()
462                   .render(String.format("   @|bold %-" + length + "s|@ %s%n",
463                       command.getName(),
464                       command.getDescription())));
465         }
466         builder
467             .append(System.lineSeparator())
468             .append('\'')
469             .append(getExec())
470             .append(" <command> --help' will show help on that specific command.")
471             .append(System.lineSeparator());
472         retval = builder.toString();
473         assert retval != null;
474       }
475       return retval;
476     }

478     /**
479      * Get the CLI syntax.
480      *
481      * @return the CLI syntax to display in help output
482      */
483     protected String buildHelpCliSyntax() {
484 
485       StringBuilder builder = new StringBuilder(64);
486       builder.append(getExec());
487 
488       Deque<Command> calledCommands = getCalledCommands();
489       if (!calledCommands.isEmpty()) {
490         builder.append(calledCommands.stream()
491             .map(Command::getName)
492             .collect(Collectors.joining(" ", " ", "")));
493       }
494 
495       // output calling commands
496       Command targetCommand = getTargetCommand();
497       if (targetCommand == null) {
498         builder.append(" <command>");
499       } else {
500         Collection<Command> subCommands = targetCommand.getSubCommands();
501 
502         if (!subCommands.isEmpty()) {
503           builder.append(' ');
504           if (!targetCommand.isSubCommandRequired()) {
505             builder.append('[');
506           }
507 
508           builder.append("<command>");
509 
510           if (!targetCommand.isSubCommandRequired()) {
511             builder.append(']');
512           }
513         }
514       }
515 
516       // output required options
517       getOptionsList().stream()
518           .filter(option -> option.isRequired())
519           .forEach(option -> {
520             builder
521                 .append(' ')
522                 .append(OptionUtils.toArgument(ObjectUtils.notNull(option)));
523             if (option.hasArg()) {
524               builder
525                   .append('=')
526                   .append(option.getArgName());
527             }
528           });
529 
530       // output non-required option placeholder
531       builder.append(" [<options>]");
532 
533       // output extra arguments
534       if (targetCommand != null) {
535         // handle extra arguments
536         for (ExtraArgument argument : targetCommand.getExtraArguments()) {
537           builder.append(' ');
538           if (!argument.isRequired()) {
539             builder.append('[');
540           }
541 
542           builder.append('<');
543           builder.append(argument.getName());
544           builder.append('>');
545 
546           if (argument.getNumber() > 1) {
547             builder.append("...");
548           }
549 
550           if (!argument.isRequired()) {
551             builder.append(']');
552           }
553         }
554       }
555 
556       String retval = builder.toString();
557       assert retval != null;
558       return retval;
559     }

561     public void showHelp() {
562 
563       HelpFormatter formatter = new HelpFormatter();
564       formatter.setLongOptSeparator("=");
565 
566       AnsiPrintStream out = AnsiConsole.out();
567       int terminalWidth = Math.max(out.getTerminalWidth(), 40);
568 
569       @SuppressWarnings("resource") PrintWriter writer = new PrintWriter(out, true, StandardCharsets.UTF_8); // NOPMD -
570                                                                                                              // not
571                                                                                                              // owned
572       formatter.printHelp(
573           writer,
574           terminalWidth,
575           buildHelpCliSyntax(),
576           buildHelpHeader(),
577           toOptions(),
578           HelpFormatter.DEFAULT_LEFT_PAD,
579           HelpFormatter.DEFAULT_DESC_PAD,
580           buildHelpFooter(),
581           false);
582       writer.flush();
583     }
