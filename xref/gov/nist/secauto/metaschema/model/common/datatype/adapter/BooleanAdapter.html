27  package gov.nist.secauto.metaschema.model.common.datatype.adapter;
28  
29  import com.fasterxml.jackson.core.JsonGenerator;
30  import com.fasterxml.jackson.core.JsonParser;
31  import com.fasterxml.jackson.databind.jsonFormatVisitors.JsonFormatTypes;
32  
33  import gov.nist.secauto.metaschema.model.common.datatype.AbstractDataTypeAdapter;
34  import gov.nist.secauto.metaschema.model.common.metapath.function.InvalidValueForCastFunctionException;
35  import gov.nist.secauto.metaschema.model.common.metapath.item.IAnyAtomicItem;
36  import gov.nist.secauto.metaschema.model.common.metapath.item.IBooleanItem;
37  import gov.nist.secauto.metaschema.model.common.metapath.item.INumericItem;
38  import gov.nist.secauto.metaschema.model.common.metapath.item.IStringItem;
39  import gov.nist.secauto.metaschema.model.common.util.ObjectUtils;
40  
41  import java.io.IOException;
42  import java.util.List;
43  
44  import edu.umd.cs.findbugs.annotations.NonNull;
45  
46  public class BooleanAdapter
47      extends AbstractDataTypeAdapter<Boolean, IBooleanItem> {
48    @NonNull
49    private static final List<String> NAMES = ObjectUtils.notNull(
50        List.of("boolean"));
51  
52    BooleanAdapter() {
53      super(Boolean.class);
54    }
55  
56    @Override
57    public List<String> getNames() {
58      return NAMES;
59    }
60  
61    @Override
62    public JsonFormatTypes getJsonRawType() {
63      return JsonFormatTypes.BOOLEAN;
64    }
65  
66    @SuppressWarnings("null")
67    @Override
68    public Boolean parse(String value) {
69      return Boolean.valueOf(value);
70    }
71  
72    @Override
73    public Boolean parse(JsonParser parser) throws IOException {
74      Boolean value = parser.getBooleanValue();
75      // skip over value
76      parser.nextToken();
77      return value;
78    }
79  
80    @Override
81    public void writeJsonValue(Object value, JsonGenerator generator)
82        throws IOException {
83      try {
84        generator.writeBoolean((Boolean) value);
85      } catch (ClassCastException ex) {
86        throw new IOException(ex);
87      }
88    }
89  
90    @Override
91    public Boolean copy(Object obj) {
92      // the value is immutable
93      return (Boolean) obj;
94    }
95  
96    @Override
97    public Class<IBooleanItem> getItemClass() {
98      return IBooleanItem.class;
99    }
100 
101   @Override
102   public IBooleanItem newItem(Object value) {
103     Boolean item = toValue(value);
104     return IBooleanItem.valueOf(item);
105   }
106 
107   @Override
108   protected IBooleanItem castInternal(@NonNull IAnyAtomicItem item) {
109     IBooleanItem retval;
110     if (item instanceof INumericItem) {
111       retval = castToBoolean((INumericItem) item);
112     } else if (item instanceof IStringItem) {
113       retval = castToBoolean((IStringItem) item);
114     } else {
115       retval = castToBoolean(item.newStringItem());
116     }
117     return retval;
118   }
119 
120   /**
121    * Cast the provided numeric value to a boolean. Any non-zero value will be {@code true}, or
122    * {@code false} otherwise.
123    *
124    * @param item
125    *          the item to cast
126    * @return {@code true} if the item value is non-zero, or {@code false} otherwise
127    */
128   @NonNull
129   protected IBooleanItem castToBoolean(@NonNull INumericItem item) {
130     return IBooleanItem.valueOf(item.toEffectiveBoolean());
131   }
132 
133   /**
134    * If the string is a numeric value, treat it as so. Otherwise parse the value as a boolean string.
135    *
136    * @param item
137    *          the item to cast
138    * @return the effective boolean value of the string
139    * @throws InvalidValueForCastFunctionException
140    *           if the provided item cannot be cast to a boolean value by any means
141    */
142   @NonNull
143   protected IBooleanItem castToBoolean(@NonNull IStringItem item) {
144     IBooleanItem retval;
145     try {
146       INumericItem numeric = INumericItem.cast(item);
147       retval = castToBoolean(numeric);
148     } catch (InvalidValueForCastFunctionException ex) {
149       retval = super.castInternal(item);
150     }
151     return retval;
152   }
153 
154 }
