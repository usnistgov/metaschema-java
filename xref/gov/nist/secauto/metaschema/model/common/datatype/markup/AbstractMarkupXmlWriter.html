73  public abstract class AbstractMarkupXmlWriter<WRITER> {
74    private static final Pattern ENTITY_PATTERN = Pattern.compile("^&([^;]+);$");
75    private static final Map<String, String> ENTITY_MAP;
76  
77    static {
78      ENTITY_MAP = new HashMap<>();
79      ENTITY_MAP.put("&amp;", "&");
80      ENTITY_MAP.put("&lsquo;", "‘");
81      ENTITY_MAP.put("&rsquo;", "’");
82      ENTITY_MAP.put("&hellip;", "…");
83      ENTITY_MAP.put("&mdash;", "—");
84      ENTITY_MAP.put("&ndash;", "–");
85      ENTITY_MAP.put("&ldquo;", "“");
86      ENTITY_MAP.put("&rdquo;", "”");
87      ENTITY_MAP.put("&laquo;", "«");
88      ENTITY_MAP.put("&raquo;", "»");
89    }

91    @NotNull
92    private final String namespace;
93    private final boolean handleBlockElements;

95    public AbstractMarkupXmlWriter(@NotNull String namespace, boolean handleBlockElements) {
96      this.namespace = namespace;
97      this.handleBlockElements = handleBlockElements;

100   @NotNull
101   protected String getNamespace() {
102     return namespace;

105   protected boolean isHandleBlockElements() {
106     return handleBlockElements;
107   }

109   public void visitChildren(Node parentNode, WRITER writer) throws XMLStreamException {
110     for (Node node : parentNode.getChildren()) {
111       visit(node, writer);
112     }
113   }
114 
115   protected void visit(Node node, WRITER writer) throws XMLStreamException {
116     boolean handled = processInlineElements(node, writer);
117     if (!handled && node instanceof Block) {
118       if (isHandleBlockElements()) {
119         handled = processBlockElements(node, writer);
120       } else {
121         visitChildren(node, writer);
122         handled = true;
123       }
124     }
125 
126     if (!handled) {
127       throw new UnsupportedOperationException(
128           String.format("Node '%s' not handled. AST: %s", node.getNodeName(), node.toAstString(true)));
129     }
130   }

132   protected abstract void writeText(WRITER writer, String text) throws XMLStreamException;
133 
134   protected abstract void writeHtmlEntity(WRITER writer, String entityText) throws XMLStreamException;

136   protected boolean processInlineElements(Node node, WRITER writer) throws XMLStreamException {
137     boolean retval = false;
138     if (node instanceof Text) {
139       writeText(writer, node.getChars().toString());
140       retval = true;
141     } else if (node instanceof EscapedCharacter) {
142       handleEscapedCharacter((EscapedCharacter) node, writer);
143       retval = true;
144     } else if (node instanceof HtmlEntity) {
145       handleHtmlEntity(((HtmlEntity) node).getChars().toString(), writer);
146       retval = true;
147     } else if (node instanceof TypographicSmarts) {
148       handleTypographicSmarts((TypographicSmarts) node, writer);
149       retval = true;
150     } else if (node instanceof TypographicQuotes) {
151       if (node instanceof DoubleQuoteNode) {
152         handleBasicElement(node, writer, "q");
153       } else {
154         handleTypographicSmarts((TypographicQuotes) node, writer);
155       }
156       retval = true;
157     } else if (node instanceof Code) {
158       handleBasicElement(node, writer, "code");
159       retval = true;
160     } else if (node instanceof StrongEmphasis) {
161       handleBasicElement(node, writer, "strong");
162       retval = true;
163     } else if (node instanceof Emphasis) {
164       handleBasicElement(node, writer, "em");
165       retval = true;
166     } else if (node instanceof ListItem) {
167       handleBasicElement(node, writer, "li");
168       retval = true;
169     } else if (node instanceof Link || node instanceof AutoLink) {
170       handleLink((LinkNode) node, writer);
171       retval = true;
172     } else if (node instanceof TextBase) {
173       // ignore these, but process their children
174       visitChildren(node, writer);
175       retval = true;
176     } else if (node instanceof Subscript) {
177       handleBasicElement(node, writer, "sub");
178       retval = true;
179     } else if (node instanceof Superscript) {
180       handleBasicElement(node, writer, "sup");
181       retval = true;
182     } else if (node instanceof Image) {
183       handleImage((Image) node, writer);
184       retval = true;
185     } else if (node instanceof InsertAnchorNode) {
186       handleInsertAnchor((InsertAnchorNode) node, writer);
187       retval = true;
188     } else if (node instanceof SoftLineBreak) {
189       handleSoftLineBreak((SoftLineBreak) node, writer);
190       retval = true;
191     } else if (node instanceof HtmlInline) {
192       handleHtmlInline((HtmlInline) node, writer);
193       retval = true;
194     }
195     return retval;
196   }
197 
198   protected boolean processBlockElements(Node node, WRITER writer) throws XMLStreamException {
199     boolean retval = false;
200     if (node instanceof Paragraph) {
201       handleBasicElement(node, writer, "p");
202       retval = true;
203     } else if (node instanceof Heading) {
204       handleHeading((Heading) node, writer);
205       retval = true;
206     } else if (node instanceof OrderedList) {
207       handleBasicElement(node, writer, "ol");
208       retval = true;
209     } else if (node instanceof BulletList) {
210       handleBasicElement(node, writer, "ul");
211       retval = true;
212     } else if (node instanceof TableBlock) {
213       handleTable((TableBlock) node, writer);
214       retval = true;
215     } else if (node instanceof HtmlBlock) {
216       handleHtmlBlock((HtmlBlock) node, writer);
217       retval = true;
218     }
219     return retval;
220   }

222   protected void handleImage(Image node, WRITER writer) throws XMLStreamException {
223     QName name = new QName(getNamespace(), "img");
224     handleImage(node, writer, name, node.getUrl().toString(), node.getText().toString());
225   }

227   protected abstract void handleImage(Image node, WRITER writer, QName name, String href, String alt)
228       throws XMLStreamException;
229 
230   protected void handleTable(TableBlock node, WRITER writer) throws XMLStreamException {
231     QName tableQName = new QName(getNamespace(), "table");
232     handleBasicElementStart(node, writer, tableQName);
233 
234     TableHead head = (TableHead) node.getChildOfType(TableHead.class);
235 
236     if (head != null) {
237       for (Node childNode : head.getChildren()) {
238         if (childNode instanceof TableRow) {
239           handleTableRow((TableRow) childNode, writer);
240         }
241       }
242     }
243 
244     TableBody body = (TableBody) node.getChildOfType(TableBody.class);
245 
246     if (body != null) {
247       for (Node childNode : body.getChildren()) {
248         if (childNode instanceof TableRow) {
249           handleTableRow((TableRow) childNode, writer);
250         }
251       }
252     }
253 
254     handleBasicElementEnd(node, writer, tableQName);
255   }
256 
257   private void handleTableRow(TableRow node, WRITER writer) throws XMLStreamException {
258     QName trQName = new QName(getNamespace(), "tr");
259     handleBasicElementStart(node, writer, trQName);
260 
261     for (Node childNode : node.getChildren()) {
262       if (childNode instanceof TableCell) {
263         handleTableCell((TableCell) childNode, writer);
264       }
265     }
266 
267     handleBasicElementEnd(node, writer, trQName);
268   }
269 
270   private void handleTableCell(TableCell node, WRITER writer) throws XMLStreamException {
271     QName cellQName;
272     if (node.isHeader()) {
273       cellQName = new QName(getNamespace(), "th");
274     } else {
275       cellQName = new QName(getNamespace(), "td");
276     }
277     handleBasicElementStart(node, writer, cellQName);
278     visitChildren(node, writer);
279     handleBasicElementEnd(node, writer, cellQName);
280   }
281 
282   protected void handleInsertAnchor(InsertAnchorNode node, WRITER writer) throws XMLStreamException {
283     QName name = new QName(getNamespace(), "insert");
284     handleInsertAnchor(node, writer, name);
285   }

287   protected abstract void handleInsertAnchor(InsertAnchorNode node, WRITER writer, QName name)
288       throws XMLStreamException;
289 
290   protected void handleHeading(Heading node, WRITER writer) throws XMLStreamException {
291     int level = node.getLevel();
292 
293     handleBasicElement(node, writer, String.format("h%d", level));
294   }
295 
296   protected void handleEscapedCharacter(EscapedCharacter node, WRITER writer) throws XMLStreamException {
297     writeText(writer, node.getChars().unescape());
298   }
299 
300   protected void handleTypographicSmarts(TypographicQuotes node, WRITER writer) throws XMLStreamException {
301     if (node.getTypographicOpening() != null && !node.getTypographicOpening().isEmpty()) {
302       handleHtmlEntity(node.getTypographicOpening(), writer);
303     }
304     visitChildren(node, writer);
305     if (node.getTypographicClosing() != null && !node.getTypographicClosing().isEmpty()) {
306       handleHtmlEntity(node.getTypographicClosing(), writer);
307     }
308   }
309 
310   protected void handleTypographicSmarts(TypographicSmarts node, WRITER writer) throws XMLStreamException {
311     handleHtmlEntity(node.getTypographicText(), writer);
312   }
313 
314   protected void handleHtmlEntity(String entityText, WRITER writer) throws XMLStreamException {
315     String replacement = ENTITY_MAP.get(entityText);
316     if (replacement != null) {
317       writeText(writer, replacement);
318     } else {
319       Matcher matcher = ENTITY_PATTERN.matcher(entityText);
320       if (matcher.matches()) {
321         writeHtmlEntity(writer, matcher.group(1));
322       } else {
323         writeText(writer, replacement);
324       }
325     }
326   }
327 
328   protected void handleBasicElement(Node node, WRITER writer, String localName) throws XMLStreamException {
329     QName name = new QName(getNamespace(), localName);
330     handleBasicElementStart(node, writer, name);
331     visitChildren(node, writer);
332     handleBasicElementEnd(node, writer, name);
333   }

339   protected void handleLink(LinkNode node, WRITER writer) throws XMLStreamException {
340     QName name = new QName(getNamespace(), "a");
341     handleLinkStart(node, writer, name, node.getUrl().toString());
342 
343     visitChildren(node, writer);
344 
345     handleLinkEnd(node, writer, name);
346   }
