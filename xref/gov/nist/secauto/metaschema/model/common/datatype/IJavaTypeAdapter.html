50  public interface IJavaTypeAdapter<TYPE> {
51  
52    /**
53     * Get the metaschema type name associated with this adapter. This name must be unique with respect
54     * to all other metaschema types.
55     * 
56     * @return the name
57     */
58    @NotNull
59    String getName();
60  
61    /**
62     * Get the Java class supported by this adapter.
63     * 
64     * @return the Java class
65     */
66    @NotNull
67    Class<TYPE> getJavaClass();
68  
69    /**
70     * Casts the provided value to the type associated with this adapter.
71     * 
72     * @param value
73     *          a value of the provided type
74     * @return the typed value
75     */
76    @NotNull
77    TYPE toValue(@NotNull Object value);
78  
79    /**
80     * Gets the value as a string suitable for writing as text. This is intended for data types that
81     * have a simple string-based structure in XML and JSON, such as for XML attributes or JSON keys. An
82     * adapter for a complex data structures that consist of XML elements will throw an
83     * {@link UnsupportedOperationException} when this is called.
84     * 
85     * @param value
86     *          the data to formatted as a string
87     * @return a string
88     * @throws UnsupportedOperationException
89     *           if the data type cannot be represented as a string
90     */
91    @NotNull
92    String asString(@NotNull Object value) throws UnsupportedOperationException;
93  
94    /**
95     * Create a copy of the provided value.
96     * 
97     * @param obj
98     *          the value to copy
99     * @return the copy
100    */
101   @NotNull
102   TYPE copy(@NotNull Object obj);
103 
104   /**
105    * Determines if the data type is an atomic, scalar value. Complex structures such as Markup are not
106    * considered atomic.
107    * 
108    * @return {@code true} if the data type is an atomic scalar value, or {@code false} otherwise
109    */
110   default boolean isAtomic() {
111     return true;
112   }
113 
114   /**
115    * Get the java type of the associated item.
116    * 
117    * @return the java associated item type
118    */
119   @NotNull
120   Class<? extends IAnyAtomicItem> getItemClass();
121 
122   /**
123    * Construct a new item of this type using the provided value.
124    * 
125    * @param value
126    *          the item's value
127    * @return a new item
128    */
129   // TODO: markup types are not atomic values. Figure out a better base type (i.e., IValuedItem)
130   @NotNull
131   IAnyAtomicItem newItem(@NotNull Object value);
132 
133   /**
134    * Cast the provided item to an item of this type, if possible.
135    * 
136    * @param item
137    *          the atomic item to cast
138    * @return an atomic item of this type
139    * @throws InvalidValueForCastFunctionMetapathException
140    *           if the provided item type cannot be cast to this item type
141    */
142   @NotNull
143   IAnyAtomicItem cast(IAnyAtomicItem item) throws InvalidValueForCastFunctionMetapathException;
144 
145   /**
146    * Indicates if the adapter will parse the {@link XMLEvent#START_ELEMENT} before parsing the value
147    * data.
148    * 
149    * @return {@code true} if the adapter requires the start element for parsing, or {@code false}
150    *         otherwise
151    */
152   // TODO; implement or remove this
153   boolean isParsingStartElement();
154 
155   /**
156    * Determines if adapter can parse the next element. The next element's {@link QName} is provided
157    * for this purpose.
158    * <p>
159    * This will be called when the parser encounter's an element it does not recognize. This gives the
160    * adapter a chance to request parsing of the data.
161    * 
162    * @param nextElementQName
163    *          the next element's namespace-qualified name
164    * @return {@code true} if the adapter will parse the element, or {@code false} otherwise
165    */
166   // TODO: implement this
167   boolean canHandleQName(@NotNull QName nextElementQName);
168 
169   /**
170    * Parses a provided string. Used to parse XML attributes, simple XML character data, and JSON/YAML
171    * property values.
172    * 
173    * @param value
174    *          the string value to parse
175    * @return the parsed data as the adapter's type
176    * @throws IllegalArgumentException
177    *           if the data is not valid to the data type
178    */
179   @NotNull
180   TYPE parse(@NotNull String value) throws IllegalArgumentException;
181 
182   /**
183    * This method is expected to parse content starting at the next event. Parsing will continue until
184    * the next event represents content that is not handled by this adapter. This means the event
185    * stream should be positioned after any {@link XMLEvent#END_ELEMENT} that corresponds to an
186    * {@link XMLEvent#START_ELEMENT} parsed by this adapter.
187    * <p>
188    * If {@link #isParsingStartElement()} returns {@code true}, then the first event to parse will be
189    * the {@link XMLEvent#START_ELEMENT} for the element that contains the value data, then the value
190    * data. If this is the case, this method must also parse the corresponding
191    * {@link XMLEvent#END_ELEMENT}. Otherwise, the first event to parse will be the value data.
192    * <p>
193    * The value data is expected to be parsed completely, leaving the event stream on a peeked event
194    * corresponding to content that is not handled by this method.
195    * 
196    * @param eventReader
197    *          the XML parser used to read the parsed value
198    * @return the parsed value
199    * @throws IOException
200    *           if a parsing error occurs
201    */
202   @NotNull
203   TYPE parse(@NotNull XMLEventReader2 eventReader) throws IOException;
204 
205   /**
206    * Parses a JSON property value.
207    * 
208    * @param parser
209    *          the JSON parser used to read the parsed value
210    * @return the parsed value
211    * @throws IOException
212    *           if a parsing error occurs
213    */
214   @NotNull
215   TYPE parse(@NotNull JsonParser parser) throws IOException;
216 
217   /**
218    * Parses a provided string using {@link #parse(String)}.
219    * <p>
220    * This method may pre-parse the data and then return copies, since the data can only be parsed
221    * once, but the supplier might be called multiple times.
222    * 
223    * @param value
224    *          the string value to parse
225    * @return a supplier that will provide new instances of the parsed data
226    * @throws IOException
227    *           if an error occurs while parsing
228    * @see #parse(String)
229    */
230   @NotNull
231   default Supplier<TYPE> parseAndSupply(@NotNull String value) throws IOException {
232     TYPE retval = parse(value);
233     return () -> copy(retval);
234   }
235 
236   /**
237    * Parses a provided string using {@link IJavaTypeAdapter#parse(XMLEventReader2)}.
238    * <p>
239    * This method may pre-parse the data and then return copies, since the data can only be parsed
240    * once, but the supplier might be called multiple times.
241    * 
242    * @param eventReader
243    *          the XML parser used to read the parsed value
244    * @return a supplier that will provide new instances of the parsed data
245    * @throws IOException
246    *           if an error occurs while parsing
247    * @see #parse(String)
248    * @see #parse(XMLEventReader2)
249    */
250   @NotNull
251   default Supplier<TYPE> parseAndSupply(@NotNull XMLEventReader2 eventReader) throws IOException {
252     TYPE retval = parse(eventReader);
253     return () -> copy(retval);
254   }
255 
256   /**
257    * Parses a provided string using {@link #parse(JsonParser)}.
258    * <p>
259    * This method may pre-parse the data and then return copies, since the data can only be parsed
260    * once, but the supplier might be called multiple times.
261    * 
262    * @param parser
263    *          the JSON parser used to read the parsed value
264    * @return a supplier that will provide new instances of the parsed data
265    * @throws IOException
266    *           if an error occurs while parsing
267    * @see #parse(String)
268    * @see #parse(JsonParser)
269    */
270   @NotNull
271   default Supplier<TYPE> parseAndSupply(@NotNull JsonParser parser) throws IOException {
272     TYPE retval = parse(parser);
273     return () -> copy(retval);
274   }
275 
276   /**
277    * Writes the provided Java class instance data as XML. The parent element information is provided
278    * as a {@link StartElement} event, which allows namespace information to be obtained from the
279    * parent element using the {@link StartElement#getName()} and
280    * {@link StartElement#getNamespaceContext()} methods, which can be used when writing the provided
281    * instance value.
282    * 
283    * @param instance
284    *          the {@link Field} instance value to write
285    * @param parent
286    *          the {@link StartElement} XML event that is the parent of the data to write
287    * @param eventFactory
288    *          the XML event factory used to generate XML writing events
289    * @param eventWriter
290    *          the XML writer used to output XML as events
291    * @throws XMLStreamException
292    *           if an unexpected error occurred while processing the XML output
293    * @throws IOException
294    *           if an unexpected error occurred while writing to the output stream
295    */
296   void writeXml(@NotNull Object instance, @NotNull StartElement parent, @NotNull XMLEventFactory2 eventFactory,
297       @NotNull XMLEventWriter eventWriter)
298       throws IOException, XMLStreamException;
299 
300   /**
301    * Writes the provided Java class instance data as an XML character sequence. The parent element
302    * information is provided as an XML {@link QName}, which allows namespace information to be
303    * obtained from the parent element. Additional namespace information can be gathered using the
304    * {@link XMLStreamWriter2#getNamespaceContext()} method, which can be used when writing the
305    * provided instance value.
306    * 
307    * @param instance
308    *          the {@link Field} instance value to write
309    * @param parentName
310    *          the qualified name of the XML data's parent element
311    * @param writer
312    *          the XML writer used to output the XML data
313    * @throws XMLStreamException
314    *           if an unexpected error occurred while processing the XML output
315    */
316   void writeXmlCharacters(@NotNull Object instance, @NotNull QName parentName, @NotNull XMLStreamWriter2 writer)
317       throws XMLStreamException;
318 
319   /**
320    * Writes the provided Java class instance as a JSON/YAML field value.
321    * 
322    * @param instance
323    *          the {@link Field} instance value to write
324    * @param writer
325    *          the JSON/YAML writer used to output the JSON/YAML data
326    * @throws IOException
327    *           if an unexpected error occurred while writing the JSON/YAML output
328    */
329   void writeJsonValue(@NotNull Object instance, @NotNull JsonGenerator writer) throws IOException;
330 
331   /**
332    * Gets the default value to use as the JSON/YAML field name for a Metaschema field value if no JSON
333    * value key flag or name is configured.
334    * 
335    * @return the default field name to use
336    */
337   @NotNull
338   String getDefaultJsonValueKey();
339 
340   /**
341    * Determines if the data type's value is allowed to be unwrapped in XML when the value is a field
342    * value.
343    * 
344    * @return {@code true} if allowed, or {@code false} otherwise.
345    */
346   boolean isUnrappedValueAllowedInXml();
347 }
