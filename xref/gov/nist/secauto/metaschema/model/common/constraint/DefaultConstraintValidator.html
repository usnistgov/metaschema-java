106    * Validate the provided flag item against any associated constraints.
107    * 
108    * @param item
109    *          the flag item to validate
110    * @throws MetapathException
111    *           if an error occurred while evaluating a Metapath used in a constraint
112    */
113   protected void validateFlag(@NonNull IFlagNodeItem item) {
114     IFlagDefinition definition = item.getDefinition();
115 
116     validateExpect(definition.getExpectConstraints(), item);
117     validateAllowedValues(definition.getAllowedValuesConstraints(), item);
118     validateIndexHasKey(definition.getIndexHasKeyConstraints(), item);
119     validateMatches(definition.getMatchesConstraints(), item);
120   }

159   protected void validateHasCardinality(@NonNull List<? extends ICardinalityConstraint> constraints,
160       @NonNull List<? extends IAssemblyNodeItem> items) {
161 
162     items.stream().forEachOrdered(item -> {
163       assert item != null;
164       validateHasCardinality(constraints, item);
165     });
166   }
167 
168   protected void validateHasCardinality(@NonNull List<? extends ICardinalityConstraint> constraints,
169       @NonNull IAssemblyNodeItem item) {
170     for (ICardinalityConstraint constraint : constraints) {
171       ISequence<? extends IDefinitionNodeItem> targets = constraint.matchTargets(item, getMetapathContext());
172       try {
173         validateHasCardinality(constraint, item, targets);
174       } catch (MetapathException ex) {
175         rethrowConstraintError(constraint, item, ex);
176       }
177     }
178   }
179 
180   protected void validateHasCardinality(@NonNull ICardinalityConstraint constraint, @NonNull IAssemblyNodeItem node,
181       ISequence<? extends INodeItem> targets) {
182     int itemCount = targets.size();
183 
184     Integer minOccurs = constraint.getMinOccurs();
185     if (minOccurs != null && itemCount < minOccurs) {
186       getConstraintValidationHandler().handleCardinalityMinimumViolation(constraint, node, targets);
187     }
188 
189     Integer maxOccurs = constraint.getMaxOccurs();
190     if (maxOccurs != null && itemCount > maxOccurs) {
191       getConstraintValidationHandler().handleCardinalityMaximumViolation(constraint, node, targets);
192     }
193   }

215   protected void validateIndex(@NonNull IIndexConstraint constraint, @NonNull IAssemblyNodeItem node,
216       @NonNull ISequence<? extends INodeItem> targets) {
217     String indexName = constraint.getName();
218     if (indexNameToIndexMap.containsKey(indexName)) {
219       getConstraintValidationHandler().handleIndexDuplicateViolation(constraint, node);
220       return; // NOPMD - readability
221     }
222 
223     IIndex index = IIndex.newInstance(constraint.getKeyFields());
224     targets.asStream()
225         .forEachOrdered(item -> {
226           assert item != null;
227 
228           try {
229             INodeItem oldItem = index.put(item, metapathContext);
230             if (oldItem != null) {
231               getConstraintValidationHandler().handleIndexDuplicateKeyViolation(constraint, node, oldItem, item);
232             }
233           } catch (MetapathException ex) {
234             getConstraintValidationHandler().handleKeyMatchError(constraint, node, item, ex);
235           }
236         });
237     indexNameToIndexMap.put(indexName, index);
238   }

261   protected void validateUnique(@NonNull IUniqueConstraint constraint,
262       @NonNull IAssemblyNodeItem node, @NonNull ISequence<? extends INodeItem> targets) {
263     IIndex index = IIndex.newInstance(constraint.getKeyFields());
264     targets.asStream()
265         .forEachOrdered(item -> {
266           assert item != null;
267           try {
268             INodeItem oldItem = index.put(item, metapathContext);
269             if (oldItem != null) {
270               getConstraintValidationHandler().handleUniqueKeyViolation(constraint, node, oldItem, item);
271             }
272           } catch (MetapathException ex) {
273             getConstraintValidationHandler().handleKeyMatchError(constraint, node, item, ex);
274             throw ex;
275           }
276         });
277   }

279   protected void validateMatches(@NonNull List<? extends IMatchesConstraint> constraints,
280       @NonNull IDefinitionNodeItem item) {
281 
282     for (IMatchesConstraint constraint : constraints) {
283       ISequence<? extends IDefinitionNodeItem> targets = constraint.matchTargets(item, getMetapathContext());
284       try {
285         validateMatches(constraint, item, targets);
286       } catch (MetapathException ex) {
287         rethrowConstraintError(constraint, item, ex);
288       }
289     }

292   protected void validateMatches(@NonNull IMatchesConstraint constraint, @NonNull INodeItem node,
293       ISequence<? extends INodeItem> targets) {
294     targets.asStream()
295         .forEachOrdered(item -> {
296           assert item != null;
297           String value = FnData.fnDataItem(item).asString();
298 
299           Pattern pattern = constraint.getPattern();
300           if (pattern != null) {
301             // validate pattern
302             Predicate<String> predicate = pattern.asMatchPredicate();
303             if (!predicate.test(value)) {
304               getConstraintValidationHandler().handleMatchPatternViolation(constraint, node, item, value);
305             }
306           }
307 
308           IDataTypeAdapter<?> adapter = constraint.getDataType();
309           if (adapter != null) {
310             try {
311               adapter.parse(value);
312             } catch (IllegalArgumentException ex) {
313               getConstraintValidationHandler().handleMatchDatatypeViolation(constraint, node, item, value, ex);
314             }
315           }
316         });
317   }

329   protected void validateIndexHasKey(
330       @NonNull IIndexHasKeyConstraint constraint,
331       @NonNull IDefinitionNodeItem node,
332       @NonNull ISequence<? extends INodeItem> targets) {
333     String indexName = constraint.getIndexName();
334 
335     List<KeyRef> keyRefItems = indexNameToKeyRefMap.get(indexName);
336     if (keyRefItems == null) {
337       keyRefItems = new LinkedList<>();
338       indexNameToKeyRefMap.put(indexName, keyRefItems);
339     }
340 
341     KeyRef keyRef = new KeyRef(constraint, node, new ArrayList<>(targets.asList()));
342     keyRefItems.add(keyRef);
343   }

345   protected void validateExpect(@NonNull List<? extends IExpectConstraint> constraints,
346       @NonNull IDefinitionNodeItem item) {
347     for (IExpectConstraint constraint : constraints) {
348       ISequence<? extends IDefinitionNodeItem> targets = constraint.matchTargets(item, getMetapathContext());
349       validateExpect(constraint, item, targets);

353   protected void validateExpect(@NonNull IExpectConstraint constraint, @NonNull INodeItem node,
354       @NonNull ISequence<? extends INodeItem> targets) {
355     targets.asStream()
356         .map(item -> (INodeItem) item)
357         .forEachOrdered(item -> {
358           assert item != null;
359           MetapathExpression metapath = constraint.getTest();
360           try {
361             ISequence<?> result = metapath.evaluate(item, getMetapathContext());
362             if (!FnBoolean.fnBoolean(result).toBoolean()) {
363               getConstraintValidationHandler().handleExpectViolation(constraint, node, item, getMetapathContext());
364             }
365           } catch (MetapathException ex) {
366             rethrowConstraintError(constraint, item, ex);
367           }
368         });
369   }

371   protected void validateAllowedValues(@NonNull List<? extends IAllowedValuesConstraint> constraints,
372       @NonNull IDefinitionNodeItem item) {
373     for (IAllowedValuesConstraint constraint : constraints) {
374       ISequence<? extends IDefinitionNodeItem> targets = constraint.matchTargets(item, getMetapathContext());
375       validateAllowedValues(constraint, targets);
376     }

379   protected void validateAllowedValues(@NonNull IAllowedValuesConstraint constraint,
380       ISequence<? extends IDefinitionNodeItem> targets) {
381     targets.asStream().forEachOrdered(item -> {
382       assert item != null;
383       try {
384         updateValueStatus(item, constraint);
385       } catch (MetapathException ex) {
386         rethrowConstraintError(constraint, item, ex);
387       }
388     });
389   }

391   private static void rethrowConstraintError(@NonNull IConstraint constraint, INodeItem item,
392       MetapathException ex) {
393     StringBuilder builder = new StringBuilder(94);
394     builder.append("A ")
395         .append(constraint.getClass().getName())
396         .append(" constraint");
397 
398     String id = constraint.getId();
399     if (id == null) {
400       builder.append(" targeting the metapath '")
401           .append(constraint.getTarget().getPath())
402           .append('\'');
403     } else {
404       builder.append(" with id '")
405           .append(id)
406           .append('\'');
407     }
408 
409     builder.append(", matching the item at path '")
410         .append(item.getMetapath())
411         .append("', resulted in an unexpected error. The error was: ")
412         .append(ex.getLocalizedMessage());
413 
414     throw new MetapathException(builder.toString(), ex);
415   }

417   /**
418    * Add a new allowed value to the value status tracker.
419    * 
420    * @param targetItem
421    *          the item whose value is targeted by the constraint
422    * @param allowedValues
423    *          the set of allowed values
424    */
425   protected void updateValueStatus(@NonNull INodeItem targetItem, @NonNull IAllowedValuesConstraint allowedValues) {
426     // constraint.getAllowedValues().containsKey(value)
427 
428     @Nullable
429     ValueStatus valueStatus = valueMap.get(targetItem);
430     if (valueStatus == null) {
431       valueStatus = new ValueStatus(targetItem);
432       valueMap.put(targetItem, valueStatus);
433     }
434 
435     valueStatus.registerAllowedValue(allowedValues);
436   }
437 
438   protected void handleAllowedValues(@NonNull INodeItem targetItem) {
439     ValueStatus valueStatus = valueMap.remove(targetItem);
440     if (valueStatus != null) {
441       valueStatus.validate();
442     }
443   }

445   @Override
446   public void finalizeValidation() {
447     // key references
448     for (Map.Entry<String, List<KeyRef>> entry : indexNameToKeyRefMap.entrySet()) {
449       String indexName = entry.getKey();
450       IIndex index = indexNameToIndexMap.get(indexName);
451 
452       List<KeyRef> keyRefs = entry.getValue();
453 
454       for (KeyRef keyRef : keyRefs) {
455         IIndexHasKeyConstraint constraint = keyRef.getConstraint();
456         for (INodeItem item : keyRef.getTargets()) {
457           assert item != null;
458 
459           try {
460             INodeItem referencedItem = index.get(item, constraint.getKeyFields(), getMetapathContext());
461 
462             if (referencedItem == null) {
463               getConstraintValidationHandler().handleIndexMiss(constraint, keyRef.getNode(), item);
464             }
465           } catch (MetapathException ex) {
466             getConstraintValidationHandler().handleKeyMatchError(constraint, keyRef.getNode(), item, ex);
467           }
468         }
469       }
470     }

473   private class ValueStatus {
474     @NonNull
475     private final List<IAllowedValuesConstraint> constraints = new LinkedList<>();
476     @NonNull
477     private final String value;
478     @NonNull
479     private final INodeItem item;
480     private boolean allowOthers = true;
481     @NonNull
482     private IAllowedValuesConstraint.Extensible extensible = IAllowedValuesConstraint.Extensible.EXTERNAL;
483 
484     public ValueStatus(@NonNull INodeItem item) {
485       this.item = item;
486       this.value = FnData.fnDataItem(item).asString();
487     }
488 
489     public void registerAllowedValue(@NonNull IAllowedValuesConstraint allowedValues) {
490       this.constraints.add(allowedValues);
491       if (!allowedValues.isAllowedOther()) {
492         // record the most restrictive value
493         allowOthers = false;
494       }
495 
496       IAllowedValuesConstraint.Extensible newExtensible = allowedValues.getExtensible();
497       if (newExtensible.ordinal() > extensible.ordinal()) {
498         // record the most restrictive value
499         extensible = allowedValues.getExtensible();
500       } else if (IAllowedValuesConstraint.Extensible.NONE.equals(newExtensible)
501           && IAllowedValuesConstraint.Extensible.NONE.equals(extensible)) {
502         // this is an error, where there are two none constraints that conflict
503         throw new MetapathException(
504             String.format("Multiple constraints have extensibility scope=none at path '%s'", item.getMetapath()));
505       } else if (allowedValues.getExtensible().ordinal() < extensible.ordinal()) {
506         String msg = String.format(
507             "An allowed values constraint with an extensibility scope '%s'"
508                 + " exceeds the allowed scope '%s' at path '%s'",
509             allowedValues.getExtensible().name(), extensible.name(), item.getMetapath());
510         LOGGER.atError().log(msg);
511         throw new MetapathException(msg);
512       }
513     }
514 
515     public void validate() {
516       if (!constraints.isEmpty()) {
517         boolean match = false;
518         List<IAllowedValuesConstraint> failedConstraints = new LinkedList<>();
519         for (IAllowedValuesConstraint allowedValues : constraints) {
520           IAllowedValue matchingValue = allowedValues.getAllowedValue(value);
521           if (matchingValue != null) {
522             match = true;
523           } else if (IAllowedValuesConstraint.Extensible.NONE.equals(allowedValues.getExtensible())) {
524             // hard failure, since no other values can satisfy this constraint
525             failedConstraints = CollectionUtil.singletonList(allowedValues);
526             match = false;
527             break;
528           } else {
529             failedConstraints.add(allowedValues);
530           } // this constraint passes, but we need to make sure other constraints do as well
531         }
532 
533         // it's not a failure if allow others is true
534         if (!match && !allowOthers) {
535           getConstraintValidationHandler().handleAllowedValuesViolation(failedConstraints, item);
536         }
537       }
538     }

585     private final IIndexHasKeyConstraint constraint;
586     @NonNull
587     private final INodeItem node;
588     @NonNull
589     private final List<INodeItem> targets;

591     public KeyRef(
592         @NonNull IIndexHasKeyConstraint constraint,
593         @NonNull INodeItem node,
594         @NonNull List<INodeItem> targets) {
595       this.node = node;
596       this.constraint = constraint;
597       this.targets = targets;
598     }

602       return constraint;
603     }
604 
605     @NonNull
606     protected INodeItem getNode() {

608     }
609

611     public List<INodeItem> getTargets() {
