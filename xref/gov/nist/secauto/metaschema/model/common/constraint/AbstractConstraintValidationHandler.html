27  package gov.nist.secauto.metaschema.model.common.constraint;
28  
29  import gov.nist.secauto.metaschema.model.common.datatype.IJavaTypeAdapter;
30  import gov.nist.secauto.metaschema.model.common.metapath.DynamicContext;
31  import gov.nist.secauto.metaschema.model.common.metapath.evaluate.ISequence;
32  import gov.nist.secauto.metaschema.model.common.metapath.format.IPathFormatter;
33  import gov.nist.secauto.metaschema.model.common.metapath.item.INodeItem;
34  
35  import org.jetbrains.annotations.NotNull;
36  
37  public abstract class AbstractConstraintValidationHandler implements IConstraintValidationHandler {
38    @NotNull
39    public abstract IPathFormatter getPathFormatter();
40  
41    protected String toPath(@NotNull INodeItem nodeItem) {
42      return nodeItem.toPath(getPathFormatter());
43    }
44  
45    @SuppressWarnings("null")
46    @NotNull
47    protected String newCardinalityMinimumViolationMessage(
48        @NotNull ICardinalityConstraint constraint,
49        @SuppressWarnings("unused") @NotNull INodeItem node,
50        @NotNull ISequence<? extends INodeItem> targets) {
51      // TODO: render the item paths instead of the expression
52      return String.format(
53          "The cardinality '%d' is below the required minimum '%d' for items matching the expression '%s'.",
54          targets.size(), constraint.getMinOccurs(), constraint.getTarget().getPath());
55    }
56  
57    @SuppressWarnings("null")
58    @NotNull
59    protected String newCardinalityMaximumViolationMessage(
60        @NotNull ICardinalityConstraint constraint,
61        @SuppressWarnings("unused") @NotNull INodeItem node,
62        @NotNull ISequence<? extends INodeItem> targets) {
63      // TODO: render the item paths instead of the expression
64      return String.format(
65          "The cardinality '%d' is greater than the required maximum '%d' for items matching the expression '%s'.",
66          targets.size(), constraint.getMinOccurs(), constraint.getTarget().getPath());
67    }
68  
69    @SuppressWarnings("null")
70    @NotNull
71    protected String newIndexDuplicateKeyViolationMessage(
72        @NotNull IIndexConstraint constraint,
73        @SuppressWarnings("unused") @NotNull INodeItem node,
74        @NotNull INodeItem oldItem,
75        @NotNull INodeItem target) {
76      // TODO: render the key paths
77      return String.format("Index '%s' has duplicate key for items at paths '%s' and '%s'", constraint.getName(),
78          toPath(oldItem), toPath(target));
79    }
80  
81    @SuppressWarnings("null")
82    @NotNull
83    protected String newUniqueKeyViolationMessage(
84        @SuppressWarnings("unused") @NotNull IUniqueConstraint constraint,
85        @SuppressWarnings("unused") @NotNull INodeItem node,
86        @NotNull INodeItem oldItem,
87        @NotNull INodeItem target) {
88      // TODO: render the key paths
89      return String.format("Unique constraint violation at paths '%s' and '%s'",
90          toPath(oldItem), toPath(target));
91    }
92  
93    @SuppressWarnings("null")
94    @NotNull
95    protected String newMatchPatternViolationMessage(
96        @NotNull IMatchesConstraint constraint,
97        @SuppressWarnings("unused") @NotNull INodeItem node,
98        @NotNull INodeItem target,
99        @NotNull String value) {
100     return String.format("Value '%s' did not match the pattern '%s' at path '%s'",
101         value,
102         constraint.getPattern().pattern(),
103         toPath(target));
104   }
105 
106   @SuppressWarnings("null")
107   @NotNull
108   protected String newMatchDatatypeViolationMessage(
109       @NotNull IMatchesConstraint constraint,
110       @SuppressWarnings("unused") @NotNull INodeItem node,
111       @NotNull INodeItem target,
112       @NotNull String value) {
113     IJavaTypeAdapter<?> adapter = constraint.getDataType();
114     return String.format("Value '%s' did not conform to the data type '%s' at path '%s'", value,
115         adapter.getName(), toPath(target));
116   }
117 
118   @SuppressWarnings("null")
119   @NotNull
120   protected CharSequence newExpectViolationMessage(
121       @NotNull IExpectConstraint constraint,
122       @SuppressWarnings("unused") @NotNull INodeItem node,
123       @NotNull INodeItem target,
124       @NotNull DynamicContext dynamicContext) {
125     CharSequence message;
126     if (constraint.getMessage() != null) {
127       message = constraint.generateMessage(target, dynamicContext);
128     } else {
129       message = String.format("Expect constraint '%s' did not match the data at path '%s'",
130           constraint.getTest().getPath(),
131           toPath(target));
132     }
133     return message;
134   }
135 }
