49    /**
50     * A callback used to evaluate the provided {@code expression} in the context of a set of context
51     * items.
52     *
53     * @param expression
54     *          the expression to evaluate
55     * @param dynamicContext
56     *          the evaluation context
57     * @param contextItems
58     *          the set of items to evaluate the expression against
59     * @return the resulting set of items produced by evaluating each context item against the
60     *         expression
61     */
62    @SuppressWarnings("null")
63    @NonNull
64    protected ISequence<?> evaluateInNodeContext(@NonNull IExpression expression, @NonNull DynamicContext dynamicContext,
65        @NonNull ISequence<? extends INodeItem> contextItems) {
66      ISequence<?> retval;
67      if (contextItems.isEmpty()) {
68        retval = ISequence.empty();
69      } else {
70        // evaluate the right path in the context of the left's children
71        Stream<? extends IItem> result = contextItems.asStream()
72            .flatMap(node -> expression.accept(dynamicContext, node).asStream());
73        retval = ISequence.of(result);
74      }
75      return retval;
76    }

78    /**
79     * Evaluate the {@code nodeContext} and its ancestors against the provided {@code expression},
80     * keeping any matching nodes.
81     *
82     * @param expression
83     *          the expression to evaluate
84     * @param dynamicContext
85     *          the evaluation context
86     * @param nodeContext
87     *          the current context node
88     * @return the matching nodes
89     */
90    @NonNull
91    protected Stream<? extends INodeItem> searchExpression(@NonNull IExpression expression,
92        @NonNull DynamicContext dynamicContext, @NonNull INodeContext nodeContext) {
93  
94      // check the current node
95      @SuppressWarnings("unchecked") Stream<? extends INodeItem> nodeMatches
96          = (Stream<? extends INodeItem>) expression.accept(dynamicContext, nodeContext).asStream();
97  
98      Stream<? extends INodeItem> childMatches;
99      if (nodeContext instanceof ICycledAssemblyNodeItem) {
100       // hack to prevent stack overflow
101       childMatches = Stream.empty();
102     } else {
103       // create a stream of flags and model elements to check
104       Stream<? extends IFlagNodeItem> flags = nodeContext.flags();
105       Stream<? extends INodeItem> modelItems = nodeContext.modelItems();
106       childMatches = Stream.concat(flags, modelItems)
107           .flatMap(instance -> {
108             assert instance != null;
109             return searchExpression(expression, dynamicContext, instance);
110           });
111     }
112 
113     @SuppressWarnings("null")
114     @NonNull Stream<? extends INodeItem> result = Stream.concat(nodeMatches, childMatches);
115     return result;
116   }
