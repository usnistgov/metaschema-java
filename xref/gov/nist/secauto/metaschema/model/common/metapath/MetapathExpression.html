58    public enum ResultType {
59      NUMBER,
60      STRING,
61      BOOLEAN,
62      SEQUENCE,
63      NODE;
64    }

66    private static final Logger LOGGER = LogManager.getLogger(MetapathExpression.class);
67  
68    @NonNull
69    public static final MetapathExpression CONTEXT_NODE = new MetapathExpression(".", ContextItem.instance());
70  
71    private final String path;
72    @NonNull
73    private final IExpression node;

75    /**
76     * Compiles a Metapath expression string.
77     *
78     * @param path
79     *          the metapath expression
80     * @return the compiled expression object
81     * @throws MetapathException
82     *           if an error occurred while compiling the Metapath expression
83     */
84    @NonNull
85    public static MetapathExpression compile(@NonNull String path) {
86      @NonNull MetapathExpression retval;
87      if (".".equals(path)) {
88        retval = CONTEXT_NODE;
89      } else {
90        try {
91          metapath10Lexer lexer = new metapath10Lexer(CharStreams.fromString(path));
92          CommonTokenStream tokens = new CommonTokenStream(lexer);
93          metapath10Parser parser = new metapath10Parser(tokens);
94          parser.removeErrorListeners();
95          parser.addErrorListener(new FailingErrorListener());
96  
97          ParseTree tree = ObjectUtils.notNull(parser.expr());
98  
99          if (LOGGER.isDebugEnabled()) {
100           try (ByteArrayOutputStream os = new ByteArrayOutputStream()) {
101             try (PrintStream ps = new PrintStream(os, true, StandardCharsets.UTF_8)) {
102               CSTPrinter printer = new CSTPrinter(ps);
103               printer.print(tree, Arrays.asList(metapath10Parser.ruleNames));
104               ps.flush();
105             }
106             LOGGER.atDebug().log(String.format("Metapath CST:%n%s", os.toString(StandardCharsets.UTF_8)));
107           } catch (IOException ex) {
108             LOGGER.atError().withThrowable(ex).log("An unexpected error occured while closing the steam.");
109           }
110         }
111 
112         IExpression expr = new BuildAstVisitor().visit(tree);
113 
114         if (LOGGER.isDebugEnabled()) {
115           LOGGER.atDebug().log(String.format("Metapath AST:%n%s", ASTPrinter.instance().visit(expr)));
116         }
117         retval = new MetapathExpression(path, expr);
118       } catch (MetapathException | ParseCancellationException ex) {
119         String msg = String.format("Unable to compile Metapath '%s'", path);
120         LOGGER.atError().withThrowable(ex).log(msg);
121         throw new MetapathException(msg, ex);
122       }
123     }
124     return retval;
125   }

127   /**
128    * Construct a new Metapath expression.
129    *
130    * @param path
131    *          the Metapath as a string
132    * @param expr
133    *          the Metapath as a compiled abstract syntax tree (AST)
134    */
135   protected MetapathExpression(@NonNull String path, @NonNull IExpression expr) {
136     this.path = path;
137     this.node = expr;
138   }

140   /**

143    * @return the expression
144    */
145   public String getPath() {
146     return path;
147   }

149   /**

154   @NonNull
155   protected IExpression getASTNode() {

164   /**
165    * Evaluate this Metapath expression using the provided {@code nodeContext} as the initial
166    * evaluation context. The specific result type will be determined by the {@code resultType}
167    * argument.
168    *
169    * @param <T>
170    *          the expected result type
171    * @param nodeContext
172    *          the initial evaluation context
173    * @param resultType
174    *          the type of result to produce
175    * @return the converted result
176    * @throws TypeMetapathException
177    *           if the provided sequence is incompatible with the requested result type
178    * @throws MetapathException
179    *           if an error occurred during evaluation
180    * @see #toResultType(ISequence, ResultType)
181    */
182   @Nullable
183   public <T> T evaluateAs(@NonNull INodeContext nodeContext, @NonNull ResultType resultType) {
184     ISequence<?> result = evaluate(nodeContext);
185     return toResultType(result, resultType);
186   }

188   /**
189    * Evaluate this Metapath expression using the provided {@code nodeContext} as the initial
190    * evaluation context. The specific result type will be determined by the {@code resultType}
191    * argument.
192    * <p>
193    * This variant allow for reuse of a provided {@code dynamicContext}.
194    *
195    * @param <T>
196    *          the expected result type
197    * @param nodeContext
198    *          the initial evaluation context
199    * @param resultType
200    *          the type of result to produce
201    * @param dynamicContext
202    *          the dynamic context to use for evaluation
203    * @return the converted result
204    * @throws TypeMetapathException
205    *           if the provided sequence is incompatible with the requested result type
206    * @throws MetapathException
207    *           if an error occurred during evaluation
208    * @see #toResultType(ISequence, ResultType)
209    */
210   @Nullable
211   public <T> T evaluateAs(@NonNull INodeContext nodeContext, @NonNull ResultType resultType,
212       @NonNull DynamicContext dynamicContext) {
213     ISequence<?> result = evaluate(nodeContext, dynamicContext);
214     return toResultType(result, resultType);
215   }

217   /**
218    * Converts the provided {@code sequence} to the requested {@code resultType}.
219    * <p>
220    * The {@code resultType} determines the returned result, which is derived from the evaluation
221    * result sequence, as follows:
222    * <ul>
223    * <li>BOOLEAN - the effective boolean result is produced using
224    * {@link FnBoolean#fnBoolean(ISequence)}.</li>
225    * <li>NODE - the first result item in the sequence is returned.</li>
226    * <li>NUMBER - the sequence is cast to a number using
227    * {@link IDecimalItem#cast(IAnyAtomicItem)}.</li>
228    * <li>SEQUENCE - the evaluation result sequence.</li>
229    * <li>STRING - the string value of the first result item in the sequence.</li>
230    * </ul>
231    *
232    * @param <T>
233    *          the requested return value
234    * @param sequence
235    *          the sequence to convert
236    * @param resultType
237    *          the type of result to produce
238    * @return the converted result
239    * @throws TypeMetapathException
240    *           if the provided sequence is incompatible with the requested result type
241    */
242   @SuppressWarnings("PMD.NullAssignment") // for readability
243   @Nullable
244   protected <T> T toResultType(@NonNull ISequence<?> sequence, @NonNull ResultType resultType) {
245     Object result;
246     switch (resultType) {
247     case BOOLEAN:
248       result = FnBoolean.fnBoolean(sequence).toBoolean();
249       break;
250     case NODE:
251       result = FunctionUtils.getFirstItem(sequence, true);
252       break;
253     case NUMBER:
254       INumericItem numeric = FunctionUtils.toNumeric(sequence, true);
255       result = numeric == null ? null : numeric.asDecimal();
256       break;
257     case SEQUENCE:
258       result = sequence;
259       break;
260     case STRING:
261       IItem item = FunctionUtils.getFirstItem(sequence, true);
262       result = item == null ? "" : FnData.fnDataItem(item).asString();
263       break;
264     default:
265       throw new InvalidTypeMetapathException(null, String.format("unsupported result type '%s'", resultType.name()));
266     }
267 
268     @SuppressWarnings("unchecked") T retval = (T) result;
269     return retval;
270   }

272   /**
273    * Evaluate this Metapath expression using the provided {@code nodeContext} as the initial
274    * evaluation context.
275    *
276    * @param <T>
277    *          the type of items contained in the resulting sequence
278    * @param nodeContext
279    *          the initial evaluation context
280    * @return a sequence of Metapath items representing the result of the evaluation
281    * @throws MetapathException
282    *           if an error occurred during evaluation
283    */
284   @SuppressWarnings("unchecked")
285   @NonNull
286   public <T extends IItem> ISequence<T> evaluate(@NonNull INodeContext nodeContext) {
287     return (ISequence<T>) evaluate(nodeContext, new StaticContext().newDynamicContext());
288   }
289 
290   /**
291    * Evaluate this Metapath expression using the provided {@code nodeContext} as the initial
292    * evaluation context.
293    * <p>
294    * This variant allow for reuse of a provided {@code dynamicContext}.
295    *
296    * @param <T>
297    *          the type of items contained in the resulting sequence
298    * @param nodeContext
299    *          the initial evaluation context
300    * @param dynamicContext
301    *          the dynamic context to use for evaluation
302    * @return a sequence of Metapath items representing the result of the evaluation
303    * @throws MetapathException
304    *           if an error occurred during evaluation
305    */
306   @SuppressWarnings("unchecked")
307   @NonNull
308   public <T extends IItem> ISequence<T> evaluate(@NonNull INodeContext nodeContext,
309       @NonNull DynamicContext dynamicContext) {
310     try {
311       return (ISequence<T>) getASTNode().accept(dynamicContext,
312           nodeContext);
313     } catch (MetapathException ex) { // NOPMD - intentional
314       throw new MetapathException(
315           String.format("An error occurred while evaluating the expression '%s'.", getPath()), ex);
316     }
317   }
