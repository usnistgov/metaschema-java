48  public final class FnBoolean {
49    @NotNull
50    static final IFunction SIGNATURE = IFunction.builder()
51        .name("boolean")
52        .deterministic()
53        .contextIndependent()
54        .focusIndependent()
55        .argument(IArgument.newBuilder()
56            .name("arg")
57            .type(IItem.class)
58            .zeroOrMore()
59            .build())
60        .returnType(IBooleanItem.class)
61        .returnOne()
62        .functionHandler(FnBoolean::execute)
63        .build();
64  
65    private FnBoolean() {
66      // disable construction
67    }
68  
69    @SuppressWarnings("unused")
70    @NotNull
71    private static ISequence<IBooleanItem> execute(@NotNull IFunction function,
72        @NotNull List<@NotNull ISequence<?>> arguments,
73        @NotNull DynamicContext dynamicContext,
74        INodeItem focus) {
75  
76      @SuppressWarnings("null")
77      ISequence<?> items = arguments.iterator().next();
78  
79      IBooleanItem result = fnBoolean(items);
80      return ISequence.of(result);
81    }
82  
83    /**
84     * Get the effective boolean value of the provided sequence.
85     * <p>
86     * Based on the XPath 3.1
87     * <a href="https://www.w3.org/TR/xpath-functions-31/#func-boolean">fn:boolean</a> function.
88     * 
89     * @param sequence
90     *          the sequence to evaluate
91     * @return the effective boolean value of the sequence
92     */
93    @NotNull
94    public static IBooleanItem fnBoolean(@Nullable ISequence<?> sequence) {
95      IBooleanItem retval;
96      if (sequence == null) {
97        retval = IBooleanItem.FALSE;
98      } else {
99        retval = IBooleanItem.valueOf(fnBooleanAsPrimitive(sequence));
100     }
101     return retval;
102   }
103 
104   /**
105    * A helper method that gets the effective boolean value of the provided sequence based on
106    * <a href="https://www.w3.org/TR/xpath-31/#id-ebv">XPath 3.1</a>.
107    * 
108    * @param sequence
109    *          the sequence to evaluate
110    * @return the effective boolean value
111    */
112   // TODO: fix misspelled Primative
113   public static boolean fnBooleanAsPrimitive(@NotNull ISequence<?> sequence) {
114     boolean retval = false;
115     if (!sequence.isEmpty()) {
116       List<? extends IItem> items = sequence.asList();
117       IItem first = ObjectUtils.notNull(items.iterator().next());
118       if (first instanceof INodeItem) {
119         retval = true;
120       } else if (items.size() == 1) {
121         retval = fnBooleanAsPrimitive(first);
122       }
123     }
124     return retval;
125   }
126 
127   /**
128    * A helper method that gets the effective boolean value of the provided item based on
129    * <a href="https://www.w3.org/TR/xpath-31/#id-ebv">XPath 3.1</a>.
130    * 
131    * @param item
132    *          the item to evaluate
133    * @return the effective boolean value
134    */
135   public static boolean fnBooleanAsPrimitive(@NotNull IItem item) {
136     boolean retval;
137     if (item instanceof IBooleanItem) {
138       retval = ((IBooleanItem) item).toBoolean();
139     } else if (item instanceof IStringItem) {
140       String string = ((IStringItem) item).asString();
141       retval = !string.isBlank();
142     } else if (item instanceof INumericItem) {
143       retval = ((INumericItem) item).toEffectiveBoolean();
144     } else if (item instanceof IUntypedAtomicItem) {
145       String string = ((IUntypedAtomicItem) item).asString();
146       retval = !string.isBlank();
147     } else if (item instanceof IAnyUriItem) {
148       String string = ((IAnyUriItem) item).asString();
149       retval = !string.isBlank();
150     } else {
151       throw new InvalidArgumentFunctionException(InvalidArgumentFunctionException.INVALID_ARGUMENT_TYPE,
152           String.format("Invalid argument type '%s'", item.getClass().getName()));
153     }
154     return retval;
155   }
