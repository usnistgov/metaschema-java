49  /**
50   * Provides a common, abstract implementation of a {@link IMetaschema}.
51   */
52  public abstract class AbstractMetaschema implements IMetaschema {
53    private static final Logger LOGGER = LogManager.getLogger(AbstractMetaschema.class);
54  
55    @NotNull
56    private final URI location;
57    @NotNull
58    private final Map<@NotNull URI, ? extends IMetaschema> importedMetaschemaByUri;
59    @NotNull
60    private final Map<@NotNull String, ? extends IMetaschema> importedMetaschemaByName;
61    private Map<@NotNull String, IFlagDefinition> exportedFlagDefinitions;
62    private Map<@NotNull String, IFieldDefinition> exportedFieldDefinitions;
63    private Map<@NotNull String, IAssemblyDefinition> exportedAssemblyDefinitions;
64  
65    /**
66     * Constructs a new {@link IMetaschema} instance.
67     * 
68     * @param metaschemaResource
69     *          the resource to load the Metaschema from
70     * @param importedMetaschema
71     *          all previously imported Metaschema that this Metaschema might import
72     */
73    @SuppressWarnings("null")
74    public AbstractMetaschema(
75        @NotNull URI metaschemaResource,
76        @NotNull Map<@NotNull URI, ? extends IMetaschema> importedMetaschema) {
77      Objects.requireNonNull(metaschemaResource, "metaschemaResource");
78      Objects.requireNonNull(importedMetaschema, "importedMetaschema");
79      this.location = metaschemaResource;
80      this.importedMetaschemaByUri = Collections.unmodifiableMap(importedMetaschema);
81      this.importedMetaschemaByName = Collections.unmodifiableMap(
82          importedMetaschema.values().stream().collect(Collectors.toMap(IMetaschema::getShortName, Function.identity())));
83      LOGGER.trace("Creating metaschema '{}'", metaschemaResource);
84    }
85  
86    @Override
87    public Map<@NotNull URI, ? extends IMetaschema> getImportedMetaschemaMap() {
88      return importedMetaschemaByUri;
89    }
90  
91    @Override
92    public Map<@NotNull String, ? extends IMetaschema> getImportedMetaschemaByShortNames() {
93      return importedMetaschemaByName;
94    }
95  
96    @SuppressWarnings("null")
97    @Override
98    public Map<@NotNull String, ? extends IFlagDefinition> getExportedFlagDefinitionMap() {
99      return Collections.unmodifiableMap(exportedFlagDefinitions);
100   }
101 
102   @SuppressWarnings("null")
103   @Override
104   public Map<@NotNull String, ? extends IFieldDefinition> getExportedFieldDefinitionMap() {
105     return Collections.unmodifiableMap(exportedFieldDefinitions);
106   }
107 
108   @SuppressWarnings("null")
109   @Override
110   public Map<@NotNull String, ? extends IAssemblyDefinition> getExportedAssemblyDefinitionMap() {
111     return Collections.unmodifiableMap(exportedAssemblyDefinitions);
112   }

114   private static class ExportedDefinitionsFilter<DEF extends IDefinition> implements Predicate<@NotNull DEF> {
115     @Override
116     public boolean test(@NotNull DEF definition) {
117       return ModuleScopeEnum.INHERITED.equals(definition.getModuleScope())
118           || ModelType.ASSEMBLY.equals(definition.getModelType()) && ((IAssemblyDefinition) definition).isRoot();
119     }
120   }
121 
122   @Override
123   public Collection<@NotNull ? extends IModelElement> getInfoElementsByMetapath(String path) {
124     // TODO: implement path evaluation
125     throw new UnsupportedOperationException();
126   }
127 
128   private static <DEF extends IDefinition> void addToMap(@NotNull Collection<@NotNull ? extends DEF> items,
129       @NotNull Map<@NotNull String, DEF> existingMap) {
130     for (DEF item : items) {
131       DEF oldItem = existingMap.put(item.getName(), item);
132       if (oldItem != null && oldItem != item && LOGGER.isWarnEnabled()) {
133         LOGGER.warn("The {} '{}' from metaschema '{}' is shadowing '{}' from metaschema '{}'",
134             item.getModelType().name().toLowerCase(Locale.ROOT), item.getName(),
135             item.getContainingMetaschema().getShortName(),
136             oldItem.getName(), oldItem.getContainingMetaschema().getShortName());
137       }
138     }
139   }
140 
141   /**
142    * Processes the definitions exported by the Metaschema, saving a list of all exported by specific
143    * model types.
144    * 
145    * @throws MetaschemaException
146    *           if a parsing error occurs
147    */
148   @SuppressWarnings("null")
149   protected void processExportedDefinitions() throws MetaschemaException {
150     if (LOGGER.isDebugEnabled()) {
151       LOGGER.debug("Processing metaschema '{}'", this.getLocation());
152     }
153 
154     this.exportedFlagDefinitions = new LinkedHashMap<>();
155     this.exportedFieldDefinitions = new LinkedHashMap<>();
156     this.exportedAssemblyDefinitions = new LinkedHashMap<>();
157 
158     // handle definitions from any included metaschema first
159     if (!getImportedMetaschemas().isEmpty()) {
160       for (IMetaschema metaschema : getImportedMetaschemas()) {
161         addToMap(metaschema.getExportedFlagDefinitions(), this.exportedFlagDefinitions);
162         addToMap(metaschema.getExportedFieldDefinitions(), this.exportedFieldDefinitions);
163         addToMap(metaschema.getExportedAssemblyDefinitions(), this.exportedAssemblyDefinitions);
164       }
165     }
166 
167     // now handle top-level definitions from this metaschema. Start first with filtering for globals,
168     // then add these to the maps
169     addToMap(getFlagDefinitions().stream().filter(new ExportedDefinitionsFilter<>()).collect(Collectors.toList()),
170         this.exportedFlagDefinitions);
171     addToMap(getFieldDefinitions().stream().filter(new ExportedDefinitionsFilter<>()).collect(Collectors.toList()),
172         this.exportedFieldDefinitions);
173     addToMap(getAssemblyDefinitions().stream().filter(new ExportedDefinitionsFilter<>()).collect(Collectors.toList()),
174         this.exportedAssemblyDefinitions);
175   }
176 
177   @Override
178   public URI getLocation() {
179     return location;
180   }
181 }
