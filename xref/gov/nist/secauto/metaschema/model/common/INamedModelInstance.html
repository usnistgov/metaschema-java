27  package gov.nist.secauto.metaschema.model.common;
28  
29  import java.util.Collection;
30  
31  import edu.umd.cs.findbugs.annotations.NonNull;
32  import edu.umd.cs.findbugs.annotations.Nullable;
33  
34  public interface INamedModelInstance extends INamedInstance, IModelInstance {
35    @Override
36    IModelDefinition getDefinition();
37  
38    /**
39     * Get the item values for the provided {@code instanceValue}. An instance may be singular or many
40     * valued.
41     * 
42     * @param instanceValue
43     *          the instance
44     * @return the item values or an empty collection if no item values exist
45     */
46    @NonNull
47    Collection<?> getItemValues(Object instanceValue);
48  
49    /**
50     * Indicates if a flag's value can be used as a property name in the containing object in JSON who's
51     * value will be the object containing the flag. In such cases, the flag will not appear in the
52     * object. This is only allowed if the flag is required, as determined by a {@code true} result from
53     * {@link IFlagInstance#isRequired()}. The {@link IFlagInstance} can be retrieved using
54     * {@link #getJsonKeyFlagInstance()}.
55     * 
56     * @return {@code true} if the flag's value can be used as a property name, or {@code false}
57     *         otherwise
58     * @see #getJsonKeyFlagInstance()
59     */
60    default boolean hasJsonKey() {
61      return getDefinition().hasJsonKey();
62    }
63  
64    /**
65     * Retrieves the flag instance to use as as the property name for the containing object in JSON
66     * who's value will be the object containing the flag.
67     * 
68     * @return the flag instance if a JSON key is configured, or {@code null} otherwise
69     * @see #hasJsonKey()
70     */
71    @Nullable
72    default IFlagInstance getJsonKeyFlagInstance() {
73      return getDefinition().getJsonKeyFlagInstance();
74    }
75  }
