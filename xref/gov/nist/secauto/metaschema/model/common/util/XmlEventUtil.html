81    @SuppressWarnings("null")
82    @NonNull
83    private static Object escape(@NonNull String data) {
84      return data.chars().mapToObj(c -> (char) c).map(c -> escape(c)).collect(Collectors.joining());
85    }

87    @SuppressWarnings("null")
88    @NonNull
89    private static String escape(char ch) {
90      String retval;
91      switch (ch) {
92      case '\n':
93        retval = "\\n";
94        break;
95      case '\r':
96        retval = "\\r";
97        break;
98      default:
99        retval = String.valueOf(ch);
100       break;
101     }
102     return retval;
103   }

105   /**
106    * Generate a message suitable for logging that describes the provided {@link XMLEvent}.
107    * 
108    * @param xmlEvent
109    *          the event to generate the message for
110    * @return the message
111    */
112   @NonNull
113   public static CharSequence toString(XMLEvent xmlEvent) {
114     CharSequence retval;
115     if (xmlEvent == null) {
116       retval = "EOF";
117     } else {
118       @SuppressWarnings("null")
119       @NonNull
120       StringBuilder builder = new StringBuilder()
121           .append(toEventName(xmlEvent));
122       QName name = toQName(xmlEvent);
123       if (name != null) {
124         builder.append(": ").append(name.toString());
125       }
126       if (xmlEvent.isCharacters()) {
127         String text = xmlEvent.asCharacters().getData();
128         if (text != null) {
129           builder.append(" '").append(escape(text)).append('\'');
130         }
131       }
132       Location location = toLocation(xmlEvent);
133       if (location != null) {
134         builder.append(" at ").append(toString(location));
135       }
136       retval = builder;
137     }
138     return retval;
139   }

158   /**
159    * Generates a string containing the current event and location of the stream reader.
160    * 
161    * @param reader
162    *          the stream reader
163    * @return the generated string
164    */
165   @NonNull
166   public static CharSequence toString(@NonNull XMLStreamReader2 reader) { // NO_UCD (unused code)
167     int type = reader.getEventType();
168 
169     @SuppressWarnings("null")
170     @NonNull
171     StringBuilder builder = new StringBuilder().append(toEventName(type));
172     QName name = reader.getName();
173     if (name != null) {
174       builder.append(": ").append(name.toString());
175     }
176     if (XMLStreamConstants.CHARACTERS == type) {
177       String text = reader.getText();
178       if (text != null) {
179         builder.append(" '").append(escape(text)).append('\'');
180       }
181     }
182     Location location = reader.getLocation();
183     if (location != null) {
184       builder.append(" at ").append(toString(location));
185     }
186     return builder;
187   }

189   /**
190    * Retrieve the resource location of {@code event}.
191    * 
192    * @param event
193    *          the event to identify the location for
194    * @return the location or {@code null} if the location is unknown
195    */
196   @Nullable
197   public static Location toLocation(@NonNull XMLEvent event) {
198     Location retval = null;
199     if (event.isStartElement()) {
200       StartElement start = event.asStartElement();
201       retval = start.getLocation();
202     } else if (event.isEndElement()) {
203       EndElement end = event.asEndElement();
204       retval = end.getLocation();
205     } else if (event.isCharacters()) {
206       Characters characters = event.asCharacters();
207       retval = characters.getLocation();
208     }
209     return retval;
210   }

214    * 
215    * @param event
216    *          the event to get the {@link QName} for
217    * @return the name of the node or {@code null} if the event is not a start or end element
218    */
219   @Nullable
220   public static QName toQName(@NonNull XMLEvent event) {
221     QName retval = null;
222     if (event.isStartElement()) {
223       StartElement start = event.asStartElement();
224       retval = start.getName();
225     } else if (event.isEndElement()) {
226       EndElement end = event.asEndElement();
227       retval = end.getName();
228     }
229     return retval;
230   }

261   /**
262    * Advance through XMLEvents until the event type identified by {@code eventType} is reached or the
263    * end of stream is found.
264    * 
265    * @param reader
266    *          the event reader to advance
267    * @param eventType
268    *          the event type to stop on as defined by {@link XMLStreamConstants}
269    * @return the next event of the specified type or {@code null} if the end of stream is reached
270    * @throws XMLStreamException
271    *           if an error occurred while advancing the stream
272    */
273   @Nullable
274   public static XMLEvent advanceTo(@NonNull XMLEventReader2 reader, int eventType)
275       throws XMLStreamException { // NO_UCD (unused code)
276     XMLEvent xmlEvent;
277     do {
278       xmlEvent = reader.nextEvent();
279       // if (LOGGER.isWarnEnabled()) {
280       // LOGGER.warn("skipping over: {}", XmlEventUtil.toString(xmlEvent));
281       // }
282       if (xmlEvent.isStartElement()) {
283         advanceTo(reader, XMLStreamConstants.END_ELEMENT);
284         // skip this end element
285         xmlEvent = reader.nextEvent();
286         // if (LOGGER.isDebugEnabled()) {
287         // LOGGER.debug("skipping over: {}", XmlEventUtil.toString(xmlEvent));
288         // }
289       }
290     } while (reader.hasNext() && (xmlEvent = reader.peek()).getEventType() != eventType);
291     return xmlEvent;
292   }

294   /**
295    * Skip over any processing instructions.
296    * 
297    * @param reader
298    *          the event reader to advance
299    * @return the last processing instruction event or the reader's next event if no processing
300    *         instruction was found
301    * @throws XMLStreamException
302    *           if an error occurred while advancing the stream
303    */
304   @NonNull
305   public static XMLEvent skipProcessingInstructions(@NonNull XMLEventReader2 reader) throws XMLStreamException {
306     XMLEvent nextEvent;
307     while ((nextEvent = reader.peek()).isProcessingInstruction()) {
308       nextEvent = reader.nextEvent();
309     }
310     return nextEvent;
311   }

313   /**
314    * Skip over any whitespace.
315    * 
316    * @param reader
317    *          the event reader to advance
318    * @return the last character event containing whitespace or the reader's next event if no character
319    *         event was found
320    * @throws XMLStreamException
321    *           if an error occurred while advancing the stream
322    */
323   @SuppressWarnings("null")
324   @NonNull
325   public static XMLEvent skipWhitespace(@NonNull XMLEventReader2 reader) throws XMLStreamException {
326     @NonNull
327     XMLEvent nextEvent;
328     while ((nextEvent = reader.peek()).isCharacters()) {
329       Characters characters = nextEvent.asCharacters();
330       String data = characters.getData();
331       if (WHITESPACE_ONLY.matcher(data).matches()) {
332         nextEvent = reader.nextEvent();
333       } else {
334         break;
335       }
336     }
337     return nextEvent;
338   }

358   /**
359    * Determine if the next event from {@code reader} is an end element whose name matches the provided
360    * {@code expectedLocalName} and {@code expectedNamespace}.
361    * 
362    * @param reader
363    *          the event reader
364    * @param expectedLocalName
365    *          the expected element name
366    * @param expectedNamespace
367    *          the expected element namespace
368    * @return {@code true} if the next event matches the {@code name}
369    * @throws XMLStreamException
370    *           if an error occurred while looking at the next event
371    */
372   public static boolean isNextEventEndElement(@NonNull XMLEventReader2 reader, @NonNull String expectedLocalName,
373       String expectedNamespace) throws XMLStreamException {
374     Objects.requireNonNull(reader, "reader");
375     Objects.requireNonNull(expectedLocalName, "expectedLocalName");
376     XMLEvent event = reader.peek();
377 
378     boolean retval;
379     if (event.isEndElement()) {
380       EndElement endElement = event.asEndElement();
381       QName name = endElement.getName();
382       retval = expectedLocalName.equals(name.getLocalPart())
383           && (expectedNamespace == null || expectedNamespace.equals(name.getNamespaceURI()));
384     } else {
385       retval = false;
386     }
387     return retval;
388   }

438    * Consume the next event from {@code reader} and assert that this event is of the type identified
439    * by {@code presumedEventType} and has the name identified by {@code presumedName}.
440    * 
441    * @param reader
442    *          the event reader
443    * @param presumedEventType
444    *          the expected event type as defined by {@link XMLStreamConstants}
445    * @param presumedName

447    * @return the next event
448    * @throws XMLStreamException
449    *           if an error occurred while looking at the next event
450    */
451   public static XMLEvent consumeAndAssert(XMLEventReader2 reader, int presumedEventType, QName presumedName)
452       throws XMLStreamException {
453     XMLEvent retval = reader.nextEvent();
454 
455     int eventType = retval.getEventType();
456     QName name = toQName(retval);
457     assert eventType == presumedEventType
458         && (presumedName == null
459             || presumedName.equals(name)) : generateAssertMessage(
460                 retval,
461                 presumedEventType,
462                 presumedName);
463     return retval;
464   }

506   private static CharSequence generateAssertMessage(XMLEvent retval, int presumedEventType, QName presumedName) {
507     StringBuilder builder = new StringBuilder(30);
508     builder
509         .append("Expected XML ")
510         .append(toEventName(presumedEventType));
511 
512     if (presumedName != null) {
513       builder.append(" for QName '")
514           .append(presumedName.toString());
515     }
516     builder.append("', instead found ")
517         .append(toString(retval));
518     return builder;
519   }

521   /**
522    * Skips events specified by {@code events}.
523    * 
524    * @param reader
525    *          the event reader
526    * @param events
527    *          the events to skip
528    * @return the next non-mataching event returned by {@link XMLEventReader2#peek()}, or {@code null}
529    *         if there was no next event
530    * @throws XMLStreamException
531    *           if an error occurred while reading
532    */
533   public static XMLEvent skipEvents(XMLEventReader2 reader, int... events) throws XMLStreamException {
534     Set<Integer> skipEvents = IntStream.of(events).boxed().collect(Collectors.toSet());
535 
536     XMLEvent nextEvent = null;
537     while (reader.hasNext()) {
538       nextEvent = reader.peek();
539       if (!skipEvents.contains(nextEvent.getEventType())) {
540         break;
541       }
542       reader.nextEvent();
543     }
544     return nextEvent;
545   }
