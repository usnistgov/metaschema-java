62  /**
63   * Provides methods to load a constraint set expressed in XML.
64   * <p>
65   * Loaded constraint instances are cached to avoid the need to load them for every use. Any
66   * constraint set imported is also loaded and cached automatically.
67   */
68  public class ConstraintLoader
69      extends AbstractLoader<IConstraintSet> {
70  
71    @Override
72    protected IConstraintSet parseResource(@NonNull URI resource, @NonNull Deque<URI> visitedResources)
73        throws IOException {
74  
75      // parse this metaschema
76      METASCHEMACONSTRAINTSDocument xmlObject = parseConstraintSet(resource);
77  
78      // now check if this constraint set imports other constraint sets
79      int size = xmlObject.getMETASCHEMACONSTRAINTS().sizeOfImportArray();
80      @NonNull Map<URI, IConstraintSet> importedMetaschema;
81      if (size == 0) {
82        importedMetaschema = ObjectUtils.notNull(Collections.emptyMap());
83      } else {
84        try {
85          importedMetaschema = new LinkedHashMap<>();
86          for (METASCHEMACONSTRAINTSDocument.METASCHEMACONSTRAINTS.Import imported : xmlObject.getMETASCHEMACONSTRAINTS()
87              .getImportList()) {
88            URI importedResource = URI.create(imported.getHref());
89            importedResource = ObjectUtils.notNull(resource.resolve(importedResource));
90            importedMetaschema.put(importedResource, loadInternal(importedResource, visitedResources));
91          }
92        } catch (MetaschemaException ex) {
93          throw new IOException(ex);
94        }
95      }
96  
97      // now create this metaschema
98      Collection<IConstraintSet> values = importedMetaschema.values();
99      return new DefaultConstraintSet(resource, parseScopedConstraints(xmlObject, resource), new LinkedHashSet<>(values));
100   }

102   /**
103    * Parse the provided XML resource as a Metaschema.
104    *
105    * @param resource
106    *          the resource to parse
107    * @return the XMLBeans representation of the Metaschema
108    * @throws IOException
109    *           if a parsing error occurred
110    */
111   @NonNull
112   protected METASCHEMACONSTRAINTSDocument parseConstraintSet(@NonNull URI resource) throws IOException {
113     try {
114       XmlOptions options = new XmlOptions();
115       options.setBaseURI(resource);
116       options.setLoadLineNumbers();
117       return ObjectUtils.notNull(METASCHEMACONSTRAINTSDocument.Factory.parse(resource.toURL(), options));
118     } catch (XmlException ex) {
119       throw new IOException(ex);
120     }
121   }
122 
123   /**
124    * Parse individual constraint definitions from the provided XMLBeans object.
125    *
126    * @param xmlObject
127    *          the XMLBeans object
128    * @param source
129    *          the source of the constraint content
130    * @return the scoped constraint definitions
131    */
132   @NonNull
133   protected List<IScopedContraints> parseScopedConstraints(
134       @NonNull METASCHEMACONSTRAINTSDocument xmlObject,
135       @NonNull URI source) {
136     List<IScopedContraints> scopedConstraints = new LinkedList<>();
137     ISource constraintSource = ExternalSource.instance(source);
138 
139     for (Scope scope : xmlObject.getMETASCHEMACONSTRAINTS().getScopeList()) {
140       URI namespace = ObjectUtils.notNull(URI.create(scope.getMetaschemaNamespace()));
141       String shortName = ObjectUtils.requireNonNull(scope.getMetaschemaShortName());
142 
143       try (XmlCursor cursor = scope.newCursor()) {
144         cursor.selectPath("declare namespace m='http://csrc.nist.gov/ns/oscal/metaschema/1.0';"
145             + "$this/m:assembly|$this/m:field|$this/m:flag");
146 
147         List<ITargetedConstaints> targetedConstraints = new LinkedList<>(); // NOPMD - intentional
148         while (cursor.toNextSelection()) {
149           XmlObject obj = cursor.getObject();
150           if (obj instanceof Scope.Assembly) {
151             Scope.Assembly assembly = (Scope.Assembly) obj;
152             MetapathExpression expression = ObjectUtils.requireNonNull(assembly.getTarget());
153             AssemblyConstraintSupport constraints
154                 = new AssemblyConstraintSupport(assembly, constraintSource); // NOPMD - intentional
155             targetedConstraints.add(new AssemblyTargetedConstraints(expression, constraints));
156           } else if (obj instanceof Scope.Field) {
157             Scope.Field field = (Scope.Field) obj;
158             MetapathExpression expression = ObjectUtils.requireNonNull(field.getTarget());
159             ValueConstraintSupport constraints
160                 = new ValueConstraintSupport(field, constraintSource); // NOPMD - intentional
161             targetedConstraints.add(new FieldTargetedConstraints(expression, constraints));
162           } else if (obj instanceof ScopedIndexHasKeyConstraintType) {
163             Scope.Flag flag = (Scope.Flag) obj;
164             MetapathExpression expression = ObjectUtils.requireNonNull(flag.getTarget());
165             ValueConstraintSupport constraints
166                 = new ValueConstraintSupport(flag, constraintSource); // NOPMD - intentional
167             targetedConstraints.add(new FlagTargetedConstraints(expression, constraints));
168           }
169         }
170         scopedConstraints.add(
171             new DefaultScopedContraints(namespace, shortName, CollectionUtil.unmodifiableList(targetedConstraints)));
172       }
173     }
174     return CollectionUtil.unmodifiableList(scopedConstraints);
175   }
