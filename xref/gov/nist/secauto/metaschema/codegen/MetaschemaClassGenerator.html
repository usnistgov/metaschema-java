57  class MetaschemaClassGenerator {
58    @NonNull
59    private final IMetaschema metaschema;
60    @NonNull
61    private final ITypeResolver typeResolver;
62  
63    public MetaschemaClassGenerator(@NonNull IMetaschema metaschema, @NonNull ITypeResolver typeResolver) {
64      this.metaschema = ObjectUtils.requireNonNull(metaschema, "metaschema");
65      this.typeResolver = ObjectUtils.requireNonNull(typeResolver, "typeResolver");
66    }
67  
68    @NonNull
69    protected IMetaschema getMetaschema() {
70      return metaschema;
71    }
72  
73    @NonNull
74    protected ITypeResolver getTypeResolver() {
75      return typeResolver;
76    }
77  
78    @NonNull
79    public ClassName getClassName() {
80      return getTypeResolver().getClassName(getMetaschema());
81    }
82  
83    @NonNull
84    public IGeneratedClass generateClass(Path outputDir) throws IOException {
85      ClassName className = getClassName();
86  
87      TypeSpec.Builder builder = generateClass(className);
88  
89      JavaFile javaFile = JavaFile.builder(className.packageName(), builder.build()).build();
90      Path classFile = ObjectUtils.notNull(javaFile.writeToPath(outputDir));
91  
92      return new DefaultGeneratedClass(classFile, className);
93    }
94  
95    @NonNull
96    protected TypeSpec.Builder generateClass(@NonNull ClassName className) { // NOPMD - long, but readable
97      // create the class
98      TypeSpec.Builder builder = TypeSpec.classBuilder(className).addModifiers(Modifier.PUBLIC, Modifier.FINAL);
99  
100     builder.superclass(AbstractBoundMetaschema.class);
101 
102     AnnotationSpec.Builder metaschemaAnnotation = AnnotationSpec.builder(Metaschema.class);
103 
104     IMetaschema metaschema = getMetaschema();
105 
106     ITypeResolver typeResolver = getTypeResolver();
107     for (IFieldDefinition definition : metaschema.getFieldDefinitions()) {
108       if (!definition.isSimple()) {
109         metaschemaAnnotation.addMember("fields", "$T.class", typeResolver.getClassName(definition));
110       }
111     }
112 
113     for (IAssemblyDefinition definition : metaschema.getAssemblyDefinitions()) {
114       metaschemaAnnotation.addMember(
115           "assemblies",
116           "$T.class",
117           typeResolver.getClassName(ObjectUtils.notNull(definition)));
118     }
119 
120     for (IMetaschema metaschemaImport : metaschema.getImportedMetaschemas()) {
121       metaschemaAnnotation.addMember(
122           "imports",
123           "$T.class",
124           typeResolver.getClassName(ObjectUtils.notNull(metaschemaImport)));
125     }
126 
127     {
128       MarkupMultiline remarks = metaschema.getRemarks();
129       if (remarks != null) {
130         metaschemaAnnotation.addMember("remarks", "$S", remarks.toMarkdown());
131       }
132     }
133 
134     builder.addAnnotation(metaschemaAnnotation.build());
135 
136     builder.addField(
137         FieldSpec.builder(MarkupLine.class, "NAME", Modifier.PRIVATE, Modifier.STATIC, Modifier.FINAL)
138             .initializer("$T.fromMarkdown($S)", MarkupLine.class, metaschema.getName().toMarkdown())
139             .build());
140 
141     builder.addField(
142         FieldSpec.builder(String.class, "SHORT_NAME", Modifier.PRIVATE, Modifier.STATIC, Modifier.FINAL)
143             .initializer("$S", metaschema.getShortName())
144             .build());
145 
146     builder.addField(
147         FieldSpec.builder(String.class, "VERSION", Modifier.PRIVATE, Modifier.STATIC, Modifier.FINAL)
148             .initializer("$S", metaschema.getVersion())
149             .build());
150 
151     builder.addField(
152         FieldSpec.builder(URI.class, "XML_NAMESPACE", Modifier.PRIVATE, Modifier.STATIC, Modifier.FINAL)
153             .initializer("$T.create($S)", URI.class, metaschema.getXmlNamespace())
154             .build());
155 
156     builder.addField(
157         FieldSpec.builder(URI.class, "JSON_BASE_URI", Modifier.PRIVATE, Modifier.STATIC, Modifier.FINAL)
158             .initializer("$T.create($S)", URI.class, metaschema.getJsonBaseUri())
159             .build());
160 
161     MarkupMultiline remarks = metaschema.getRemarks();
162     if (remarks != null) {
163       builder.addField(
164           FieldSpec.builder(MarkupMultiline.class, "REMARKS", Modifier.PRIVATE, Modifier.STATIC, Modifier.FINAL)
165               .initializer("$T.fromMarkdown($S)", MarkupMultiline.class, remarks.toMarkdown())
166               .build());
167     }
168 
169     builder.addMethod(
170         MethodSpec.constructorBuilder()
171             .addModifiers(Modifier.PUBLIC)
172             .addParameter(
173                 ParameterizedTypeName.get(ClassName.get(List.class),
174                     WildcardTypeName.subtypeOf(IMetaschema.class).box()),
175                 "importedMetaschema")
176             .addParameter(IBindingContext.class, "bindingContext")
177             .addStatement("super($N, $N)", "importedMetaschema", "bindingContext")
178             .build());
179     builder.addMethod(
180         MethodSpec.methodBuilder("getName")
181             .addAnnotation(Override.class)
182             .addModifiers(Modifier.PUBLIC)
183             .returns(MarkupLine.class)
184             .addStatement("return NAME")
185             .build());
186 
187     builder.addMethod(
188         MethodSpec.methodBuilder("getShortName")
189             .addAnnotation(Override.class)
190             .addModifiers(Modifier.PUBLIC)
191             .returns(String.class)
192             .addStatement("return SHORT_NAME")
193             .build());
194 
195     builder.addMethod(
196         MethodSpec.methodBuilder("getVersion")
197             .addAnnotation(Override.class)
198             .addModifiers(Modifier.PUBLIC)
199             .returns(String.class)
200             .addStatement("return VERSION")
201             .build());
202 
203     builder.addMethod(
204         MethodSpec.methodBuilder("getXmlNamespace")
205             .addAnnotation(Override.class)
206             .addModifiers(Modifier.PUBLIC)
207             .returns(URI.class)
208             .addStatement("return XML_NAMESPACE")
209             .build());
210 
211     builder.addMethod(
212         MethodSpec.methodBuilder("getJsonBaseUri")
213             .addAnnotation(Override.class)
214             .addModifiers(Modifier.PUBLIC)
215             .returns(URI.class)
216             .addStatement("return JSON_BASE_URI")
217             .build());
218 
219     MethodSpec.Builder methodBuilder = MethodSpec.methodBuilder("getRemarks")
220         .addAnnotation(Override.class)
221         .addModifiers(Modifier.PUBLIC)
222         .returns(MarkupMultiline.class);
223 
224     if (remarks == null) {
225       methodBuilder.addStatement("return null");
226     } else {
227       methodBuilder.addStatement("return REMARKS");
228     }
229 
230     builder.addMethod(methodBuilder.build());
231 
232     return builder;
233   }
