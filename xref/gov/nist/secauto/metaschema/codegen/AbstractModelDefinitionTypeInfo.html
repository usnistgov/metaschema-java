58  class AbstractModelDefinitionTypeInfo<DEF extends INamedModelDefinition>
59      extends AbstractDefinitionTypeInfo<DEF>
60      implements INamedModelDefinitionTypeInfo {
61    @NotNull
62    private final ClassName className;
63    @Nullable
64    private final ClassName baseClassName;
65    private Map<@NotNull String, IFlagInstanceTypeInfo> flagTypeInfos;
66  
67    public AbstractModelDefinitionTypeInfo(@NotNull DEF definition,
68        @NotNull ITypeResolver typeResolver) {
69      super(definition, typeResolver);
70      this.className = typeResolver.getClassName(definition);
71      this.baseClassName = typeResolver.getBaseClassName(definition);
72    }
73  
74    @Override
75    public ClassName getClassName() {
76      return className;
77    }
78  
79    @Override
80    public ClassName getBaseClassName() {
81      return baseClassName;
82    }
83  
84    @Override
85    protected boolean initInstanceTypeInfos() {
86      synchronized (this) {
87        boolean retval;
88        if (flagTypeInfos == null) {
89          // create Java properties for the definition's flags
90          flagTypeInfos = getDefinition().getFlagInstances().stream()
91              .map(instance -> {
92                return newFlagTypeInfo(instance);
93              })
94              .collect(Collectors.toUnmodifiableMap(IFlagInstanceTypeInfo::getPropertyName, Function.identity()));
95          retval = true;
96        } else {
97          retval = false;
98        }
99        return retval;
100     }
101   }
102 
103   @Override
104   public IFlagInstanceTypeInfo getFlagInstanceTypeInfo(@NotNull IFlagInstance instance) {
105     initInstanceTypeInfos();
106     return (IFlagInstanceTypeInfo) getInstanceTypeInfo(instance);
107   }
108 
109   @SuppressWarnings("null")
110   @Override
111   public Collection<@NotNull IFlagInstanceTypeInfo> getFlagInstanceTypeInfos() {
112     initInstanceTypeInfos();
113     return flagTypeInfos.values();
114   }
115 
116   protected IFlagInstanceTypeInfo newFlagTypeInfo(@NotNull IFlagInstance instance) {
117     IFlagInstanceTypeInfo retval = new FlagInstanceTypeInfoImpl(instance, this);
118     addPropertyTypeInfo(retval);
119     return retval;
120   }
121 
122   @Override
123   public TypeSpec generateChildClass() throws IOException {
124     return generateClass(getClassName(), true);
125   }
126 
127   protected void buildCommonProperties(@NotNull AnnotationSpec.Builder annotation) {
128     annotation.addMember("name", "$S", getDefinition().getName());
129     IMetaschema metaschema = getDefinition().getContainingMetaschema();
130     annotation.addMember("metaschema", "$T.class", getTypeResolver().getClassName(metaschema));
131   }
132 
133   protected void buildConstraints(@NotNull AnnotationSpec.Builder annotation) {
134     INamedModelDefinition definition = getDefinition();
135     AnnotationUtils.applyAllowedValuesConstraints(annotation, definition.getAllowedValuesConstraints());
136     AnnotationUtils.applyIndexHasKeyConstraints(annotation, definition.getIndexHasKeyConstraints());
137     AnnotationUtils.applyMatchesConstraints(annotation, definition.getMatchesConstraints());
138     AnnotationUtils.applyExpectConstraints(annotation, definition.getExpectConstraints());
139   }
140 
141   @Override
142   public DefaultGeneratedDefinitionClass generateClass(Path outputDir) throws IOException {
143     ClassName className = getClassName();
144 
145     TypeSpec classSpec = generateClass(className, false);
146 
147     JavaFile javaFile = JavaFile.builder(className.packageName(), classSpec).build();
148     Path classFile = ObjectUtils.notNull(javaFile.writeToPath(outputDir));
149 
150     return new DefaultGeneratedDefinitionClass(classFile, className, getDefinition());
151   }
152 
153   /**
154    * Supports the building of Java classes.
155    * 
156    * @param className
157    *          the type info for the class
158    * @param isChild
159    *          {@code true} if the class to be generated is a child class, or {@code false} otherwise
160    * @return the class definition
161    * @throws IOException
162    *           if a building error occurred while generating the Java class
163    */
164   @NotNull
165   protected TypeSpec generateClass(@NotNull ClassName className, boolean isChild) throws IOException {
166     // create the class
167     TypeSpec.Builder builder = TypeSpec.classBuilder(className).addModifiers(Modifier.PUBLIC);
168     if (isChild) {
169       builder.addModifiers(Modifier.STATIC);
170     }
171 
172     ClassName baseClassName = getBaseClassName();
173     if (baseClassName != null) {
174       builder.superclass(baseClassName);
175     }
176 
177     Set<@NotNull INamedModelDefinition> additionalChildClasses = buildClass(builder, className);
178 
179     ITypeResolver typeResolver = getTypeResolver();
180 
181     for (INamedModelDefinition definition : additionalChildClasses) {
182       INamedModelDefinitionTypeInfo typeInfo = typeResolver.getTypeInfo(definition);
183       TypeSpec childClass = typeInfo.generateChildClass();
184       builder.addType(childClass);
185     }
186     return ObjectUtils.notNull(builder.build());
187   }
188 
189   /**
190    * Generate the class contents.
191    * 
192    * @param builder
193    *          the builder to use for generating the class
194    * @param className
195    *          the name info for the class to build
196    * @return the set of additional definitions for which child classes need to be generated
197    * @throws IOException
198    *           if an error occurred while building the class
199    */
200   @NotNull
201   protected Set<@NotNull INamedModelDefinition> buildClass(@NotNull TypeSpec.Builder builder,
202       @NotNull ClassName className)
203       throws IOException {
204     MarkupLine description = getDefinition().getDescription();
205     if (description != null) {
206       builder.addJavadoc(description.toHtml());
207     }
208 
209     Set<@NotNull INamedModelDefinition> additionalChildClasses = new HashSet<>();
210 
211     // generate a no-arg constructor
212     builder.addMethod(MethodSpec.constructorBuilder().addModifiers(Modifier.PUBLIC).build());
213 
214     // // generate a copy constructor
215     // MethodSpec.Builder copyBuilder = MethodSpec.constructorBuilder().addModifiers(Modifier.PUBLIC);
216     // copyBuilder.addParameter(className, "that", Modifier.FINAL);
217     // for (IPropertyGenerator property : getPropertyGenerators()) {
218     // additionalChildClasses.addAll(property.buildCopyStatements(copyBuilder, getTypeResolver()));
219     // }
220     // builder.addMethod(copyBuilder.build());
221 
222     // generate all the properties and access methods
223     for (ITypeInfo property : getPropertyTypeInfos()) {
224       additionalChildClasses.addAll(property.build(builder, getTypeResolver()));
225     }
226 
227     // generate a toString method that will help with debugging
228     MethodSpec.Builder toString = MethodSpec.methodBuilder("toString").addModifiers(Modifier.PUBLIC)
229         .returns(String.class).addAnnotation(Override.class);
230     toString.addStatement("return new $T(this, $T.MULTI_LINE_STYLE).toString()", ReflectionToStringBuilder.class,
231         MultilineRecursiveToStringStyle.class);
232     builder.addMethod(toString.build());
233     return CollectionUtil.unmodifiableSet(additionalChildClasses);
234   }
