88    public static Object getDefaultValue(Class<?> annotation, String member) {
89      Method method;
90      try {
91        method = annotation.getDeclaredMethod(member);
92      } catch (NoSuchMethodException ex) {
93        throw new IllegalArgumentException(ex);
94      }
95      Object retval;
96      try {
97        retval = method.getDefaultValue();
98      } catch (TypeNotPresentException ex) {
99        retval = null;
100     }
101     return retval;
102   }

104   private static void buildConstraint(Class<?> annotationType, AnnotationSpec.Builder annotation,
105       IConstraint constraint) {
106     String id = constraint.getId();
107     if (id != null) {
108       annotation.addMember("id", "$S", id);
109     }
110 
111     String formalName = constraint.getFormalName();
112     if (formalName != null) {
113       annotation.addMember("formalName", "$S", formalName);
114     }
115 
116     MarkupLine description = constraint.getDescription();
117     if (description != null) {
118       annotation.addMember("description", "$S", description.toMarkdown());
119     }
120 
121     annotation.addMember("level", "$T.$L", IConstraint.Level.class, constraint.getLevel());
122 
123     MetapathExpression target = constraint.getTarget();
124     String path = target.getPath();
125     if (!path.equals(getDefaultValue(annotationType, "target"))) {
126       annotation.addMember("target", "$S", path);
127     }
128   }

130   public static void buildValueConstraints(
131       @NonNull FieldSpec.Builder builder,
132       @NonNull IValuedDefinition definition) {
133     if (!definition.getConstraints().isEmpty()) {
134       AnnotationSpec.Builder valueConstraintsAnnotation = AnnotationSpec.builder(ValueConstraints.class);
135 
136       applyAllowedValuesConstraints(valueConstraintsAnnotation, definition.getAllowedValuesConstraints());
137       applyIndexHasKeyConstraints(valueConstraintsAnnotation, definition.getIndexHasKeyConstraints());
138       applyMatchesConstraints(valueConstraintsAnnotation, definition.getMatchesConstraints());
139       applyExpectConstraints(valueConstraintsAnnotation, definition.getExpectConstraints());
140 
141       builder.addAnnotation(valueConstraintsAnnotation.build());
142     }
143   }

145   public static void buildValueConstraints(
146       @NonNull TypeSpec.Builder builder,
147       @NonNull IModelDefinition definition) {
148 
149     List<? extends IAllowedValuesConstraint> allowedValues = definition.getAllowedValuesConstraints();
150     List<? extends IIndexHasKeyConstraint> indexHasKey = definition.getIndexHasKeyConstraints();
151     List<? extends IMatchesConstraint> matches = definition.getMatchesConstraints();
152     List<? extends IExpectConstraint> expects = definition.getExpectConstraints();
153 
154     if (!allowedValues.isEmpty() || !indexHasKey.isEmpty() || !matches.isEmpty() || !expects.isEmpty()) {
155       AnnotationSpec.Builder annotation = AnnotationSpec.builder(ValueConstraints.class);
156 
157       applyAllowedValuesConstraints(annotation, allowedValues);
158       applyIndexHasKeyConstraints(annotation, indexHasKey);
159       applyMatchesConstraints(annotation, matches);
160       applyExpectConstraints(annotation, expects);
161 
162       builder.addAnnotation(annotation.build());
163     }
164   }

166   public static void buildAssemblyConstraints(
167       @NonNull TypeSpec.Builder builder,
168       @NonNull IAssemblyDefinition definition) {
169 
170     List<? extends IIndexConstraint> index = definition.getIndexConstraints();
171     List<? extends IUniqueConstraint> unique = definition.getUniqueConstraints();
172     List<? extends ICardinalityConstraint> cardinality = definition.getHasCardinalityConstraints();
173 
174     if (!index.isEmpty() || !unique.isEmpty() || !cardinality.isEmpty()) {
175       AnnotationSpec.Builder annotation = ObjectUtils.notNull(AnnotationSpec.builder(AssemblyConstraints.class));
176 
177       applyIndexConstraints(annotation, index);
178       applyUniqueConstraints(annotation, unique);
179       applyHasCardinalityConstraints(definition, annotation, cardinality);
180 
181       builder.addAnnotation(annotation.build());
182     }
183   }

185   private static void applyAllowedValuesConstraints(AnnotationSpec.Builder annotation,
186       List<? extends IAllowedValuesConstraint> constraints) {
187     for (IAllowedValuesConstraint constraint : constraints) {
188       AnnotationSpec.Builder constraintAnnotation = AnnotationSpec.builder(AllowedValues.class);
189       buildConstraint(AllowedValues.class, constraintAnnotation, constraint);
190 
191       boolean isAllowedOther = constraint.isAllowedOther();
192       if (!Boolean.valueOf(isAllowedOther).equals(getDefaultValue(AllowedValues.class, "allowOthers"))) {
193         constraintAnnotation.addMember("allowOthers", "$L", isAllowedOther);
194       }
195 
196       for (IAllowedValue value : constraint.getAllowedValues().values()) {
197         AnnotationSpec.Builder valueAnnotation = AnnotationSpec.builder(AllowedValue.class);
198 
199         valueAnnotation.addMember("value", "$S", value.getValue());
200         valueAnnotation.addMember("description", "$S", value.getDescription().toMarkdown());
201 
202         constraintAnnotation.addMember("values", "$L", valueAnnotation.build());
203       }
204 
205       MarkupMultiline remarks = constraint.getRemarks();
206       if (remarks != null) {
207         constraintAnnotation.addMember("remarks", "$S", remarks.toMarkdown());
208       }
209       annotation.addMember("allowedValues", "$L", constraintAnnotation.build());
210     }
211   }

220 
221       buildKeyFields(constraintAnnotation, constraint.getKeyFields());
222 
223       MarkupMultiline remarks = constraint.getRemarks();
224       if (remarks != null) {
225         constraintAnnotation.addMember("remarks", "$S", remarks.toMarkdown());
226       }
227 
228       annotation.addMember("indexHasKey", "$L", constraintAnnotation.build());
229     }

232   private static void buildKeyFields(@NonNull Builder constraintAnnotation,
233       @NonNull List<? extends IKeyField> keyFields) {
234     for (IKeyField key : keyFields) {
235       AnnotationSpec.Builder keyAnnotation = AnnotationSpec.builder(KeyField.class);
236 
237       MetapathExpression target = key.getTarget();
238       String path = target.getPath();
239       if (!path.equals(getDefaultValue(KeyField.class, "target"))) {
240         keyAnnotation.addMember("target", "$S", path);
241       }
242 
243       Pattern pattern = key.getPattern();
244       if (pattern != null) {
245         keyAnnotation.addMember("pattern", "$S", pattern.pattern());
246       }
247 
248       MarkupMultiline remarks = key.getRemarks();
249       if (remarks != null) {
250         keyAnnotation.addMember("remarks", "$S", remarks.toMarkdown());
251       }
252 
253       constraintAnnotation.addMember("keyFields", "$L", keyAnnotation.build());
254     }
255   }

257   private static void applyMatchesConstraints(AnnotationSpec.Builder annotation,
258       List<? extends IMatchesConstraint> constraints) {
259     for (IMatchesConstraint constraint : constraints) {
260       AnnotationSpec.Builder constraintAnnotation = AnnotationSpec.builder(Matches.class);
261       buildConstraint(Matches.class, constraintAnnotation, constraint);
262 
263       Pattern pattern = constraint.getPattern();
264       if (pattern != null) {
265         constraintAnnotation.addMember("pattern", "$S", pattern.pattern());
266       }
267 
268       IDataTypeAdapter<?> dataType = constraint.getDataType();
269       if (dataType != null) {
270         constraintAnnotation.addMember("typeAdapter", "$T.class", dataType.getClass());
271       }
272 
273       MarkupMultiline remarks = constraint.getRemarks();
274       if (remarks != null) {
275         constraintAnnotation.addMember("remarks", "$S", remarks.toMarkdown());
276       }
277       annotation.addMember("matches", "$L", constraintAnnotation.build());
278     }
279   }

286       buildConstraint(Expect.class, constraintAnnotation, constraint);
287 
288       MetapathExpression test = constraint.getTest();
289       constraintAnnotation.addMember("test", "$S", test.getPath());
290 
291       if (constraint.getMessage() != null) {
292         constraintAnnotation.addMember("message", "$S", constraint.getMessage());
293       }
294 
295       MarkupMultiline remarks = constraint.getRemarks();
296       if (remarks != null) {
297         constraintAnnotation.addMember("remarks", "$S", remarks.toMarkdown());
298       }
299 
300       annotation.addMember("expect", "$L", constraintAnnotation.build());
301     }

342   private static void applyHasCardinalityConstraints(
343       @NonNull IAssemblyDefinition definition,
344       @NonNull AnnotationSpec.Builder annotation,
345       @NonNull List<? extends ICardinalityConstraint> constraints) {
346 
347     DynamicContext dynamicContext = new StaticContext().newDynamicContext();
348     dynamicContext.disablePredicateEvaluation();
349 
350     for (ICardinalityConstraint constraint : constraints) {
351       Integer minOccurs = constraint.getMinOccurs();
352       Integer maxOccurs = constraint.getMaxOccurs();
353 
354       IAssemblyNodeItem definitionNodeItem = DefaultNodeItemFactory.instance().newAssemblyNodeItem(definition, null);
355 
356       ISequence<? extends IDefinitionNodeItem> instanceSet
357           = constraint.matchTargets(definitionNodeItem, dynamicContext);
358 
359       for (IDefinitionNodeItem item : instanceSet.asList()) {
360         INamedInstance instance = item.getInstance();
361         if (instance instanceof INamedModelInstance) {
362           INamedModelInstance modelInstance = (INamedModelInstance) instance;
363 
364           if (minOccurs != null) {
365             if (minOccurs == modelInstance.getMinOccurs()) {
366               if (LOGGER.isWarnEnabled()) {
367                 LOGGER.warn(String.format(
368                     "Definition '%s' has min-occurs=%d cardinality constraint targeting '%s' that is redundant with a"
369                         + " targeted instance named '%s' that requires min-occurs=%d",
370                     definition.getName(), minOccurs, constraint.getTarget().getPath(),
371                     modelInstance.getName(),
372                     modelInstance.getMinOccurs()));
373               }
374             } else if (minOccurs < modelInstance.getMinOccurs()) {
375               if (LOGGER.isWarnEnabled()) { // NOPMD - readability
376                 LOGGER.warn(String.format(
377                     "Definition '%s' has min-occurs=%d cardinality constraint targeting '%s' that conflicts with a"
378                         + " targeted instance named '%s' that requires min-occurs=%d",
379                     definition.getName(), minOccurs, constraint.getTarget().getPath(),
380                     modelInstance.getName(),
381                     modelInstance.getMinOccurs()));
382               }
383             }
384           }
385 
386           if (maxOccurs != null) {
387             if (maxOccurs == modelInstance.getMaxOccurs()) {
388               if (LOGGER.isWarnEnabled()) {
389                 LOGGER.warn(String.format(
390                     "Definition '%s' has max-occurs=%d cardinality constraint targeting '%s' that is redundant with a"
391                         + " targeted instance named '%s' that requires max-occurs=%d",
392                     definition.getName(), maxOccurs, constraint.getTarget().getPath(),
393                     modelInstance.getName(),
394                     modelInstance.getMaxOccurs()));
395               }
396             } else if (maxOccurs < modelInstance.getMaxOccurs()) {
397               if (LOGGER.isWarnEnabled()) { // NOPMD - readability
398                 LOGGER.warn(String.format(
399                     "Definition '%s' has max-occurs=%d cardinality constraint targeting '%s' that conflicts with a"
400                         + " targeted instance named '%s' that requires max-occurs=%d",
401                     definition.getName(), maxOccurs, constraint.getTarget().getPath(),
402                     modelInstance.getName(),
403                     modelInstance.getMaxOccurs()));
404               }
405             }
406           }
407         } else if (LOGGER.isWarnEnabled()) {
408           LOGGER.warn(String.format(
409               "Definition '%s' has min-occurs=%d cardinality constraint targeting '%s' that is not a model instance",
410               definition.getName(), minOccurs, constraint.getTarget().getPath()));
411         }
412       }
413 
414       AnnotationSpec.Builder constraintAnnotation = AnnotationSpec.builder(HasCardinality.class);
415 
416       buildConstraint(HasCardinality.class, constraintAnnotation, constraint);
417 
418       if (minOccurs != null && !minOccurs.equals(getDefaultValue(HasCardinality.class, "minOccurs"))) {
419         constraintAnnotation.addMember("minOccurs", "$L", minOccurs);
420       }
421 
422       if (maxOccurs != null && !maxOccurs.equals(getDefaultValue(HasCardinality.class, "maxOccurs"))) {
423         constraintAnnotation.addMember("maxOccurs", "$L", maxOccurs);
424       }
425 
426       annotation.addMember("hasCardinality", "$L", constraintAnnotation.build());
427 
428       MarkupMultiline remarks = constraint.getRemarks();
429       if (remarks != null) {
430         constraintAnnotation.addMember("remarks", "$S", remarks.toMarkdown());
431       }
432     }
