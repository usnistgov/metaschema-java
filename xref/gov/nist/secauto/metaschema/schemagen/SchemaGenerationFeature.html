27  package gov.nist.secauto.metaschema.schemagen;
28  
29  import gov.nist.secauto.metaschema.model.common.configuration.IConfigurationFeature;
30  
31  /**
32   * Configuration options for schema generation.
33   */
34  public enum SchemaGenerationFeature implements IConfigurationFeature {
35    /**
36     * If enabled, definitions that are defined inline will be generated as inline types. If disabled,
37     * definitions will always be generated as global types.
38     */
39    INLINE_DEFINITIONS(false),
40    /**
41     * If enabled, child definitions of a choice that are defined inline will be generated as inline
42     * types. If disabled, child definitions of a choice will always be generated as global types. This
43     * option will only be used if {@link #INLINE_DEFINITIONS} is also enabled.
44     */
45    INLINE_CHOICE_DEFINITIONS(false);
46  
47    private final boolean enabledByDefault;
48  
49    SchemaGenerationFeature(boolean enabled) {
50      this.enabledByDefault = enabled;
51    }
52  
53    @Override
54    public boolean isEnabledByDefault() {
55      return enabledByDefault;
56    }
57  }
