27  package gov.nist.secauto.metaschema.schemagen;
28  
29  import gov.nist.secauto.metaschema.model.common.IMetaschema;
30  import gov.nist.secauto.metaschema.model.common.configuration.IConfiguration;
31  import gov.nist.secauto.metaschema.schemagen.json.JsonSchemaGenerator;
32  import gov.nist.secauto.metaschema.schemagen.xml.XmlSchemaGenerator;
33  
34  import java.io.IOException;
35  import java.io.OutputStream;
36  import java.io.OutputStreamWriter;
37  import java.io.Writer;
38  import java.nio.charset.StandardCharsets;
39  import java.nio.file.Files;
40  import java.nio.file.Path;
41  import java.nio.file.StandardOpenOption;
42  
43  import edu.umd.cs.findbugs.annotations.NonNull;
44  
45  public interface ISchemaGenerator {
46    /**
47     * Generate and write a schema for the provided {@code metaschema} to the {@link Writer} provided by
48     * {@code writer} using the provided {@code configuration}.
49     *
50     * @param metaschema
51     *          the Metaschema to generate the schema for
52     * @param writer
53     *          the writer to use to write the schema
54     * @param configuration
55     *          the schema generation configuration
56     * @throws SchemaGenerationException
57     *           if an error occurred while writing the schema
58     */
59    void generateFromMetaschema(
60        @NonNull IMetaschema metaschema,
61        @NonNull Writer writer,
62        @NonNull IConfiguration<SchemaGenerationFeature> configuration);
63  
64    static void generateSchema(
65        @NonNull IMetaschema metaschema,
66        @NonNull Path destination,
67        @NonNull SchemaFormat asFormat,
68        @NonNull IConfiguration<SchemaGenerationFeature> configuration)
69        throws IOException {
70      ISchemaGenerator schemaGenerator = asFormat.getSchemaGenerator();
71  
72      try (Writer writer = Files.newBufferedWriter(
73          destination,
74          StandardCharsets.UTF_8,
75          StandardOpenOption.CREATE,
76          StandardOpenOption.WRITE,
77          StandardOpenOption.TRUNCATE_EXISTING)) {
78        assert writer != null;
79        schemaGenerator.generateFromMetaschema(metaschema, writer, configuration);
80        writer.flush();
81      }
82    }
83  
84    static void generateSchema(
85        @NonNull IMetaschema metaschema,
86        @NonNull OutputStream os,
87        @NonNull SchemaFormat asFormat,
88        @NonNull IConfiguration<SchemaGenerationFeature> configuration)
89        throws IOException {
90      ISchemaGenerator schemaGenerator = asFormat.getSchemaGenerator();
91  
92      Writer writer = new OutputStreamWriter(os, StandardCharsets.UTF_8);
93      schemaGenerator.generateFromMetaschema(metaschema, writer, configuration);
94      writer.flush();
95      // we don't want to close os, since we do not own it
96    }
97  
98    enum SchemaFormat {
99      /**
100      * a JSON Schema.
101      */
102     JSON(new JsonSchemaGenerator()),
103     /**
104      * an XML Schema.
105      */
106     XML(new XmlSchemaGenerator());
107 
108     private final ISchemaGenerator schemaGenerator;
109 
110     SchemaFormat(@NonNull ISchemaGenerator schemaGenerator) {
111       this.schemaGenerator = schemaGenerator;
112     }
113 
114     public ISchemaGenerator getSchemaGenerator() {
115       return schemaGenerator;
116     }
117   }
118 }
