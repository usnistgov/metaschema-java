27  package gov.nist.secauto.metaschema.schemagen;
28  
29  import gov.nist.secauto.metaschema.model.common.util.CollectionUtil;
30  
31  import org.jdom2.Element;
32  import org.jetbrains.annotations.NotNull;
33  
34  import java.io.InputStream;
35  import java.util.List;
36  import java.util.Map;
37  import java.util.stream.Collectors;
38  
39  public class XmlMarkupLineDatatypeProvider
40      extends AbstractXmlDatatypeProvider {
41    private static final String DATATYPE_NAME = "MarkupLineDatatype";
42  
43    @Override
44    protected InputStream getSchemaResource() {
45      return JDom2XmlSchemaLoader.class.getClassLoader()
46          .getResourceAsStream("schema/xml/metaschema-markup-line.xsd");
47    }
48  
49    @Override
50    protected List<@NotNull Element> queryElements(JDom2XmlSchemaLoader loader) {
51      return loader.getContent(
52          "/xs:schema/*",
53          CollectionUtil.singletonMap("xs", JDom2XmlSchemaLoader.NS_XML_SCHEMA));
54    }
55  
56    @Override
57    protected @NotNull Map<@NotNull String, IDatatypeContent> handleResults(@NotNull List<@NotNull Element> items) {
58      return CollectionUtil.singletonMap(
59          DATATYPE_NAME,
60          new JDom2DatatypeContent(
61              DATATYPE_NAME,
62              items.stream()
63                  .filter(element -> !("include".equals(element.getName())))
64                  .collect(Collectors.toList()),
65              CollectionUtil.emptyList()));
66    }
67  }
