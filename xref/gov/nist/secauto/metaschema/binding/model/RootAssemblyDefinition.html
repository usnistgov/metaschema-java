176   @Override
177   public Object readRoot(IXmlParsingContext context) throws IOException, XMLStreamException {
178 
179     XMLEventReader2 reader = context.getReader();
180 
181     // we may be at the START_DOCUMENT
182     if (reader.peek().isStartDocument()) {
183       XmlEventUtil.consumeAndAssert(reader, XMLEvent.START_DOCUMENT);
184     }
185 
186     XmlEventUtil.skipEvents(reader, XMLStreamConstants.CHARACTERS, XMLStreamConstants.PROCESSING_INSTRUCTION);
187 
188     QName rootQName = getRootXmlQName();
189     if (!reader.peek().isStartElement()) {
190       throw new IOException(
191           String.format("Expected an element named '%s', but found a '%s' instead.",
192               rootQName,
193               XmlEventUtil.toString(reader.peek())));
194     }
195 
196     XmlEventUtil.assertNext(reader, XMLEvent.START_ELEMENT, rootQName);
197 
198     StartElement start = ObjectUtils.notNull(reader.nextEvent().asStartElement());
199     Object result = ObjectUtils.requireNonNull(readItem(null, start, context));
200 
201     XmlEventUtil.consumeAndAssert(reader, XMLEvent.END_ELEMENT, rootQName);
202 
203     // if (reader.hasNext() && LOGGER.isDebugEnabled()) {
204     // LOGGER.debug("After Parse: {}", XmlEventUtil.toString(reader.peek()));
205     // }
206 
207     return result;
208   }

210   @Override
211   public Object readRoot(IJsonParsingContext context) throws IOException {
212     if (!isRoot()) {
213       throw new IOException(
214           String.format("The bound assembly '%s' does not have a root defined in the '%s' annotation.",
215               getBoundClass().getName(),
216               MetaschemaAssembly.class.getName()));
217     }
218 
219     JsonParser parser = context.getReader(); // NOPMD - intentional
220 
221     boolean objectWrapper = false;
222     if (parser.currentToken() == null) {
223       parser.nextToken();
224     }
225 
226     if (JsonToken.START_OBJECT.equals(parser.currentToken())) {
227       // advance past the start object to the field name
228       JsonUtil.assertAndAdvance(parser, JsonToken.START_OBJECT);
229       objectWrapper = true;
230     }
231 
232     String rootFieldName = getRootJsonName();
233     JsonToken token;
234     Object instance = null;
235     while (!(JsonToken.END_OBJECT.equals(token = parser.currentToken()) || token == null)) {
236       if (!JsonToken.FIELD_NAME.equals(token)) {
237         throw new IOException(String.format("Expected FIELD_NAME token, found '%s'", token.toString()));
238       }
239 
240       String fieldName = parser.currentName();
241       if (fieldName.equals(rootFieldName)) {
242         // process the object value, bound to the requested class
243         JsonUtil.assertAndAdvance(parser, JsonToken.FIELD_NAME);
244         instance = readObject(context);
245 
246         // stop now, since we found the root field
247         break;
248       }
249 
250       if (!context.getProblemHandler().handleUnknownRootProperty(this, fieldName, context)) {
251         LOGGER.warn("Skipping unhandled top-level JSON field '{}'.", fieldName);
252         JsonUtil.skipNextValue(parser);
253       }
254     }
255 
256     if (instance == null) {
257       throw new IOException(String.format("Failed to find root field '%s'.", rootFieldName));
258     }
259 
260     if (objectWrapper) {
261       // advance past the end object
262       JsonUtil.assertAndAdvance(parser, JsonToken.END_OBJECT);
263     }
264 
265     return instance;
266   }

268   @Override
269   public void writeRoot(Object instance, IXmlWritingContext context) throws XMLStreamException, IOException {
270 
271     XMLStreamWriter2 writer = context.getWriter();
272 
273     writer.writeStartDocument("UTF-8", "1.0");
274 
275     QName rootQName = getRootXmlQName();
276 
277     NamespaceContext nsContext = writer.getNamespaceContext();
278     String prefix = nsContext.getPrefix(rootQName.getNamespaceURI());
279     if (prefix == null) {
280       prefix = "";
281     }
282 
283     writer.writeStartElement(prefix, rootQName.getLocalPart(), rootQName.getNamespaceURI());
284 
285     writeItem(instance, rootQName, context);
286 
287     writer.writeEndElement();
288   }

290   @Override
291   public void writeRoot(Object instance, IJsonWritingContext context) throws IOException {
292 
293     JsonGenerator writer = context.getWriter(); // NOPMD - intentional
294 
295     // first read the initial START_OBJECT
296     writer.writeStartObject();
297 
298     writer.writeFieldName(getRootJsonName());
299 
300     writeItems(CollectionUtil.singleton(instance), true, context);
301 
302     // end of root object
303     writer.writeEndObject();
304   }
