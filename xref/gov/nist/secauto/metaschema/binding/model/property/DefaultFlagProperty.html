134   @Override
135   public boolean read(Object parentInstance, StartElement parent, IXmlParsingContext context) throws IOException {
136 
137     // when reading an attribute:
138     // - "parent" will contain the attributes to read
139     // - the event reader "peek" will be on the end element or the next start element
140     boolean handled = false;
141     if (parent != null) {
142       Attribute attribute = parent.getAttributeByName(getXmlQName());
143       if (attribute != null) {
144         // get the attribute value
145         Object value = getJavaTypeAdapter().parse(attribute.getValue());
146         // apply the value to the parentObject
147         setValue(parentInstance, value);
148 
149         handled = true;
150       }
151     }
152     return handled;
153   }

160   @Override
161   protected Object readInternal(Object parentInstance, IJsonParsingContext context) throws IOException {
162     JsonParser parser = context.getReader();
163 
164     // advance past the property name
165     parser.nextFieldName();
166 
167     // parse the value
168     return readValueAndSupply(context).get();
169   }

187   @Override
188   public boolean write(Object instance, QName parentName, IXmlWritingContext context)
189       throws XMLStreamException, IOException {
190     QName name = getXmlQName();
191     String value;
192 
193     Object objectValue = getValue(instance);
194     if (objectValue != null) {
195       value = objectValue.toString();
196     } else {
197       value = null;
198     }
199 
200     if (value != null) {
201       XMLStreamWriter2 writer = context.getWriter();
202       if (name.getNamespaceURI().isEmpty()) {
203         writer.writeAttribute(name.getLocalPart(), value);
204       } else {
205         writer.writeAttribute(name.getNamespaceURI(), name.getLocalPart(), value);
206       }
207     }
208     return true;
209   }

211   @Override
212   public void write(Object instance, IJsonWritingContext context) throws IOException {
213     JsonGenerator writer = context.getWriter();
214 
215     Object value = getValue(instance);
216     if (value != null) {
217       // write the field name
218       writer.writeFieldName(getJsonName());
219 
220       // write the value
221       writeValue(value, context);
222     }
223   }
