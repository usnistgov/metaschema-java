132   public boolean isNextProperty(@NotNull IXmlParsingContext context) throws XMLStreamException {
133     XMLEventReader2 eventReader = context.getReader();
134 
135     XmlEventUtil.skipWhitespace(eventReader);
136 
137     boolean handled = false;
138     QName groupQName = getXmlGroupAsQName();
139     if (groupQName != null) {
140       // we are to parse the grouping element, if the next token matches
141       XMLEvent event = eventReader.peek();
142       if (event.isStartElement() && groupQName.equals(event.asStartElement().getName())) {
143         handled = true;
144       }
145     }
146 
147     if (!handled) {
148       XMLEvent event = eventReader.peek();
149       QName xmlQName = getXmlQName();
150       if (xmlQName != null && event.isStartElement() && xmlQName.equals(event.asStartElement().getName())) {
151         handled = true;
152       }
153     }
154     return handled;
155   }

157   @Override
158   public boolean read(Object parentInstance, StartElement start, IXmlParsingContext context)
159       throws IOException, XMLStreamException {
160     boolean handled = isNextProperty(context);
161     if (handled) {
162       Object value = readInternal(parentInstance, start, context);
163       setValue(parentInstance, value);
164     }
165     return handled;
166   }

168   @Override
169   protected Object readInternal(Object parentInstance, IJsonParsingContext context)
170       throws IOException {
171     JsonParser parser = context.getReader(); // NOPMD - intentional
172 
173     // the parser's current token should be the JSON field name
174     // advance past the property name
175     JsonUtil.assertAndAdvance(parser, JsonToken.FIELD_NAME);
176 
177     // parse the value
178     IPropertyCollector collector = newPropertyCollector();
179     IModelPropertyInfo info = getPropertyInfo();
180     info.readValue(collector, parentInstance, context);
181 
182     JsonUtil.assertCurrent(context.getReader(), JsonToken.FIELD_NAME, JsonToken.END_OBJECT);
183 
184     return collector.getValue();
185   }
186 
187   protected Object readInternal(@Nullable Object parentInstance, @NotNull StartElement start,
188       @NotNull IXmlParsingContext context)
189       throws IOException, XMLStreamException {
190     XMLEventReader2 eventReader = context.getReader();
191 
192     XmlEventUtil.skipWhitespace(eventReader);
193 
194     StartElement currentStart = start;
195 
196     QName groupQName = getXmlGroupAsQName();
197     if (groupQName != null) {
198       // we are to parse the grouping element, if the next token matches
199       XMLEvent groupEvent = XmlEventUtil.consumeAndAssert(eventReader, XMLEvent.START_ELEMENT, groupQName);
200       currentStart = ObjectUtils.notNull(groupEvent.asStartElement());
201     }
202 
203     IPropertyCollector collector = newPropertyCollector();
204     // There are zero or more named values based on cardinality
205     getPropertyInfo().readValue(collector, parentInstance, currentStart, context);
206 
207     Object value = collector.getValue();
208 
209     // consume extra whitespace between elements
210     XmlEventUtil.skipWhitespace(eventReader);
211 
212     if (groupQName != null) {
213       // consume the end of the group
214       XmlEventUtil.consumeAndAssert(eventReader, XMLEvent.END_ELEMENT, groupQName);
215     }
216 
217     return value;
218   }

232   @Override
233   public boolean write(Object parentInstance, QName parentName, IXmlWritingContext context)
234       throws XMLStreamException, IOException {
235     Object value = getValue(parentInstance);
236     if (value == null) {
237       return false; // NOPMD - intentional
238     }
239 
240     QName currentStart = parentName;
241     XMLStreamWriter2 writer = context.getWriter();
242     QName groupQName = getXmlGroupAsQName();
243     if (groupQName != null) {
244       // write the grouping element
245       writer.writeStartElement(groupQName.getNamespaceURI(), groupQName.getLocalPart());
246       currentStart = groupQName;
247     }
248 
249     // There are one or more named values based on cardinality
250     getPropertyInfo().writeValue(value, currentStart, context);
251 
252     if (groupQName != null) {
253       writer.writeEndElement();
254     }
255     return true;
256   }

270   @Override
271   public void write(Object parentInstance, IJsonWritingContext context) throws IOException {
272     if (getPropertyInfo().isValueSet(parentInstance)) {
273       // write the field name
274       context.getWriter().writeFieldName(getJsonName());
275 
276       // dispatch to the property info implementation to address cardinality
277       getPropertyInfo().writeValue(parentInstance, context);
278     }
279   }

281   @Override
282   public void copyBoundObject(@NotNull Object fromInstance, @NotNull Object toInstance) throws BindingException {
283     Object value = getValue(fromInstance);
284     if (value != null) {
285       IModelPropertyInfo propertyInfo = getPropertyInfo();
286       IPropertyCollector collector = newPropertyCollector();
287 
288       propertyInfo.copy(fromInstance, toInstance, collector);
289 
290       value = collector.getValue();
291     }
292     setValue(toInstance, value);
293   }

295   @Override
296   public Object copyItem(Object fromItem, Object toInstance) throws BindingException {
297     return getDataTypeHandler().copyItem(fromItem, toInstance);
