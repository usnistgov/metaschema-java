60    @NonNull
61    static Class<?> getItemType(@NonNull Field field) {
62      Type fieldType = field.getGenericType();
63      Class<?> rawType = ObjectUtils.notNull(
64          (Class<?>) (fieldType instanceof ParameterizedType ? ((ParameterizedType) fieldType).getRawType() : fieldType));
65  
66      Class<?> itemType;
67      if (Map.class.isAssignableFrom(rawType)) {
68        // this is a Map so the second generic type is the value
69        itemType = ObjectUtils.notNull((Class<?>) ((ParameterizedType) fieldType).getActualTypeArguments()[1]);
70      } else if (List.class.isAssignableFrom(rawType)) {
71        // this is a List so there is only a single generic type
72        itemType = ObjectUtils.notNull((Class<?>) ((ParameterizedType) fieldType).getActualTypeArguments()[0]);
73      } else {
74        // non-collection
75        itemType = rawType;
76      }
77      return itemType;
78    }

91    @NonNull
92    default IModelPropertyInfo newPropertyInfo() { // NOPMD - cyclomatic complexity is unavoidable
93      // create the property info
94      Type type = getType();
95  
96      IModelPropertyInfo retval;
97      if (getMaxOccurs() == -1 || getMaxOccurs() > 1) {
98        // collection case
99        // expect a ParameterizedType
100       if (!(type instanceof ParameterizedType)) {
101         switch (getJsonGroupAsBehavior()) {
102         case KEYED:
103           throw new IllegalStateException(
104               String.format("The field '%s' on class '%s' has data type of '%s'," + " but should have a type of '%s'.",
105                   getField().getName(), getParentClassBinding().getBoundClass().getName(),
106                   getField().getType().getName(), Map.class.getName()));
107         case LIST:
108         case SINGLETON_OR_LIST:
109           throw new IllegalStateException(
110               String.format("The field '%s' on class '%s' has data type of '%s'," + " but should have a type of '%s'.",
111                   getField().getName(), getParentClassBinding().getBoundClass().getName(),
112                   getField().getType().getName(), List.class.getName()));
113         default:
114           // this should not occur
115           throw new IllegalStateException(getJsonGroupAsBehavior().name());
116         }
117       }
118 
119       // collection case
120       Class<?> rawType = (Class<?>) ((ParameterizedType) type).getRawType();
121       if (JsonGroupAsBehavior.KEYED.equals(getJsonGroupAsBehavior())) {
122         if (!Map.class.isAssignableFrom(rawType)) {
123           throw new IllegalArgumentException(String.format(
124               "The field '%s' on class '%s' has data type '%s', which is not the expected '%s' derived data type.",
125               getField().getName(), getParentClassBinding().getBoundClass().getName(),
126               getField().getType().getName(), Map.class.getName()));
127         }
128         retval = new MapPropertyInfo(this);
129       } else {
130         if (!List.class.isAssignableFrom(rawType)) {
131           throw new IllegalArgumentException(String.format(
132               "The field '%s' on class '%s' has data type '%s', which is not the expected '%s' derived data type.",
133               getField().getName(), getParentClassBinding().getBoundClass().getName(),
134               getField().getType().getName(), List.class.getName()));
135         }
136         retval = new ListPropertyInfo(this);
137       }
138     } else {
139       // single value case
140       if (type instanceof ParameterizedType) {
141         throw new IllegalStateException(String.format(
142             "The field '%s' on class '%s' has a data parmeterized type of '%s',"
143                 + " but the occurance is not multi-valued.",
144             getField().getName(), getParentClassBinding().getBoundClass().getName(), getField().getType().getName()));
145       }
146       retval = new SingletonPropertyInfo(this);
147     }
148     return retval;
149   }
