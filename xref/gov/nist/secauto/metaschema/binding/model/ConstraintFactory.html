65  final class ConstraintFactory {
66    private ConstraintFactory() {
67      // disable
68    }
69  
70    @NotNull
71    static Map<@NotNull String, DefaultAllowedValue> toAllowedValues(@NotNull AllowedValues constraint) {
72      AllowedValue[] values = constraint.values();
73  
74      Map<@NotNull String, DefaultAllowedValue> allowedValues = new LinkedHashMap<>(values.length); // NOPMD - intentional
75      for (AllowedValue value : values) {
76        DefaultAllowedValue allowedValue
77            = new DefaultAllowedValue(value.value(), MarkupLine.fromMarkdown(value.description())); // NOPMD - intentional
78        allowedValues.put(allowedValue.getValue(), allowedValue);
79      }
80      return CollectionUtil.unmodifiableMap(allowedValues);
81    }
82  
83    static String toId(String id) {
84      return id.isBlank() ? null : id;
85    }
86  
87    static MarkupMultiline toRemarks(String remarks) {
88      return remarks.isBlank() ? null : MarkupMultiline.fromMarkdown(remarks);
89    }
90  
91    static Pattern toPattern(String pattern) {
92      return pattern.isBlank() ? null : Pattern.compile(pattern);
93    }
94  
95    static String toMessage(String message) {
96      return message.isBlank() ? null : message;
97    }
98  
99    @Nullable
100   static IDataTypeAdapter<?> toDataType(@NotNull Class<? extends IDataTypeAdapter<?>> adapterClass) {
101     return adapterClass.isAssignableFrom(NullJavaTypeAdapter.class) ? null
102         : DataTypeService.getInstance().getJavaTypeAdapterByClass(adapterClass);
103   }
104 
105   @NotNull
106   static MetapathExpression toMetapath(@NotNull String metapath) {
107     return metapath.isBlank() ? MetapathExpression.CONTEXT_NODE : MetapathExpression.compile(metapath);
108   }
109 
110   @NotNull
111   static DefaultAllowedValuesConstraint newAllowedValuesConstraint(@NotNull AllowedValues constraint,
112       @NotNull ISource source) {
113     return new DefaultAllowedValuesConstraint(
114         toId(constraint.id()),
115         source,
116         constraint.level(),
117         toMetapath(constraint.target()),
118         toAllowedValues(constraint),
119         constraint.allowOthers(),
120         constraint.extensible(),
121         toRemarks(constraint.remarks()));
122   }
123 
124   @NotNull
125   static DefaultMatchesConstraint newMatchesConstraint(Matches constraint, @NotNull ISource source) {
126     return new DefaultMatchesConstraint(
127         toId(constraint.id()),
128         source,
129         constraint.level(),
130         toMetapath(constraint.target()),
131         toPattern(constraint.pattern()),
132         toDataType(constraint.typeAdapter()),
133         toRemarks(constraint.remarks()));
134   }
135 
136   @NotNull
137   static List<@NotNull DefaultKeyField> toKeyFields(@NotNull KeyField... keyFields) {
138     List<@NotNull DefaultKeyField> retval;
139     if (keyFields == null || keyFields.length == 0) {
140       retval = CollectionUtil.emptyList();
141     } else {
142       retval = new ArrayList<>(keyFields.length);
143       for (KeyField keyField : keyFields) {
144         DefaultKeyField field = new DefaultKeyField(toMetapath(keyField.target()), // NOPMD - intentional
145             toPattern(keyField.pattern()), toRemarks(keyField.remarks()));
146         retval.add(field);
147       }
148       retval = CollectionUtil.unmodifiableList(retval);
149     }
150     return retval;
151   }

153   @Nullable
154   static Integer toCardinality(int value) {
155     return value < 0 ? null : Integer.valueOf(value);
156   }
