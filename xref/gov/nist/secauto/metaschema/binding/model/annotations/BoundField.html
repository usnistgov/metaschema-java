27  package gov.nist.secauto.metaschema.binding.model.annotations;
28  
29  import static java.lang.annotation.ElementType.FIELD;
30  import static java.lang.annotation.RetentionPolicy.RUNTIME;
31  
32  import gov.nist.secauto.metaschema.model.common.JsonGroupAsBehavior;
33  import gov.nist.secauto.metaschema.model.common.MetaschemaModelConstants;
34  import gov.nist.secauto.metaschema.model.common.XmlGroupAsBehavior;
35  import gov.nist.secauto.metaschema.model.common.datatype.adapter.IDataTypeAdapter;
36  
37  import java.lang.annotation.Documented;
38  import java.lang.annotation.Retention;
39  import java.lang.annotation.Target;
40  
41  /**
42   * Identifies that the annotation target is a bound property that references a Metaschema field.
43   * <p>
44   * For XML serialization, the {@link #useName()} identifies the name of the element to use and the
45   * {@link #namespace()} identifies the namespace of this element.
46   * <p>
47   * For JSON and YAML serializations, the {@link #useName()} identifies the property/item name to
48   * use.
49   */
50  @Documented
51  @Retention(RUNTIME)
52  @Target({ FIELD })
53  public @interface BoundField {
54    /**
55     * The model name to use for singleton values. This name will be used for associated XML elements.
56     * <p>
57     * If the value is "##default", then element name is derived from the JavaBean property name.
58     * 
59     * @return the name
60     */
61    String useName() default "##default";
62  
63    /**
64     */
65    /**
66     * The namespace to use for associated XML elements.
67     * <p>
68     * If the value is "##default", then element name is derived from the namespace provided in the
69     * package-info.
70     * 
71     * @return the namespace
72     */
73    String namespace() default "##default";
74  
75    /**
76     * If the data type allows it, determines if the field's value must be wrapped with an element
77     * having the specified {@link #useName()} and {@link #namespace()}.
78     * 
79     * @return {@code true} if the field must be wrapped, or {@code false} otherwise
80     */
81    boolean inXmlWrapped() default MetaschemaModelConstants.DEFAULT_FIELD_IN_XML_WRAPPED;
82  
83    /**
84     * The Metaschema data type adapter for the field's value.
85     * 
86     * @return the data type adapter
87     */
88    Class<? extends IDataTypeAdapter<?>> typeAdapter() default NullJavaTypeAdapter.class;
89    //
90    // /**
91    // * The name of the JSON property that contains the field's value. If this value is provided, the
92    // * name will be used as the property name. Use of this annotation is mutually exclusive with the
93    // * {@link JsonFieldValueKeyFlag} annotation.
94    // *
95    // * @return the name
96    // */
97    // String valueName() default "##none";
98  
99    /**
100    * The name to use for an XML element wrapper or a JSON/YAML property.
101    * 
102    * @return the name
103    */
104   String groupName() default "##none";
105 
106   /**
107    * XML target namespace of the XML Schema element.
108    * <p>
109    * If the value is "##default", then element name is derived from the namespace provided in the
110    * package-info.
111    * 
112    * @return the namespace
113    */
114   String groupNamespace() default "##default";
115 
116   /**
117    * A non-negative number that indicates the minimum occurrence of the element.
118    * 
119    * @return a non-negative number
120    */
121   int minOccurs() default MetaschemaModelConstants.DEFAULT_GROUP_AS_MIN_OCCURS;
122 
123   /**
124    * A number that indicates the maximum occurrence of the element.
125    * 
126    * @return a positive number or {@code -1} to indicate "unbounded"
127    */
128   int maxOccurs() default MetaschemaModelConstants.DEFAULT_GROUP_AS_MAX_OCCURS;
129 
130   /**
131    * Describes how to handle collections in JSON/YAML.
132    * 
133    * @return the JSON collection strategy
134    */
135   JsonGroupAsBehavior inJson() default JsonGroupAsBehavior.NONE;
136 
137   /**
138    * Describes how to handle collections in XML.
139    * 
140    * @return the XML collection strategy
141    */
142   XmlGroupAsBehavior inXml() default XmlGroupAsBehavior.UNGROUPED;
143 
144   /**
145    * Get the allowed value constraints for this field.
146    * 
147    * @return the allowed values or an empty array if no allowed values constraints are defined
148    */
149   AllowedValues[] allowedValues() default {};
150 
151   /**
152    * Get the matches constraints for this field.
153    * 
154    * @return the allowed values or an empty array if no allowed values constraints are defined
155    */
156   Matches[] matches() default {};
157 
158   /**
159    * Get the index-has-key constraints for this field.
160    * 
161    * @return the allowed values or an empty array if no allowed values constraints are defined
162    */
163   IndexHasKey[] indexHasKey() default {};
164 
165   /**
166    * Get the expect constraints for this field.
167    * 
168    * @return the expected constraints or an empty array if no expected constraints are defined
169    */
170   Expect[] expect() default {};
171 }
