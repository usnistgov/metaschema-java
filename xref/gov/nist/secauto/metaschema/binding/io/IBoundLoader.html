48  /**
49   * A common interface for loading Metaschema based instance resources.
50   */
51  public interface IBoundLoader extends IDocumentLoader, IMutableConfiguration<DeserializationFeature> {
52    /**
53     * Determine the format of the provided resource.
54     * 
55     * @param url
56     *          the resource
57     * @return the format of the provided resource
58     * @throws IOException
59     *           if an error occurred while reading the resource
60     * @throws URISyntaxException
61     *           if the provided {@code url} is malformed
62     */
63    @NonNull
64    default Format detectFormat(@NonNull URL url) throws IOException, URISyntaxException {
65      return detectFormat(toInputSource(ObjectUtils.notNull(url.toURI())));
66    }
67  
68    /**
69     * Determine the format of the provided resource.
70     * 
71     * @param path
72     *          the resource
73     * @return the format of the provided resource
74     * @throws IOException
75     *           if an error occurred while reading the resource
76     */
77    @NonNull
78    default Format detectFormat(@NonNull Path path) throws IOException {
79      return detectFormat(toInputSource(ObjectUtils.notNull(path.toUri())));
80    }
81  
82    /**
83     * Determine the format of the provided resource.
84     * 
85     * @param file
86     *          the resource
87     * @return the format of the provided resource
88     * @throws IOException
89     *           if an error occurred while reading the resource
90     */
91    @NonNull
92    default Format detectFormat(@NonNull File file) throws IOException {
93      return detectFormat(ObjectUtils.notNull(file.toPath()));
94    }
95  
96    /**
97     * Determine the format of the provided resource.
98     * <p>
99     * This method will consume data from the provided {@link InputStream}. If the caller of this method
100    * intends to read data from the stream after determining the format, the caller should pass in a
101    * stream that can be reset.
102    * <p>
103    * This method will not close the provided {@link InputStream}, since it does not own the stream.
104    * 
105    * @param is
106    *          an input stream for the resource
107    * @return the format of the provided resource
108    * @throws IOException
109    *           if an error occurred while reading the resource
110    */
111   @NonNull
112   default Format detectFormat(@NonNull InputStream is) throws IOException {
113     return detectFormat(new InputSource(is));
114   }
115 
116   /**
117    * Determine the format of the provided resource.
118    * <p>
119    * This method will consume data from any {@link InputStream} provided by the {@link InputSource}.
120    * If the caller of this method intends to read data from the stream after determining the format,
121    * the caller should pass in a stream that can be reset.
122    * <p>
123    * This method will not close any {@link InputStream} provided by the {@link InputSource}, since it
124    * does not own the stream.
125    * 
126    * @param source
127    *          information about how to access the resource
128    * @return the format of the provided resource
129    * @throws IOException
130    *           if an error occurred while reading the resource
131    */
132   @NonNull
133   Format detectFormat(@NonNull InputSource source) throws IOException;
134 
135   /**
136    * Load data from the provided resource into a bound object.
137    * <p>
138    * This method will auto-detect the format of the provided resource.
139    * 
140    * @param <CLASS>
141    *          the type of the bound object to return
142    * @param url
143    *          the resource
144    * @return a bound object containing the loaded data
145    * @throws IOException
146    *           if an error occurred while reading the resource
147    * @throws URISyntaxException
148    *           if the provided {@code url} is malformed
149    * @see #detectFormat(URL)
150    */
151   @SuppressWarnings("unchecked")
152   @NonNull
153   default <CLASS> CLASS load(@NonNull URL url) throws IOException, URISyntaxException {
154     return (CLASS) loadAsNodeItem(url).getValue();
155   }
156 
157   /**
158    * Load data from the provided resource into a bound object.
159    * <p>
160    * This method will auto-detect the format of the provided resource.
161    * 
162    * @param <CLASS>
163    *          the type of the bound object to return
164    * @param path
165    *          the resource
166    * @return a bound object containing the loaded data
167    * @throws IOException
168    *           if an error occurred while reading the resource
169    * @see #detectFormat(File)
170    */
171   @SuppressWarnings("unchecked")
172   @NonNull
173   default <CLASS> CLASS load(@NonNull Path path) throws IOException {
174     return (CLASS) loadAsNodeItem(path).getValue();
175   }
176 
177   /**
178    * Load data from the provided resource into a bound object.
179    * <p>
180    * This method will auto-detect the format of the provided resource.
181    * 
182    * @param <CLASS>
183    *          the type of the bound object to return
184    * @param file
185    *          the resource
186    * @return a bound object containing the loaded data
187    * @throws IOException
188    *           if an error occurred while reading the resource
189    * @see #detectFormat(File)
190    */
191   @SuppressWarnings("unchecked")
192   @NonNull
193   default <CLASS> CLASS load(@NonNull File file) throws IOException {
194     return (CLASS) loadAsNodeItem(file).getValue();
195   }
196 
197   /**
198    * Load data from the provided resource into a bound object.
199    * <p>
200    * This method should auto-detect the format of the provided resource.
201    * <p>
202    * This method will not close the provided {@link InputStream}, since it does not own the stream.
203    * 
204    * @param <CLASS>
205    *          the type of the bound object to return
206    * @param is
207    *          the resource
208    * @param documentUri
209    *          the URI of the resource
210    * @return a bound object containing the loaded data
211    * @throws IOException
212    *           if an error occurred while reading the resource
213    * @see #detectFormat(InputStream)
214    */
215   @SuppressWarnings("unchecked")
216   @NonNull
217   default <CLASS> CLASS load(@NonNull InputStream is, @NonNull URI documentUri) throws IOException {
218     return (CLASS) loadAsNodeItem(is, documentUri).getValue();
219   }
220 
221   /**
222    * Load data from the provided resource into a bound object.
223    * <p>
224    * This method will auto-detect the format of the provided resource.
225    * <p>
226    * This method will not close any {@link InputStream} provided by the {@link InputSource}, since it
227    * does not own the stream.
228    * 
229    * @param <CLASS>
230    *          the type of the bound object to return
231    * @param source
232    *          information about how to access the resource
233    * @return a bound object containing the loaded data
234    * @throws IOException
235    *           if an error occurred while reading the resource
236    * @see #detectFormat(InputSource)
237    */
238   @SuppressWarnings("unchecked")
239   @NonNull
240   default <CLASS> CLASS load(@NonNull InputSource source) throws IOException {
241     return (CLASS) loadAsNodeItem(source).getValue();
242   }
243 
244   /**
245    * Load data from the specified resource into a bound object with the type of the specified Java
246    * class.
247    * 
248    * @param <CLASS>
249    *          the Java type to load data into
250    * @param clazz
251    *          the class for the java type
252    * @param path
253    *          the resource to load
254    * @return the loaded instance data
255    * @throws IOException
256    *           if an error occurred while loading the data in the specified file
257    */
258   @NonNull
259   default <CLASS> CLASS load(@NonNull Class<CLASS> clazz, @NonNull Path path) throws IOException {
260     try (InputStream is = Files.newInputStream(path)) {
261       return load(clazz, toInputSource(ObjectUtils.notNull(path.toUri())));
262     }
263   }
264 
265   /**
266    * Load data from the specified resource into a bound object with the type of the specified Java
267    * class.
268    * 
269    * @param <CLASS>
270    *          the Java type to load data into
271    * @param clazz
272    *          the class for the java type
273    * @param file
274    *          the resource to load
275    * @return the loaded instance data
276    * @throws IOException
277    *           if an error occurred while loading the data in the specified file
278    */
279   @NonNull
280   default <CLASS> CLASS load(@NonNull Class<CLASS> clazz, @NonNull File file) throws IOException {
281     return load(clazz, ObjectUtils.notNull(file.toPath()));
282   }
283 
284   /**
285    * Load data from the specified resource into a bound object with the type of the specified Java
286    * class.
287    * 
288    * @param <CLASS>
289    *          the Java type to load data into
290    * @param clazz
291    *          the class for the java type
292    * @param url
293    *          the resource to load
294    * @return the loaded instance data
295    * @throws IOException
296    *           if an error occurred while loading the data in the specified file
297    * @throws URISyntaxException
298    *           if the provided {@code url} is malformed
299    */
300   @NonNull
301   default <CLASS> CLASS load(@NonNull Class<CLASS> clazz, @NonNull URL url) throws IOException, URISyntaxException {
302     return load(clazz, toInputSource(ObjectUtils.notNull(url.toURI())));
303   }
304 
305   /**
306    * Load data from the specified resource into a bound object with the type of the specified Java
307    * class.
308    * <p>
309    * This method will not close the provided {@link InputStream}, since it does not own the stream.
310    * <p>
311    * Implementations of this method will do format detection. This process might leave the provided
312    * {@link InputStream} at a position beyond the last parsed location. If you want to avoid this
313    * possibility, use and implementation of {@link IDeserializer#deserialize(InputStream, URI)}
314    * instead, such as what is provided by
315    * {@link DefaultBindingContext#newDeserializer(Format, Class)}.
316    * 
317    * @param <CLASS>
318    *          the Java type to load data into
319    * @param clazz
320    *          the class for the java type
321    * @param source
322    *          information about how to access the resource
323    * @return the loaded instance data
324    * @throws IOException
325    *           if an error occurred while loading the data in the specified file
326    */
327   @NonNull
328   <CLASS> CLASS load(@NonNull Class<CLASS> clazz, @NonNull InputSource source) throws IOException;
329 
330   IDocumentNodeItem loadAsNodeItem(@NonNull Format format, @NonNull InputSource source) throws IOException;
331 }
