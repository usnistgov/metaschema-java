27  package gov.nist.secauto.metaschema.binding.io;
28  
29  import gov.nist.secauto.metaschema.binding.IBindingContext;
30  import gov.nist.secauto.metaschema.binding.model.IAssemblyClassBinding;
31  import gov.nist.secauto.metaschema.model.common.util.ObjectUtils;
32  
33  import org.jetbrains.annotations.NotNull;
34  
35  import java.util.Map;
36  
37  abstract class AbstractSerializationBase implements IMutableConfiguration {
38    @NotNull
39    private final IBindingContext bindingContext;
40    @NotNull
41    private final IAssemblyClassBinding classBinding;
42    @NotNull
43    private final IMutableConfiguration configuration;
44  
45    protected AbstractSerializationBase(IBindingContext bindingContext, IAssemblyClassBinding classBinding) {
46      this.bindingContext = ObjectUtils.requireNonNull(bindingContext, "bindingContext");
47      this.classBinding = ObjectUtils.requireNonNull(classBinding, "classBinding");
48      this.configuration = new DefaultMutableConfiguration();
49    }
50  
51    /**
52     * Retrieve the binding context associated with the serializer.
53     * 
54     * @return the binding context
55     */
56    @NotNull
57    protected IBindingContext getBindingContext() {
58      return bindingContext;
59    }
60  
61    /**
62     * Retrieve the bound class information associated with the assembly that the
63     * serializer/deserializer will write/read data from.
64     * 
65     * @return the class binding for the Metaschema assembly
66     */
67    @NotNull
68    protected IAssemblyClassBinding getClassBinding() {
69      return classBinding;
70    }
71  
72    /**
73     * Get the current configuration of the serializer/deserializer.
74     * 
75     * @return the configuration
76     */
77    @NotNull
78    protected IConfiguration getConfiguration() {
79      return configuration;
80    }
81  
82    @Override
83    public IMutableConfiguration enableFeature(Feature feature) {
84      return configuration.enableFeature(feature);
85    }
86  
87    @Override
88    public IMutableConfiguration disableFeature(Feature feature) {
89      return configuration.disableFeature(feature);
90    }
91  
92    @Override
93    public boolean isFeatureEnabled(Feature feature) {
94      return configuration.isFeatureEnabled(feature);
95    }
96  
97    @Override
98    public Map<@NotNull Feature, Boolean> getFeatureSettings() {
99      return configuration.getFeatureSettings();
100   }
101 }
