62    private static final String MARKUP_HTML_NAMESPACE = "http://www.w3.org/1999/xhtml";
63    private static final String MARKUP_HTML_PREFIX = "";

65    MarkupXmlStreamWriter newMarkupXmlStreamWriter(boolean handleBlockElements) {
66      return new MarkupXmlStreamWriter(MARKUP_HTML_NAMESPACE, handleBlockElements);
67    }
68  
69    XMLStreamWriter2 newXmlStreamWriter(StringWriter stringWriter) throws XMLStreamException {
70      XMLOutputFactory2 factory = (XMLOutputFactory2) XMLOutputFactory.newInstance();
71      assert factory instanceof WstxOutputFactory;
72      factory.setProperty(WstxOutputProperties.P_OUTPUT_VALIDATE_STRUCTURE, false);
73      XMLStreamWriter2 xmlStreamWriter = (XMLStreamWriter2) factory.createXMLStreamWriter(stringWriter);
74      NamespaceContext nsContext = MergedNsContext.construct(xmlStreamWriter.getNamespaceContext(),
75          List.of(NamespaceEventImpl.constructNamespace(null, MARKUP_HTML_PREFIX, MARKUP_HTML_NAMESPACE)));
76      xmlStreamWriter.setNamespaceContext(nsContext);
77      return xmlStreamWriter;
78    }

80    @Test
81    void markupLineFromMarkdownTest() {
82      String markdown = "Some \\**more* **text** and a param: {{ insert: param, insert }}.";
83  
84      MarkupLine ms = MarkupLine.fromMarkdown(markdown);
85      AstCollectingVisitor visitor = new AstCollectingVisitor();
86      Document document = ms.getDocument();
87  
88      assertNotNull(document);
89      visitor.collect(document);
90      // System.out.println("Markup AST");
91      // System.out.println("==========");
92      // System.out.println(visitor.getAst());
93      // System.out.println("HTML Output");
94      // System.out.println("===========");
95      // System.out.println(ms.toHtml());
96      // System.out.println("Markdown Output");
97      // System.out.println("===============");
98      // System.out.println(ms.toMarkdown());
99  
100     // Document[0, 49]
101     List<Node> documentChildren = CollectionUtil.toList(document.getChildren());
102     {
103       // Paragraph[0, 49]
104       // ensure there is a single paragraph
105       assertEquals(1, documentChildren.size());
106       Node paragraph = documentChildren.get(0);
107       assertTrue(paragraph instanceof Paragraph);
108 
109       List<Node> paragraphChildren = CollectionUtil.toList(paragraph.getChildren());
110       // TextBase[0, 7] chars:[0, 7, "Some \*"]
111       {
112         TextBase textBase = (TextBase) paragraphChildren.get(0);
113         List<Node> textBaseChildren = CollectionUtil.toList(textBase.getChildren());
114         // Text[0, 5] chars:[0, 5, "Some "]
115         {
116           Text text = (Text) textBaseChildren.get(0);
117           assertEquals("Some ", text.getChars().toString());
118         }
119         // EscapedCharacter[5, 7] textOpen:[5, 6, "\"] text:[6, 7, "*"]
120         {
121           EscapedCharacter text = (EscapedCharacter) textBaseChildren.get(1);
122           assertEquals("*", text.getText().toString());
123         }
124       }
125       // Emphasis[7, 13] textOpen:[7, 8, "*"] text:[8, 12, "more"] textClose:[12, 13, "*"]
126       {
127         Emphasis emphasis = (Emphasis) paragraphChildren.get(1);
128         List<Node> emphasisChildren = CollectionUtil.toList(emphasis.getChildren());
129         // Text[8, 12] chars:[8, 12, "more"]
130         {
131           Text text = (Text) emphasisChildren.get(0);
132           assertEquals("more", text.getChars().toString());
133         }
134       }
135       // Text[13, 14] chars:[13, 14, " "]
136       {
137         Text text = (Text) paragraphChildren.get(2);
138         assertEquals(" ", text.getChars().toString());
139       }
140       // StrongEmphasis[14, 22] textOpen:[14, 16, "**"] text:[16, 20, "text"] textClose:[20, 22, "**"]
141       {
142         StrongEmphasis strongEmphasis = (StrongEmphasis) paragraphChildren.get(3);
143         List<Node> strongEmphasisChildren = CollectionUtil.toList(strongEmphasis.getChildren());
144         // Text[16, 20] chars:[16, 20, "text"]
145         {
146           Text text = (Text) strongEmphasisChildren.get(0);
147           assertEquals("text", text.getChars().toString());
148         }
149 
150       }
151       // Text[22, 36] chars:[22, 36, " and … ram: "]
152       {
153         Text text = (Text) paragraphChildren.get(4);
154         assertEquals(" and a param: ", text.getChars().toString());
155       }
156       // InsertAnchorNode[0, 0] name:[39, 45, "insert"]
157       {
158         @SuppressWarnings("unused")
159         InsertAnchorNode insert = (InsertAnchorNode) paragraphChildren.get(5);
160       }
161       // Text[48, 49] chars:[48, 49, "."]
162       {
163         Text text = (Text) paragraphChildren.get(6);
164         assertEquals(".", text.getChars().toString());
165       }
166     }
167 
168     assertEquals(markdown, ms.toMarkdown());
169 
170     String html
171         = "<p>Some *<em>more</em> <strong>text</strong> and a param: <insert type=\"param\" id-ref=\"insert\" />.</p>";
172 
173     assertEquals(html, ms.toHtml());
174   }

319   @Test
320   void testEntityRoundTrip() throws XMLStreamException {
321     String markdown = "hijacked was used (e.g., the &lt;CTRL&gt; + &lt;ALT&gt; + &lt;DEL&gt; keys).";
322 
323     MarkupLine ms = MarkupLine.fromMarkdown(markdown);
324     AstCollectingVisitor visitor = new AstCollectingVisitor();
325     Document document = ms.getDocument();
326 
327     assertNotNull(document);
328     visitor.collect(document);
329     // System.out.println("Markup AST");
330     // System.out.println("==========");
331     // System.out.println(visitor.getAst());
332     // System.out.println("HTML Output");
333     // System.out.println("===========");
334     // System.out.println(ms.toHtml());
335     // System.out.println("Markdown Output");
336     // System.out.println("===============");
337     // System.out.println(ms.toMarkdown());
338 
339     assertEquals(markdown, ms.toMarkdown());
340 
341     String html = "<p>hijacked was used (e.g., the &lt;CTRL&gt; + &lt;ALT&gt; + &lt;DEL&gt; keys).</p>";
342     assertEquals(html, ms.toHtml().trim());
343 
344     MarkupXmlStreamWriter writer = newMarkupXmlStreamWriter(true);
345     StringWriter stringWriter = new StringWriter();
346     XMLStreamWriter2 xmlStreamWriter = newXmlStreamWriter(stringWriter);
347 
348     writer.visitChildren(ms.getDocument(), xmlStreamWriter);
349     xmlStreamWriter.close();
350 
351     assertEquals(html, stringWriter.toString());
352   }
