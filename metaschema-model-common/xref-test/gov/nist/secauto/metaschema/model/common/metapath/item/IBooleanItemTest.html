46  class IBooleanItemTest {
47    @Test
48    void testValueOf() {
49      Assertions.assertAll(
50          () -> assertEquals(IBooleanItem.valueOf(true), IBooleanItem.TRUE),
51          () -> assertEquals(IBooleanItem.valueOf(false), IBooleanItem.FALSE),
52          () -> assertEquals(IBooleanItem.valueOf(ObjectUtils.notNull(Boolean.TRUE)), IBooleanItem.TRUE),
53          () -> assertEquals(IBooleanItem.valueOf(ObjectUtils.notNull(Boolean.FALSE)), IBooleanItem.FALSE),
54          () -> assertEquals(IBooleanItem.valueOf("1"), IBooleanItem.TRUE, "1"),
55          () -> assertEquals(IBooleanItem.valueOf("0"), IBooleanItem.FALSE, "0"),
56          () -> assertEquals(IBooleanItem.valueOf(""), IBooleanItem.FALSE, ""),
57          () -> assertEquals(IBooleanItem.valueOf("true"), IBooleanItem.TRUE),
58          () -> assertEquals(IBooleanItem.valueOf("false"), IBooleanItem.FALSE));
59    }
60  
61    private static Stream<Arguments> provideValuesForCast() {
62      return Stream.of(
63          Arguments.of(IBooleanItem.TRUE, IBooleanItem.TRUE),
64          Arguments.of(IBooleanItem.FALSE, IBooleanItem.FALSE),
65          Arguments.of(integer(1), IBooleanItem.TRUE),
66          Arguments.of(integer(0), IBooleanItem.FALSE),
67          Arguments.of(decimal("1"), IBooleanItem.TRUE),
68          Arguments.of(decimal("0"), IBooleanItem.FALSE),
69          Arguments.of(string("1"), IBooleanItem.TRUE),
70          Arguments.of(string("654321"), IBooleanItem.TRUE),
71          Arguments.of(string("0"), IBooleanItem.FALSE),
72          Arguments.of(string(""), IBooleanItem.FALSE),
73          Arguments.of(string("true"), IBooleanItem.TRUE),
74          Arguments.of(string("false"), IBooleanItem.FALSE));
75    }
76  
77    @ParameterizedTest
78    @MethodSource("provideValuesForCast")
79    void testCast(@NonNull IAnyAtomicItem item, @NonNull IBooleanItem expected) {
80      IBooleanItem result = IBooleanItem.cast(item);
81      assertEquals(expected, result);
82    }

84  }
