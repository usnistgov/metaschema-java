027package gov.nist.secauto.metaschema.model.common.metapath.function.library;
028
029import gov.nist.secauto.metaschema.model.common.metapath.DynamicContext;
030import gov.nist.secauto.metaschema.model.common.metapath.ISequence;
031import gov.nist.secauto.metaschema.model.common.metapath.function.IArgument;
032import gov.nist.secauto.metaschema.model.common.metapath.function.IFunction;
033import gov.nist.secauto.metaschema.model.common.metapath.function.InvalidArgumentFunctionException;
034import gov.nist.secauto.metaschema.model.common.metapath.item.IAnyUriItem;
035import gov.nist.secauto.metaschema.model.common.metapath.item.IBooleanItem;
036import gov.nist.secauto.metaschema.model.common.metapath.item.IItem;
037import gov.nist.secauto.metaschema.model.common.metapath.item.INodeItem;
038import gov.nist.secauto.metaschema.model.common.metapath.item.INumericItem;
039import gov.nist.secauto.metaschema.model.common.metapath.item.IStringItem;
040import gov.nist.secauto.metaschema.model.common.metapath.item.IUntypedAtomicItem;
041import gov.nist.secauto.metaschema.model.common.util.ObjectUtils;
042
043import java.util.List;
044
045import edu.umd.cs.findbugs.annotations.NonNull;
046import edu.umd.cs.findbugs.annotations.Nullable;
047
048public final class FnBoolean {
049  @NonNull
050  static final IFunction SIGNATURE = IFunction.builder()
051      .name("boolean")
052      .deterministic()
053      .contextIndependent()
054      .focusIndependent()
055      .argument(IArgument.newBuilder()
056          .name("arg")
057          .type(IItem.class)
058          .zeroOrMore()
059          .build())
060      .returnType(IBooleanItem.class)
061      .returnOne()
062      .functionHandler(FnBoolean::execute)
063      .build();
064
065  private FnBoolean() {
066    // disable construction
067  }
068
069  @SuppressWarnings("unused")
070  @NonNull
071  private static ISequence<IBooleanItem> execute(@NonNull IFunction function,
072      @NonNull List<ISequence<?>> arguments,
073      @NonNull DynamicContext dynamicContext,
074      INodeItem focus) {
075
076    ISequence<?> items = ObjectUtils.requireNonNull(arguments.iterator().next());
077
078    IBooleanItem result = fnBoolean(items);
079    return ISequence.of(result);
080  }
081
082  /**
083   * Get the effective boolean value of the provided sequence.
084   * <p>
085   * Based on the XPath 3.1
086   * <a href="https://www.w3.org/TR/xpath-functions-31/#func-boolean">fn:boolean</a> function.
087   *
088   * @param sequence
089   *          the sequence to evaluate
090   * @return the effective boolean value of the sequence
091   */
092  @NonNull
093  public static IBooleanItem fnBoolean(@Nullable ISequence<?> sequence) {
094    IBooleanItem retval;
095    if (sequence == null) {
096      retval = IBooleanItem.FALSE;
097    } else {
098      retval = IBooleanItem.valueOf(fnBooleanAsPrimitive(sequence));
099    }
100    return retval;
101  }
102
103  /**
104   * A helper method that gets the effective boolean value of the provided sequence based on
105   * <a href="https://www.w3.org/TR/xpath-31/#id-ebv">XPath 3.1</a>.
106   *
107   * @param sequence
108   *          the sequence to evaluate
109   * @return the effective boolean value
110   */
111  public static boolean fnBooleanAsPrimitive(@NonNull ISequence<?> sequence) {
112    boolean retval = false;
113    if (!sequence.isEmpty()) {
114      List<? extends IItem> items = sequence.asList();
115      IItem first = ObjectUtils.notNull(items.iterator().next());
116      if (first instanceof INodeItem) {
117        retval = true;
118      } else if (items.size() == 1) {
119        retval = fnBooleanAsPrimitive(first);
120      }
121    }
122    return retval;
123  }
124
125  /**
126   * A helper method that gets the effective boolean value of the provided item based on
127   * <a href="https://www.w3.org/TR/xpath-31/#id-ebv">XPath 3.1</a>.
128   *
129   * @param item
130   *          the item to evaluate
131   * @return the effective boolean value
132   */
133  public static boolean fnBooleanAsPrimitive(@NonNull IItem item) {
134    boolean retval;
135    if (item instanceof IBooleanItem) {
136      retval = ((IBooleanItem) item).toBoolean();
137    } else if (item instanceof IStringItem) {
138      String string = ((IStringItem) item).asString();
139      retval = !string.isBlank();
140    } else if (item instanceof INumericItem) {
141      retval = ((INumericItem) item).toEffectiveBoolean();
142    } else if (item instanceof IUntypedAtomicItem) {
143      String string = ((IUntypedAtomicItem) item).asString();
144      retval = !string.isBlank();
145    } else if (item instanceof IAnyUriItem) {
146      String string = ((IAnyUriItem) item).asString();
147      retval = !string.isBlank();
148    } else {
149      throw new InvalidArgumentFunctionException(InvalidArgumentFunctionException.INVALID_ARGUMENT_TYPE,
150          String.format("Invalid argument type '%s'", item.getClass().getName()));
151    }
152    return retval;
153  }
154}
