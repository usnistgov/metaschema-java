027package gov.nist.secauto.metaschema.model.common.constraint;
028
029import gov.nist.secauto.metaschema.model.common.datatype.markup.MarkupLine;
030import gov.nist.secauto.metaschema.model.common.datatype.markup.MarkupMultiline;
031import gov.nist.secauto.metaschema.model.common.metapath.MetapathExpression;
032import gov.nist.secauto.metaschema.model.common.util.CollectionUtil;
033import gov.nist.secauto.metaschema.model.common.util.ObjectUtils;
034
035import java.util.LinkedHashMap;
036import java.util.LinkedHashSet;
037import java.util.Map;
038import java.util.Objects;
039import java.util.Set;
040
041import javax.xml.namespace.QName;
042
043import edu.umd.cs.findbugs.annotations.NonNull;
044import edu.umd.cs.findbugs.annotations.Nullable;
045
046public abstract class AbstractConstraint implements IConstraint { // NOPMD - intentional data class
047  @Nullable
048  private final String id;
049  @Nullable
050  private final String formalName;
051  @Nullable
052  private final MarkupLine description;
053  @NonNull
054  private final ISource source;
055  @NonNull
056  private final Level level;
057  @NonNull
058  private final MetapathExpression target;
059  @Nullable
060  private final MarkupMultiline remarks;
061  @NonNull
062  private final Map<QName, Set<String>> properties;
063
064  /**
065   * Construct a new Metaschema constraint.
066   *
067   * @param id
068   *          the optional identifier for the constraint
069   * @param formalName
070   *          the constraint's formal name or {@code null} if not provided
071   * @param description
072   *          the constraint's semantic description or {@code null} if not provided
073   * @param source
074   *          information about the constraint source
075   * @param level
076   *          the significance of a violation of this constraint
077   * @param target
078   *          the Metapath expression identifying the nodes the constraint targets
079   * @param properties
080   *          a collection of associated properties
081   * @param remarks
082   *          optional remarks describing the intent of the constraint
083   */
084  protected AbstractConstraint(
085      @Nullable String id,
086      @Nullable String formalName,
087      @Nullable MarkupLine description,
088      @NonNull ISource source,
089      @NonNull Level level,
090      @NonNull MetapathExpression target,
091      @NonNull Map<QName, Set<String>> properties,
092      @Nullable MarkupMultiline remarks) {
093    Objects.requireNonNull(target);
094    this.id = id;
095    this.formalName = formalName;
096    this.description = description;
097    this.source = source;
098    this.level = ObjectUtils.requireNonNull(level, "level");
099    this.target = ObjectUtils.requireNonNull(target, "target");
100    this.properties = properties;
101    this.remarks = remarks;
102  }
103
104  @Override
105  public String getId() {
106    return id;
107  }
108
109  @Override
110  public MarkupLine getDescription() {
111    return description;
112  }
113
114  @Override
115  public String getFormalName() {
116    return formalName;
117  }
118
119  @Override
120  public ISource getSource() {
121    return source;
122  }
123
124  @Override
125  @NonNull
126  public Level getLevel() {
127    return level;
128  }
129
130  @Override
131  public MetapathExpression getTarget() {
132    return target;
133  }
134
135  @Override
136  public Map<QName, Set<String>> getProperties() {
137    return CollectionUtil.unmodifiableMap(properties);
138  }
139
140  @Override
141  public MarkupMultiline getRemarks() {
142    return remarks;
143  }
144
145  public abstract static class AbstractConstraintBuilder<T extends AbstractConstraintBuilder<T,
146      R>, R extends AbstractConstraint> {
147    private String id;
148    private String formalName;
149    private MarkupLine description;
150    private ISource source;
151    @NonNull
152    private Level level = IConstraint.DEFAULT_LEVEL;
153    @NonNull
154    private MetapathExpression target = IConstraint.DEFAULT_TARGET;
155    @NonNull
156    private Map<QName, Set<String>> properties = new LinkedHashMap<>(); // NOPMD not thread safe
157    private MarkupMultiline remarks;
158
159    protected abstract T getThis();
160
161    public T identifier(@NonNull String id) {
162      this.id = id;
163      return getThis();
164    }
165
166    public T formalName(@NonNull String name) {
167      this.formalName = name;
168      return getThis();
169    }
170
171    public T description(@NonNull MarkupLine description) {
172      this.description = description;
173      return getThis();
174    }
175
176    public T source(@NonNull ISource source) {
177      this.source = source;
178      return getThis();
179    }
180
181    public T level(@NonNull Level level) {
182      this.level = level;
183      return getThis();
184    }
185
186    public T target(@NonNull MetapathExpression target) {
187      this.target = target;
188      return getThis();
189    }
190
191    public T properties(@NonNull Map<QName, Set<String>> properties) {
192      this.properties = properties;
193      return getThis();
194    }
195
196    public T property(@NonNull QName name, @NonNull String value) {
197      return property(name, CollectionUtil.singleton(value));
198    }
199
200    public T property(@NonNull QName name, @NonNull Set<String> newValues) {
201      Set<String> existingValues = properties.get(name);
202      if (existingValues == null) {
203        existingValues = new LinkedHashSet<>();
204        properties.put(name, existingValues);
205      }
206
207      existingValues.addAll(newValues);
208      return getThis();
209    }
210
211    public T remarks(@NonNull MarkupMultiline remarks) {
212      this.remarks = remarks;
213      return getThis();
214    }
215
216    protected void validate() {
217      ObjectUtils.requireNonNull(getSource());
218    }
219
220    @NonNull
221    protected abstract R newInstance();
222
223    @NonNull
224    public R build() {
225      validate();
226      return newInstance();
227    }
228
229    protected String getId() {
230      return id;
231    }
232
233    protected String getFormalName() {
234      return formalName;
235    }
236
237    protected MarkupLine getDescription() {
238      return description;
239    }
240
241    protected ISource getSource() {
242      return source;
243    }
244
245    @NonNull
246    protected Level getLevel() {
247      return level;
248    }
249
250    @NonNull
251    protected MetapathExpression getTarget() {
252      return target;
253    }
254
255    @NonNull
256    protected Map<QName, Set<String>> getProperties() {
257      return properties;
258    }
259
260    protected MarkupMultiline getRemarks() {
261      return remarks;
262    }
263  }
264}
