164  /**
165   * Converts arguments in an attempt to align with the function's signature.
166   *
167   * @param function
168   *          the function
169   * @param parameters
170   *          the argument parameters
171   * @return the converted argument list
172   */
173  @NonNull
174  public static List<ISequence<?>> convertArguments(
175      @NonNull IFunction function,
176      @NonNull List<ISequence<?>> parameters) {
177    @NonNull List<ISequence<?>> retval = new ArrayList<>(parameters.size());
178
179    Iterator<IArgument> argumentIterator = function.getArguments().iterator();
180    Iterator<ISequence<?>> parametersIterator = parameters.iterator();
181
182    IArgument argument = null;
183    while (parametersIterator.hasNext()) {
184      if (argumentIterator.hasNext()) {
185        argument = argumentIterator.next();
186      } else if (!function.isArityUnbounded()) {
187        throw new InvalidTypeMetapathException(
188            null,
189            String.format("argument signature doesn't match '%s'", function.toSignature()));
190      }
191
192      assert argument != null;
193
194      ISequence<?> parameter = parametersIterator.next();
195
196      int size = parameter.size();
197      Occurrence occurrence = argument.getSequenceType().getOccurrence();
198      switch (occurrence) {
199      case ONE: {
200        if (size != 1) {
201          throw new InvalidTypeMetapathException(
202              null,
203              String.format("a sequence of one expected, but found '%d'", size));
204        }
205
206        IItem item = FunctionUtils.getFirstItem(parameter, true);
207        parameter = item == null ? ISequence.empty() : ISequence.of(item);
208        break;
209      }
210      case ZERO_OR_ONE: {
211        if (size > 1) {
212          throw new InvalidTypeMetapathException(
213              null,
214              String.format("a sequence of zero or one expected, but found '%d'", size));
215        }
216
217        IItem item = FunctionUtils.getFirstItem(parameter, false);
218        parameter = item == null ? ISequence.empty() : ISequence.of(item);
219        break;
220      }
221      case ONE_OR_MORE:
222        if (size < 1) {
223          throw new InvalidTypeMetapathException(
224              null,
225              String.format("a sequence of zero or more expected, but found '%d'", size));
226        }
227        break;
228      case ZERO:
229        if (size != 0) {
230          throw new InvalidTypeMetapathException(
231              null,
232              String.format("an empty sequence expected, but found '%d'", size));
233        }
234        break;
235      case ZERO_OR_MORE:
236      default:
237        // do nothing
238      }
239
240      Class<? extends IItem> argumentClass = argument.getSequenceType().getType();
241
242      // apply function conversion and type promotion to the parameter
243      parameter = convertSequence(argument, parameter);
244
245      // check resulting values
246      for (IItem item : parameter.asList()) {
247        Class<? extends IItem> itemClass = item.getClass();
248        if (!argumentClass.isAssignableFrom(itemClass)) {
249          throw new InvalidTypeMetapathException(
250              item,
251              String.format("The type '%s' is not a subtype of '%s'", itemClass.getName(), argumentClass.getName()));
252        }
253      }
254
255      retval.add(parameter);
256    }
257    return retval;
258  }

260  /**
261   * Based on XPath 3.1 <a href="https://www.w3.org/TR/xpath-31/#dt-function-conversion">function
262   * conversion</a> rules.
263   *
264   * @param argument
265   *          the function argument signature details
266   * @param sequence
267   *          the sequence to convert
268   * @return the converted sequence
269   */
270  @NonNull
271  protected static ISequence<?> convertSequence(@NonNull IArgument argument, @NonNull ISequence<?> sequence) {
272    @NonNull ISequence<?> retval;
273    if (sequence.isEmpty()) {
274      retval = ISequence.empty();
275    } else {
276      ISequenceType requiredSequenceType = argument.getSequenceType();
277      Class<? extends IItem> requiredSequenceTypeClass = requiredSequenceType.getType();
278
279      List<IItem> result = new ArrayList<>(sequence.size());
280
281      boolean atomize = IAnyAtomicItem.class.isAssignableFrom(requiredSequenceTypeClass);
282
283      for (IItem item : sequence.asList()) {
284        assert item != null;
285        if (atomize) {
286          item = FnData.fnDataItem(item); // NOPMD - intentional
287
288          // if (IUntypedAtomicItem.class.isInstance(item)) { // NOPMD
289          // // TODO: apply cast to atomic type
290          // }
291
292          // promote URIs to strings if a string is required
293          if (IStringItem.class.equals(requiredSequenceTypeClass) && IAnyUriItem.class.isInstance(item)) {
294            item = IStringItem.cast((IAnyUriItem) item); // NOPMD - intentional
295          }
296        }
297
298        // item = requiredSequenceType.
299        if (!requiredSequenceTypeClass.isInstance(item)) {
300          throw new InvalidTypeMetapathException(
301              item,
302              String.format("The type '%s' is not a subtype of '%s'", item.getClass().getName(),
303                  requiredSequenceTypeClass.getName()));
304        }
305        result.add(item);
306      }
307      retval = ISequence.of(result);
308    }
309    return retval;
310  }

312  @Override
313  public ISequence<?> execute(@NonNull List<ISequence<?>> arguments, @NonNull DynamicContext dynamicContext,
314      INodeContext focus) {
315    try {
316      List<ISequence<?>> convertedArguments = convertArguments(this, arguments);
317
318      CallingContext callingContext = null;
319      ISequence<?> result = null;
320      if (isDeterministic()) {
321        // check cache
322        callingContext = newCallingContext(arguments, focus);
323        // attempt to get the result from the cache
324        result = dynamicContext.getCachedResult(callingContext);
325      }
326
327      if (result == null) {
328        // logger.info(String.format("Executing function '%s' with arguments '%s'.", toSignature(),
329        // convertedArguments.toString()));
330        result = handler.execute(this, convertedArguments, dynamicContext, focus.getNodeItem());
331
332        if (callingContext != null) {
333          // add result to cache
334          dynamicContext.cacheResult(callingContext, result);
335        }
336      }
337
338      // logger.info(String.format("Executed function '%s' with arguments '%s' producing result '%s'",
339      // toSignature(), convertedArguments.toString(), result.asList().toString()));
340      return result;
341    } catch (MetapathException ex) {
342      throw new MetapathException(String.format("Unable to execute function '%s'", toSignature()), ex);
343    }
344  }

351  @Override
352  public boolean equals(Object obj) {
353    if (this == obj) {
354      return true; // NOPMD - readability
355    }
356    if (obj == null) {
357      return false; // NOPMD - readability
358    }
359    if (getClass() != obj.getClass()) {
360      return false; // NOPMD - readability
361    }
362    DefaultFunction other = (DefaultFunction) obj;
363    return Objects.equals(getArguments(), other.getArguments()) && Objects.equals(handler, other.handler)
364        && Objects.equals(getName(), other.getName()) && Objects.equals(properties, other.properties)
365        && Objects.equals(result, other.result);
366  }

373  @Override
374  public String toSignature() {
375    StringBuilder builder = new StringBuilder()
376        .append(getName()) // name
377        .append('('); // arguments
378
379    List<IArgument> arguments = getArguments();
380    if (arguments.isEmpty()) {
381      builder.append("()");
382    } else {
383      builder.append(arguments.stream().map(argument -> argument.toSignature()).collect(Collectors.joining(",")));
384
385      if (isArityUnbounded()) {
386        builder.append(", ...");
387      }
388    }
389
390    builder.append(") as ")
391        .append(getResult().toSignature());// return type
392
393    return builder.toString();
394  }

396  /**
397   * Set up the execution context for this function.
398   *
399   * @param arguments
400   *          the function arguments
401   * @param focus
402   *          the current node context
403   * @return the calling context
404   */
405  @NonNull
406  public CallingContext newCallingContext(@NonNull List<ISequence<?>> arguments, @NonNull INodeContext focus) {
407    return new CallingContext(arguments, focus);
408  }
409
410  public final class CallingContext {
411    @Nullable
412    private final INodeItem contextNodeItem;
413    @NonNull
414    private final List<ISequence<?>> arguments;
415
416    private CallingContext(@NonNull List<ISequence<?>> arguments, @NonNull INodeContext focus) {
417      if (isFocusDepenent()) {
418        contextNodeItem = focus.getNodeItem();
419      } else {
420        contextNodeItem = null;
421      }
422      this.arguments = arguments;
423    }
424
425    /**
426     * Get the function instance associated with the calling context.
427     *
428     * @return the function instance
429     */
430    @NonNull
431    public DefaultFunction getFunction() {
432      return DefaultFunction.this;
433    }
434
435    /**
436     * Get the node item focus associated with the calling context.
437     *
438     * @return the function instance
439     */
440    @Nullable
441    public INodeItem getContextNodeItem() {
442      return contextNodeItem;
443    }
444
445    /**
446     * Get the arguments associated with the calling context.
447     *
448     * @return the arguments
449     */
450    @NonNull
451    public List<ISequence<?>> getArguments() {
452      return arguments;
453    }
454
455    @Override
456    public int hashCode() {
457      final int prime = 31;
458      int result = 1;
459      result = prime * result + getFunction().hashCode();
460      result = prime * result + Objects.hash(contextNodeItem, arguments);
461      return result;
462    }
463
464    @Override
465    public boolean equals(Object obj) {
466      if (this == obj) {
467        return true; // NOPMD - readability
468      }
469      if (obj == null) {
470        return false; // NOPMD - readability
471      }
472      if (getClass() != obj.getClass()) {
473        return false; // NOPMD - readability
474      }
475      CallingContext other = (CallingContext) obj;
476      if (!getFunction().equals(other.getFunction())) {
477        return false; // NOPMD - readability
478      }
479      return Objects.equals(arguments, other.arguments) && Objects.equals(contextNodeItem, other.contextNodeItem);
480    }
481  }
