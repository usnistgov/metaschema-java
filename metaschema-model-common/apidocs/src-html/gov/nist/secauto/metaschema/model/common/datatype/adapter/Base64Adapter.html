027package gov.nist.secauto.metaschema.model.common.datatype.adapter;
028
029import com.fasterxml.jackson.databind.jsonFormatVisitors.JsonFormatTypes;
030
031import gov.nist.secauto.metaschema.model.common.datatype.AbstractDataTypeAdapter;
032import gov.nist.secauto.metaschema.model.common.metapath.item.IBase64BinaryItem;
033import gov.nist.secauto.metaschema.model.common.util.ObjectUtils;
034
035import java.nio.ByteBuffer;
036import java.util.Base64;
037import java.util.List;
038
039import edu.umd.cs.findbugs.annotations.NonNull;
040
041public class Base64Adapter
042    extends AbstractDataTypeAdapter<ByteBuffer, IBase64BinaryItem> {
043  @NonNull
044  private static final List<String> NAMES = ObjectUtils.notNull(
045      List.of(
046          "base64",
047          // for backwards compatibility with original type name
048          "base64Binary"));
049
050  Base64Adapter() {
051    super(ByteBuffer.class);
052  }
053
054  @Override
055  public List<String> getNames() {
056    return NAMES;
057  }
058
059  @Override
060  public JsonFormatTypes getJsonRawType() {
061    return JsonFormatTypes.STRING;
062  }
063
064  @SuppressWarnings("null")
065  @Override
066  public ByteBuffer parse(String value) {
067    Base64.Decoder decoder = Base64.getDecoder();
068    byte[] result = decoder.decode(value);
069    return ByteBuffer.wrap(result);
070  }
071
072  @Override
073  public ByteBuffer copy(Object obj) {
074    ByteBuffer buffer = (ByteBuffer) obj;
075    final ByteBuffer clone
076        = buffer.isDirect() ? ByteBuffer.allocateDirect(buffer.capacity()) : ByteBuffer.allocate(buffer.capacity());
077    final ByteBuffer readOnlyCopy = buffer.asReadOnlyBuffer();
078    readOnlyCopy.flip();
079    clone.put(readOnlyCopy);
080    return clone;
081  }
082
083  @SuppressWarnings("null")
084  @Override
085  public String asString(Object value) {
086    Base64.Encoder encoder = Base64.getEncoder();
087    return encoder.encodeToString(((ByteBuffer) value).array());
088  }
089
090  @Override
091  public Class<IBase64BinaryItem> getItemClass() {
092    return IBase64BinaryItem.class;
093  }
094
095  @Override
096  public IBase64BinaryItem newItem(Object value) {
097    ByteBuffer item = toValue(value);
098    return IBase64BinaryItem.valueOf(item);
099  }
100
101}
