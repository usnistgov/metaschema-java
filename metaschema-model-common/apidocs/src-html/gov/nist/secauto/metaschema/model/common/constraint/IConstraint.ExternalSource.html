046/**
047 * Represents a rule constraining the model of a Metaschema assembly, field or flag. Provides a
048 * common interface for all constraint definitions.
049 */
050public interface IConstraint {
051  /**
052   * The degree to which a constraint violation is significant.
053   * <p>
054   * These values are ordered from least significant to most significant.
055   */
056  enum Level {
057    /**
058     * A violation of the constraint represents a point of interest.
059     */
060    INFORMATIONAL,
061    /**
062     * A violation of the constraint represents a potential issue with the content.
063     */
064    WARNING,
065    /**
066     * A violation of the constraint represents a fault in the content. This may include issues around
067     * compatibility, integrity, consistency, etc.
068     */
069    ERROR,
070    /**
071     * A violation of the constraint represents a serious fault in the content that will prevent typical
072     * use of the content.
073     */
074    CRITICAL;
075  }
076
077  /**
078   * The default level to use if no level is provided.
079   */
080  @NonNull
081  Level DEFAULT_LEVEL = Level.ERROR;
082  /**
083   * The default target Metapath expression to use if no target is provided.
084   */
085  @NonNull
086  MetapathExpression DEFAULT_TARGET = MetapathExpression.CONTEXT_NODE;
087  /**
088   * The default target Metapath expression to use if no target is provided.
089   */
090  @NonNull
091  String DEFAULT_TARGET_METAPATH = ".";
092
093  /**
094   * Retrieve the unique identifier for the constraint.
095   *
096   * @return the identifier or {@code null} if no identifier is defined
097   */
098  @Nullable
099  String getId();
100
101  @Nullable
102  MarkupLine getDescription();
103
104  @Nullable
105  String getFormalName();
106
107  /**
108   * Get information about the source of the constraint.
109   *
110   * @return the source information
111   */
112  @NonNull
113  ISource getSource();
114
115  /**
116   * The significance of a violation of this constraint.
117   *
118   * @return the level
119   */
120  @NonNull
121  Level getLevel();
122
123  @NonNull
124  Map<QName, Set<String>> getProperties();
125
126  /**
127   * Retrieve the Metapath expression to use to query the targets of the constraint.
128   *
129   * @return a Metapath expression
130   */
131  @NonNull
132  MetapathExpression getTarget();
133
134  /**
135   * Based on the provided {@code contextNodeItem}, find all nodes matching the target expression.
136   *
137   * @param contextNodeItem
138   *          the node item to evaluate the target expression against
139   * @return the matching nodes as a sequence
140   * @see #getTarget()
141   */
142  @NonNull
143  default ISequence<? extends IDefinitionNodeItem> matchTargets(@NonNull IDefinitionNodeItem contextNodeItem) {
144    return getTarget().evaluate(contextNodeItem);
145  }
146
147  /**
148   * Based on the provided {@code contextNodeItem}, find all nodes matching the target expression.
149   *
150   * @param contextNodeItem
151   *          the node item to evaluate the target expression against
152   * @param dynamicContext
153   *          the Metapath evaluation context to use
154   * @return the matching nodes as a sequence
155   * @see #getTarget()
156   */
157  @NonNull
158  default ISequence<? extends IDefinitionNodeItem> matchTargets(@NonNull IDefinitionNodeItem contextNodeItem,
159      @NonNull DynamicContext dynamicContext) {
160    return getTarget().evaluate(contextNodeItem, dynamicContext);
161  }
162
163  /**
164   * Retrieve the remarks associated with the constraint.
165   *
166   * @return the remarks or {@code null} if no remarks are defined
167   */
168  MarkupMultiline getRemarks();
169
170  <T, R> R accept(@NonNull IConstraintVisitor<T, R> visitor, T state);
171
172  interface ISource {
173    enum SourceType {
174      /**
175       * A constraint embedded in a model.
176       */
177      MODEL,
178      /**
179       * A constraint defined externally from a model.
180       */
181      EXTERNAL;
182    }
183
184    @NonNull
185    SourceType getSourceType();
186
187    @Nullable
188    URI getSource();
189  }
190
191  final class InternalModelSource implements ISource {
192    @NonNull
193    private static final ISource SINGLETON = new InternalModelSource();
194
195    @NonNull
196    public static ISource instance() {
197      return SINGLETON;
198    }
199
200    private InternalModelSource() {
201      // reduce visibility
202    }
203
204    @Override
205    public SourceType getSourceType() {
206      return SourceType.MODEL;
207    }
208
209    @Override
210    public URI getSource() {
211      // always null
212      return null;
213    }
214  }
215
216  final class ExternalModelSource implements IConstraint.ISource {
217    @NonNull
218    private static final Map<URI, ExternalModelSource> sources = new HashMap<>(); // NOPMD - intentional
219    @NonNull
220    private final URI modelUri;
221
222    @NonNull
223    public static ISource instance(@NonNull URI location) {
224      ISource retval;
225      synchronized (sources) {
226        retval = sources.get(location);
227        if (retval == null) {
228          retval = new ExternalModelSource(location);
229        }
230      }
231      return retval;
232    }
233
234    private ExternalModelSource(@NonNull URI modelSource) {
235      this.modelUri = modelSource;
236    }
237
238    @Override
239    public SourceType getSourceType() {
240      return SourceType.MODEL;
241    }
242
243    @NonNull
244    @Override
245    public URI getSource() {
246      return modelUri;
247    }
248  }
249
250  final class ExternalSource implements IConstraint.ISource {
251    @NonNull
252    private static final Map<URI, ExternalSource> sources = new HashMap<>(); // NOPMD - intentional
253
254    @NonNull
255    private final URI modelUri;
256
257    @NonNull
258    public static ISource instance(@NonNull URI location) {
259      ISource retval;
260      synchronized (sources) {
261        retval = sources.get(location);
262        if (retval == null) {
263          retval = new ExternalModelSource(location);
264        }
265      }
266      return retval;
267    }
268
269    private ExternalSource(@NonNull URI modelSource) {
270      this.modelUri = modelSource;
271    }
272
273    @Override
274    public SourceType getSourceType() {
275      return SourceType.EXTERNAL;
276    }
277
278    @NonNull
279    @Override
280    public URI getSource() {
281      return modelUri;
282    }
283  }
