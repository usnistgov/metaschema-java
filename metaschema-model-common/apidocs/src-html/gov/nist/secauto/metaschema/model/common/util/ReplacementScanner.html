027package gov.nist.secauto.metaschema.model.common.util;
028
029import java.util.function.Function;
030import java.util.regex.Matcher;
031import java.util.regex.Pattern;
032
033import edu.umd.cs.findbugs.annotations.NonNull;
034
035public final class ReplacementScanner {
036  private ReplacementScanner() {
037    // disable construction
038  }
039
040  /**
041   * Search for instances of {@code pattern} in {@code text}. Replace each matching occurrence using
042   * the {@code replacementFunction}.
043   *
044   * @param text
045   *          the text to search
046   * @param pattern
047   *          the pattern to search for
048   * @param replacementFunction
049   *          a function that will provided the replacement text
050   * @return the resulting text after replacing matching occurrences in {@code text}
051   */
052  public static CharSequence replaceTokens(@NonNull CharSequence text, @NonNull Pattern pattern,
053      Function<Matcher, CharSequence> replacementFunction) {
054    int lastIndex = 0;
055    StringBuilder retval = new StringBuilder();
056    Matcher matcher = pattern.matcher(text);
057    while (matcher.find()) {
058      retval.append(text, lastIndex, matcher.start())
059          .append(replacementFunction.apply(matcher));
060
061      lastIndex = matcher.end();
062    }
063    if (lastIndex < text.length()) {
064      retval.append(text, lastIndex, text.length());
065    }
066    return retval;
067  }
068}
