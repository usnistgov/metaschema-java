033import gov.nist.secauto.metaschema.model.common.metapath.function.InvalidValueForCastFunctionException;
034import gov.nist.secauto.metaschema.model.common.metapath.item.IAnyAtomicItem;

036

038import org.codehaus.stax2.XMLStreamWriter2;
039import org.codehaus.stax2.evt.XMLEventFactory2;
040
041import java.io.IOException;
042import java.lang.reflect.Field;
043import java.util.List;
044import java.util.function.Supplier;
045
046import javax.xml.namespace.QName;
047import javax.xml.stream.XMLEventWriter;
048import javax.xml.stream.XMLStreamException;

052import edu.umd.cs.findbugs.annotations.NonNull;
053
054public interface IDataTypeAdapter<TYPE> {
055
056  /**
057   * Get the metaschema type names associated with this adapter. This name must be unique with respect
058   * to all other metaschema types.
059   * <p>
060   * At least one name must be provided, with the first name being the most preferred name.
061   *
062   * @return the name
063   */
064  @NonNull
065  List<String> getNames();
066
067  /**
068   * The JSON primative type of the data type.
069   *
070   * @return the JSON data type
071   */
072  JsonFormatTypes getJsonRawType();
073
074  /**
075   * Get the most preferred name for this data type.
076   *
077   * @return the name
078   */
079  @NonNull
080  default String getPreferredName() {
081    return ObjectUtils.notNull(getNames().iterator().next());
082  }
083
084  /**
085   * Get the Java class supported by this adapter.
086   *
087   * @return the Java class
088   */
089  @NonNull
090  Class<TYPE> getJavaClass();
091
092  /**
093   * Casts the provided value to the type associated with this adapter.
094   *
095   * @param value
096   *          a value of the provided type
097   * @return the typed value
098   */
099  @NonNull
100  TYPE toValue(@NonNull Object value);
101
102  /**
103   * Gets the value as a string suitable for writing as text. This is intended for data types that
104   * have a simple string-based structure in XML and JSON, such as for XML attributes or JSON keys. An
105   * adapter for a complex data structures that consist of XML elements will throw an
106   * {@link UnsupportedOperationException} when this is called.
107   *
108   * @param value
109   *          the data to formatted as a string
110   * @return a string
111   * @throws UnsupportedOperationException
112   *           if the data type cannot be represented as a string
113   */
114  @NonNull
115  String asString(@NonNull Object value);
116
117  /**
118   * Create a copy of the provided value.
119   *
120   * @param obj
121   *          the value to copy
122   * @return the copy
123   */
124  @NonNull
125  TYPE copy(@NonNull Object obj);
126
127  /**
128   * Determines if the data type is an atomic, scalar value. Complex structures such as Markup are not
129   * considered atomic.
130   *
131   * @return {@code true} if the data type is an atomic scalar value, or {@code false} otherwise
132   */
133  default boolean isAtomic() {
134    return true;
135  }
136
137  /**
138   * Get the java type of the associated item.
139   *
140   * @return the java associated item type
141   */
142  @NonNull
143  Class<? extends IAnyAtomicItem> getItemClass();
144
145  /**
146   * Construct a new item of this type using the provided value.
147   *
148   * @param value
149   *          the item's value
150   * @return a new item
151   */
152  // TODO: markup types are not atomic values. Figure out a better base type (i.e., IValuedItem)
153  @NonNull
154  IAnyAtomicItem newItem(@NonNull Object value);
155
156  /**
157   * Cast the provided item to an item of this type, if possible.
158   *
159   * @param item
160   *          the atomic item to cast
161   * @return an atomic item of this type
162   * @throws InvalidValueForCastFunctionException
163   *           if the provided item type cannot be cast to this item type
164   */
165  @NonNull
166  IAnyAtomicItem cast(IAnyAtomicItem item);
167
168  /**
169   * Indicates if the adapter will parse the {@link XMLEvent#START_ELEMENT} before parsing the value
170   * data.
171   *
172   * @return {@code true} if the adapter requires the start element for parsing, or {@code false}
173   *         otherwise
174   */
175  // TODO; implement or remove this
176  boolean isParsingStartElement();
177
178  /**
179   * Determines if adapter can parse the next element. The next element's {@link QName} is provided
180   * for this purpose.
181   * <p>
182   * This will be called when the parser encounter's an element it does not recognize. This gives the
183   * adapter a chance to request parsing of the data.
184   *
185   * @param nextElementQName
186   *          the next element's namespace-qualified name
187   * @return {@code true} if the adapter will parse the element, or {@code false} otherwise
188   */
189  // TODO: implement this
190  boolean canHandleQName(@NonNull QName nextElementQName);
191
192  /**
193   * Parses a provided string. Used to parse XML attributes, simple XML character data, and JSON/YAML
194   * property values.
195   *
196   * @param value
197   *          the string value to parse
198   * @return the parsed data as the adapter's type
199   * @throws IllegalArgumentException
200   *           if the data is not valid to the data type
201   */
202  @NonNull
203  TYPE parse(@NonNull String value);
204
205  /**
206   * This method is expected to parse content starting at the next event. Parsing will continue until
207   * the next event represents content that is not handled by this adapter. This means the event
208   * stream should be positioned after any {@link XMLEvent#END_ELEMENT} that corresponds to an
209   * {@link XMLEvent#START_ELEMENT} parsed by this adapter.
210   * <p>
211   * If {@link #isParsingStartElement()} returns {@code true}, then the first event to parse will be
212   * the {@link XMLEvent#START_ELEMENT} for the element that contains the value data, then the value
213   * data. If this is the case, this method must also parse the corresponding
214   * {@link XMLEvent#END_ELEMENT}. Otherwise, the first event to parse will be the value data.
215   * <p>
216   * The value data is expected to be parsed completely, leaving the event stream on a peeked event
217   * corresponding to content that is not handled by this method.
218   *
219   * @param eventReader
220   *          the XML parser used to read the parsed value
221   * @return the parsed value
222   * @throws IOException
223   *           if a parsing error occurs
224   */
225  @NonNull
226  TYPE parse(@NonNull XMLEventReader2 eventReader) throws IOException;
227
228  /**
229   * Parses a JSON property value.
230   *
231   * @param parser
232   *          the JSON parser used to read the parsed value
233   * @return the parsed value
234   * @throws IOException
235   *           if a parsing error occurs
236   */
237  @NonNull
238  TYPE parse(@NonNull JsonParser parser) throws IOException;
239
240  /**
241   * Parses a provided string using {@link #parse(String)}.
242   * <p>
243   * This method may pre-parse the data and then return copies, since the data can only be parsed
244   * once, but the supplier might be called multiple times.
245   *
246   * @param value
247   *          the string value to parse
248   * @return a supplier that will provide new instances of the parsed data
249   * @throws IOException
250   *           if an error occurs while parsing
251   * @see #parse(String)
252   */
253  @NonNull
254  default Supplier<TYPE> parseAndSupply(@NonNull String value) throws IOException {
255    TYPE retval = parse(value);
256    return () -> copy(retval);
257  }
258
259  /**
260   * Parses a provided string using {@link IDataTypeAdapter#parse(XMLEventReader2)}.
261   * <p>
262   * This method may pre-parse the data and then return copies, since the data can only be parsed
263   * once, but the supplier might be called multiple times.
264   *
265   * @param eventReader
266   *          the XML parser used to read the parsed value
267   * @return a supplier that will provide new instances of the parsed data
268   * @throws IOException
269   *           if an error occurs while parsing
270   * @see #parse(String)
271   * @see #parse(XMLEventReader2)
272   */
273  @NonNull
274  default Supplier<TYPE> parseAndSupply(@NonNull XMLEventReader2 eventReader) throws IOException {
275    TYPE retval = parse(eventReader);
276    return () -> copy(retval);
277  }
278
279  /**
280   * Parses a provided string using {@link #parse(JsonParser)}.
281   * <p>
282   * This method may pre-parse the data and then return copies, since the data can only be parsed
283   * once, but the supplier might be called multiple times.
284   *
285   * @param parser
286   *          the JSON parser used to read the parsed value
287   * @return a supplier that will provide new instances of the parsed data
288   * @throws IOException
289   *           if an error occurs while parsing
290   * @see #parse(String)
291   * @see #parse(JsonParser)
292   */
293  @NonNull
294  default Supplier<TYPE> parseAndSupply(@NonNull JsonParser parser) throws IOException {
295    TYPE retval = parse(parser);
296    return () -> copy(retval);
297  }
298
299  /**
300   * Writes the provided Java class instance data as XML. The parent element information is provided
301   * as a {@link StartElement} event, which allows namespace information to be obtained from the
302   * parent element using the {@link StartElement#getName()} and
303   * {@link StartElement#getNamespaceContext()} methods, which can be used when writing the provided
304   * instance value.
305   *
306   * @param instance
307   *          the {@link Field} instance value to write
308   * @param parent
309   *          the {@link StartElement} XML event that is the parent of the data to write
310   * @param eventFactory
311   *          the XML event factory used to generate XML writing events
312   * @param eventWriter
313   *          the XML writer used to output XML as events
314   * @throws XMLStreamException
315   *           if an unexpected error occurred while processing the XML output
316   * @throws IOException
317   *           if an unexpected error occurred while writing to the output stream
318   */
319  void writeXmlValue(@NonNull Object instance, @NonNull StartElement parent, @NonNull XMLEventFactory2 eventFactory,
320      @NonNull XMLEventWriter eventWriter)
321      throws IOException, XMLStreamException;
322
323  /**
324   * Writes the provided Java class instance data as XML. The parent element information is provided
325   * as an XML {@link QName}, which allows namespace information to be obtained from the parent
326   * element. Additional namespace information can be gathered using the
327   * {@link XMLStreamWriter2#getNamespaceContext()} method, which can be used when writing the
328   * provided instance value.
329   *
330   * @param instance
331   *          the {@link Field} instance value to write
332   * @param parentName
333   *          the qualified name of the XML data's parent element
334   * @param writer
335   *          the XML writer used to output the XML data
336   * @throws XMLStreamException
337   *           if an unexpected error occurred while processing the XML output
338   */
339  void writeXmlValue(@NonNull Object instance, @NonNull QName parentName, @NonNull XMLStreamWriter2 writer)
340      throws XMLStreamException;
341
342  /**
343   * Writes the provided Java class instance as a JSON/YAML field value.
344   *
345   * @param instance
346   *          the {@link Field} instance value to write
347   * @param writer
348   *          the JSON/YAML writer used to output the JSON/YAML data
349   * @throws IOException
350   *           if an unexpected error occurred while writing the JSON/YAML output
351   */
352  void writeJsonValue(@NonNull Object instance, @NonNull JsonGenerator writer) throws IOException;
353
354  /**
355   * Gets the default value to use as the JSON/YAML field name for a Metaschema field value if no JSON
356   * value key flag or name is configured.
357   *
358   * @return the default field name to use
359   */
360  @NonNull
361  String getDefaultJsonValueKey();
362
363  /**
364   * Determines if the data type's value is allowed to be unwrapped in XML when the value is a field
365   * value.
366   *
367   * @return {@code true} if allowed, or {@code false} otherwise.
368   */
369  boolean isUnrappedValueAllowedInXml();
370
371  /**
372   * Determines if the datatype uses mixed text and element content in XML.
373   *
374   * @return {@code true} if the datatype uses mixed text and element content in XML, or {@code false}
375   *         otherwise
376   */
377  boolean isXmlMixed();
378}
