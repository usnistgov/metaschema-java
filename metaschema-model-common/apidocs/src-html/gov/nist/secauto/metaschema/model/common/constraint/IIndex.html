027package gov.nist.secauto.metaschema.model.common.constraint;
028
029import gov.nist.secauto.metaschema.model.common.metapath.DynamicContext;
030import gov.nist.secauto.metaschema.model.common.metapath.item.INodeItem;
031
032import java.util.List;
033
034import edu.umd.cs.findbugs.annotations.NonNull;
035import edu.umd.cs.findbugs.annotations.Nullable;
036
037public interface IIndex {
038
039  /**
040   * Construct a new index using the provided key field components to generate keys.
041   *
042   * @param keyFields
043   *          the key field components to use to generate keys by default
044   * @return the new index
045   */
046  @NonNull
047  static IIndex newInstance(@NonNull List<? extends IKeyField> keyFields) {
048    return new DefaultIndex(keyFields);
049  }
050
051  /**
052   * Check if a key contains information other than {@code null} Strings.
053   *
054   * @param key
055   *          the key to check
056   * @return {@code true} if the series of key values contains only {@code null} values, or
057   *         {@code false} otherwise
058   */
059  static boolean isAllNulls(@NonNull Iterable<String> key) {
060    for (String value : key) {
061      if (value != null) {
062        return false; // NOPMD readability
063      }
064    }
065    return true;
066  }
067
068  /**
069   * Retrieve the key field components used to generate a key for this index.
070   *
071   * @return the key field components
072   */
073  @NonNull
074  List<IKeyField> getKeyFields();
075
076  /**
077   * Store the provided item in the index using the index's key field components to generate the key.
078   *
079   * @param item
080   *          the item to store in the index
081   * @param dynamicContext
082   *          the Metapath evaluation context
083   * @return the previous item stored in the index, or {@code null} otherwise
084   */
085  @Nullable
086  default INodeItem put(@NonNull INodeItem item, @NonNull DynamicContext dynamicContext) {
087    List<String> key = toKey(item, getKeyFields(), dynamicContext);
088    return put(item, key);
089  }
090
091  /**
092   * Store the provided item using the provided key.
093   *
094   * @param item
095   *          the item to store
096   * @param key
097   *          the key to store the item with
098   * @return the previous item stored in the index using the key, or {@code null} otherwise
099   */
100  @Nullable
101  INodeItem put(@NonNull INodeItem item, @NonNull List<String> key);
102
103  /**
104   * Retrieve the item from the index that matches the key generated by evaluating the index's default
105   * key field components against the provided item.
106   *
107   * @param item
108   *          the item to store in the index
109   * @param dynamicContext
110   *          the Metapath evaluation context
111   * @return the previous item stored in the index, or {@code null} otherwise
112   */
113  @Nullable
114  default INodeItem get(@NonNull INodeItem item, @NonNull DynamicContext dynamicContext) {
115    return get(item, getKeyFields(), dynamicContext);
116  }
117
118  /**
119   * Retrieve the item from the index that matches the key generated by evaluating the provided key
120   * field components against the provided item.
121   *
122   * @param item
123   *          the item to generate the key from
124   * @param keyFields
125   *          the key field components used to generate the key
126   * @param dynamicContext
127   *          the Metapath evaluation context
128   * @return the item with the matching key or {@code null} if no matching item was found
129   */
130  @Nullable
131  default INodeItem get(
132      @NonNull INodeItem item,
133      @NonNull List<? extends IKeyField> keyFields,
134      @NonNull DynamicContext dynamicContext) {
135    List<String> key = toKey(item, keyFields, dynamicContext);
136    return get(key);
137  }
138
139  /**
140   * Retrieve the item from the index that matches the provided key.
141   *
142   * @param key
143   *          the key to use for lookup
144   * @return the item with the matching key or {@code null} if no matching item was found
145   */
146  INodeItem get(List<String> key);
147
148  /**
149   * Construct a key by evaluating the provided key field components against the provided item.
150   *
151   * @param item
152   *          the item to generate the key from
153   * @param keyFields
154   *          the key field components used to generate the key
155   * @param dynamicContext
156   *          the Metapath evaluation context
157   * @return a new key
158   */
159  @NonNull
160  List<String> toKey(@NonNull INodeItem item, @NonNull List<? extends IKeyField> keyFields,
161      @NonNull DynamicContext dynamicContext);
162}
