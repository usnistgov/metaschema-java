027package gov.nist.secauto.metaschema.model.common.datatype.adapter;
028
029import com.google.auto.service.AutoService;
030
031import gov.nist.secauto.metaschema.model.common.datatype.AbstractDataTypeProvider;
032import gov.nist.secauto.metaschema.model.common.datatype.IDataTypeProvider;
033
034import edu.umd.cs.findbugs.annotations.NonNull;
035
036/**
037 * Provides for runtime discovery of built-in implementations of the core Metaschema data types.
038 */
039@AutoService(IDataTypeProvider.class)
040public final class MetaschemaDataTypeProvider // NOPMD - Used for service initialization
041    extends AbstractDataTypeProvider {
042  @NonNull
043  public static final Base64Adapter BASE64 = new Base64Adapter();
044  @NonNull
045  public static final BooleanAdapter BOOLEAN = new BooleanAdapter();
046  @NonNull
047  public static final DateAdapter DATE = new DateAdapter();
048  @NonNull
049  public static final DateWithTZAdapter DATE_WITH_TZ = new DateWithTZAdapter();
050  @NonNull
051  public static final DateTimeAdapter DATE_TIME = new DateTimeAdapter();
052  @NonNull
053  public static final DateTimeWithTZAdapter DATE_TIME_WITH_TZ = new DateTimeWithTZAdapter();
054  @NonNull
055  public static final IPv4AddressAdapter IP_V4_ADDRESS = new IPv4AddressAdapter();
056  @NonNull
057  public static final IPv6AddressAdapter IP_V6_ADDRESS = new IPv6AddressAdapter();
058  @NonNull
059  public static final UriAdapter URI = new UriAdapter();
060  @NonNull
061  public static final UriReferenceAdapter URI_REFERENCE = new UriReferenceAdapter();
062  @NonNull
063  public static final UuidAdapter UUID = new UuidAdapter();
064
065  @NonNull
066  public static final DayTimeAdapter DAY_TIME_DURATION = new DayTimeAdapter();
067  @NonNull
068  public static final YearMonthAdapter YEAR_MONTH_DURATION = new YearMonthAdapter();
069
070  @NonNull
071  public static final DecimalAdapter DECIMAL = new DecimalAdapter();
072  @NonNull
073  public static final IntegerAdapter INTEGER = new IntegerAdapter();
074  @NonNull
075  public static final NonNegativeIntegerAdapter NON_NEGATIVE_INTEGER = new NonNegativeIntegerAdapter();
076  @NonNull
077  public static final PositiveIntegerAdapter POSITIVE_INTEGER = new PositiveIntegerAdapter();
078
079  @NonNull
080  public static final EmailAddressAdapter EMAIL_ADDRESS = new EmailAddressAdapter();
081  @NonNull
082  public static final HostnameAdapter HOSTNAME = new HostnameAdapter();
083  @NonNull
084  public static final NcNameAdapter NCNAME = new NcNameAdapter();
085  @NonNull
086  public static final StringAdapter STRING = new StringAdapter();
087  @NonNull
088  public static final TokenAdapter TOKEN = new TokenAdapter();
089
090  @NonNull
091  public static final StringAdapter DEFAULT_DATA_TYPE = STRING;
092
093  /**
094   * Initialize the built-in data types.
095   */
096  public MetaschemaDataTypeProvider() {
097    // The data type "string" must be first since this is the default data type for the {@link String}
098    // Java type. This ensures that when a data type is resolved that this data type is matched first
099    // before other String-based data types.
100    registerDatatype(STRING);
101
102    registerDatatype(BASE64);
103    registerDatatype(BOOLEAN);
104    registerDatatype(DATE);
105    registerDatatype(DATE_WITH_TZ);
106    registerDatatype(DATE_TIME);
107    registerDatatype(DATE_TIME_WITH_TZ);
108    registerDatatype(DAY_TIME_DURATION);
109    registerDatatype(DECIMAL);
110    registerDatatype(EMAIL_ADDRESS);
111    registerDatatype(HOSTNAME);
112    registerDatatype(INTEGER);
113    registerDatatype(IP_V4_ADDRESS);
114    registerDatatype(IP_V6_ADDRESS);
115
116    registerDatatype(NON_NEGATIVE_INTEGER);
117    registerDatatype(POSITIVE_INTEGER);
118    registerDatatype(TOKEN);
119    registerDatatype(URI);
120    registerDatatype(URI_REFERENCE);
121    registerDatatype(UUID);
122    registerDatatype(YEAR_MONTH_DURATION);
123  }
124}
