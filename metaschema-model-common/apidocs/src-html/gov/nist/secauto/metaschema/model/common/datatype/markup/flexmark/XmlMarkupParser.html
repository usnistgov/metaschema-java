055  @NonNull

078    return SINGLETON;

081  public MarkupLine parseMarkupline(XMLEventReader2 reader) throws XMLStreamException { // NOPMD - acceptable
082    StringBuilder buffer = new StringBuilder();
083    parseContents(reader, null, buffer);
084    String html = buffer.toString().trim();
085    return html.isEmpty() ? null : MarkupLine.fromHtml(html);
086  }
087
088  public MarkupMultiline parseMarkupMultiline(XMLEventReader2 reader) throws XMLStreamException {
089    StringBuilder buffer = new StringBuilder();
090    parseToString(reader, buffer);
091    String html = buffer.toString().trim();
092
093    if (LOGGER.isDebugEnabled()) {
094      LOGGER.debug("XML->HTML: {}", html);
095    }
096    return html.isEmpty() ? null : MarkupMultiline.fromHtml(html);
097  }
098
099  protected void parseToString(XMLEventReader2 reader, StringBuilder buffer) // NOPMD - acceptable
100      throws XMLStreamException {
101    // if (LOGGER.isDebugEnabled()) {
102    // LOGGER.debug("parseToString(enter): {}", XmlEventUtil.toString(reader.peek()));
103    // }
104
105    outer: while (reader.hasNextEvent() && !reader.peek().isEndElement()) {
106      // skip whitespace before the next block element
107      XMLEvent nextEvent = XmlEventUtil.skipWhitespace(reader);
108
109      // if (LOGGER.isDebugEnabled()) {
110      // LOGGER.debug("parseToString: {}", XmlEventUtil.toString(nextEvent));
111      // }
112
113      if (nextEvent.isStartElement()) {
114        StartElement start = nextEvent.asStartElement();
115        QName name = start.getName();
116
117        // Note: the next element is not consumed. The called method is expected to
118        // consume it
119        if (BLOCK_ELEMENTS.contains(name.getLocalPart())) {
120          parseStartElement(reader, start, buffer);
121
122          // the next event should be the event after the start's END_ELEMENT
123          // assert XmlEventUtil.isNextEventEndElement(reader, name) : XmlEventUtil.toString(reader.peek());
124        } else {
125          // throw new IllegalStateException();
126          // stop parsing on first unrecognized event
127          break outer;
128        }
129      }
130      // reader.nextEvent();
131
132      // skip whitespace before the next block element
133      XmlEventUtil.skipWhitespace(reader);
134    }
135
136    // if (LOGGER.isDebugEnabled()) {
137    // LOGGER.debug("parseToString(exit): {}", reader.peek() != null ?
138    // XmlEventUtil.toString(reader.peek()) : "");
139    // }
140  }

142  private void parseStartElement(XMLEventReader2 reader, StartElement start, StringBuilder buffer)
143      throws XMLStreamException {
144    if (LOGGER.isDebugEnabled()) {
145      LOGGER.debug("parseStartElement(enter): {}", XmlEventUtil.toString(start));
146    }
147
148    // consume the start event
149    reader.nextEvent();
150
151    QName name = start.getName();
152    buffer.append('<')
153        .append(name.getLocalPart());
154    for (Attribute attribute : CollectionUtil.toIterable(start.getAttributes())) {
155      buffer
156          .append(' ')
157          .append(attribute.getName().getLocalPart())
158          .append("=\"")
159          .append(attribute.getValue())
160          .append('"');
161    }
162
163    XMLEvent next = reader.peek();
164    if (next != null && next.isEndElement()) {
165      buffer.append("/>");
166      // consume end element event
167      reader.nextEvent();
168    } else {
169      buffer.append('>');
170
171      // parse until the start's END_ELEMENT is reached
172      parseContents(reader, start, buffer);
173
174      buffer
175          .append("</")
176          .append(name.getLocalPart())
177          .append('>');
178
179      // the next event should be the start's END_ELEMENT
180      assert XmlEventUtil.isNextEventEndElement(reader, name) : XmlEventUtil.toString(next);
181
182      // consume the start's END_ELEMENT
183      reader.nextEvent();
184    }
185
186    if (LOGGER.isDebugEnabled()) {
187      LOGGER.debug("parseStartElement(exit): {}", reader.peek() != null ? XmlEventUtil.toString(reader.peek()) : "");
188    }
189  }
190
191  private void parseContents(XMLEventReader2 reader, StartElement start, StringBuilder buffer)
192      throws XMLStreamException {
193    // if (LOGGER.isDebugEnabled()) {
194    // LOGGER.debug("parseContents(enter): {}", XmlEventUtil.toString(reader.peek()));
195    // }
196
197    XMLEvent event;
198    while (reader.hasNextEvent() && !(event = reader.peek()).isEndElement()) {
199      // // skip whitespace before the next list item
200      // event = XmlEventUtil.skipWhitespace(reader);
201
202      // if (LOGGER.isDebugEnabled()) {
203      // LOGGER.debug("parseContents(before): {}", XmlEventUtil.toString(event));
204      // }
205
206      if (event.isStartElement()) {
207        StartElement nextStart = event.asStartElement();
208        // QName nextName = nextStart.getName();
209        parseStartElement(reader, nextStart, buffer);
210
211        // if (LOGGER.isDebugEnabled()) {
212        // LOGGER.debug("parseContents(after): {}", XmlEventUtil.toString(reader.peek()));
213        // }
214
215        // assert XmlEventUtil.isNextEventEndElement(reader, nextName) :
216        // XmlEventUtil.toString(reader.peek());
217
218        // reader.nextEvent();
219      } else if (event.isCharacters()) {
220        Characters characters = event.asCharacters();
221        buffer.append(Escaping.escapeHtml(characters.getData(), true));
222        reader.nextEvent();
223      }
224    }
225
226    assert start == null
227        || XmlEventUtil.isNextEventEndElement(reader, ObjectUtils.notNull(start.getName())) : XmlEventUtil
228            .toString(reader.peek());
229
230    // if (LOGGER.isDebugEnabled()) {
231    // LOGGER.debug("parseContents(exit): {}", reader.peek() != null ?
232    // XmlEventUtil.toString(reader.peek()) : "");
233    // }
234  }
