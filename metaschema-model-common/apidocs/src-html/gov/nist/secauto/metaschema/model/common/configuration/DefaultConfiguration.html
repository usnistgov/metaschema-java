027package gov.nist.secauto.metaschema.model.common.configuration;
028
029import gov.nist.secauto.metaschema.model.common.util.CollectionUtil;
030
031import java.util.HashMap;
032import java.util.Map;
033
034import edu.umd.cs.findbugs.annotations.NonNull;
035
036/**
037 * Provides immutable access to configuration state.
038 *
039 * @param <T>
040 *          the type of managed features
041 */
042@SuppressWarnings({ "PMD.ReplaceVectorWithList", "PMD.DoNotUseThreads" })
043public class DefaultConfiguration<T extends IConfigurationFeature<?>>
044    implements IMutableConfiguration<T> {
045  @NonNull
046  private Map<T, Object> featureValues;
047
048  /**
049   * Create a new configuration.
050   *
051   */
052  @SuppressWarnings("PMD.CloseResource")
053  public DefaultConfiguration() {
054    this.featureValues = new HashMap<>();
055  }
056
057  /**
058   * Create a new configuration based on the provided feature value map.
059   *
060   * @param featureValues
061   *          the set of enabled features
062   */
063  public DefaultConfiguration(@NonNull Map<T, Object> featureValues) {
064    this.featureValues = new HashMap<>(featureValues);
065  }
066
067  /**
068   * Create a new configuration based on the provided configuration.
069   *
070   * @param original
071   *          the original configuration
072   */
073  public DefaultConfiguration(@NonNull DefaultConfiguration<T> original) {
074    this(original.getFeatureValues());
075  }
076
077  @Override
078  public Map<T, Object> getFeatureValues() {
079    return CollectionUtil.unmodifiableMap(featureValues);
080  }
081
082  private void ensureBooleanValue(@NonNull T feature) {
083    Class<?> valueClass = feature.getValueClass();
084    if (!Boolean.class.isAssignableFrom(valueClass)) {
085      throw new UnsupportedOperationException(
086          String.format("Feature value class '%s' is boolean valued.", valueClass.getName()));
087    }
088  }
089
090  @Override
091  public boolean isFeatureEnabled(@NonNull T feature) {
092    ensureBooleanValue(feature);
093    return get(feature);
094  }
095
096  @Override
097  public IMutableConfiguration<T> enableFeature(@NonNull T feature) {
098    ensureBooleanValue(feature);
099    featureValues.put(feature, true);
100    return this;
101  }
102
103  @Override
104  public IMutableConfiguration<T> disableFeature(@NonNull T feature) {
105    ensureBooleanValue(feature);
106    featureValues.put(feature, false);
107    return this;
108  }
109
110  @Override
111  public IMutableConfiguration<T> applyConfiguration(@NonNull IConfiguration<T> original) {
112    this.featureValues.putAll(original.getFeatureValues());
113    return this;
114  }
115
116  @SuppressWarnings("unchecked")
117  @Override
118  public <V> V get(T feature) {
119    V value = (V) featureValues.get(feature);
120    if (value == null) {
121      value = (V) feature.getDefault();
122    }
123    return value;
124  }
125
126  @Override
127  public IMutableConfiguration<T> set(T feature, Object value) {
128    Class<?> featureValueClass = feature.getValueClass();
129    Class<?> valueClass = value.getClass();
130    if (!featureValueClass.isAssignableFrom(valueClass)) {
131      throw new UnsupportedOperationException(
132          String.format("Provided value of class '%s' is not assignment compatible with feature value class '%s'.",
133              valueClass.getName(),
134              featureValueClass.getName()));
135    }
136    featureValues.put(feature, value);
137    return this;
138  }
139}
