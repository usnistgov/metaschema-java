027package gov.nist.secauto.metaschema.model.common.constraint;
028
029import gov.nist.secauto.metaschema.model.common.datatype.markup.MarkupLine;
030import gov.nist.secauto.metaschema.model.common.datatype.markup.MarkupMultiline;
031import gov.nist.secauto.metaschema.model.common.metapath.MetapathExpression;
032import gov.nist.secauto.metaschema.model.common.util.ObjectUtils;
033
034import java.util.Map;
035import java.util.Set;
036
037import javax.xml.namespace.QName;
038
039import edu.umd.cs.findbugs.annotations.NonNull;
040import edu.umd.cs.findbugs.annotations.Nullable;
041
042public final class DefaultCardinalityConstraint
043    extends AbstractConstraint
044    implements ICardinalityConstraint {
045  @Nullable
046  private final Integer minOccurs;
047  @Nullable
048  private final Integer maxOccurs;
049
050  /**
051   * Construct a new cardinality constraint which enforces that the number of items matching the
052   * target fall within the inclusive range described by the {@code minOccurs} or {@code maxOccurs}
053   * values.
054   *
055   * @param id
056   *          the optional identifier for the constraint
057   * @param formalName
058   *          the constraint's formal name or {@code null} if not provided
059   * @param description
060   *          the constraint's semantic description or {@code null} if not provided
061   * @param source
062   *          information about the constraint source
063   * @param level
064   *          the significance of a violation of this constraint
065   * @param target
066   *          the Metapath expression identifying the nodes the constraint targets
067   * @param properties
068   *          a collection of associated properties
069   * @param minOccurs
070   *          if provided, the constraint ensures that the count of targets is at least this value
071   * @param maxOccurs
072   *          if provided, the constraint ensures that the count of targets is at most this value
073   * @param remarks
074   *          optional remarks describing the intent of the constraint
075   */
076  private DefaultCardinalityConstraint(
077      @Nullable String id,
078      @Nullable String formalName,
079      @Nullable MarkupLine description,
080      @NonNull ISource source,
081      @NonNull Level level,
082      @NonNull MetapathExpression target,
083      @NonNull Map<QName, Set<String>> properties,
084      @Nullable Integer minOccurs,
085      @Nullable Integer maxOccurs,
086      MarkupMultiline remarks) {
087    super(id, formalName, description, source, level, target, properties, remarks);
088    if (minOccurs == null && maxOccurs == null) {
089      throw new IllegalArgumentException("at least one of minOccurs or maxOccurs must be provided");
090    }
091    this.minOccurs = minOccurs;
092    this.maxOccurs = maxOccurs;
093  }
094
095  @Override
096  public Integer getMinOccurs() {
097    return minOccurs;
098  }
099
100  @Override
101  public Integer getMaxOccurs() {
102    return maxOccurs;
103  }
104
105  @Override
106  public <T, R> R accept(IConstraintVisitor<T, R> visitor, T state) {
107    return visitor.visitCardinalityConstraint(this, state);
108  }
109
110  @NonNull
111  public static Builder builder() {
112    return new Builder();
113  }
114
115  public static final class Builder
116      extends AbstractConstraintBuilder<Builder, DefaultCardinalityConstraint> {
117    private Integer minOccurs;
118    private Integer maxOccurs;
119
120    private Builder() {
121      // disable construction
122    }
123
124    public Builder minOccurs(int value) {
125      this.minOccurs = value;
126      return this;
127    }
128
129    public Builder maxOccurs(int value) {
130      this.maxOccurs = value;
131      return this;
132    }
133
134    @Override
135    protected Builder getThis() {
136      return this;
137    }
138
139    @Override
140    protected void validate() {
141      super.validate();
142
143      if (getMinOccurs() == null && getMaxOccurs() == null) {
144        throw new IllegalStateException("At least one of minOccurs or maxOccurs must be provided.");
145      }
146    }
147
148    protected Integer getMinOccurs() {
149      return minOccurs;
150    }
151
152    protected Integer getMaxOccurs() {
153      return maxOccurs;
154    }
155
156    @Override
157    protected DefaultCardinalityConstraint newInstance() {
158      return new DefaultCardinalityConstraint(
159          getId(),
160          getFormalName(),
161          getDescription(),
162          ObjectUtils.notNull(getSource()),
163          getLevel(),
164          getTarget(),
165          getProperties(),
166          getMinOccurs(),
167          getMaxOccurs(),
168          getRemarks());
169    }
170  }
171}
