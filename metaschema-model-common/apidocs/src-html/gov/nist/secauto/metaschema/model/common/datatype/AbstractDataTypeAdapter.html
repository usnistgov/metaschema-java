027package gov.nist.secauto.metaschema.model.common.datatype;
028
029import com.fasterxml.jackson.core.JsonGenerator;
030import com.fasterxml.jackson.core.JsonParser;
031
032import gov.nist.secauto.metaschema.model.common.metapath.function.InvalidValueForCastFunctionException;
033import gov.nist.secauto.metaschema.model.common.metapath.item.IAnyAtomicItem;
034import gov.nist.secauto.metaschema.model.common.util.XmlEventUtil;
035
036import org.codehaus.stax2.XMLEventReader2;
037import org.codehaus.stax2.XMLStreamWriter2;
038import org.codehaus.stax2.evt.XMLEventFactory2;
039
040import java.io.IOException;
041
042import javax.xml.namespace.QName;
043import javax.xml.stream.XMLEventWriter;
044import javax.xml.stream.XMLStreamException;
045import javax.xml.stream.events.Characters;
046import javax.xml.stream.events.StartElement;
047import javax.xml.stream.events.XMLEvent;
048
049import edu.umd.cs.findbugs.annotations.NonNull;
050
051/**
052 * Provides a basic Java type adapter implementation. This implementation should be the parent class
053 * of all Java type adapter implementations.
054 *
055 * @param <TYPE>
056 *          the raw Java type this adapter supports
057 * @param <ITEM_TYPE>
058 *          the metapath item type corresponding to the raw Java type supported by the adapter
059 */
060public abstract class AbstractDataTypeAdapter<TYPE, ITEM_TYPE extends IAnyAtomicItem>
061    implements IDataTypeAdapter<TYPE> {
062  public static final String DEFAULT_JSON_FIELD_NAME = "STRVALUE";
063
064  @NonNull
065  private final Class<TYPE> clazz;
066
067  /**
068   * Construct a new Java type adapter for a provided class.
069   *
070   * @param clazz
071   *          the Java type this adapter supports
072   */
073  protected AbstractDataTypeAdapter(@NonNull Class<TYPE> clazz) {
074    this.clazz = clazz;
075  }
076
077  @Override
078  @SuppressWarnings("unchecked")
079  public TYPE toValue(Object value) {
080    return (TYPE) value;
081  }
082
083  @Override
084  public Class<TYPE> getJavaClass() {
085    return clazz;
086  }
087
088  @Override
089  public boolean isParsingStartElement() {
090    return false;
091  }
092
093  @Override
094  public boolean canHandleQName(QName nextQName) {
095    return false;
096  }
097
098  @Override
099  public String getDefaultJsonValueKey() {
100    return DEFAULT_JSON_FIELD_NAME;
101  }
102
103  @Override
104  public boolean isUnrappedValueAllowedInXml() {
105    return false;
106  }
107
108  @Override
109  public boolean isXmlMixed() {
110    return false;
111  }
112
113  @Override
114  public TYPE parse(XMLEventReader2 eventReader) throws IOException {
115    StringBuilder builder = new StringBuilder();
116    try {
117      XMLEvent nextEvent;
118      while (!(nextEvent = eventReader.peek()).isEndElement()) {
119        if (nextEvent.isCharacters()) {
120          Characters characters = nextEvent.asCharacters();
121          builder.append(characters.getData());
122          // advance past current event
123          eventReader.nextEvent();
124        } else {
125          throw new IOException(String.format("Invalid content '%s' at %s", XmlEventUtil.toString(nextEvent),
126              XmlEventUtil.toString(nextEvent.getLocation())));
127        }
128      }
129
130      // trim leading and trailing whitespace
131      @SuppressWarnings("null")
132      @NonNull String value = builder.toString().trim();
133      return parse(value);
134    } catch (XMLStreamException ex) {
135      throw new IOException(ex);
136    }
137  }
138
139  /**
140   * This default implementation will parse the value as a string and delegate to the string-based
141   * parsing method.
142   */
143  @Override
144  public TYPE parse(JsonParser parser) throws IOException {
145    String value = parser.getValueAsString();
146    if (value == null) {
147      throw new IOException("Unable to parse field value as text");
148    }
149    // skip over value
150    parser.nextToken();
151    return parse(value);
152  }
153
154  @SuppressWarnings("null")
155  @Override
156  public String asString(Object value) {
157    return value.toString();
158  }
159
160  @Override
161  public void writeXmlValue(
162      Object value,
163      StartElement parent,
164      XMLEventFactory2 eventFactory,
165      XMLEventWriter eventWriter)
166      throws IOException, XMLStreamException {
167    try {
168      String content = asString(value);
169      Characters characters = eventFactory.createCharacters(content);
170      eventWriter.add(characters);
171    } catch (XMLStreamException ex) {
172      throw new IOException(ex);
173    }
174  }
175
176  @Override
177  public void writeXmlValue(Object value, QName parentName, XMLStreamWriter2 writer) throws XMLStreamException {
178    String content = asString(value);
179    writer.writeCharacters(content);
180  }
181
182  @Override
183  public void writeJsonValue(Object value, JsonGenerator generator) throws IOException {
184    generator.writeString(asString(value));
185  }
186
187  @Override
188  public abstract Class<ITEM_TYPE> getItemClass();
189
190  @Override
191  public abstract ITEM_TYPE newItem(Object value);
192
193  @Override
194  public ITEM_TYPE cast(IAnyAtomicItem item) {
195    ITEM_TYPE retval;
196    if (item == null) {
197      throw new InvalidValueForCastFunctionException("item is null");
198    } else if (getItemClass().isAssignableFrom(item.getClass())) {
199      @SuppressWarnings("unchecked") ITEM_TYPE typedItem = (ITEM_TYPE) item;
200      retval = typedItem;
201    } else {
202      retval = castInternal(item);
203    }
204    return retval;
205  }
206
207  /**
208   * Attempt to cast the provided item to this adapter's item type.
209   * <p>
210   * The default implementation of this will attempt to parse the provided item as a string using the
211   * {@link #parse(String)} method. If this behavior is undesirable, then a subclass should override
212   * this method.
213   *
214   * @param item
215   *          the item to cast
216   * @return the item casted to this adapter's item type
217   * @throws InvalidValueForCastFunctionException
218   *           if the casting of the item is not possible because the item represents an invalid value
219   *           for this adapter's item type
220   */
221  @NonNull
222  protected ITEM_TYPE castInternal(@NonNull IAnyAtomicItem item) {
223    // try string based casting as a fallback
224    String itemString = item.asString();
225    try {
226      TYPE value = parse(itemString);
227      return newItem(value);
228    } catch (IllegalArgumentException ex) {
229      throw new InvalidValueForCastFunctionException(
230          String.format("The value '%s' is not compatible with the type '%s'", itemString, getItemClass().getName()),
231          ex);
232    }
233  }
234}
