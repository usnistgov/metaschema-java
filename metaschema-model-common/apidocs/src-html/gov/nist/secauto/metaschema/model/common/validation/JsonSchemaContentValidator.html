027package gov.nist.secauto.metaschema.model.common.validation;
028
029import gov.nist.secauto.metaschema.model.common.constraint.IConstraint;
030import gov.nist.secauto.metaschema.model.common.util.ObjectUtils;
031
032import org.everit.json.schema.Schema;
033import org.everit.json.schema.ValidationException;
034import org.everit.json.schema.loader.SchemaLoader;
035import org.json.JSONException;
036import org.json.JSONObject;
037import org.json.JSONTokener;
038import org.xml.sax.InputSource;
039
040import java.io.IOException;
041import java.io.InputStream;
042import java.io.Reader;
043import java.net.URI;
044import java.net.URL;
045import java.util.Collections;
046import java.util.List;
047import java.util.Objects;
048import java.util.stream.Collectors;
049import java.util.stream.Stream;
050
051import edu.umd.cs.findbugs.annotations.NonNull;
052
053public class JsonSchemaContentValidator implements IContentValidator {
054  @NonNull
055  private final Schema schema;
056
057  @NonNull
058  public static JSONObject toJsonObject(@NonNull InputStream schemaInputStream) {
059    return new JSONObject(new JSONTokener(schemaInputStream));
060  }
061
062  @NonNull
063  public static JSONObject toJsonObject(@NonNull Reader reader) {
064    return new JSONObject(new JSONTokener(reader));
065  }
066
067  public JsonSchemaContentValidator(@NonNull InputStream schemaInputStream) {
068    this(toJsonObject(Objects.requireNonNull(schemaInputStream, "schemaInputStream")));
069  }
070
071  @SuppressWarnings("null")
072  public JsonSchemaContentValidator(@NonNull JSONObject jsonSchema) {
073    this(SchemaLoader.load(Objects.requireNonNull(jsonSchema, "jsonSchema")));
074  }
075
076  protected JsonSchemaContentValidator(@NonNull Schema schema) {
077    this.schema = ObjectUtils.requireNonNull(schema, "schema");
078  }
079
080  @NonNull
081  public Schema getSchema() {
082    return schema;
083  }
084
085  @Override
086  public IValidationResult validate(@NonNull InputSource source) throws IOException {
087    URI uri = ObjectUtils.notNull(URI.create(source.getSystemId()));
088
089    JSONObject json;
090    try {
091      if (source.getCharacterStream() != null) {
092        // attempt to use a provided character stream
093        json = new JSONObject(new JSONTokener(source.getCharacterStream()));
094      } else if (source.getByteStream() != null) {
095        // attempt to use a provided byte stream stream
096        json = new JSONObject(new JSONTokener(source.getByteStream()));
097      } else {
098        // fall back to a URL-based connection
099        URL url = uri.toURL();
100        try (InputStream is = url.openStream()) {
101          json = new JSONObject(new JSONTokener(is));
102        }
103      }
104    } catch (JSONException ex) {
105      throw new IOException(String.format("Unable to parse JSON from '%s'", uri), ex);
106    }
107    return validate(json, uri);
108  }
109
110  @SuppressWarnings("null")
111  @NonNull
112  public IValidationResult validate(@NonNull JSONObject json, @NonNull URI documentUri) {
113    IValidationResult retval;
114    try {
115      schema.validate(json);
116      retval = IValidationResult.PASSING_RESULT;
117    } catch (ValidationException ex) {
118      retval = new JsonValidationResult(
119          handleValidationException(ex, documentUri)
120              .collect(Collectors.toList()));
121    }
122
123    return retval;
124  }
125
126  @SuppressWarnings("null")
127  @NonNull
128  protected Stream<JsonValidationFinding> handleValidationException(@NonNull ValidationException ex,
129      @NonNull URI documentUri) {
130    JsonValidationFinding finding = new JsonValidationFinding(ex, documentUri);
131    Stream<JsonValidationFinding> childFindings = ex.getCausingExceptions().stream()
132        .flatMap(exception -> {
133          return handleValidationException(exception, documentUri);
134        });
135    return Stream.concat(Stream.of(finding), childFindings);
136  }
137
138  public static class JsonValidationFinding implements IValidationFinding {
139    @NonNull
140    private final ValidationException exception;
141    @NonNull
142    private final URI documentUri;
143
144    public JsonValidationFinding(@NonNull ValidationException exception, @NonNull URI documentUri) {
145      this.exception = ObjectUtils.requireNonNull(exception, "exception");
146      this.documentUri = ObjectUtils.requireNonNull(documentUri, "documentUri");
147    }
148
149    @Override
150    public IConstraint.Level getSeverity() {
151      return IConstraint.Level.ERROR;
152    }
153
154    @Override
155    public URI getDocumentUri() {
156      return documentUri;
157    }
158
159    @SuppressWarnings("null")
160    @Override
161    public String getMessage() {
162      return getCause().getLocalizedMessage();
163    }
164
165    @NonNull
166    @Override
167    public ValidationException getCause() {
168      return exception;
169    }
170  }
171
172  private static class JsonValidationResult implements IValidationResult {
173    @NonNull
174    private final List<JsonValidationFinding> findings;
175
176    @SuppressWarnings("null")
177    public JsonValidationResult(@NonNull List<JsonValidationFinding> findings) {
178      this.findings = Collections.unmodifiableList(Objects.requireNonNull(findings, "findings"));
179    }
180
181    @Override
182    public IConstraint.Level getHighestSeverity() {
183      return findings.isEmpty() ? IConstraint.Level.INFORMATIONAL : IConstraint.Level.ERROR;
184    }
185
186    @Override
187    public List<? extends IValidationFinding> getFindings() {
188      return findings;
189    }
190
191  }
