027package gov.nist.secauto.metaschema.model.common.constraint;
028
029import gov.nist.secauto.metaschema.model.common.metapath.DynamicContext;
030import gov.nist.secauto.metaschema.model.common.metapath.InvalidTypeMetapathException;
031import gov.nist.secauto.metaschema.model.common.metapath.MetapathException;
032import gov.nist.secauto.metaschema.model.common.metapath.MetapathExpression;
033import gov.nist.secauto.metaschema.model.common.metapath.MetapathExpression.ResultType;
034import gov.nist.secauto.metaschema.model.common.metapath.function.library.FnData;
035import gov.nist.secauto.metaschema.model.common.metapath.item.INodeItem;
036import gov.nist.secauto.metaschema.model.common.util.CollectionUtil;
037import gov.nist.secauto.metaschema.model.common.util.ObjectUtils;
038
039import java.util.ArrayList;
040import java.util.List;
041import java.util.Map;
042import java.util.concurrent.ConcurrentHashMap;
043import java.util.regex.Matcher;
044import java.util.regex.Pattern;
045import java.util.stream.Collectors;
046
047import edu.umd.cs.findbugs.annotations.NonNull;
048import edu.umd.cs.findbugs.annotations.Nullable;
049
050public class DefaultIndex implements IIndex {
051  @NonNull
052  private final List<IKeyField> keyFields;
053  @NonNull
054  private final Map<List<String>, INodeItem> keyToItemMap = new ConcurrentHashMap<>();
055
056  /**
057   * Construct a new index.
058   *
059   * @param keyFields
060   *          the key field components to use to generate keys by default
061   */
062  protected DefaultIndex(@NonNull List<? extends IKeyField> keyFields) {
063    this.keyFields = CollectionUtil.unmodifiableList(new ArrayList<>(keyFields));
064  }
065
066  @Override
067  public List<IKeyField> getKeyFields() {
068    return keyFields;
069  }
070
071  @Override
072  public INodeItem put(@NonNull INodeItem item, @NonNull List<String> key) {
073    INodeItem oldItem = null;
074    if (!IIndex.isAllNulls(key)) {
075      // only add keys with some information (values)
076      oldItem = keyToItemMap.put(key, item);
077    }
078    return oldItem;
079  }
080
081  @Override
082  public INodeItem get(List<String> key) {
083    return keyToItemMap.get(key);
084  }
085
086  @Override
087  public List<String> toKey(@NonNull INodeItem item, @NonNull List<? extends IKeyField> keyFields,
088      @NonNull DynamicContext dynamicContext) {
089    if (getKeyFields().size() != keyFields.size()) {
090      throw new IllegalArgumentException("Provided key fields are not the same size as the index requires.");
091    }
092    return CollectionUtil.unmodifiableList(
093        ObjectUtils.notNull(keyFields.stream()
094            .map(keyField -> {
095              assert keyField != null;
096              return buildKeyItem(item, keyField, dynamicContext);
097            })
098            .collect(Collectors.toCollection(ArrayList::new))));
099  }
100
101  /**
102   * Evaluates the provided key field component against the item to generate a key value.
103   *
104   * @param item
105   *          the item to generate the key value from
106   * @param keyField
107   *          the key field component used to generate the key value
108   * @param dynamicContext
109   *          the Metapath evaluation context
110   * @return the key value or {@code null} if the evaluation resulted in no value
111   */
112  @Nullable
113  protected static String buildKeyItem(
114      @NonNull INodeItem item,
115      @NonNull IKeyField keyField,
116      @NonNull DynamicContext dynamicContext) {
117    MetapathExpression keyPath = keyField.getTarget();
118
119    INodeItem keyItem;
120    try {
121      keyItem = keyPath.evaluateAs(item, ResultType.NODE, dynamicContext);
122    } catch (InvalidTypeMetapathException ex) {
123      throw new MetapathException("Key path did not result in a single node", ex);
124    }
125
126    String keyValue = null;
127    if (keyItem != null) {
128      keyValue = FnData.fnDataItem(keyItem).asString();
129      Pattern pattern = keyField.getPattern();
130      if (pattern != null) {
131        keyValue = applyPattern(keyItem, keyValue, pattern);
132      }
133    } // empty key
134    return keyValue;
135  }
136
137  /**
138   * Apply the key value pattern, if configured, to generate the final key value.
139   *
140   * @param keyItem
141   *          the node item used to form the key field
142   * @param pattern
143   *          the key field pattern configuration from the constraint
144   * @param keyValue
145   *          the current key value
146   * @return the final key value
147   */
148  protected static String applyPattern(@NonNull INodeItem keyItem, @NonNull String keyValue,
149      @NonNull Pattern pattern) {
150    Matcher matcher = pattern.matcher(keyValue);
151    if (!matcher.matches()) {
152      throw new MetapathException(
153          String.format("Key field declares the pattern '%s' which does not match the value '%s' of node '%s'",
154              pattern.pattern(), keyValue, keyItem.getMetapath()));
155    }
156
157    if (matcher.groupCount() != 1) {
158      throw new MetapathException(String.format(
159          "The first group was not a match for value '%s' of node '%s' for key field pattern '%s'",
160          keyValue, keyItem.getMetapath(), pattern.pattern()));
161    }
162    return matcher.group(1);
163  }
164
165}
