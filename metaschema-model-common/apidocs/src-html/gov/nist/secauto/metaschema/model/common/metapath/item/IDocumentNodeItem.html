027package gov.nist.secauto.metaschema.model.common.metapath.item;
028
029import gov.nist.secauto.metaschema.model.common.metapath.format.IPathFormatter;
030import gov.nist.secauto.metaschema.model.common.util.ObjectUtils;
031
032import java.net.URI;
033import java.util.Collection;
034import java.util.Collections;
035import java.util.List;
036import java.util.stream.Stream;
037
038import edu.umd.cs.findbugs.annotations.NonNull;
039
040public interface IDocumentNodeItem extends IRequiredValueNodeItem {
041  @Override
042  default NodeItemType getNodeItemType() {
043    return NodeItemType.DOCUMENT;
044  }
045
046  @Override
047  default IDocumentNodeItem getNodeItem() {
048    return this;
049  }
050
051  /**
052   * Get the root assembly associated with this document.
053   *
054   * @return the root assembly
055   */
056  @NonNull
057  IRootAssemblyNodeItem getRootAssemblyNodeItem();
058
059  @Override
060  default IRequiredValueModelNodeItem getParentContentNodeItem() {
061    // there is no parent
062    return null;
063  }
064
065  @Override
066  default IRequiredValueNodeItem getParentNodeItem() {
067    // there is no parent
068    return null;
069  }
070
071  /**
072   * Get the URI associated with this document.
073   *
074   * @return the document's URI
075   */
076  @NonNull
077  URI getDocumentUri();
078
079  @Override
080  @NonNull
081  default URI getBaseUri() {
082    return getDocumentUri();
083  }
084
085  /**
086   * Documents do not have flag items. This call should return an empty collection.
087   */
088  @SuppressWarnings("null")
089  @Override
090  default Collection<? extends IRequiredValueFlagNodeItem> getFlags() {
091    // a document does not have flags
092    return Collections.emptyList();
093  }
094
095  /**
096   * Documents do not have flag items. This call should return {@code null}.
097   */
098  @Override
099  default IRequiredValueFlagNodeItem getFlagByName(@NonNull String name) {
100    // a document does not have flags
101    return null;
102  }
103
104  /**
105   * Documents do not have flag items. This call should return an empty stream.
106   */
107  @SuppressWarnings("null")
108  @Override
109  default @NonNull Stream<? extends IRequiredValueFlagNodeItem> flags() {
110    // a document does not have flags
111    return Stream.empty();
112  }
113
114  @SuppressWarnings("null")
115  @Override
116  default Stream<? extends IModelNodeItem> modelItems() {
117    return Stream.of(getRootAssemblyNodeItem());
118  }
119
120  @SuppressWarnings("null")
121  @Override
122  default @NonNull List<? extends IRequiredValueModelNodeItem> getModelItemsByName(String name) {
123    IRootAssemblyNodeItem root = getRootAssemblyNodeItem();
124    return root.getName().equals(name) ? Collections.singletonList(root) : Collections.emptyList();
125  }
126
127  @SuppressWarnings("null")
128  @Override
129  default @NonNull Collection<? extends List<? extends IRequiredValueModelNodeItem>> getModelItems() {
130    return Collections.singletonList(Collections.singletonList(getRootAssemblyNodeItem()));
131  }
132
133  @Override
134  default @NonNull String format(@NonNull IPathFormatter formatter) {
135    return formatter.formatDocument(this);
136  }
137
138  @Override
139  default <RESULT, CONTEXT> RESULT accept(@NonNull INodeItemVisitor<RESULT, CONTEXT> visitor, CONTEXT context) {
140    return visitor.visitDocument(this, context);
141  }
142
143  @Override
144  @NonNull
145  default Object getValue() {
146    return ObjectUtils.requireNonNull(getRootAssemblyNodeItem().getValue());
147  }
148}
