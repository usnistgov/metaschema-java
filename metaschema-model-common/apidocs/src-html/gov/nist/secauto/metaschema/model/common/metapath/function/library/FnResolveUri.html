031import gov.nist.secauto.metaschema.model.common.metapath.function.FunctionUtils;
032import gov.nist.secauto.metaschema.model.common.metapath.function.IArgument;
033import gov.nist.secauto.metaschema.model.common.metapath.function.IFunction;
034import gov.nist.secauto.metaschema.model.common.metapath.function.InvalidArgumentFunctionException;
035import gov.nist.secauto.metaschema.model.common.metapath.function.UriFunctionException;
036import gov.nist.secauto.metaschema.model.common.metapath.item.IAnyUriItem;

041import java.net.URI;
042import java.util.List;
043
044import edu.umd.cs.findbugs.annotations.NonNull;
045import edu.umd.cs.findbugs.annotations.Nullable;
046
047public final class FnResolveUri {
048  @NonNull
049  static final IFunction SIGNATURE_ONE_ARG = IFunction.builder()
050      .name("resolve-uri")
051      .deterministic()
052      .contextDependent()
053      .focusIndependent()
054      .argument(IArgument.newBuilder()
055          .name("relative")
056          .type(IStringItem.class)
057          .zeroOrOne()
058          .build())
059      .returnType(IAnyUriItem.class)
060      .returnZeroOrOne()
061      .functionHandler(FnResolveUri::executeOneArg)
062      .build();
063
064  @NonNull
065  static final IFunction SIGNATURE_TWO_ARG = IFunction.builder()
066      .name("resolve-uri")
067      .deterministic()
068      .contextIndependent()
069      .focusIndependent()
070      .argument(IArgument.newBuilder()
071          .name("relative")
072          .type(IStringItem.class)
073          .zeroOrOne()
074          .build())
075      .argument(IArgument.newBuilder()
076          .name("base")
077          .type(IStringItem.class)
078          .one()
079          .build())
080      .returnType(IAnyUriItem.class)
081      .returnZeroOrOne()
082      .functionHandler(FnResolveUri::executeTwoArg)
083      .build();
084
085  private FnResolveUri() {
086    // disable construction
087  }
088
089  @SuppressWarnings("unused")
090  @NonNull
091  private static ISequence<IAnyUriItem> executeOneArg(
092      @NonNull IFunction function,
093      @NonNull List<ISequence<?>> arguments,
094      @NonNull DynamicContext dynamicContext,
095      INodeItem focus) {
096
097    ISequence<? extends IStringItem> relativeSequence
098        = FunctionUtils.asType(ObjectUtils.requireNonNull(arguments.get(0)));
099    if (relativeSequence.isEmpty()) {
100      return ISequence.empty(); // NOPMD - readability
101    }
102
103    IAnyUriItem baseUri = FnStaticBaseUri.fnStaticBaseUri(dynamicContext);
104    if (baseUri == null) {
105      throw new UriFunctionException(UriFunctionException.BASE_URI_NOT_DEFINED_IN_STATIC_CONTEXT,
106          "The base-uri is not defined in the static context");
107    }
108
109    IStringItem relativeString = FunctionUtils.getFirstItem(relativeSequence, true);
110
111    IAnyUriItem resolvedUri = fnResolveUri(relativeString, baseUri);
112    return resolvedUri == null ? ISequence.empty() : ISequence.of(resolvedUri);
113  }
114
115  /**
116   * Implements the two argument version of the XPath 3.1 function
117   * <a href="https://www.w3.org/TR/xpath-functions-31/#func-resolve-uri">resolve-uri</a>.
118   *
119   * @param function
120   *          the function definition
121   * @param arguments
122   *          a list of sequence arguments with an expected size of 2
123   * @param dynamicContext
124   *          the evaluation context
125   * @param focus
126   *          the current focus item
127   * @return a sequence containing the resolved URI or and empty sequence if either the base or
128   *         relative URI is {@code null}
129   */
130  @SuppressWarnings("PMD.UnusedPrivateMethod") // used in lambda
131  @NonNull
132  private static ISequence<IAnyUriItem> executeTwoArg(
133      @NonNull IFunction function, // NOPMD - ok
134      @NonNull List<ISequence<?>> arguments,
135      @NonNull DynamicContext dynamicContext, // NOPMD - ok
136      INodeItem focus) { // NOPMD - ok
137
138    /* there will always be two arguments */
139    assert arguments.size() == 2;
140
141    ISequence<? extends IStringItem> relativeSequence = FunctionUtils.asType(
142        ObjectUtils.requireNonNull(arguments.get(0)));
143    if (relativeSequence.isEmpty()) {
144      return ISequence.empty(); // NOPMD - readability
145    }
146
147    ISequence<? extends IStringItem> baseSequence = FunctionUtils.asType(ObjectUtils.requireNonNull(arguments.get(1)));
148    IStringItem baseString = FunctionUtils.getFirstItem(baseSequence, true);
149
150    if (baseString == null) {
151      throw new InvalidArgumentFunctionException(
152          InvalidArgumentFunctionException.INVALID_ARGUMENT_TO_RESOLVE_URI,
153          "Invalid argument to fn:resolve-uri().");
154    }
155    IAnyUriItem baseUri = IAnyUriItem.cast(baseString);
156
157    IStringItem relativeString = FunctionUtils.getFirstItem(relativeSequence, true);
158
159    IAnyUriItem resolvedUri = fnResolveUri(relativeString, baseUri);
160    return resolvedUri == null ? ISequence.empty() : ISequence.of(resolvedUri);
161  }
162
163  /**
164   * Resolve the {@code relative} URI against the provided {@code base} URI.
165   *
166   * @param relative
167   *          the relative URI to resolve
168   * @param base
169   *          the base URI to resolve against
170   * @return the resolved URI or {@code null} if the {@code relative} URI in {@code null}
171   */
172  @Nullable
173  public static IAnyUriItem fnResolveUri(@Nullable IStringItem relative, @NonNull IAnyUriItem base) {
174    IAnyUriItem relativeUri = relative == null ? null : IAnyUriItem.cast(relative);
175
176    return fnResolveUri(relativeUri, base);
177  }
178
179  /**
180   * Resolve the {@code relative} URI against the provided {@code base} URI.
181   *
182   * @param relative
183   *          the relative URI to resolve
184   * @param base
185   *          the base URI to resolve against
186   * @return the resolved URI or {@code null} if the {@code relative} URI in {@code null}
187   */
188  @Nullable
189  public static IAnyUriItem fnResolveUri(@Nullable IAnyUriItem relative, @NonNull IAnyUriItem base) {
190    if (relative == null) {
191      return null; // NOPMD - readability
192    }
193
194    @SuppressWarnings("null")
195    @NonNull URI resolvedUri = base.getValue().resolve(relative.getValue());
196    return IAnyUriItem.valueOf(resolvedUri);
197  }
