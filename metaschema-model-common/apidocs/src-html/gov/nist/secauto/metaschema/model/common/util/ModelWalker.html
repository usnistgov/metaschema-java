027package gov.nist.secauto.metaschema.model.common.util;
028
029import gov.nist.secauto.metaschema.model.common.IAssemblyDefinition;
030import gov.nist.secauto.metaschema.model.common.IAssemblyInstance;
031import gov.nist.secauto.metaschema.model.common.IChoiceInstance;
032import gov.nist.secauto.metaschema.model.common.IDefinition;
033import gov.nist.secauto.metaschema.model.common.IFieldDefinition;
034import gov.nist.secauto.metaschema.model.common.IFieldInstance;
035import gov.nist.secauto.metaschema.model.common.IFlagDefinition;
036import gov.nist.secauto.metaschema.model.common.IFlagInstance;
037import gov.nist.secauto.metaschema.model.common.IModelInstance;
038
039import java.util.Collection;
040
041import edu.umd.cs.findbugs.annotations.NonNull;

043/**
044 * Walks a Metaschema model. The "visit" methods can be implemented by child classes to perform
045 * processing on a visited node.
046 *
047 * @param <DATA>
048 *          state information that is carried through the walk
049 */
050public abstract class ModelWalker<DATA> {
051  /**
052   * Generate default state information.
053   *
054   * @return the state information
055   */
056  protected abstract DATA getDefaultData();
057
058  /**
059   * Will visit the provided metaschema flag definition.
060   *
061   * @param flag
062   *          the metaschema flag definition to walk
063   */
064  public void walk(@NonNull IFlagDefinition flag) {
065    walk(flag, getDefaultData());
066  }
067
068  /**
069   * Will visit the provided metaschema flag definition.
070   *
071   * @param flag
072   *          the metaschema flag definition to walk
073   * @param data
074   *          additional state information to operate on
075   */
076  public void walk(@NonNull IFlagDefinition flag, DATA data) {
077    visit(flag, data);
078  }
079
080  /**
081   * Will visit the provided metaschema field definition, and then walk the associated flag instances.
082   *
083   * @param field
084   *          the metaschema field definition to walk
085   */
086  public void walk(@NonNull IFieldDefinition field) {
087    walk(field, getDefaultData());
088  }
089
090  /**
091   * Will visit the provided metaschema field definition, and then walk the associated flag instances.
092   *
093   * @param field
094   *          the metaschema field definition to walk
095   * @param data
096   *          additional state information to operate on
097   */
098  public void walk(@NonNull IFieldDefinition field, DATA data) {
099    if (visit(field, data)) {
100      walkFlagInstances(field.getFlagInstances(), data);
101    }
102  }
103
104  /**
105   * Will visit the provided metaschema assembly definition, and then walk the associated flag and
106   * model instances.
107   *
108   * @param assembly
109   *          the metaschema assembly definition to walk
110   */
111  public void walk(@NonNull IAssemblyDefinition assembly) {
112    walk(assembly, getDefaultData());
113  }
114
115  /**
116   * Will visit the provided metaschema assembly definition, and then walk the associated flag and
117   * model instances.
118   *
119   * @param assembly
120   *          the metaschema assembly definition to walk
121   * @param data
122   *          additional state information to operate on
123   */
124  public void walk(@NonNull IAssemblyDefinition assembly, DATA data) {
125    if (visit(assembly, data)) {
126      walkFlagInstances(assembly.getFlagInstances(), data);
127      walkModelInstances(assembly.getModelInstances(), data);
128    }
129  }
130
131  /**
132   * Will visit the provided metaschema flag instance, and then walk the associated flag definition.
133   *
134   * @param instance
135   *          the metaschema flag instance to walk
136   * @param data
137   *          additional state information to operate on
138   */
139  public void walk(@NonNull IFlagInstance instance, DATA data) {
140    if (visit(instance, data)) {
141      walk(instance.getDefinition(), data);
142    }
143  }
144
145  /**
146   * Will visit the provided metaschema field instance, and then walk the associated field definition.
147   *
148   * @param instance
149   *          the metaschema field instance to walk
150   * @param data
151   *          additional state information to operate on
152   */
153  public void walk(@NonNull IFieldInstance instance, DATA data) {
154    if (visit(instance, data)) {
155      walk(instance.getDefinition(), data);
156    }
157  }
158
159  /**
160   * Will visit the provided metaschema assembly instance, and then walk the associated assembly
161   * definition.
162   *
163   * @param instance
164   *          the metaschema assembly instance to walk
165   * @param data
166   *          additional state information to operate on
167   */
168  public void walk(@NonNull IAssemblyInstance instance, DATA data) {
169    if (visit(instance, data)) {
170      walk(instance.getDefinition(), data);
171    }
172  }
173
174  /**
175   * Will visit the provided metaschema choice instance, and then walk the choice's child model
176   * instances.
177   *
178   * @param instance
179   *          the metaschema choice instance to walk
180   * @param data
181   *          additional state information to operate on
182   */
183  public void walk(@NonNull IChoiceInstance instance, DATA data) {
184    if (visit(instance, data)) {
185      walkModelInstances(instance.getModelInstances(), data);
186    }
187  }
188
189  /**
190   * Will walk the provided model definition.
191   *
192   * @param definition
193   *          the definition to walk
194   */
195  public void walkDefinition(@NonNull IDefinition definition) {
196    walkDefinition(definition, getDefaultData());
197  }
198
199  /**
200   * Will walk the provided model definition.
201   *
202   * @param definition
203   *          the definition to walk
204   * @param data
205   *          additional state information to operate on
206   */
207  public void walkDefinition(@NonNull IDefinition definition, DATA data) {
208    if (definition instanceof IAssemblyDefinition) {
209      walk((IAssemblyDefinition) definition, data);
210    } else if (definition instanceof IFieldDefinition) {
211      walk((IFieldDefinition) definition, data);
212    } else if (definition instanceof IFlagDefinition) {
213      walk((IFlagDefinition) definition, data);
214    }
215  }
216
217  /**
218   * Will walk each of the provided flag instances.
219   *
220   * @param instances
221   *          a collection of flag instances to visit
222   * @param data
223   *          additional state information to operate on
224   */
225  protected void walkFlagInstances(@NonNull Collection<? extends IFlagInstance> instances, DATA data) {
226    for (IFlagInstance instance : instances) {
227      assert instance != null;
228      walk(instance, data);
229    }
230  }
231
232  /**
233   * Will walk each of the provided model instances.
234   *
235   * @param instances
236   *          a collection of model instances to visit
237   * @param data
238   *          additional state information to operate on
239   */
240  protected void walkModelInstances(@NonNull Collection<? extends IModelInstance> instances, DATA data) {
241    for (IModelInstance instance : instances) {
242      assert instance != null;
243      walkModelInstance(instance, data);
244    }
245  }
246
247  /**
248   * Will walk the provided model instance.
249   *
250   * @param instance
251   *          the instance to walk
252   * @param data
253   *          additional state information to operate on
254   */
255  protected void walkModelInstance(@NonNull IModelInstance instance, DATA data) {
256    if (instance instanceof IAssemblyInstance) {
257      walk((IAssemblyInstance) instance, data);
258    } else if (instance instanceof IFieldInstance) {
259      walk((IFieldInstance) instance, data);
260    } else if (instance instanceof IChoiceInstance) {
261      walk((IChoiceInstance) instance, data);
262    }
263  }
264
265  /**
266   * Called when the provided definition is walked. This can be overridden by child classes to enable
267   * processing of the visited definition.
268   *
269   * @param def
270   *          the definition that is visited
271   * @param data
272   *          additional state information to operate on
273   */
274  protected abstract void visit(@NonNull IFlagDefinition def, DATA data);
275
276  /**
277   * Called when the provided definition is walked. This can be overridden by child classes to enable
278   * processing of the visited definition.
279   *
280   * @param def
281   *          the definition that is visited
282   * @param data
283   *          additional state information to operate on
284   * @return {@code true} if child instances are to be walked, or {@code false} otherwise
285   */
286  protected boolean visit(@NonNull IFieldDefinition def, DATA data) {
287    return true;
288  }
289
290  /**
291   * Called when the provided definition is walked. This can be overridden by child classes to enable
292   * processing of the visited definition.
293   *
294   * @param def
295   *          the definition that is visited
296   * @param data
297   *          additional state information to operate on
298   * @return {@code true} if child instances are to be walked, or {@code false} otherwise
299   */
300  protected boolean visit(@NonNull IAssemblyDefinition def, DATA data) {
301    return true;
302  }
303
304  /**
305   * Called when the provided instance is walked. This can be overridden by child classes to enable
306   * processing of the visited instance.
307   *
308   * @param instance
309   *          the instance that is visited
310   * @param data
311   *          additional state information to operate on
312   * @return {@code true} if the associated definition is to be walked, or {@code false} otherwise
313   */
314  protected boolean visit(@NonNull IFlagInstance instance, DATA data) {
315    return true;
316  }
317
318  /**
319   * Called when the provided instance is walked. This can be overridden by child classes to enable
320   * processing of the visited instance.
321   *
322   * @param instance
323   *          the instance that is visited
324   * @param data
325   *          additional state information to operate on
326   * @return {@code true} if the associated definition is to be walked, or {@code false} otherwise
327   */
328  protected boolean visit(@NonNull IFieldInstance instance, DATA data) {
329    return true;
330  }
331
332  /**
333   * Called when the provided instance is walked. This can be overridden by child classes to enable
334   * processing of the visited instance.
335   *
336   * @param instance
337   *          the instance that is visited
338   * @param data
339   *          additional state information to operate on
340   * @return {@code true} if the associated definition is to be walked, or {@code false} otherwise
341   */
342  protected boolean visit(@NonNull IAssemblyInstance instance, DATA data) {
343    return true;
344  }
345
346  /**
347   * Called when the provided instance is walked. This can be overridden by child classes to enable
348   * processing of the visited instance.
349   *
350   * @param instance
351   *          the instance that is visited
352   * @param data
353   *          additional state information to operate on
354   * @return {@code true} if the child instances are to be walked, or {@code false} otherwise
355   */
356  protected boolean visit(@NonNull IChoiceInstance instance, DATA data) {
357    return true;
358  }
359}
