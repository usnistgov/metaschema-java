030import gov.nist.secauto.metaschema.model.common.util.CustomCollectors;
031import gov.nist.secauto.metaschema.model.common.util.ObjectUtils;
032
033import org.apache.logging.log4j.LogManager;
034import org.apache.logging.log4j.Logger;
035
036import java.util.Collection;
037import java.util.List;
038import java.util.Locale;
039import java.util.Map;
040import java.util.function.Function;
041import java.util.function.Predicate;
042import java.util.stream.Collectors;
043import java.util.stream.Stream;
044
045import edu.umd.cs.findbugs.annotations.NonNull;
046import edu.umd.cs.findbugs.annotations.SuppressFBWarnings;

048/**
049 * Provides a common, abstract implementation of a {@link IMetaschema}.
050 */
051@SuppressWarnings("PMD.CouplingBetweenObjects")
052public abstract class AbstractMetaschema
053    implements IMetaschema {
054  private static final Logger LOGGER = LogManager.getLogger(AbstractMetaschema.class);
055
056  @NonNull
057  private final List<? extends IMetaschema> importedMetaschemas;
058  private Map<String, IFlagDefinition> exportedFlagDefinitions;
059  private Map<String, IFieldDefinition> exportedFieldDefinitions;
060  private Map<String, IAssemblyDefinition> exportedAssemblyDefinitions;
061
062  /**
063   * Construct a new Metaschema object.
064   *
065   * @param importedMetaschemas
066   *          the collection of Metaschema objects this Metaschema imports
067   */
068  public AbstractMetaschema(@NonNull List<? extends IMetaschema> importedMetaschemas) {
069    this.importedMetaschemas
070        = CollectionUtil.unmodifiableList(ObjectUtils.requireNonNull(importedMetaschemas, "importedMetaschema"));
071  }
072
073  @Override
074  @SuppressFBWarnings(value = "EI_EXPOSE_REP", justification = "interface doesn't allow modification")
075  public List<? extends IMetaschema> getImportedMetaschemas() {
076    return importedMetaschemas;
077  }
078
079  private Map<String, ? extends IMetaschema> getImportedMetaschemaByShortNames() {
080    return importedMetaschemas.stream().collect(Collectors.toMap(IMetaschema::getShortName, Function.identity()));
081  }
082
083  @Override
084  public IMetaschema getImportedMetaschemaByShortName(String name) {
085    return getImportedMetaschemaByShortNames().get(name);
086  }
087
088  @SuppressWarnings("null")
089  @Override
090  public Collection<? extends IFlagDefinition> getExportedFlagDefinitions() {
091    return getExportedFlagDefinitionMap().values();
092  }
093
094  @Override
095  public IFlagDefinition getExportedFlagDefinitionByName(String name) {
096    return getExportedFlagDefinitionMap().get(name);
097  }
098
099  private Map<String, ? extends IFlagDefinition> getExportedFlagDefinitionMap() {
100    initExports();
101    return exportedFlagDefinitions;
102  }
103
104  @SuppressWarnings("null")
105  @Override
106  public Collection<? extends IFieldDefinition> getExportedFieldDefinitions() {
107    return getExportedFieldDefinitionMap().values();
108  }
109
110  @Override
111  public IFieldDefinition getExportedFieldDefinitionByName(String name) {
112    return getExportedFieldDefinitionMap().get(name);
113  }
114
115  private Map<String, ? extends IFieldDefinition> getExportedFieldDefinitionMap() {
116    initExports();
117    return exportedFieldDefinitions;
118  }
119
120  @SuppressWarnings("null")
121  @Override
122  public Collection<? extends IAssemblyDefinition> getExportedAssemblyDefinitions() {
123    return getExportedAssemblyDefinitionMap().values();
124  }
125
126  @Override
127  public IAssemblyDefinition getExportedAssemblyDefinitionByName(String name) {
128    return getExportedAssemblyDefinitionMap().get(name);
129  }
130
131  private Map<String, ? extends IAssemblyDefinition> getExportedAssemblyDefinitionMap() {
132    initExports();
133    return exportedAssemblyDefinitions;
134  }
135
136  /**
137   * Processes the definitions exported by the Metaschema, saving a list of all exported by specific
138   * model types.
139   */
140  protected void initExports() {
141    synchronized (this) {
142      if (exportedFlagDefinitions == null) {
143        // Populate the stream with the definitions from this metaschema
144        Predicate<IDefinition> filter = IMetaschema.allNonLocalDefinitions();
145        Stream<? extends IFlagDefinition> flags = getFlagDefinitions().stream()
146            .filter(filter);
147        Stream<? extends IFieldDefinition> fields = getFieldDefinitions().stream()
148            .filter(filter);
149        Stream<? extends IAssemblyDefinition> assemblies = getAssemblyDefinitions().stream()
150            .filter(filter);
151
152        // handle definitions from any included metaschema
153        if (!getImportedMetaschemas().isEmpty()) {
154          Stream<? extends IFlagDefinition> importedFlags = Stream.empty();
155          Stream<? extends IFieldDefinition> importedFields = Stream.empty();
156          Stream<? extends IAssemblyDefinition> importedAssemblies = Stream.empty();
157
158          for (IMetaschema metaschema : getImportedMetaschemas()) {
159            importedFlags = Stream.concat(importedFlags, metaschema.getExportedFlagDefinitions().stream());
160            importedFields = Stream.concat(importedFields, metaschema.getExportedFieldDefinitions().stream());
161            importedAssemblies
162                = Stream.concat(importedAssemblies, metaschema.getExportedAssemblyDefinitions().stream());
163          }
164
165          flags = Stream.concat(importedFlags, flags);
166          fields = Stream.concat(importedFields, fields);
167          assemblies = Stream.concat(importedAssemblies, assemblies);
168        }
169
170        // Build the maps. Definitions from this Metaschema will take priority, with shadowing being
171        // reported when a definition from this Metaschema has the same name as an imported one
172        Map<String, IFlagDefinition> exportedFlagDefinitions = flags.collect(
173            CustomCollectors.toMap(
174                IFlagDefinition::getName,
175                CustomCollectors.identity(),
176                AbstractMetaschema::handleShadowedDefinitions));
177        Map<String, IFieldDefinition> exportedFieldDefinitions = fields.collect(
178            CustomCollectors.toMap(
179                IFieldDefinition::getName,
180                CustomCollectors.identity(),
181                AbstractMetaschema::handleShadowedDefinitions));
182        Map<String, IAssemblyDefinition> exportedAssemblyDefinitions = assemblies.collect(
183            CustomCollectors.toMap(
184                IAssemblyDefinition::getName,
185                CustomCollectors.identity(),
186                AbstractMetaschema::handleShadowedDefinitions));
187
188        this.exportedFlagDefinitions = exportedFlagDefinitions.isEmpty()
189            ? CollectionUtil.emptyMap()
190            : CollectionUtil.unmodifiableMap(exportedFlagDefinitions);
191        this.exportedFieldDefinitions = exportedFieldDefinitions.isEmpty()
192            ? CollectionUtil.emptyMap()
193            : CollectionUtil.unmodifiableMap(exportedFieldDefinitions);
194        this.exportedAssemblyDefinitions = exportedAssemblyDefinitions.isEmpty()
195            ? CollectionUtil.emptyMap()
196            : CollectionUtil.unmodifiableMap(exportedAssemblyDefinitions);
197      }
198    }
199  }
200
201  @SuppressWarnings({ "unused", "PMD.UnusedPrivateMethod" }) // used by lambda
202  private static <DEF extends IDefinition> DEF handleShadowedDefinitions(
203      @SuppressWarnings("unused") @NonNull String key, @NonNull DEF oldDef, @NonNull DEF newDef) {
204    if (!oldDef.equals(newDef) && LOGGER.isWarnEnabled()) {
205      LOGGER.warn("The {} '{}' from metaschema '{}' is shadowing '{}' from metaschema '{}'",
206          newDef.getModelType().name().toLowerCase(Locale.ROOT),
207          newDef.getName(),
208          newDef.getContainingMetaschema().getShortName(),
209          oldDef.getName(),
210          oldDef.getContainingMetaschema().getShortName());
211    }
212    return newDef;
213  }
