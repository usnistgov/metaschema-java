037import org.codehaus.stax2.XMLStreamWriter2;
038import org.codehaus.stax2.evt.XMLEventFactory2;
039
040import java.io.IOException;
041import java.util.List;
042import java.util.function.Predicate;
043import java.util.stream.Stream;
044
045import javax.xml.stream.XMLEventWriter;
046import javax.xml.stream.XMLStreamException;
047
048import edu.umd.cs.findbugs.annotations.NonNull;
049
050public interface IMarkupString<TYPE extends IMarkupString<TYPE>>
051    extends ICustomJavaDataType<TYPE> {
052  @NonNull
053  FlexmarkFactory getFlexmarkFactory();
054
055  @NonNull
056  Document getDocument();
057
058  // /**
059  // * Write HTML content to the provided {@code xmlStreamWriter} using the provided {@code
060  // namespace}.
061  // *
062  // * @param writer
063  // * the writer
064  // * @param namespace
065  // * the XML namespace for the HTML
066  // * @throws XMLStreamException
067  // * if an error occurred while writing
068  // */
069  // void writeHtml(@NonNull XMLStreamWriter2 writer, @NonNull String namespace) throws
070  // XMLStreamException;
071
072  @NonNull
073  String toHtml();
074
075  @NonNull
076  String toXHtml(@NonNull String namespace) throws XMLStreamException, IOException;
077
078  @NonNull
079  String toMarkdown();
080
081  @NonNull
082  String toMarkdown(@NonNull Formatter formatter);
083
084  /**
085   * Retrieve all nodes contained within this markup text as a stream.
086   *
087   * @return a depth first stream
088   */
089  @NonNull
090  Stream<Node> getNodesAsStream();
091
092  @NonNull
093  default List<InsertAnchorNode> getInserts() {
094    return getInserts(insert -> true);
095  }
096
097  /**
098   * Retrieve all insert statements that are contained within this markup text that match the provided
099   * filter.
100   *
101   * @param filter
102   *          a filter used to identify matching insert statements
103   * @return the matching insert statements
104   */
105  @NonNull
106  List<InsertAnchorNode> getInserts(
107      @NonNull Predicate<InsertAnchorNode> filter);
108
109  /**
110   * Determine if the Markup consists of block elements.
111   *
112   * @return {@code true} if the markup consists of block elements, or {@code false} otherwise
113   */
114  boolean isBlock();
115
116  void writeXHtml(
117      @NonNull String namespace,
118      @NonNull XMLStreamWriter2 streamWriter) throws XMLStreamException;
119
120  void writeXHtml(
121      @NonNull String namespace,
122      @NonNull XMLEventFactory2 eventFactory,
123      @NonNull XMLEventWriter eventWriter) throws XMLStreamException;
124}
