027package gov.nist.secauto.metaschema.model.common.metapath.item;
028
029import gov.nist.secauto.metaschema.model.common.IFlagDefinition;
030import gov.nist.secauto.metaschema.model.common.IFlagInstance;
031import gov.nist.secauto.metaschema.model.common.metapath.format.IPathFormatter;
032
033import java.net.URI;
034import java.util.Collection;
035import java.util.Collections;
036import java.util.List;
037import java.util.stream.Stream;
038
039import edu.umd.cs.findbugs.annotations.NonNull;
040import edu.umd.cs.findbugs.annotations.Nullable;
041
042public interface IFlagNodeItem extends IDefinitionNodeItem, IAtomicValuedItem {
043  @Override
044  default NodeItemType getNodeItemType() {
045    return NodeItemType.FLAG;
046  }
047
048  @Override
049  default IFlagNodeItem getNodeItem() {
050    return this;
051  }
052
053  @Override
054  default IModelNodeItem getParentContentNodeItem() {
055    return getParentNodeItem();
056  }
057
058  @Override
059  IModelNodeItem getParentNodeItem();
060
061  @Override
062  IFlagDefinition getDefinition();
063
064  @Override
065  IFlagInstance getInstance();
066
067  @Override
068  @Nullable
069  default URI getBaseUri() {
070    INodeItem parent = getParentNodeItem();
071    return parent == null ? null : parent.getBaseUri();
072  }
073
074  /**
075   * Flags do not have flag items. This call should return an empty collection.
076   */
077  @SuppressWarnings("null")
078  @Override
079  default Collection<? extends IFlagNodeItem> getFlags() {
080    // a flag does not have flags
081    return Collections.emptyList();
082  }
083
084  /**
085   * Flags do not have flag items. This call should return {@code null}.
086   */
087  @Override
088  default IFlagNodeItem getFlagByName(@NonNull String name) {
089    // a flag does not have flags
090    return null;
091  }
092
093  /**
094   * Flags do not have flag items. This call should return an empty stream.
095   */
096  @SuppressWarnings("null")
097  @Override
098  default @NonNull Stream<? extends IFlagNodeItem> flags() {
099    // a flag does not have flags
100    return Stream.empty();
101  }
102
103  /**
104   * Flags do not have model items. This call should return an empty collection.
105   */
106  @SuppressWarnings("null")
107  @Override
108  default @NonNull Collection<? extends List<? extends IModelNodeItem>> getModelItems() {
109    // a flag does not have model items
110    return Collections.emptyList();
111  }
112
113  /**
114   * Flags do not have model items. This call should return an empty list.
115   */
116  @SuppressWarnings("null")
117  @Override
118  default @NonNull List<? extends IModelNodeItem> getModelItemsByName(String name) {
119    // a flag does not have model items
120    return Collections.emptyList();
121  }
122
123  /**
124   * Flags do not have model items. This call should return an empty stream.
125   */
126  @SuppressWarnings("null")
127  @NonNull
128  @Override
129  default Stream<? extends IModelNodeItem> modelItems() {
130    // a flag does not have model items
131    return Stream.empty();
132  }
133
134  @Override
135  default @NonNull String format(@NonNull IPathFormatter formatter) {
136    return formatter.formatFlag(this);
137  }
138
139  @Override
140  default <RESULT, CONTEXT> RESULT accept(@NonNull INodeItemVisitor<RESULT, CONTEXT> visitor, CONTEXT context) {
141    return visitor.visitFlag(this, context);
142  }
143}
