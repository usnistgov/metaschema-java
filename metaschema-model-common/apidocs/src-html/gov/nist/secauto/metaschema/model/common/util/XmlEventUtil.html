081  @SuppressWarnings("null")
082  @NonNull
083  private static Object escape(@NonNull String data) {
084    return data.chars().mapToObj(c -> (char) c).map(c -> escape(c)).collect(Collectors.joining());
085  }
086
087  @SuppressWarnings("null")
088  @NonNull
089  private static String escape(char ch) {
090    String retval;
091    switch (ch) {
092    case '\n':
093      retval = "\\n";
094      break;
095    case '\r':
096      retval = "\\r";
097      break;
098    default:
099      retval = String.valueOf(ch);
100      break;
101    }
102    return retval;
103  }

105  /**
106   * Generate a message suitable for logging that describes the provided {@link XMLEvent}.
107   *
108   * @param xmlEvent
109   *          the event to generate the message for
110   * @return the message
111   */
112  @NonNull
113  public static CharSequence toString(XMLEvent xmlEvent) {
114    CharSequence retval;
115    if (xmlEvent == null) {
116      retval = "EOF";
117    } else {
118      @SuppressWarnings("null")
119      @NonNull StringBuilder builder = new StringBuilder()
120          .append(toEventName(xmlEvent));
121      QName name = toQName(xmlEvent);
122      if (name != null) {
123        builder.append(": ").append(name.toString());
124      }
125      if (xmlEvent.isCharacters()) {
126        String text = xmlEvent.asCharacters().getData();
127        if (text != null) {
128          builder.append(" '").append(escape(text)).append('\'');
129        }
130      }
131      Location location = toLocation(xmlEvent);
132      if (location != null) {
133        builder.append(" at ").append(toString(location));
134      }
135      retval = builder;
136    }
137    return retval;
138  }

140  /**
141   * Generates a message for the provided {@link Location}.
142   *
143   * @param location
144   *          the location to generate the message for
145   * @return the message
146   */
147  @SuppressWarnings("null")
148  @NonNull
149  public static CharSequence toString(@Nullable Location location) {
150    return location == null ? "unknown"
151        : new StringBuilder()
152            .append(location.getLineNumber())
153            .append(':')
154            .append(location.getColumnNumber());
155  }

157  /**
158   * Generates a string containing the current event and location of the stream reader.
159   *
160   * @param reader
161   *          the stream reader
162   * @return the generated string
163   */
164  @NonNull
165  public static CharSequence toString(@NonNull XMLStreamReader2 reader) { // NO_UCD (unused code)
166    int type = reader.getEventType();
167
168    @SuppressWarnings("null")
169    @NonNull StringBuilder builder = new StringBuilder().append(toEventName(type));
170    QName name = reader.getName();
171    if (name != null) {
172      builder.append(": ").append(name.toString());
173    }
174    if (XMLStreamConstants.CHARACTERS == type) {
175      String text = reader.getText();
176      if (text != null) {
177        builder.append(" '").append(escape(text)).append('\'');
178      }
179    }
180    Location location = reader.getLocation();
181    if (location != null) {
182      builder.append(" at ").append(toString(location));
183    }
184    return builder;
185  }

187  /**
188   * Retrieve the resource location of {@code event}.
189   *
190   * @param event
191   *          the event to identify the location for
192   * @return the location or {@code null} if the location is unknown
193   */
194  @Nullable
195  public static Location toLocation(@NonNull XMLEvent event) {
196    Location retval = null;
197    if (event.isStartElement()) {
198      StartElement start = event.asStartElement();
199      retval = start.getLocation();
200    } else if (event.isEndElement()) {
201      EndElement end = event.asEndElement();
202      retval = end.getLocation();
203    } else if (event.isCharacters()) {
204      Characters characters = event.asCharacters();
205      retval = characters.getLocation();
206    }
207    return retval;
208  }

210  /**
211   * Retrieve the name of the node associated with {@code event}.
212   *
213   * @param event
214   *          the event to get the {@link QName} for
215   * @return the name of the node or {@code null} if the event is not a start or end element
216   */
217  @Nullable
218  public static QName toQName(@NonNull XMLEvent event) {
219    QName retval = null;
220    if (event.isStartElement()) {
221      StartElement start = event.asStartElement();
222      retval = start.getName();
223    } else if (event.isEndElement()) {
224      EndElement end = event.asEndElement();
225      retval = end.getName();
226    }
227    return retval;
228  }

259  /**
260   * Advance through XMLEvents until the event type identified by {@code eventType} is reached or the
261   * end of stream is found.
262   *
263   * @param reader
264   *          the event reader to advance
265   * @param eventType
266   *          the event type to stop on as defined by {@link XMLStreamConstants}
267   * @return the next event of the specified type or {@code null} if the end of stream is reached
268   * @throws XMLStreamException
269   *           if an error occurred while advancing the stream
270   */
271  @Nullable
272  public static XMLEvent advanceTo(@NonNull XMLEventReader2 reader, int eventType)
273      throws XMLStreamException { // NO_UCD (unused code)
274    XMLEvent xmlEvent;
275    do {
276      xmlEvent = reader.nextEvent();
277      // if (LOGGER.isWarnEnabled()) {
278      // LOGGER.warn("skipping over: {}", XmlEventUtil.toString(xmlEvent));
279      // }
280      if (xmlEvent.isStartElement()) {
281        advanceTo(reader, XMLStreamConstants.END_ELEMENT);
282        // skip this end element
283        xmlEvent = reader.nextEvent();
284        // if (LOGGER.isDebugEnabled()) {
285        // LOGGER.debug("skipping over: {}", XmlEventUtil.toString(xmlEvent));
286        // }
287      }
288    } while (reader.hasNext() && (xmlEvent = reader.peek()).getEventType() != eventType);
289    return xmlEvent;
290  }

292  /**
293   * Skip over any processing instructions.
294   *
295   * @param reader
296   *          the event reader to advance
297   * @return the last processing instruction event or the reader's next event if no processing
298   *         instruction was found
299   * @throws XMLStreamException
300   *           if an error occurred while advancing the stream
301   */
302  @NonNull
303  public static XMLEvent skipProcessingInstructions(@NonNull XMLEventReader2 reader) throws XMLStreamException {
304    XMLEvent nextEvent;
305    while ((nextEvent = reader.peek()).isProcessingInstruction()) {
306      nextEvent = reader.nextEvent();
307    }
308    return nextEvent;
309  }

311  /**
312   * Skip over any whitespace.
313   *
314   * @param reader
315   *          the event reader to advance
316   * @return the last character event containing whitespace or the reader's next event if no character
317   *         event was found
318   * @throws XMLStreamException
319   *           if an error occurred while advancing the stream
320   */
321  @SuppressWarnings("null")
322  @NonNull
323  public static XMLEvent skipWhitespace(@NonNull XMLEventReader2 reader) throws XMLStreamException {
324    @NonNull XMLEvent nextEvent;
325    while ((nextEvent = reader.peek()).isCharacters()) {
326      Characters characters = nextEvent.asCharacters();
327      String data = characters.getData();
328      if (WHITESPACE_ONLY.matcher(data).matches()) {
329        nextEvent = reader.nextEvent();
330      } else {
331        break;
332      }
333    }
334    return nextEvent;
335  }

352    return isNextEventEndElement(reader, name.getLocalPart(), name.getNamespaceURI());
353  }

355  /**
356   * Determine if the next event from {@code reader} is an end element whose name matches the provided
357   * {@code expectedLocalName} and {@code expectedNamespace}.
358   *
359   * @param reader
360   *          the event reader
361   * @param expectedLocalName
362   *          the expected element name
363   * @param expectedNamespace
364   *          the expected element namespace
365   * @return {@code true} if the next event matches the {@code name}
366   * @throws XMLStreamException
367   *           if an error occurred while looking at the next event
368   */
369  public static boolean isNextEventEndElement(@NonNull XMLEventReader2 reader, @NonNull String expectedLocalName,
370      String expectedNamespace) throws XMLStreamException {
371    Objects.requireNonNull(reader, "reader");
372    Objects.requireNonNull(expectedLocalName, "expectedLocalName");
373    XMLEvent event = reader.peek();
374
375    boolean retval;
376    if (event.isEndElement()) {
377      EndElement endElement = event.asEndElement();
378      QName name = endElement.getName();
379      retval = expectedLocalName.equals(name.getLocalPart())
380          && (expectedNamespace == null || expectedNamespace.equals(name.getNamespaceURI()));
381    } else {
382      retval = false;
383    }
384    return retval;
385  }

434  /**
435   * Consume the next event from {@code reader} and assert that this event is of the type identified
436   * by {@code presumedEventType} and has the name identified by {@code presumedName}.
437   *
438   * @param reader
439   *          the event reader
440   * @param presumedEventType
441   *          the expected event type as defined by {@link XMLStreamConstants}
442   * @param presumedName
443   *          the expected name of the node associated with the event
444   * @return the next event
445   * @throws XMLStreamException
446   *           if an error occurred while looking at the next event
447   */
448  public static XMLEvent consumeAndAssert(XMLEventReader2 reader, int presumedEventType, QName presumedName)
449      throws XMLStreamException {
450    XMLEvent retval = reader.nextEvent();
451
452    int eventType = retval.getEventType();
453    QName name = toQName(retval);
454    assert eventType == presumedEventType
455        && (presumedName == null
456            || presumedName.equals(name)) : generateAssertMessage(
457                retval,
458                presumedEventType,
459                presumedName);
460    return retval;
461  }

497    QName name = toQName(nextEvent);

503  private static CharSequence generateAssertMessage(
504      @NonNull XMLEvent event,
505      int presumedEventType,
506      @Nullable QName presumedName) {
507    StringBuilder builder = new StringBuilder(64);
508    builder
509        .append("Expected XML ")
510        .append(toEventName(presumedEventType));
511
512    if (presumedName != null) {
513      builder.append(" for QName '")
514          .append(presumedName.toString());
515    }
516    builder.append("', instead found ")
517        .append(toString(event));
518
519    Location location = toLocation(event);
520    if (location != null) {
521      builder.append(" at ").append(toString(location));
522    }
523    return builder;
524  }

526  /**
527   * Skips events specified by {@code events}.
528   *
529   * @param reader
530   *          the event reader
531   * @param events
532   *          the events to skip
533   * @return the next non-mataching event returned by {@link XMLEventReader2#peek()}, or {@code null}
534   *         if there was no next event
535   * @throws XMLStreamException
536   *           if an error occurred while reading
537   */
538  public static XMLEvent skipEvents(XMLEventReader2 reader, int... events) throws XMLStreamException {
539    Set<Integer> skipEvents = IntStream.of(events).boxed().collect(Collectors.toSet());
540
541    XMLEvent nextEvent = null;
542    while (reader.hasNext()) {
543      nextEvent = reader.peek();
544      if (!skipEvents.contains(nextEvent.getEventType())) {
545        break;
546      }
547      reader.nextEvent();
548    }
549    return nextEvent;
550  }
