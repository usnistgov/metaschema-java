045public final class FunctionUtils {
046  public static final MathContext MATH_CONTEXT = MathContext.DECIMAL64;
047
048  private FunctionUtils() {

052  /**
053   * Converts a {@link INumericItem} value to an integer value.
054   *
055   * @param value
056   *          the value to convert
057   * @return the integer value
058   * @throws ArithmeticException
059   *           if the provided value will not exactly fit in an {@code int}
060   */
061  public static int asInteger(@NonNull INumericItem value) {
062    return asInteger(value.asInteger());
063  }
064
065  /**
066   * Converts a {@link BigInteger} value to an integer value.
067   *
068   * @param value
069   *          the value to convert
070   * @return the integer value
071   * @throws ArithmeticException
072   *           if the provided value will not exactly fit in an {@code int}
073   */
074  public static int asInteger(@NonNull BigInteger value) {
075    return value.intValueExact();
076  }
077
078  /**
079   * Converts a {@link INumericItem} value to a long value.
080   *
081   * @param value
082   *          the value to convert
083   * @return the long value
084   * @throws ArithmeticException
085   *           if the provided value will not exactly fit in an {@code long}
086   */
087  public static long asLong(@NonNull INumericItem value) {
088    return asLong(value.asInteger());
089  }
090
091  /**
092   * Converts a {@link BigInteger} value to a long value.
093   *
094   * @param value
095   *          the value to convert
096   * @return the long value
097   * @throws ArithmeticException
098   *           if the provided value will not exactly fit in an {@code long}
099   */
100  public static long asLong(@NonNull BigInteger value) {
101    return value.longValueExact();
102  }
103
104  /**
105   * Retrieves the first item in a sequence. If the sequence is empty, a {@code null} result is
106   * returned. If requireSingleton is {@code true} and the sequence contains more than one item, a
107   * {@link TypeMetapathException} is thrown.
108   *
109   * @param <ITEM>
110   *          the item type to return derived from the provided sequence
111   * @param sequence
112   *          the sequence to retrieve the first item from
113   * @param requireSingleton
114   *          if {@code true} then a {@link TypeMetapathException} is thrown if the sequence contains
115   *          more than one item
116   * @return {@code null} if the sequence is empty, or the item otherwise
117   * @throws TypeMetapathException
118   *           if the sequence contains more than one item and requireSingleton is {@code true}
119   */
120  @Nullable
121  public static <ITEM extends IItem> ITEM getFirstItem(@NonNull ISequence<ITEM> sequence, boolean requireSingleton) {
122    @Nullable ITEM retval = null;
123    if (!sequence.isEmpty()) {
124      List<ITEM> items = sequence.asList();
125      if (requireSingleton && items.size() != 1) {
126        throw new InvalidTypeMetapathException(
127            null,
128            String.format("sequence expected to contain one item, but found '%d'", items.size()));
129      }
130      retval = items.iterator().next();
131    }
132    return retval;
133  }
134
135  /**
136   * Gets the first item of the provided sequence as a {@link INumericItem} value. If the sequence is
137   * empty, then a {@code null} value is returned.
138   *
139   * @param sequence
140   *          a Metapath sequence containing the value to convert
141   * @param requireSingleton
142   *          if {@code true} then a {@link TypeMetapathException} is thrown if the sequence contains
143   *          more than one item
144   * @return the numeric item value, or {@code null} if the result is an empty sequence
145   * @throws TypeMetapathException
146   *           if the sequence contains more than one item, or the item cannot be cast to a numeric
147   *           value
148   *
149   */
150  @Nullable
151  public static INumericItem toNumeric(@NonNull ISequence<?> sequence, boolean requireSingleton) {
152    IItem item = getFirstItem(sequence, requireSingleton);
153    return item == null ? null : toNumeric(item);
154  }
155
156  /**
157   * Gets the provided item value as a {@link INumericItem} value.
158   *
159   * @param item
160   *          the value to convert
161   * @return the numeric item value
162   * @throws TypeMetapathException
163   *           if the sequence contains more than one item, or the item cannot be cast to a numeric
164   *           value
165   */
166  @NonNull
167  public static INumericItem toNumeric(@NonNull IItem item) {
168    // atomize
169    IAnyAtomicItem atomicItem = FnData.fnDataItem(item);
170    return toNumeric(atomicItem);
171  }
172
173  /**
174   * Gets the provided item value as a {@link INumericItem} value.
175   *
176   * @param item
177   *          the value to convert
178   * @return the numeric item value
179   * @throws TypeMetapathException
180   *           if the item cannot be cast to a numeric value
181   */
182  @NonNull
183  public static INumericItem toNumeric(@NonNull IAnyAtomicItem item) {
184    try {
185      return IDecimalItem.cast(item);
186    } catch (InvalidValueForCastFunctionException ex) {
187      throw new InvalidTypeMetapathException(item, ex.getLocalizedMessage(), ex);
188    }
189  }
190
191  /**
192   * Gets the provided item value as a {@link INumericItem} value. If the item is {@code null}, then a
193   * {@code null} value is returned.
194   *
195   * @param item
196   *          the value to convert
197   * @return the numeric item value
198   * @throws TypeMetapathException
199   *           if the item cannot be cast to a numeric value
200   */
201  @Nullable
202  public static INumericItem toNumericOrNull(@Nullable IAnyAtomicItem item) {
203    return item == null ? null : toNumeric(item);
204  }
205
206  @SuppressWarnings("unchecked")
207  @NonNull
208  public static <TYPE extends IItem> TYPE asType(@NonNull IItem item) {
209    return (TYPE) item;
210  }
211
212  @SuppressWarnings("unchecked")
213  @NonNull
214  public static <TYPE extends IItem> ISequence<TYPE> asType(@NonNull ISequence<?> sequence) {
215    return (ISequence<TYPE>) sequence;
216  }
