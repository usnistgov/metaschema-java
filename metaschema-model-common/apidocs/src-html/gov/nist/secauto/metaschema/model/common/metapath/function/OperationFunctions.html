218  @NonNull
219  public static IYearMonthDurationItem opDivideYearMonthDuration(@NonNull IYearMonthDurationItem arg1,
220      @NonNull INumericItem arg2)
221      throws DateTimeFunctionException {
222    IIntegerItem totalMonths = IIntegerItem.valueOf(arg1.getValue().toTotalMonths());
223    IIntegerItem result = opNumericIntegerDivide(totalMonths, arg2);
224    int months;
225    try {
226      months = FunctionUtils.asInteger(result.asInteger());
227    } catch (ArithmeticException ex) {
228      throw new DateTimeFunctionException(DateTimeFunctionException.DURATION_OVERFLOW_UNDERFLOW_ERROR,
229          "Overflow/underflow in duration operation.", ex);
230    }
231    int years = months / 12;
232    months = months % 12;
233    return IYearMonthDurationItem.valueOf(years, months, 0);
234  }

417  @NonNull
418  public static INumericItem opNumericDivide(@NonNull INumericItem dividend, @NonNull INumericItem divisor) {
419    INumericItem retval;
420    if (dividend instanceof IDecimalItem || divisor instanceof IDecimalItem) {
421      // create a decimal result
422      BigDecimal decimalDivisor = divisor.asDecimal();
423
424      if (BigDecimal.ZERO.equals(decimalDivisor)) {
425        throw new ArithmeticFunctionException(ArithmeticFunctionException.DIVISION_BY_ZERO,
426            ArithmeticFunctionException.DIVISION_BY_ZERO_MESSAGE);
427      }
428
429      BigDecimal decimalDividend = dividend.asDecimal();
430
431      @SuppressWarnings("null")
432      @NonNull BigDecimal result = decimalDividend.divide(decimalDivisor, FunctionUtils.MATH_CONTEXT);
433      retval = IDecimalItem.valueOf(result);
434    } else {
435      // create an integer result
436      BigInteger integerDivisor = divisor.asInteger();
437
438      if (BigInteger.ZERO.equals(integerDivisor)) {
439        throw new ArithmeticFunctionException(ArithmeticFunctionException.DIVISION_BY_ZERO,
440            ArithmeticFunctionException.DIVISION_BY_ZERO_MESSAGE);
441      }
442
443      BigInteger integerDividend = dividend.asInteger();
444
445      @SuppressWarnings("null")
446      @NonNull BigInteger result = integerDividend.divide(integerDivisor);
447      retval = IIntegerItem.valueOf(result);
448    }
449    return retval;
450  }

486  /**
487   * Based on XPath 3.1
488   * <a href="https://www.w3.org/TR/xpath-functions-31/#func-numeric-mod">func:numeric-mod</a>.
489   *
490   * @param dividend
491   *          the number to be divided
492   * @param divisor
493   *          the number to divide by
494   * @return the remainder
495   */
496  @NonNull
497  public static INumericItem opNumericMod(@NonNull INumericItem dividend, @NonNull INumericItem divisor) {
498    BigDecimal decimalDivisor = divisor.asDecimal();
499
500    if (BigDecimal.ZERO.equals(decimalDivisor)) {
501      throw new ArithmeticFunctionException(ArithmeticFunctionException.DIVISION_BY_ZERO,
502          ArithmeticFunctionException.DIVISION_BY_ZERO_MESSAGE);
503    }
504
505    BigDecimal decimalDividend = dividend.asDecimal();
506
507    INumericItem retval;
508    if (BigDecimal.ZERO.equals(decimalDividend)) {
509      retval = dividend;
510    } else {
511      @SuppressWarnings("null")
512      @NonNull BigDecimal result = decimalDividend.remainder(decimalDivisor, FunctionUtils.MATH_CONTEXT);
513      retval = IDecimalItem.valueOf(result);
514    }
515    return retval;
516  }

519  public static INumericItem opNumericUnaryMinus(@NonNull INumericItem item) {
520    INumericItem retval;
521    if (item instanceof IDecimalItem) {
522      // create a decimal result
523      BigDecimal decimal = item.asDecimal();
524
525      @SuppressWarnings("null")
526      @NonNull BigDecimal result = decimal.negate(FunctionUtils.MATH_CONTEXT);
527      retval = IDecimalItem.valueOf(result);
528    } else if (item instanceof IIntegerItem) {
529      // create a decimal result
530      BigInteger integer = item.asInteger();
531
532      @SuppressWarnings("null")
533      @NonNull BigInteger result = integer.negate();
534      retval = IIntegerItem.valueOf(result);
535    } else {
536      throw new InvalidTypeMetapathException(item);
537    }
538    return retval;
539  }
