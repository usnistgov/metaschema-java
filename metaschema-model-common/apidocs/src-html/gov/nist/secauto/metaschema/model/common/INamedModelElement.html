032import java.util.Map;

034
035import javax.xml.namespace.QName;
036
037import edu.umd.cs.findbugs.annotations.NonNull;
038import edu.umd.cs.findbugs.annotations.Nullable;
039
040/**
041 * A marker interface for Metaschema constructs that can be members of a Metaschema definition's
042 * model that are named.
043 */
044public interface INamedModelElement extends IModelElement {
045  /**
046   * The formal display name.
047   *
048   * @return the formal name or {@code null} if not defined
049   */
050  @Nullable
051  String getFormalName();
052
053  /**
054   * The resolved formal display name, which allows an instance to override a definition's name.
055   *
056   * @return the formal name or {@code null} if not defined
057   */
058  @Nullable
059  default String getEffectiveFormalName() {
060    return getFormalName();
061  }
062
063  /**
064   * Get the text that describes the basic use of the element.
065   *
066   * @return a line of markup text or {@code null} if not defined
067   */
068  @Nullable
069  MarkupLine getDescription();
070
071  /**
072   * Get the text that describes the basic use of the element, which allows an instance to override a
073   * definition's description.
074   *
075   * @return a line of markup text or {@code null} if not defined
076   */
077  @Nullable
078  default MarkupLine getEffectiveDescription() {
079    return getDescription();
080  }
081
082  /**
083   * Get the mapping of property name to values for the model element.
084   *
085   * @return the mapping
086   */
087  @NonNull
088  Map<QName, Set<String>> getProperties();
089
090  /**
091   * Determine if a property is defined.
092   *
093   * @param qname
094   *          the qualified name of the property
095   * @return {@code true} if the property is defined or {@code false} otherwise
096   */
097  default boolean hasProperty(@NonNull QName qname) {
098    return getProperties().containsKey(qname);
099  }
100
101  /**
102   * Get the values associated with a given property.
103   *
104   * @param qname
105   *          the qualified name of the property
106   * @return the values or an empty set
107   */
108  @NonNull
109  default Set<String> getPropertyValues(@NonNull QName qname) {
110    Set<String> retval = getProperties().get(qname);
111    if (retval == null) {
112      retval = CollectionUtil.emptySet();
113    }
114    return retval;
115  }
116
117  /**
118   * Determine if a given property, with a given {@code qname}, has the identified {@code value}.
119   *
120   * @param qname
121   *          the qualified name of the property
122   * @param value
123   *          the expected property value
124   * @return {@code true} if the property value is defined or {@code false} otherwise
125   */
126  default boolean hasPropertyValue(@NonNull QName qname, @NonNull String value) {
127    Set<String> values = getProperties().get(qname);
128    return values != null && values.contains(value);
129  }
130

132  // default QName getXmlQName() {
133  // String namespace = getXmlNamespace();
134  //
135  // @NonNull
136  // QName retval;
137  // if (namespace != null) {
138  // retval = new QName(namespace, getEffectiveName());
139  // } else {
140  // retval = new QName(getEffectiveName());
141  // }
142  // return retval;
143  // }
144
145  /**
146   * Get the name used for the associated property in JSON/YAML.
147   *
148   * @return the JSON property name
149   */
150  @NonNull
151  default String getJsonName() {
152    return getEffectiveName();
153  }
154
155  /**
156   * Get the name to use based on the provided names. This method will return the use name provided by
157   * {@link #getUseName()} if the call is not {@code null}, and fall back to the name provided by
158   * {@link #getName()} otherwise. This is the model name to use for the for an instance where the
159   * instance is referenced.
160   *
161   * @return the use name if available, or the name if not
162   *
163   * @see #getUseName()
164   * @see #getName()
165   */
166  @NonNull
167  default String getEffectiveName() {
168    @Nullable String useName = getUseName();
169    return useName == null ? getName() : useName;
170  }
171
172  /**
173   * Retrieve the name of the model element.
174   *
175   * @return the name
176   */
177  @NonNull
178  String getName();
179
180  /**
181   * Retrieve the name to use for the model element, instead of the name.
182   *
183   * @return the use name or {@code null} if no use name is defined
184   */
185  @Nullable
186  String getUseName();
187}
