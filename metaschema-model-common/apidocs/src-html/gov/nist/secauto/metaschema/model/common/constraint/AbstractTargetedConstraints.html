027package gov.nist.secauto.metaschema.model.common.constraint;
028
029import gov.nist.secauto.metaschema.model.common.IAssemblyDefinition;
030import gov.nist.secauto.metaschema.model.common.IDefinition;
031import gov.nist.secauto.metaschema.model.common.IFieldDefinition;
032import gov.nist.secauto.metaschema.model.common.IFlagDefinition;
033import gov.nist.secauto.metaschema.model.common.metapath.MetapathExpression;
034
035import java.util.List;
036import java.util.Locale;
037
038import edu.umd.cs.findbugs.annotations.NonNull;
039
040/**
041 * Provides an base implementation for a set of external constraints that target a given definition
042 * using a target Metapath expression.
043 *
044 * @param <T>
045 *          the type of the constraint container
046 */
047public abstract class AbstractTargetedConstraints<T extends IValueConstraintSupport> implements ITargetedConstaints {
048  @NonNull
049  private final MetapathExpression targetExpression;
050  @NonNull
051  private final T constraints;
052
053  public AbstractTargetedConstraints(@NonNull MetapathExpression targetExpression, @NonNull T constraints) {
054    this.targetExpression = targetExpression;
055    this.constraints = constraints;
056  }
057
058  @Override
059  public MetapathExpression getTargetExpression() {
060    return targetExpression;
061  }
062
063  @NonNull
064  protected T getConstraintSupport() {
065    return constraints;
066  }
067
068  @Override
069  public List<? extends IConstraint> getConstraints() {
070    return getConstraintSupport().getConstraints();
071  }
072
073  @Override
074  public List<? extends IAllowedValuesConstraint> getAllowedValuesConstraints() {
075    return getConstraintSupport().getAllowedValuesConstraints();
076  }
077
078  @Override
079  public List<? extends IMatchesConstraint> getMatchesConstraints() {
080    return getConstraintSupport().getMatchesConstraints();
081  }
082
083  @Override
084  public List<? extends IIndexHasKeyConstraint> getIndexHasKeyConstraints() {
085    return getConstraintSupport().getIndexHasKeyConstraints();
086  }
087
088  @Override
089  public List<? extends IExpectConstraint> getExpectConstraints() {
090    return getConstraintSupport().getExpectConstraints();
091  }
092
093  @Override
094  public void addConstraint(@NonNull IAllowedValuesConstraint constraint) {
095    getConstraintSupport().addConstraint(constraint);
096  }
097
098  @Override
099  public void addConstraint(@NonNull IMatchesConstraint constraint) {
100    getConstraintSupport().addConstraint(constraint);
101  }
102
103  @Override
104  public void addConstraint(@NonNull IIndexHasKeyConstraint constraint) {
105    getConstraintSupport().addConstraint(constraint);
106  }
107
108  @Override
109  public void addConstraint(@NonNull IExpectConstraint constraint) {
110    getConstraintSupport().addConstraint(constraint);
111  }
112
113  @SuppressWarnings("null")
114  protected void applyTo(@NonNull IDefinition definition) {
115    getAllowedValuesConstraints().forEach(constraint -> definition.addConstraint(constraint));
116    getMatchesConstraints().forEach(constraint -> definition.addConstraint(constraint));
117    getIndexHasKeyConstraints().forEach(constraint -> definition.addConstraint(constraint));
118    getExpectConstraints().forEach(constraint -> definition.addConstraint(constraint));
119  }
120
121  @Override
122  public void target(@NonNull IFlagDefinition definition) {
123    throw new IllegalStateException(
124        String.format("The targeted definition '%s' from metaschema '%s' is not a %s definition.",
125            definition.getEffectiveName(),
126            definition.getContainingMetaschema().getQName().toString(),
127            definition.getModelType().name().toLowerCase(Locale.ROOT)));
128  }
129
130  @Override
131  public void target(@NonNull IFieldDefinition definition) {
132    throw new IllegalStateException(
133        String.format("The targeted definition '%s' from metaschema '%s' is not a %s definition.",
134            definition.getEffectiveName(),
135            definition.getContainingMetaschema().getQName().toString(),
136            definition.getModelType().name().toLowerCase(Locale.ROOT)));
137  }
138
139  @Override
140  public void target(@NonNull IAssemblyDefinition definition) {
141    throw new IllegalStateException(
142        String.format("The targeted definition '%s' from metaschema '%s' is not a %s definition.",
143            definition.getEffectiveName(),
144            definition.getContainingMetaschema().getQName().toString(),
145            definition.getModelType().name().toLowerCase(Locale.ROOT)));
146  }
147}
