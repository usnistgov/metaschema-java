58  public abstract class AbstractMarkupString<TYPE extends AbstractMarkupString<TYPE>>
59      implements IMarkupText, ICustomJavaDataType<TYPE> {
60    private static final String DEFAULT_HTML_NS = "http://www.w3.org/1999/xhtml";
61    private static final String DEFAULT_HTML_PREFIX = "";
62    @NotNull
63    private final Document document;
64  
65    //
66    // public AbstractMarkupString() {
67    // this(null);
68    // }
69  
70    public AbstractMarkupString(@NotNull Document document) {
71      this.document = document;
72    }
73  
74    @Override
75    public Document getDocument() {
76      return document;
77    }
78  
79    @Override
80    public void writeHtml(@NotNull XMLStreamWriter2 xmlStreamWriter, @NotNull String namespace)
81        throws XMLStreamException {
82      MarkupXmlStreamWriter writingVisitor
83          = new MarkupXmlStreamWriter(namespace, this instanceof MarkupMultiline);
84      writingVisitor.visitChildren(getDocument(), xmlStreamWriter);
85      xmlStreamWriter.flush();
86    }
87  
88    @Override
89    public void writeHtml(@NotNull OutputStream os, @Nullable String namespace, @Nullable String prefix)
90        throws XMLStreamException {
91      XMLOutputFactory2 factory = (XMLOutputFactory2) WstxOutputFactory.newInstance();
92      factory.setProperty(WstxOutputProperties.P_OUTPUT_VALIDATE_STRUCTURE, false);
93      XMLStreamWriter2 xmlStreamWriter = (XMLStreamWriter2) factory.createXMLStreamWriter(os);
94  
95      String effectiveNamespace = namespace == null ? DEFAULT_HTML_NS : namespace;
96      String effectivePrefix = prefix == null ? DEFAULT_HTML_PREFIX : prefix;
97      NamespaceContext nsContext = MergedNsContext.construct(xmlStreamWriter.getNamespaceContext(),
98          List.of(NamespaceEventImpl.constructNamespace(null, effectivePrefix, effectiveNamespace)));
99      xmlStreamWriter.setNamespaceContext(nsContext);
100 
101     writeHtml(xmlStreamWriter, effectiveNamespace);
102   }
103 
104   @Override
105   public String toHtml() {
106     return toHTML(FlexmarkFactory.instance().getHtmlRenderer());
107   }
108 
109   @NotNull
110   protected String toHTML(HtmlRenderer renderer) {
111     return renderer.render(getDocument());
112   }
113 
114   @Override
115   public String toMarkdown() {
116     return toMarkdown(FlexmarkFactory.instance().getFormatter());
117   }
118 
119   @NotNull
120   @Override
121   public String toMarkdown(Formatter formatter) {
122     return formatter.render(getDocument());
123   }
124 
125   @Override
126   public String toMarkdownYaml() {
127     return toMarkdownYaml(FlexmarkFactory.instance().getFormatter());
128   }
129 
130   @NotNull
131   protected String toMarkdownYaml(Formatter formatter) {
132     return formatter.render(getDocument());
133   }
134 
135   @SuppressWarnings("null")
136   @Override
137   public Stream<Node> getNodesAsStream() {
138     return Stream.concat(Stream.of(getDocument()),
139         StreamSupport.stream(getDocument().getDescendants().spliterator(), false));
140   }
141 
142   @Override
143   public List<InsertAnchorNode> getInserts(
144       @NotNull Predicate<InsertAnchorNode> filter) {
145     InsertVisitor visitor = new InsertVisitor(filter);
146     visitor.visitChildren(getDocument());
147     return visitor.getInserts();
148   }
149 
150   @Override
151   public String toString() {
152     return new AstCollectingVisitor().collectAndGetAstText(getDocument());
153   }
154 }
