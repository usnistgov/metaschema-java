51    @Override
52    protected String visitChildren(IExpression expr, State context) {
53      context.push();
54      String result = super.visitChildren(expr, context);
55      context.pop();
56      return result;
57    }
58  
59    @Override
60    protected String aggregateResult(String result, String nextResult, State context) {
61      StringBuilder buffer = new StringBuilder();
62      if (result != null) {
63        buffer.append(result);
64        // buffer.append(" ar "+System.lineSeparator());
65      }
66  
67      buffer.append(context.getIndentation())
68          .append(nextResult);
69      return buffer.toString();
70    }
71  
72    @Override
73    protected String defaultResult() {
74      return "";
75    }
76  
77    /**
78     * Append the {@code childResult} to the record produced for the current node.
79     * 
80     * @param expr
81     *          the current node
82     * @param childResult
83     *          the output generated for the curren't node's children
84     * @param context
85     *          the output context state
86     * @return the string representation of the node tree for the current node and its children
87     */
88    @SuppressWarnings("static-method")
89    protected String appendNode(IExpression expr, String childResult, State context) {
90      StringBuilder buffer = new StringBuilder();
91      buffer.append(context.getIndentation())
92          .append(expr.toASTString());
93      if (childResult != null) {
94        buffer.append(System.lineSeparator())
95            .append(childResult);
96      }
97      return buffer.toString();
98    }

104    *          the node to build the node tree for
105    * @return the string representation of the node tree for the provided expression node and its
106    *         children
107    */
108   public String visit(IExpression expression) {
109     return visit(expression, new State());

277   static class State {
278     private int indentation; // 0;
279     private int lastIndentation; // 0;
280     private String indentationPadding = "";
281 
282     public String getIndentation() {
283       if (indentation != lastIndentation) {
284         StringBuilder buffer = new StringBuilder();
285         for (int i = 0; i < indentation; i++) {
286           buffer.append("  ");
287         }
288         lastIndentation = indentation;
289         indentationPadding = buffer.toString();
290       }
291       return indentationPadding;
292     }
293 
294     public State push() {
295       indentation++;
296       return this;
297     }
298 
299     public State pop() {
300       indentation--;
301       return this;
302     }
303   }
