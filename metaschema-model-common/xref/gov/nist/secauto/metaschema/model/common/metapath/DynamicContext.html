54  public class DynamicContext { // NOPMD - intentional data class
55    @NonNull
56    private final StaticContext staticContext;
57    @NonNull
58    private final ZoneId implicitTimeZone;
59    @NonNull
60    private final ZonedDateTime currentDateTime;
61    @NonNull
62    private final Map<URI, IDocumentNodeItem> availableDocuments;
63    private final Map<CallingContext, ISequence<?>> functionResultCache;
64    private CachingLoader documentLoader;
65    @NonNull
66    private final IMutableConfiguration<MetapathEvaluationFeature> configuration;
67    @NonNull
68    private final Map<String, ISequence<?>> letVariableMap;
69  
70    @SuppressWarnings("null")
71    public DynamicContext(@NonNull StaticContext staticContext) {
72      this.staticContext = staticContext;
73  
74      Clock clock = Clock.systemDefaultZone();
75  
76      this.implicitTimeZone = clock.getZone();
77      this.currentDateTime = ZonedDateTime.now(clock);
78      this.availableDocuments = new HashMap<>();
79      this.functionResultCache = new HashMap<>();
80      this.configuration = new DefaultConfiguration<>(MetapathEvaluationFeature.class);
81      this.configuration.enableFeature(MetapathEvaluationFeature.METAPATH_EVALUATE_PREDICATES);
82      this.letVariableMap = new ConcurrentHashMap<>();
83    }

93    }

95    @NonNull
96    public ZonedDateTime getCurrentDateTime() {

134   private class CachingLoader implements IDocumentLoader {
135     @NonNull
136     private final IDocumentLoader proxy;
137 
138     public CachingLoader(@NonNull IDocumentLoader proxy) {
139       this.proxy = proxy;
140     }
141 
142     @Override
143     public @Nullable EntityResolver getEntityResolver() {
144       return new ContextEntityResolver();
145     }
146 
147     @Override
148     public void setEntityResolver(@NonNull EntityResolver resolver) {
149       // we delegate to the document loader proxy, so the resolver should be set there
150       throw new UnsupportedOperationException("Set the resolver on the proxy");
151     }
152 
153     protected IDocumentLoader getProxiedDocumentLoader() {
154       return proxy;
155     }
156 
157     @Override
158     public @NonNull IDocumentNodeItem loadAsNodeItem(@NonNull InputSource source) throws IOException {
159       String systemId = source.getSystemId();
160       URI uri = ObjectUtils.notNull(URI.create(systemId));
161       IDocumentNodeItem retval = availableDocuments.get(uri);
162       if (retval == null) {
163         retval = getProxiedDocumentLoader().loadAsNodeItem(source);
164         availableDocuments.put(uri, retval);
165       }
166       return retval;
167     }

169     public class ContextEntityResolver implements EntityResolver {
170 
171       /**
172        * Provides an {@link InputSource} for the provided {@code systemId} after attempting to resolve
173        * this system identifier.
174        * <p>
175        * This implementation of an {@link EntityResolver} will perform the following operations in order:
176        * <ol>
177        * <li>Resolves the {@code systemId} against the base URI provided by the
178        * {@link StaticContext#getBaseUri()} method, if this method returns a non-{@code null} result, to
179        * get a localized resource identifier.</li>
180        * <li>It will then delegate to the EntityResolver provided by the
181        * {@link IDocumentLoader#getEntityResolver()} method, if the result is not-{@code null}, to get the
182        * {@link InputSource}.</li>
183        * <li>If no InputSource is provided by the previous step, then an InputSource will be created from
184        * the URI resolved in the first step, if possible.
185        * <li>If an InputSource is still not provided, then an InputSource will be created from the
186        * provided {@code systemId}.
187        * </ol>
188        */
189       @Override
190       public InputSource resolveEntity(String publicId, String systemId) throws SAXException, IOException {
191         URI baseUri = getStaticContext().getBaseUri();
192 
193         String uri;
194         if (baseUri == null) {
195           uri = systemId;
196         } else {
197           URI resolvedUri = baseUri.resolve(systemId);
198           uri = resolvedUri.toASCIIString();
199         }
200 
201         EntityResolver resolver = getProxiedDocumentLoader().getEntityResolver();
202         InputSource retval = resolver == null ? null : resolver.resolveEntity(null, uri);
203         if (retval == null) {
204           retval = new InputSource(uri);
205         }
206 
207         return retval;
208       }
209 
210     }
211   }
